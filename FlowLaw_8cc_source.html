37  EnthalpyConverter::Ptr ec)

44  m_standard_gravity = config.get_number("constants.standard_gravity");
45  m_ideal_gas_constant = config.get_number("constants.ideal_gas_constant");
46 
47  m_rho = config.get_number("constants.ice.density");
48  m_beta_CC_grad = config.get_number("constants.ice.beta_Clausius_Clapeyron") * m_rho * m_standard_gravity;

52  m_hardness_power = -1.0 / m_n;

54  m_A_cold = config.get_number("flow_law.Paterson_Budd.A_cold");
55  m_A_warm = config.get_number("flow_law.Paterson_Budd.A_warm");
56  m_Q_cold = config.get_number("flow_law.Paterson_Budd.Q_cold");
57  m_Q_warm = config.get_number("flow_law.Paterson_Budd.Q_warm");
58  m_crit_temp = config.get_number("flow_law.Paterson_Budd.T_critical");
59 
60  double
61  schoofLen = config.get_number("flow_law.Schoof_regularizing_length", "m"),
62  schoofVel = config.get_number("flow_law.Schoof_regularizing_velocity", "m second-1");

85 double FlowLaw::softness_paterson_budd(double T_pa) const {
86  const double A = T_pa < m_crit_temp ? m_A_cold : m_A_warm;
87  const double Q = T_pa < m_crit_temp ? m_Q_cold : m_Q_warm;
88 
89  return A * exp(-Q / (m_ideal_gas_constant * T_pa));

142 //! \brief Computes the regularized effective viscosity and its derivative with respect to the
143 //! second invariant \f$ \gamma \f$.
144 /*!
145  *
146  * @f{align*}{
147  * \nu &= \frac{1}{2} B \left( \epsilon + \gamma \right)^{(1-n)/(2n)},\\
148  * \diff{\nu}{\gamma} &= \frac{1}{2} B \cdot \frac{1-n}{2n} \cdot \left(\epsilon + \gamma \right)^{(1-n)/(2n) - 1}, \\
149  * &= \frac{1-n}{2n} \cdot \frac{1}{2} B \left( \epsilon + \gamma \right)^{(1-n)/(2n)} \cdot \frac{1}{\epsilon + \gamma}, \\
150  * &= \frac{1-n}{2n} \cdot \frac{\nu}{\epsilon + \gamma}.
151  * @f}
152  * Here @f$ \gamma @f$ is the second invariant
153  * @f{align*}{
154  * \gamma &= \frac{1}{2} D_{ij} D_{ij}\\
155  * &= \frac{1}{2}\, ((u_x)^2 + (v_y)^2 + (u_x + v_y)^2 + \frac{1}{2}\, (u_y + v_x)^2) \\
156  * @f}
157  * and
158  * @f[ D_{ij}(\mathbf{u}) = \frac{1}{2}\left(\diff{u_{i}}{x_{j}} + \diff{u_{j}}{x_{i}}\right). @f]
159  *
160  * Either one of \c nu and \c dnu can be NULL if the corresponding output is not needed.
161  *
162  * \param[in] B ice hardness
163  * \param[in] gamma the second invariant
164  * \param[out] nu effective viscosity
165  * \param[out] dnu derivative of \f$ \nu \f$ with respect to \f$ \gamma \f$
166  */
167 void FlowLaw::effective_viscosity(double B, double gamma,
168  double *nu, double *dnu) const {
169  effective_viscosity(B, gamma, m_schoofReg, nu, dnu);
170 }

172 void FlowLaw::effective_viscosity(double B, double gamma, double eps,
173  double *nu, double *dnu) const {
174  const double
175  my_nu = 0.5 * B * pow(eps + gamma, m_viscosity_power);
176 
177  if (PetscLikely(nu != NULL)) {
178  *nu = my_nu;
179  }
180 
181  if (PetscLikely(dnu != NULL)) {
182  *dnu = m_viscosity_power * my_nu / (eps + gamma);
183  }
184 }

186 void averaged_hardness_vec(const FlowLaw &ice,
187  const IceModelVec2S &thickness,
188  const IceModelVec3 &enthalpy,
189  IceModelVec2S &result) {
190 
191  const IceGrid &grid = *thickness.grid();
192 
193  IceModelVec::AccessList list{&thickness, &result, &enthalpy};
194 
195  ParallelSection loop(grid.com);
196  try {
197  for (Points p(grid); p; p.next()) {
198  const int i = p.i(), j = p.j();
199 
200  // Evaluate column integrals in flow law at every quadrature point's column
201  double H = thickness(i,j);
202  const double *enthColumn = enthalpy.get_column(i, j);
203  result(i,j) = averaged_hardness(ice, H, grid.kBelowHeight(H),
204  &(grid.z()[0]), enthColumn);
205  }
206  } catch (...) {
207  loop.failed();
208  }
209  loop.check();
210 
211  result.update_ghosts();
212 }

214 //! Computes vertical average of `B(E, p)` ice hardness, namely @f$\bar B(E, p)@f$.
215 /*!
216  * See comment for hardness(). Note `E[0], ..., E[kbelowH]` must be valid.
217  */
218 double averaged_hardness(const FlowLaw &ice,
219  double thickness, int kbelowH,
220  const double *zlevels,
221  const double *enthalpy) {
222  double B = 0;
223 
224  EnthalpyConverter &EC = *ice.EC();
225 
226  // Use trapezoidal rule to integrate from 0 to zlevels[kbelowH]:
227  if (kbelowH > 0) {
228  double
229  p0 = EC.pressure(thickness),
230  E0 = enthalpy[0],
231  h0 = ice.hardness(E0, p0); // ice hardness at the left endpoint
232 
233  for (int i = 1; i <= kbelowH; ++i) { // note the "1" and the "<="
234  const double
235  p1 = EC.pressure(thickness - zlevels[i]), // pressure at the right endpoint
236  E1 = enthalpy[i], // enthalpy at the right endpoint
237  h1 = ice.hardness(E1, p1); // ice hardness at the right endpoint
238 
239  // The trapezoid rule sans the "1/2":
240  B += (zlevels[i] - zlevels[i-1]) * (h0 + h1);
241 
242  h0 = h1;
243  }
244  }
245 
246  // Add the "1/2":
247  B *= 0.5;
248 
249  // use the "rectangle method" to integrate from
250  // zlevels[kbelowH] to thickness:
251  double
252  depth = thickness - zlevels[kbelowH],
253  p = EC.pressure(depth);
254 
255  B += depth * ice.hardness(enthalpy[kbelowH], p);
256 
257  // Now B is an integral of ice hardness; next, compute the average:
258  if (thickness > 0) {
259  B = B / thickness;
260  } else {
261  B = 0;
262  }
263 
264  return B;
265 }
