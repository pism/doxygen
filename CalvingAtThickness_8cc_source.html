91 /**
92  * Updates ice cover mask and the ice thickness according to the
93  * calving rule removing ice at the shelf front that is thinner than a
94  * given threshold.
95  *
96  * @param[in] t beginning of the time step
97  * @param[in] dt length of the time step
98  * @param[in,out] pism_mask ice cover mask
99  * @param[in,out] ice_thickness ice thickness
100  *
101  * @return 0 on success
102  */
103 void CalvingAtThickness::update(double t,
104  double dt,
105  IceModelVec2CellType &pism_mask,
106  IceModelVec2S &ice_thickness) {
107 
108  m_calving_threshold->update(t, dt);
109  m_calving_threshold->average(t, dt);
110 
111  // this call fills ghosts of m_old_mask
112  m_old_mask.copy_from(pism_mask);
113 
114  const auto &threshold = *m_calving_threshold;
115 
116  IceModelVec::AccessList list{&pism_mask, &ice_thickness, &m_old_mask, &threshold};
117  for (Points p(*m_grid); p; p.next()) {
118  const int i = p.i(), j = p.j();
119 
120  if (m_old_mask.floating_ice(i, j) &&
121  m_old_mask.next_to_ice_free_ocean(i, j) &&
122  ice_thickness(i, j) < threshold(i, j)) {
123  ice_thickness(i, j) = 0.0;
124  pism_mask(i, j) = MASK_ICE_FREE_OCEAN;
125  }
126  }
127 
128  pism_mask.update_ghosts();
129  ice_thickness.update_ghosts();
130 }
