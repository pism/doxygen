96 //! Update ice enthalpy field based on conservation of energy.
97 /*!
98 This method is documented by the page \ref bombproofenth and by [\ref
99 AschwandenBuelerKhroulevBlatter].
100 
101 This method updates IceModelVec3 m_work and IceModelVec2S basal_melt_rate.
102 No communication of ghosts is done for any of these fields.
103 
104 We use an instance of enthSystemCtx.
105 
106 Regarding drainage, see [\ref AschwandenBuelerKhroulevBlatter] and references therein.

109 void EnthalpyModel::update_impl(double t, double dt, const Inputs &inputs) {
110  // current time does not matter here
111  (void) t;
112 
113  EnthalpyConverter::Ptr EC = m_grid->ctx()->enthalpy_converter();
114 
115  const double
116  ice_density = m_config->get_number("constants.ice.density"), // kg m-3
117  bulgeEnthMax = m_config->get_number("energy.enthalpy.cold_bulge_max"), // J kg-1
118  target_water_fraction = m_config->get_number("energy.drainage_target_water_fraction");
119 
120  energy::DrainageCalculator dc(*m_config);
121 
122  inputs.check();
123 
124  // give them names that are a bit shorter...
125  const IceModelVec3
126  &strain_heating3 = *inputs.volumetric_heating_rate,
127  &u3 = *inputs.u3,
128  &v3 = *inputs.v3,
129  &w3 = *inputs.w3;
130 
131  const IceModelVec2CellType &cell_type = *inputs.cell_type;
132 
133  const IceModelVec2S
134  &basal_frictional_heating = *inputs.basal_frictional_heating,
135  &basal_heat_flux = *inputs.basal_heat_flux,
136  &ice_thickness = *inputs.ice_thickness,
137  &surface_liquid_fraction = *inputs.surface_liquid_fraction,
138  &shelf_base_temp = *inputs.shelf_base_temp,
139  &ice_surface_temp = *inputs.surface_temp,
140  &till_water_thickness = *inputs.till_water_thickness;
141 
142  energy::enthSystemCtx system(m_grid->z(), "energy.enthalpy", m_grid->dx(), m_grid->dy(), dt,
143  *m_config, m_ice_enthalpy, u3, v3, w3, strain_heating3, EC);
144 
145  const size_t Mz_fine = system.z().size();
146  const double dz = system.dz();
147  std::vector<double> Enthnew(Mz_fine); // new enthalpy in column
148 
149  IceModelVec::AccessList list{&ice_surface_temp, &shelf_base_temp, &surface_liquid_fraction,
150  &ice_thickness, &basal_frictional_heating, &basal_heat_flux, &till_water_thickness,
151  &cell_type, &u3, &v3, &w3, &strain_heating3, &m_basal_melt_rate, &m_ice_enthalpy,
152  &m_work};
153 
154  double margin_threshold = m_config->get_number("energy.margin_ice_thickness_limit");
155 
156  unsigned int liquifiedCount = 0;
157 
158  ParallelSection loop(m_grid->com);
159  try {
160  for (Points pt(*m_grid); pt; pt.next()) {
161  const int i = pt.i(), j = pt.j();
162 
163  const double H = ice_thickness(i, j);
164 
165  system.init(i, j,
166  marginal(ice_thickness, i, j, margin_threshold),
167  H);
168 
169  // enthalpy and pressures at top of ice
170  const double
171  depth_ks = H - system.ks() * dz,
172  p_ks = EC->pressure(depth_ks); // FIXME issue #15
173 
174  const double Enth_ks = EC->enthalpy_permissive(ice_surface_temp(i, j),
175  surface_liquid_fraction(i, j), p_ks);
176 
177  const bool ice_free_column = (system.ks() == 0);
178 
179  // deal completely with columns with no ice; enthalpy and basal_melt_rate need setting
180  if (ice_free_column) {
181  m_work.set_column(i, j, Enth_ks);
182  // The floating basal melt rate will be set later; cover this
183  // case and set to zero for now. Also, there is no basal melt
184  // rate on ice free land and ice free ocean
185  m_basal_melt_rate(i, j) = 0.0;
186  continue;
187  } // end of if (ice_free_column)
188 
189  if (system.lambda() < 1.0) {
190  m_stats.reduced_accuracy_counter += 1; // count columns with lambda < 1
191  }
192 
193  const bool
194  is_floating = cell_type.ocean(i, j),
195  base_is_warm = system.Enth(0) >= system.Enth_s(0),
196  above_base_is_warm = system.Enth(1) >= system.Enth_s(1);
197 
198  // set boundary conditions and update enthalpy
199  {
200  system.set_surface_dirichlet_bc(Enth_ks);
201 
202  // determine lowest-level equation at bottom of ice; see
203  // decision chart in the source code browser and page
204  // documenting BOMBPROOF
205  if (is_floating) {
206  // floating base: Dirichlet application of known temperature from ocean
207  // coupler; assumes base of ice shelf has zero liquid fraction
208  double Enth0 = EC->enthalpy_permissive(shelf_base_temp(i, j), 0.0, EC->pressure(H));
209 
210  system.set_basal_dirichlet_bc(Enth0);
211  } else {
212  // grounded ice warm and wet
213  if (base_is_warm && (till_water_thickness(i, j) > 0.0)) {
214  if (above_base_is_warm) {
215  // temperate layer at base (Neumann) case: q . n = 0 (K0 grad E . n = 0)
216  system.set_basal_heat_flux(0.0);
217  } else {
218  // only the base is warm: E = E_s(p) (Dirichlet)
219  // ( Assumes ice has zero liquid fraction. Is this a valid assumption here?
220  system.set_basal_dirichlet_bc(system.Enth_s(0));
221  }
222  } else {
223  // (Neumann) case: q . n = q_lith . n + F_b
224  // a) cold and dry base, or
225  // b) base that is still warm from the last time step, but without basal water
226  system.set_basal_heat_flux(basal_heat_flux(i, j) + basal_frictional_heating(i, j));
227  }
228  }
229 
230  // solve the system
231  system.solve(Enthnew);
232 
233  }
234 
235  // post-process (drainage and bulge-limiting)
236  double Hdrainedtotal = 0.0;
237  double Hfrozen = 0.0;
238  {
239  // drain ice segments by mechanism in [\ref AschwandenBuelerKhroulevBlatter],
240  // using DrainageCalculator dc
241  for (unsigned int k=0; k < system.ks(); k++) {
242  if (Enthnew[k] > system.Enth_s(k)) { // avoid doing any more work if cold
243 
244  const double
245  depth = H - k * dz,
246  p = EC->pressure(depth), // FIXME issue #15
247  T_m = EC->melting_temperature(p),
248  L = EC->L(T_m);
249 
250  if (Enthnew[k] >= system.Enth_s(k) + 0.5 * L) {
251  liquifiedCount++; // count these rare events...
252  Enthnew[k] = system.Enth_s(k) + 0.5 * L; // but lose the energy
253  }
254 
255  double omega = EC->water_fraction(Enthnew[k], p);
256 
257  if (omega > target_water_fraction) {
258  double fractiondrained = dc.get_drainage_rate(omega) * dt; // pure number
259 
260  fractiondrained = std::min(fractiondrained,
261  omega - target_water_fraction);
262  Hdrainedtotal += fractiondrained * dz; // always a positive contribution
263  Enthnew[k] -= fractiondrained * L;
264  }
265  }
266  }
267 
268  // apply bulge limiter
269  const double lowerEnthLimit = Enth_ks - bulgeEnthMax;
270  for (unsigned int k=0; k < system.ks(); k++) {
271  if (Enthnew[k] < lowerEnthLimit) {
272  // Count grid points which have very large cold limit advection bulge... enthalpy not
273  // too low.
274  m_stats.bulge_counter += 1;
275  Enthnew[k] = lowerEnthLimit;
276  }
277  }
278 
279  // if there is subglacial water, don't allow ice base enthalpy to be below
280  // pressure-melting; that is, assume subglacial water is at the pressure-
281  // melting temperature and enforce continuity of temperature
282  {
283  if (Enthnew[0] < system.Enth_s(0) && till_water_thickness(i,j) > 0.0) {
284  const double E_difference = system.Enth_s(0) - Enthnew[0];
285 
286  const double depth = H,
287  pressure = EC->pressure(depth),
288  T_m = EC->melting_temperature(pressure);
289 
290  Enthnew[0] = system.Enth_s(0);
291  // This adjustment creates energy out of nothing. We will
292  // freeze some basal water, subtracting an equal amount of
293  // energy, to make up for it.
294  //
295  // Note that [E_difference] = J/kg, so
296  //
297  // U_difference = E_difference * ice_density * dx * dy * (0.5*dz)
298  //
299  // is the amount of energy created (we changed enthalpy of
300  // a block of ice with the volume equal to
301  // dx*dy*(0.5*dz); note that the control volume
302  // corresponding to the grid point at the base of the
303  // column has thickness 0.5*dz, not dz).
304  //
305  // Also, [L] = J/kg, so
306  //
307  // U_freeze_on = L * ice_density * dx * dy * Hfrozen,
308  //
309  // is the amount of energy created by freezing a water
310  // layer of thickness Hfrozen (using units of ice
311  // equivalent thickness).
312  //
313  // Setting U_difference = U_freeze_on and solving for
314  // Hfrozen, we find the thickness of the basal water layer
315  // we need to freeze co restore energy conservation.
316 
317  Hfrozen = E_difference * (0.5*dz) / EC->L(T_m);
318  }
319  }
320 
321  } // end of post-processing
322 
323  // compute basal melt rate
324  {
325  bool base_is_cold = (Enthnew[0] < system.Enth_s(0)) && (till_water_thickness(i,j) == 0.0);
326  // Determine melt rate, but only preliminarily because of
327  // drainage, from heat flux out of bedrock, heat flux into
328  // ice, and frictional heating
329  if (is_floating) {
330  // The floating basal melt rate will be set later; cover
331  // this case and set to zero for now. Note that
332  // Hdrainedtotal is discarded (the ocean model determines
333  // the basal melt).
334  m_basal_melt_rate(i, j) = 0.0;
335  } else {
336  if (base_is_cold) {
337  m_basal_melt_rate(i, j) = 0.0; // zero melt rate if cold base
338  } else {
339  const double
340  p_0 = EC->pressure(H),
341  p_1 = EC->pressure(H - dz), // FIXME issue #15
342  Tpmp_0 = EC->melting_temperature(p_0);
343 
344  const bool k1_istemperate = EC->is_temperate(Enthnew[1], p_1); // level z = + \Delta z
345  double hf_up = 0.0;
346  if (k1_istemperate) {
347  const double
348  Tpmp_1 = EC->melting_temperature(p_1);
349 
350  hf_up = -system.k_from_T(Tpmp_0) * (Tpmp_1 - Tpmp_0) / dz;
351  } else {
352  double T_0 = EC->temperature(Enthnew[0], p_0);
353  const double K_0 = system.k_from_T(T_0) / EC->c();
354 
355  hf_up = -K_0 * (Enthnew[1] - Enthnew[0]) / dz;
356  }
357 
358  // compute basal melt rate from flux balance:
359  //
360  // basal_melt_rate = - Mb / rho in [\ref AschwandenBuelerKhroulevBlatter];
361  //
362  // after we compute it we make sure there is no refreeze if
363  // there is no available basal water
364  m_basal_melt_rate(i, j) = (basal_frictional_heating(i, j) + basal_heat_flux(i, j) - hf_up) / (ice_density * EC->L(Tpmp_0));
365 
366  if (till_water_thickness(i, j) <= 0 && m_basal_melt_rate(i, j) < 0) {
367  m_basal_melt_rate(i, j) = 0.0;
368  }
369  }
370 
371  // Add drained water from the column to basal melt rate.
372  m_basal_melt_rate(i, j) += (Hdrainedtotal - Hfrozen) / dt;
373  } // end of the grounded case
374  } // end of the basal melt rate computation
375 
376  system.fine_to_coarse(Enthnew, i, j, m_work);
377  }
378  } catch (...) {
379  loop.failed();
380  }
381  loop.check();
382 
383  m_stats.liquified_ice_volume = ((double) liquifiedCount) * dz * m_grid->cell_area();
384 }
