37 //! \brief Report the range of a time-series stored in `data`.
38 static void report_range(const std::vector<double> &data,
39  const units::System::Ptr unit_system,
40  const VariableMetadata &metadata,
41  const Logger &log) {
42  double min, max;
43 
44  // min_element and max_element return iterators; "*" is used to get
45  // the value corresponding to this iterator
46  min = *std::min_element(data.begin(), data.end());
47  max = *std::max_element(data.begin(), data.end());
48 
49  units::Converter c(unit_system,
50  metadata.get_string("units"),
51  metadata.get_string("glaciological_units"));
52  min = c(min);
53  max = c(max);
54 
55  std::string spacer(metadata.get_name().size(), ' ');
56 
57  log.message(2,
58  " FOUND %s / %-60s\n"
59  " %s \\ min,max = %9.3f,%9.3f (%s)\n",
60  metadata.get_name().c_str(),
61  metadata.get_string("long_name").c_str(), spacer.c_str(), min, max,
62  metadata.get_string("glaciological_units").c_str());
63 }

65 void ScalarForcing::initialize(const Context &ctx,
66  const std::string &filename,
67  const std::string &variable_name,
68  const std::string &units,
69  const std::string &glaciological_units,
70  const std::string &long_name,
71  bool periodic) {
72  try {
73  auto unit_system = ctx.unit_system();
74 
75  VariableMetadata variable(variable_name, unit_system);
76 
77  variable["units"] = units;
78  variable["glaciological_units"] = glaciological_units;
79  variable["long_name"] = long_name;
80 
81  double forcing_t0{};
82  double forcing_t1{};
83 
84  // Read data from a NetCDF file
85  {
86  ctx.log()->message(2,
87  " reading %s (%s) from file '%s'...\n",
88  long_name.c_str(), variable_name.c_str(), filename.c_str());
89 
90  File file(ctx.com(), filename, PISM_NETCDF3, PISM_READONLY);
91 
92  // Read forcing data. The read_timeseries() call will ensure that variable_name is a
93  // scalar variable.
94  std::vector<double> data{};
95  io::read_timeseries(file, variable, *ctx.log(), data);
96 
97  // The following line relies on the fact that this is a scalar variable.
98  std::string time_name = file.dimensions(variable_name)[0];
99 
100  std::vector<double> times{};
101  std::vector<double> bounds{};
102  io::read_time_info(*ctx.log(), unit_system,
103  file, time_name, ctx.time()->units_string(),
104  times, bounds);
105  size_t N = times.size();
106 
107  // Compute values used to extend data read from file
108  //
109  // Initialize using values corresponding to constant extrapolation:
110  double v0 = data.front();
111  double v1 = data.back();
112  if (periodic) {
113  double b0 = bounds.front();
114  double b1 = bounds.back();
115 
116  // compute the value at the beginning and end of the period:
117  {
118  double dt1 = times.front() - b0;
119  double dt2 = b1 - times.back();
120 
121  // interpolation factor
122  double alpha = dt1 + dt2 > 0 ? dt2 / (dt1 + dt2) : 0.0;
123 
124  v0 = (1.0 - alpha) * data.back() + alpha * data.front();
125  v1 = v0;
126  }
127 
128  {
129  m_period = b1 - b0;
130  m_period_start = b0;
131 
132  assert(m_period > 0.0);
133  }
134  }
135 
136  // Note: this should take care of file with one record as well.
137  m_times.clear();
138  m_values.clear();
139  {
140  if (bounds.front() < times.front()) {
141  m_times.emplace_back(bounds.front());
142  m_values.emplace_back(v0);
143  }
144  for (size_t k = 0; k < N; ++k) {
145  m_times.emplace_back(times[k]);
146  m_values.emplace_back(data[k]);
147  }
148  if (bounds.back() > times.back()) {
149  m_times.emplace_back(bounds.back());
150  m_values.emplace_back(v1);
151  }
152  }
153 
154  // Save the time interval covered by this forcing.
155  forcing_t0 = bounds.front();
156  forcing_t1 = bounds.back();
157  } // end of the block initializing data
158 
159  // validate resulting times
160  if (not is_increasing(m_times)) {
161  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
162  "m_times have to be strictly increasing (this is a bug)");
163  }
164 
165  report_range(m_values, unit_system, variable, *ctx.log());
166 
167  // Set up interpolation.
168  {
169  m_acc = gsl_interp_accel_alloc();
170  m_spline = gsl_spline_alloc(gsl_interp_linear, m_times.size());
171  gsl_spline_init(m_spline, m_times.data(), m_values.data(), m_times.size());
172  }
173 
174  bool extrapolate = ctx.config()->get_flag("input.forcing.time_extrapolation");
175 
176  if (not (extrapolate or periodic)) {
177  check_forcing_duration(*ctx.time(), forcing_t0, forcing_t1);
178  }
179 
180  } catch (RuntimeError &e) {
181  e.add_context("reading %s (%s) from '%s'",
182  long_name.c_str(), variable_name.c_str(), filename.c_str());
183  throw;
184  }
185 }

237 double ScalarForcing::value(double t) const {
238  double T = t;
239 
240  if (m_period > 0.0) {
241  // compute time since the period start
242  T -= m_period_start;
243 
244  // remove an integer number of periods
245  T -= std::floor(T / m_period) * m_period;
246 
247  // add the period start back
248  T += m_period_start;
249  }
250 
251  if (T > m_times.back()) {
252  return m_values.back();
253  }
254 
255  if (T < m_times.front()) {
256  return m_values.front();
257  }
258 
259  return gsl_spline_eval(m_spline, T, m_acc);
260 }

262 // Integrate from a to b, interpreting data as *not* periodic
263 double ScalarForcing::integral(double a, double b) const {
264  assert(b >= a);
265 
266  double dt = b - a;
267 
268  double t0 = m_times.front();
269  double t1 = m_times.back();
270  double v0 = m_values[0];
271  double v1 = m_values.back();
272 
273  // both points are to the left of [t0, t1]
274  if (b <= t0) {
275  return v0 * (b - a);
276  }
277 
278  // both points are to the right of [t0, t1]
279  if (a >= t1) {
280  return v1 * (b - a);
281  }
282 
283  // a is to the left of [t0, t1]
284  if (a < t0) {
285  return v0 * (t0 - a) + integral(t0, b);
286  }
287 
288  // b is to the right of [t0, t1]
289  if (b > t1) {
290  return integral(a, t1) + v1 * (b - t1);
291  }
292 
293  // both points are inside [t0, t1]
294  size_t ai = gsl_interp_bsearch(m_times.data(), a, 0, m_times.size() - 1);
295  size_t bi = gsl_interp_bsearch(m_times.data(), b, 0, m_times.size() - 1);
296 
297  // gsl_interp_bsearch() above returns the index i of the array ‘m_times’ such that
298  // ‘m_times[i] <= t < m_times[i+1]’. The index is searched for in the
299  // range [0, m_times.size() - 1] (inclusive).
300 
301  double v_a = gsl_spline_eval(m_spline, a, m_acc);
302  double v_b = gsl_spline_eval(m_spline, b, m_acc);
303 
304  if (ai == bi) {
305  return 0.5 * (v_a + v_b) * dt;
306  }
307 
308  double result = 0.0;
309  // integrate from a to the data point just to its right
310  result += 0.5 * (v_a + m_values[ai + 1]) * (m_times[ai + 1] - a);
311 
312  // integrate over (possibly zero) intervals between a and b
313  for (size_t k = ai + 1; k < bi; ++k) {
314  result += 0.5 * (m_values[k] + m_values[k + 1]) * (m_times[k + 1] - m_times[k]);
315  }
316 
317  result += 0.5 * (m_values[bi] + v_b) * (b - m_times[bi]);
318 
319  return result;
320 }

322 double ScalarForcing::average(double t, double dt) const {
323  if (dt == 0.0) {
324  return value(t);
325  }
326 
327  if (dt < 0.0) {
328  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "negative interval length");
329  }
330 
331  if (m_period > 0.0) {
332  double a = t;
333  double b = t + dt;
334  double t0 = m_period_start;
335  double P = m_period;
336 
337  double N = std::floor((a - t0) / P);
338  double M = std::floor((b - t0) / P);
339  double delta = a - t0 - P * N;
340  double gamma = b - t0 - P * M;
341 
342  double N_periods = M - (N + 1);
343 
344  if (N_periods >= 0.0) {
345  return (integral(t0 + delta, t0 + P) +
346  N_periods * integral(t0, t0 + P) +
347  integral(t0, t0 + gamma)) / dt;
348  }
349 
350  return integral(t0 + delta, t0 + gamma) / dt;
351 
352  } else {
353  return integral(t, t + dt) / dt;
354  }
355 }
