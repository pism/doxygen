68 void generate_trough_topography(IceModelVec2S &result) {
69  // computation based on code by Tony Payne, 6 March 1997:
70  // http://homepages.vub.ac.be/~phuybrec/eismint/topog2.f
71 
72  IceGrid::ConstPtr grid = result.grid();
73 
74  const double
75  b0 = 1000.0, // plateau elevation
76  L = 750.0e3, // half-width of computational domain
77  w = 200.0e3, // trough width
78  slope = b0 / L,
79  dx61 = (2.0 * L) / 60; // = 25.0e3
80 
81  IceModelVec::AccessList list(result);
82  for (Points p(*grid); p; p.next()) {
83  const int i = p.i(), j = p.j();
84 
85  const double nsd = i * grid->dx(), ewd = j * grid->dy();
86  if ((nsd >= (27 - 1) * dx61) && (nsd <= (35 - 1) * dx61) &&
87  (ewd >= (31 - 1) * dx61) && (ewd <= (61 - 1) * dx61)) {
88  result(i,j) = 1000.0 - std::max(0.0, slope * (ewd - L) * cos(M_PI * (nsd - L) / w));
89  } else {
90  result(i,j) = 1000.0;
91  }
92  }
93 }

95 void generate_mound_topography(IceModelVec2S &result) {
96  // computation based on code by Tony Payne, 6 March 1997:
97  // http://homepages.vub.ac.be/~phuybrec/eismint/topog2.f
98 
99  IceGrid::ConstPtr grid = result.grid();
100 
101  const double slope = 250.0;
102  const double w = 150.0e3; // mound width
103 
104  IceModelVec::AccessList list(result);
105  for (Points p(*grid); p; p.next()) {
106  const int i = p.i(), j = p.j();
107 
108  const double nsd = i * grid->dx(), ewd = j * grid->dy();
109  result(i,j) = fabs(slope * sin(M_PI * ewd / w) + slope * cos(M_PI * nsd / w));
110  }
111 }

113 void IceEISModel::initialize_2d() {
114 
115  m_log->message(2,
116  "initializing variables from EISMINT II experiment %c formulas... \n",
117  m_experiment);
118 
119  // set bed topography
120  switch (m_experiment) {
121  case 'I':
122  case 'J':
123  generate_trough_topography(m_geometry.bed_elevation);
124  break;
125  case 'K':
126  case 'L':
127  generate_mound_topography(m_geometry.bed_elevation);
128  break;
129  default:
130  m_geometry.bed_elevation.set(0.0);
131  break;
132  }
133 
134  m_geometry.sea_level_elevation.set(0.0);
135 
136  // set uplift
137  IceModelVec2S bed_uplift(m_grid, "uplift", WITHOUT_GHOSTS);
138  bed_uplift.set(0.0);
139 
140  // start with zero ice
141  m_geometry.ice_thickness.set(0.0);
142 
143  m_beddef->bootstrap(m_geometry.bed_elevation, bed_uplift, m_geometry.ice_thickness,
144  m_geometry.sea_level_elevation);
145 }
