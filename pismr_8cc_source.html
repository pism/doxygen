42 static void set_eismint2_config_defaults(Config &config) {
43  config.set_number("grid.Lx", 750e3);
44  config.set_number("grid.Ly", 750e3);
45  config.set_string("grid.periodicity", "none");
46  config.set_string("grid.registration", "corner");
47  config.set_string("stress_balance.sia.flow_law", "pb");
48 
49  config.set_flag("energy.temperature_based", true);
50 
51  // Set sea level elevation to -1e4 meters to remove ocean interaction
52  config.set_number("sea_level.constant.value", -1e4);
53 
54  // purely SIA, and E=1
55  config.set_number("stress_balance.sia.enhancement_factor", 1.0);
56 
57  // none use bed smoothing & bed roughness parameterization
58  config.set_number("stress_balance.sia.bed_smoother.range", 0.0);
59 
60  // basal melt does not change computation of mass continuity or vertical velocity:
61  config.set_flag("geometry.update.use_basal_melt_rate", false);
62 
63  // Make bedrock thermal material properties into ice properties. Note that
64  // zero thickness bedrock layer is the default, but we want the ice/rock
65  // interface segment to have geothermal flux applied directly to ice without
66  // jump in material properties at base.
67  config.set_number("energy.bedrock_thermal.density",
68  config.get_number("constants.ice.density"));
69  config.set_number("energy.bedrock_thermal.conductivity",
70  config.get_number("constants.ice.thermal_conductivity"));
71  config.set_number("energy.bedrock_thermal.specific_heat_capacity",
72  config.get_number("constants.ice.specific_heat_capacity"));
73 
74  // no sliding + SIA
75  config.set_string("stress_balance.model", "sia");
76 }

78 int main(int argc, char *argv[]) {
79 
80  MPI_Comm com = MPI_COMM_WORLD;
81  petsc::Initializer petsc(argc, argv, help);
82 
83  com = PETSC_COMM_WORLD;
84 
85  try {
86  // Note: EISMINT II experiments G and H are not supported.
87  auto eisII = options::Keyword("-eisII",
88  "EISMINT II experiment name",
89  "A,B,C,D,E,F,I,J,K,L", "A");
90 
91  std::shared_ptr<Context> ctx = context_from_options(com, "pismr", false);
92 
93  Logger::Ptr log = ctx->log();
94  Config::Ptr config = ctx->config();
95 
96  std::vector<std::string> required_options{};
97  if (eisII.is_set()) {
98  // set defaults:
99  set_eismint2_config_defaults(*config);
100 
101  // process -config_override
102  DefaultConfig::Ptr overrides(new DefaultConfig(com,
103  "pism_overrides",
104  "-config_override",
105  ctx->unit_system()));
106  overrides->init(*ctx->log());
107  config->import_from(*overrides);
108  // process command-line options
109  set_config_from_options(ctx->unit_system(), *config);
110  } else {
111  required_options.emplace_back("-i");
112  }
113 
114  print_config(*ctx->log(), 3, *config);
115 
116  std::string usage =
117  " pismr -i IN.nc [-bootstrap] [-regional] [OTHER PISM & PETSc OPTIONS]\n"
118  "where:\n"
119  " -i IN.nc is input file in NetCDF format: contains PISM-written model state\n"
120  " -bootstrap enable heuristics to produce an initial state from an incomplete input\n"
121  " -regional enable \"regional mode\"\n"
122  " -eisII [experiment] enable EISMINT II mode\n"
123  "notes:\n"
124  " * option -i is required\n"
125  " * if -bootstrap is used then also '-Mx A -My B -Mz C -Lz D' are required\n";
126  {
127  bool done = show_usage_check_req_opts(*log, "PISMR (basic evolution run mode)" ,
128  required_options, usage);
129  if (done) {
130  return 0;
131  }
132  }
133 
134  options::String profiling_log = options::String("-profile",
135  "Save detailed profiling data to a file.");
136 
137  if (profiling_log.is_set()) {
138  ctx->profiling().start();
139  }
140 
141  IceGrid::Ptr grid;
142  std::unique_ptr<IceModel> model;
143 
144  if (options::Bool("-regional", "enable regional (outlet glacier) mode")) {
145  grid = regional_grid_from_options(ctx);
146  model.reset(new IceRegionalModel(grid, ctx));
147  } else {
148  grid = IceGrid::FromOptions(ctx);
149 
150  if (eisII.is_set()) {
151  char experiment = eisII.value()[0];
152  model.reset(new IceEISModel(grid, ctx, experiment));
153  } else {
154  model.reset(new IceModel(grid, ctx));
155  }
156  }
157 
158  model->init();
159 
160  const bool
161  list_ascii = options::Bool("-list_diagnostics",
162  "List available diagnostic quantities and stop"),
163  list_json = options::Bool("-list_diagnostics_json",
164  "List available diagnostic quantities (JSON format) and stop");
165 
166  if (list_ascii) {
167  model->list_diagnostics();
168  } else if (list_json) {
169  model->list_diagnostics_json();
170  } else {
171  model->run();
172 
173  log->message(2, "... done with run\n");
174 
175  model->save_results();
176  }
177  print_unused_parameters(*log, 3, *config);
178 
179  if (profiling_log.is_set()) {
180  ctx->profiling().report(profiling_log);
181  }
182  }
183  catch (...) {
184  handle_fatal_errors(com);
185  return 1;
186  }
187 
188  return 0;
189 }
