51 //! @brief In a single call a driver program can provide a usage string to
52 //! the user and check if required options are given, and if not, end.
53 bool show_usage_check_req_opts(const Logger &log,
54  const std::string &execname,
55  const std::vector<std::string> &required_options,
56  const std::string &usage) {
57  const bool keep_running = false;
58  const bool terminate = true;
59 
60  log.message(2, "%s %s\n", execname.c_str(), pism::revision);
61 
62  if (options::Bool("-version", "stop after printing print PISM version")) {
63  log.message(2, pism::version());
64  return terminate;
65  }
66 
67  if (options::Bool("-usage", "print PISM usage")) {
68  show_usage(log, execname, usage);
69  return terminate;
70  }
71 
72  // go through list of required options, and if not given, fail
73  bool req_absent = false;
74  for (const auto &opt : required_options) {
75  if (not options::Bool(opt, "a required option")) {
76  req_absent = true;
77  log.error("PISM ERROR: option %s required\n", opt.c_str());
78  }
79  }
80 
81  if (req_absent) {
82  log.error("\n");
83  show_usage(log, execname, usage);
84  return terminate;
85  }
86 
87  // show usage message with -help, but don't stop
88  if (options::Bool("-help", "print help on all options")) {
89  show_usage(log, execname, usage);
90  }
91  return keep_running;
92 }
