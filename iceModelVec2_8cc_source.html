88 //! Sets an IceModelVec2 to the magnitude of a 2D vector field with components `v_x` and `v_y`.
89 /*! Computes the magnitude \b pointwise, so any of v_x, v_y and the IceModelVec
90  this is called on can be the same.
91 
92  Does not communicate.
93  */
94 void IceModelVec2S::set_to_magnitude(const IceModelVec2S &v_x,
95  const IceModelVec2S &v_y) {
96  IceModelVec::AccessList list{this, &v_x, &v_y};
97 
98  for (Points p(*m_impl->grid); p; p.next()) {
99  const int i = p.i(), j = p.j();
100 
101  (*this)(i,j) = sqrt(PetscSqr(v_x(i,j)) + PetscSqr(v_y(i,j)));
102  }
103 
104  inc_state_counter(); // mark as modified
105 }

134 void IceModelVec2::write_impl(const File &file) const {
135 
136  assert(m_impl->v != NULL);
137 
138  // The simplest case:
139  if ((m_impl->dof == 1) and (not m_impl->ghosted)) {
140  IceModelVec::write_impl(file);
141  return;
142  }
143 
144  // Get the dof=1, stencil_width=0 DMDA (components are always scalar
145  // and we just need a global Vec):
146  auto da2 = m_impl->grid->get_dm(1, 0);
147 
148  // a temporary one-component vector, distributed across processors
149  // the same way v is
150  petsc::TemporaryGlobalVec tmp(da2);
151 
152  Logger::ConstPtr log = m_impl->grid->ctx()->log();
153  log->message(4, " Writing %s...\n", m_impl->name.c_str());
154 
155  for (unsigned int j = 0; j < m_impl->dof; ++j) {
156  IceModelVec2::get_dof(da2, tmp, j);
157 
158  petsc::VecArray tmp_array(tmp);
159  io::write_spatial_variable(m_impl->metadata[j], *m_impl->grid, file,
160  tmp_array.get());
161  }

236 //! \brief View a 2D field.
237 void IceModelVec2::view(int viewer_size) const {
238  std::shared_ptr<petsc::Viewer> viewers[2];
239 
240  if (m_impl->dof > 2) {
241  throw RuntimeError(PISM_ERROR_LOCATION, "dof > 2 is not supported");
242  }
243 
244  for (unsigned int j = 0; j < std::min(m_impl->dof, 2U); ++j) {
245  std::string
246  c_name = m_impl->metadata[j].get_name(),
247  long_name = m_impl->metadata[j].get_string("long_name"),
248  glaciological_units = m_impl->metadata[j].get_string("glaciological_units"),
249  title = long_name + " (" + glaciological_units + ")";
250 
251  if (not m_impl->map_viewers[c_name]) {
252  m_impl->map_viewers[c_name].reset(new petsc::Viewer(m_impl->grid->com, title, viewer_size,
253  m_impl->grid->Lx(), m_impl->grid->Ly()));
254  }
255 
256  viewers[j] = m_impl->map_viewers[c_name];
257  }
258 
259  view(viewers[0], viewers[1]);
260 }

262 //! \brief View a 2D vector field using existing PETSc viewers.
263 void IceModelVec2::view(std::shared_ptr<petsc::Viewer> v1, std::shared_ptr<petsc::Viewer> v2) const {
264  PetscErrorCode ierr;
265 
266  std::shared_ptr<petsc::Viewer> viewers[2] = {v1, v2};
267 
268  // Get the dof=1, stencil_width=0 DMDA (components are always scalar
269  // and we just need a global Vec):
270  auto da2 = m_impl->grid->get_dm(1, 0);
271 
272  petsc::TemporaryGlobalVec tmp(da2);
273 
274  for (unsigned int i = 0; i < std::min(m_impl->dof, 2U); ++i) {
275  std::string
276  long_name = m_impl->metadata[i].get_string("long_name"),
277  units = m_impl->metadata[i].get_string("units"),
278  glaciological_units = m_impl->metadata[i].get_string("glaciological_units"),
279  title = long_name + " (" + glaciological_units + ")";
280 
281  if (not (bool)viewers[i]) {
282  continue;
283  }
284 
285  PetscViewer v = *viewers[i].get();
286 
287  PetscDraw draw = NULL;
288  ierr = PetscViewerDrawGetDraw(v, 0, &draw);
289  PISM_CHK(ierr, "PetscViewerDrawGetDraw");
290 
291  ierr = PetscDrawSetTitle(draw, title.c_str());
292  PISM_CHK(ierr, "PetscDrawSetTitle");
293 
294  IceModelVec2::get_dof(da2, tmp, i);
295 
296  convert_vec(tmp, m_impl->metadata[i].unit_system(),
297  units, glaciological_units);
298 
299  double bounds[2] = {0.0, 0.0};
300  ierr = VecMin(tmp, NULL, &bounds[0]); PISM_CHK(ierr, "VecMin");
301  ierr = VecMax(tmp, NULL, &bounds[1]); PISM_CHK(ierr, "VecMax");
302 
303  if (bounds[0] == bounds[1]) {
304  bounds[0] = -1.0;
305  bounds[1] = 1.0;
306  }
307 
308  ierr = PetscViewerDrawSetBounds(v, 1, bounds);
309  PISM_CHK(ierr, "PetscViewerDrawSetBounds");
310 
311  ierr = VecView(tmp, v);
312  PISM_CHK(ierr, "VecView");
313  }

316 //! \brief Returns the x-derivative at i,j approximated using centered finite
317 //! differences.
318 double IceModelVec2S::diff_x(int i, int j) const {
319  return ((*this)(i + 1,j) - (*this)(i - 1,j)) / (2 * m_impl->grid->dx());
320 }

322 //! \brief Returns the y-derivative at i,j approximated using centered finite
323 //! differences.
324 double IceModelVec2S::diff_y(int i, int j) const {
325  return ((*this)(i,j + 1) - (*this)(i,j - 1)) / (2 * m_impl->grid->dy());
326 }

328 //! \brief Returns the x-derivative at i,j approximated using centered finite
329 //! differences. Respects grid periodicity and uses one-sided FD at grid edges
330 //! if necessary.
331 double IceModelVec2S::diff_x_p(int i, int j) const {
332  if (m_impl->grid->periodicity() & X_PERIODIC) {
333  return diff_x(i,j);
334  }
335 
336  if (i == 0) {
337  return ((*this)(i + 1,j) - (*this)(i,j)) / (m_impl->grid->dx());
338  } else if (i == (int)m_impl->grid->Mx() - 1) {
339  return ((*this)(i,j) - (*this)(i - 1,j)) / (m_impl->grid->dx());
340  } else {
341  return diff_x(i,j);
342  }
343 }

345 //! \brief Returns the y-derivative at i,j approximated using centered finite
346 //! differences. Respects grid periodicity and uses one-sided FD at grid edges
347 //! if necessary.
348 double IceModelVec2S::diff_y_p(int i, int j) const {
349  if (m_impl->grid->periodicity() & Y_PERIODIC) {
350  return diff_y(i,j);
351  }
352 
353  if (j == 0) {
354  return ((*this)(i,j + 1) - (*this)(i,j)) / (m_impl->grid->dy());
355  } else if (j == (int)m_impl->grid->My() - 1) {
356  return ((*this)(i,j) - (*this)(i,j - 1)) / (m_impl->grid->dy());
357  } else {
358  return diff_y(i,j);
359  }
360 }

362 //! Sums up all the values in an IceModelVec2S object. Ignores ghosts.
363 /*! Avoids copying to a "global" vector.

365 double IceModelVec2S::sum() const {
366  double result = 0;
367 
368  IceModelVec::AccessList list(*this);
369 
370  // sum up the local part:
371  for (Points p(*m_impl->grid); p; p.next()) {
372  result += (*this)(p.i(), p.j());
373  }
374 
375  // find the global sum:
376  return GlobalSum(m_impl->grid->com, result);

421 IceModelVec2::IceModelVec2(IceGrid::ConstPtr grid, const std::string &name,
422  IceModelVecKind ghostedp, unsigned int stencil_width, int dof) {
423  PetscErrorCode ierr;
424 
425  m_impl->dof = dof;
426  m_impl->grid = grid;
427 
428  auto max_stencil_width = m_impl->grid->ctx()->config()->get_number("grid.max_stencil_width");
429 
430  if ((m_impl->dof != 1) or (stencil_width > max_stencil_width)) {
431  m_impl->da_stencil_width = stencil_width;
432  } else {
433  m_impl->da_stencil_width = max_stencil_width;
434  }
435 
436  // initialize the da member:
437  m_impl->da = m_impl->grid->get_dm(m_impl->dof, m_impl->da_stencil_width);
438 
439  if (ghostedp) {
440  ierr = DMCreateLocalVector(*m_impl->da, m_impl->v.rawptr());
441  PISM_CHK(ierr, "DMCreateLocalVector");
442  } else {
443  ierr = DMCreateGlobalVector(*m_impl->da, m_impl->v.rawptr());
444  PISM_CHK(ierr, "DMCreateGlobalVector");
445  }
446 
447  m_impl->ghosted = (ghostedp == WITH_GHOSTS);
448  m_impl->name = name;
449 
450  if (m_impl->dof == 1) {
451  m_impl->metadata.push_back(SpatialVariableMetadata(m_impl->grid->ctx()->unit_system(),
452  name));
453  } else {
454 
455  for (unsigned int j = 0; j < m_impl->dof; ++j) {
456  m_impl->metadata.push_back(SpatialVariableMetadata(m_impl->grid->ctx()->unit_system(),
457  pism::printf("%s[%d]",
458  m_impl->name.c_str(), j)));
459  }
460  }
