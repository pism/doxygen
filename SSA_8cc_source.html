181 /*!
182  * Compute the weight used to determine if the difference between locations `i,j` and `n`
183  * (neighbor) should be used in the computation of the surface gradient in
184  * SSA::compute_driving_stress().
185  *
186  * We avoid differencing across
187  *
188  * - ice margins if stress boundary condition at ice margins (CFBC) is active
189  * - grounding lines
190  * - ice margins next to ice free locations above the surface elevation of the ice (fjord
191  * walls, nunataks, headwalls)
192  */
193 static int weight(bool margin_bc,
194  int M_ij, int M_n,
195  double h_ij, double h_n,
196  int N_ij, int N_n) {
197  using mask::grounded;
198  using mask::icy;
199  using mask::floating_ice;
200  using mask::ice_free;
201  using mask::ice_free_ocean;
202 
203  // grounding lines and calving fronts
204  if ((grounded(M_ij) and floating_ice(M_n)) or
205  (floating_ice(M_ij) and grounded(M_n)) or
206  (floating_ice(M_ij) and ice_free_ocean(M_n))) {
207  return 0;
208  }
209 
210  // fjord walls, nunataks, headwalls
211  if ((icy(M_ij) and ice_free(M_n) and h_n > h_ij) or
212  (ice_free(M_ij) and icy(M_n) and h_ij > h_n)) {
213  return 0;
214  }
215 
216  // This condition has to match the one used to implement the calving front stress
217  // boundary condition in SSAFD::assemble_rhs().
218  if (margin_bc and
219  ((icy(M_ij) and ice_free(M_n)) or
220  (ice_free(M_ij) and icy(M_n)))) {
221  return 0;
222  }
223 
224  // boundaries of the "no model" area
225  if ((N_ij == 0 and N_n == 1) or (N_ij == 1 and N_n == 0)) {
226  return 0;
227  }
228 
229  return 1;
230 }

232 //! \brief Compute the gravitational driving stress.
233 /*!
234 Computes the gravitational driving stress at the base of the ice:
235 \f[ \tau_d = - \rho g H \nabla h \f]
236  */
237 void SSA::compute_driving_stress(const IceModelVec2S &ice_thickness,
238  const IceModelVec2S &surface_elevation,
239  const IceModelVec2CellType &cell_type,
240  const IceModelVec2Int *no_model_mask,
241  IceModelVec2V &result) const {
242 
243  bool cfbc = m_config->get_flag("stress_balance.calving_front_stress_bc");
244  bool surface_gradient_inward = m_config->get_flag("stress_balance.ssa.compute_surface_gradient_inward");
245 
246  double
247  dx = m_grid->dx(),
248  dy = m_grid->dy();
249 
250  IceModelVec::AccessList list{&surface_elevation, &cell_type, &ice_thickness, &result};
251 
252  if (no_model_mask) {
253  list.add(*no_model_mask);
254  }
255 
256  for (Points p(*m_grid); p; p.next()) {
257  const int i = p.i(), j = p.j();
258 
259  const double pressure = m_EC->pressure(ice_thickness(i, j)); // FIXME issue #15
260  if (pressure <= 0.0) {
261  result(i, j) = 0.0;
262  continue;
263  }
264 
265  // Special case for verification tests.
266  if (surface_gradient_inward) {
267  double
268  h_x = diff_x_p(surface_elevation, i, j),
269  h_y = diff_y_p(surface_elevation, i, j);
270  result(i, j) = - pressure * Vector2(h_x, h_y);
271  continue;
272  }
273 
274  // To compute the x-derivative we use
275  //
276  // * away from the grounding line, ice margins, and no_model mask transitions -- 2nd
277  // order centered difference
278  //
279  // * at the grounded cell near the grounding line -- 1st order
280  // one-sided difference using the grounded neighbor
281  //
282  // * at the floating cell near the grounding line -- 1st order
283  // one-sided difference using the floating neighbor
284  //
285  // All these cases can be combined by writing h_x as the weighted
286  // average of one-sided differences, with weights of 0 if a finite
287  // difference is not used and 1 if it is.
288  //
289  // The y derivative is handled the same way.
290 
291  auto M = cell_type.star(i, j);
292  auto h = surface_elevation.star(i, j);
293  stencils::Star<int> N(0);
294 
295  if (no_model_mask) {
296  N = no_model_mask->star(i, j);
297  }
298 
299  // x-derivative
300  double h_x = 0.0;
301  {
302  double
303  west = weight(cfbc, M.ij, M.w, h.ij, h.w, N.ij, N.w),
304  east = weight(cfbc, M.ij, M.e, h.ij, h.e, N.ij, N.e);
305 
306  if (east + west > 0) {
307  h_x = 1.0 / ((west + east) * dx) * (west * (h.ij - h.w) + east * (h.e - h.ij));
308  } else {
309  h_x = 0.0;
310  }
311  }
312 
313  // y-derivative
314  double h_y = 0.0;
315  {
316  double
317  south = weight(cfbc, M.ij, M.s, h.ij, h.s, N.ij, N.s),
318  north = weight(cfbc, M.ij, M.n, h.ij, h.n, N.ij, N.n);
319 
320  if (north + south > 0) {
321  h_y = 1.0 / ((south + north) * dy) * (south * (h.ij - h.s) + north * (h.n - h.ij));
322  } else {
323  h_y = 0.0;
324  }
325  }
326 
327  result(i, j) = - pressure * Vector2(h_x, h_y);
328  }
329 }
