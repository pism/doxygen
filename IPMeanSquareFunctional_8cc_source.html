27 //! Implicitly set the normalization constant for the functional.
28 /*! The normalization constant is selected so that if an input
29 IceModelVec2V has component vectors all of length \a scale, then the funtional value will be 1. I.e.
30 \f[
31 c_N^{-1} = \sum_{i} w_i {\tt scale}^2.
32 \f]*/
33 void IPMeanSquareFunctional2V::normalize(double scale) {
34 
35  // The local value of the weights
36  double value = 0;
37 
38  if (m_weights) {
39  IceModelVec::AccessList list{m_weights};
40 
41  for (Points p(*m_grid); p; p.next()) {
42  const int i = p.i(), j = p.j();
43 
44  value += (*m_weights)(i, j);
45  }
46  } else {
47  for (Points p(*m_grid); p; p.next()) {
48  value += 1;
49  }
50  }
51 
52  m_normalization = GlobalSum(m_grid->com, value);
53  m_normalization *= (scale*scale);
54 }
