53  m_base_pddStdDev = m_config->get_number("surface.pdd.std_dev.value");

61

64  } else if (method == "random_process") {

73  }

102  m_mass_flux.metadata()["comment"] = "positive values correspond to ice gain";

126 void TemperatureIndex::init_impl(const Geometry &geometry) {
127 
128  // call the default implementation (not the interface method init())
129  SurfaceModel::init_impl(geometry);
130 
131  // report user's modeling choices
132  {
133  m_log->message(2,
134  "* Initializing the default temperature-index, PDD-based surface processes scheme.\n"
135  " Precipitation and 2m air temperature provided by atmosphere are inputs.\n"
136  " Surface mass balance and ice upper surface temperature are outputs.\n"
137  " See PISM User's Manual for control of degree-day factors.\n");
138 
139  m_log->message(2,
140  " Computing number of positive degree-days by: %s.\n",
141  m_mbscheme->method().c_str());
142 
143  if (m_faustogreve) {
144  m_log->message(2,
145  " Setting PDD parameters from [Faustoetal2009].\n");
146  } else {
147  m_log->message(2,
148  " Using default PDD parameters.\n");
149  }
150  }
151 
152  // initialize the spatially-variable air temperature standard deviation
153  {
154  std::string sd_file = m_config->get_string("surface.pdd.std_dev.file");
155  if (sd_file.empty()) {
156  m_log->message(2,
157  " Using constant standard deviation of near-surface temperature.\n");
158 
159  SpatialVariableMetadata attributes = m_air_temp_sd->metadata();
160  // replace with a constant IceModelVec2T:
161  m_air_temp_sd = IceModelVec2T::Constant(m_grid, "air_temp_sd", m_base_pddStdDev);
162  // restore metadata:
163  m_air_temp_sd->metadata() = attributes;
164  } else {
165  m_log->message(2,
166  " Reading standard deviation of near-surface temperature from '%s'...\n",
167  sd_file.c_str());
168 
169  bool sd_periodic = m_config->get_flag("surface.pdd.std_dev.periodic");
170  m_air_temp_sd->init(sd_file, sd_periodic);
171  }
172  }
173 
174  // initializing the model state
175  InputOptions input = process_input_options(m_grid->com, m_config);
176 
177  std::string firn_file = m_config->get_string("surface.pdd.firn_depth_file");
178 
179  if (input.type == INIT_RESTART) {
180  if (not firn_file.empty()) {
181  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
182  "surface.pdd.firn_depth_file is not allowed when"
183  " re-starting from a PISM output file.");
184  }
185 
186  m_firn_depth.read(input.filename, input.record);
187  m_snow_depth.read(input.filename, input.record);
188  } else if (input.type == INIT_BOOTSTRAP) {
189 
190  m_snow_depth.regrid(input.filename, OPTIONAL, 0.0);
191 
192  if (firn_file.empty()) {
193  m_firn_depth.regrid(input.filename, OPTIONAL, 0.0);
194  } else {
195  m_firn_depth.regrid(firn_file, CRITICAL);
196  }
197  } else {
198 
199  m_snow_depth.set(0.0);
200 
201  if (firn_file.empty()) {
202  m_firn_depth.set(0.0);
203  } else {
204  m_firn_depth.regrid(firn_file, CRITICAL);
205  }
206  }
207 
208  {
209  regrid("PDD surface model", m_snow_depth);
210  regrid("PDD surface model", m_firn_depth);
211  }
212 
213  // finish up
214  {
215  m_next_balance_year_start = compute_next_balance_year_start(m_grid->ctx()->time()->current());
216 
217  m_accumulation->set(0.0);
218  m_melt->set(0.0);
219  m_runoff->set(0.0);
220  }
221 }

227 double TemperatureIndex::compute_next_balance_year_start(double time) {
228  // compute the time corresponding to the beginning of the next balance year
229  double
230  balance_year_start_day = m_config->get_number("surface.pdd.balance_year_start_day"),
231  one_day = units::convert(m_sys, 1.0, "days", "seconds"),
232  year_start = m_grid->ctx()->time()->calendar_year_start(time),
233  balance_year_start = year_start + (balance_year_start_day - 1.0) * one_day;
234 
235  if (balance_year_start > time) {
236  return balance_year_start;
237  }
238  return m_grid->ctx()->time()->increment_date(balance_year_start, 1);
239 }

241 void TemperatureIndex::update_impl(const Geometry &geometry, double t, double dt) {
242 
243  // make a copy of the pointer to convince clang static analyzer that its value does not
244  // change during the call
245  FaustoGrevePDDObject *fausto_greve = m_faustogreve.get();
246 
247  // update to ensure that temperature and precipitation time series are correct:
248  m_atmosphere->update(geometry, t, dt);
249 
250  m_temperature->copy_from(m_atmosphere->mean_annual_temp());
251 
252  // set up air temperature and precipitation time series
253  int N = m_mbscheme->get_timeseries_length(dt);
254 
255  const double dtseries = dt / N;
256  std::vector<double> ts(N), T(N), S(N), P(N), PDDs(N);
257  for (int k = 0; k < N; ++k) {
258  ts[k] = t + k * dtseries;
259  }
260 
261  // update standard deviation time series
262  if (m_sd_file_set) {
263  m_air_temp_sd->update(t, dt);
264  m_air_temp_sd->init_interpolation(ts);
265  }
266 
267  const IceModelVec2CellType &mask = geometry.cell_type;
268  const IceModelVec2S &H = geometry.ice_thickness;
269 
270  IceModelVec::AccessList list{&mask, &H, m_air_temp_sd.get(), &m_mass_flux,
271  &m_firn_depth, &m_snow_depth,
272  m_accumulation.get(), m_melt.get(), m_runoff.get()};
273 
274  const double
275  sigmalapserate = m_config->get_number("surface.pdd.std_dev.lapse_lat_rate"),
276  sigmabaselat = m_config->get_number("surface.pdd.std_dev.lapse_lat_base");
277 
278  const IceModelVec2S *latitude = nullptr;
279  if (fausto_greve or sigmalapserate != 0.0) {
280  latitude = &geometry.latitude;
281 
282  list.add(*latitude);
283  }
284 
285  if (fausto_greve) {
286  const IceModelVec2S
287  *longitude = &geometry.latitude,
288  *surface_altitude = &geometry.ice_surface_elevation;
289 
290  fausto_greve->update_temp_mj(*surface_altitude, *latitude, *longitude);
291  }
292 
293  LocalMassBalance::DegreeDayFactors ddf = m_base_ddf;
294 
295  m_atmosphere->init_timeseries(ts);
296 
297  m_atmosphere->begin_pointwise_access();
298 
299  const double ice_density = m_config->get_number("constants.ice.density");
300 
301  ParallelSection loop(m_grid->com);
302  try {
303  for (Points p(*m_grid); p; p.next()) {
304  const int i = p.i(), j = p.j();
305 
306  // the temperature time series from the AtmosphereModel and its modifiers
307  m_atmosphere->temp_time_series(i, j, T);
308 
309  if (mask.ice_free_ocean(i, j)) {
310  // ignore precipitation over ice-free ocean
311  for (int k = 0; k < N; ++k) {
312  P[k] = 0.0;
313  }
314  } else {
315  // elsewhere, get precipitation from the atmosphere model
316  m_atmosphere->precip_time_series(i, j, P);
317  }
318 
319  // convert precipitation from "kg m-2 second-1" to "m second-1" (PDDMassBalance expects
320  // accumulation in m/second ice equivalent)
321  for (int k = 0; k < N; ++k) {
322  P[k] = P[k] / ice_density;
323  // kg / (m^2 * second) / (kg / m^3) = m / second
324  }
325 
326  // interpolate temperature standard deviation time series
327  if (m_sd_file_set) {
328  m_air_temp_sd->interp(i, j, S);
329  } else {
330  double tmp = (*m_air_temp_sd)(i, j);
331  for (int k = 0; k < N; ++k) {
332  S[k] = tmp;
333  }
334  }
335 
336  if (fausto_greve) {
337  // we have been asked to set mass balance parameters according to
338  // formula (6) in [\ref Faustoetal2009]; they overwrite ddf set above
339  ddf = fausto_greve->degree_day_factors(i, j, (*latitude)(i, j));
340  }
341 
342  // apply standard deviation lapse rate on top of prescribed values
343  if (sigmalapserate != 0.0) {
344  double lat = (*latitude)(i, j);
345  for (int k = 0; k < N; ++k) {
346  S[k] += sigmalapserate * (lat - sigmabaselat);
347  }
348  (*m_air_temp_sd)(i, j) = S[0]; // ensure correct SD reporting
349  }
350 
351  // apply standard deviation param over ice if in use
352  if (m_sd_use_param and mask.icy(i, j)) {
353  for (int k = 0; k < N; ++k) {
354  S[k] = m_sd_param_a * (T[k] - 273.15) + m_sd_param_b;
355  if (S[k] < 0.0) {
356  S[k] = 0.0 ;
357  }
358  }
359  (*m_air_temp_sd)(i, j) = S[0]; // ensure correct SD reporting
360  }
361 
362  // Use temperature time series, the "positive" threshhold, and
363  // the standard deviation of the daily variability to get the
364  // number of positive degree days (PDDs)
365  if (mask.ice_free_ocean(i, j)) {
366  for (int k = 0; k < N; ++k) {
367  PDDs[k] = 0.0;
368  }
369  } else {
370  m_mbscheme->get_PDDs(dtseries, S, T, // inputs
371  PDDs); // output
372  }
373 
374  // Use temperature time series to remove rainfall from precipitation
375  m_mbscheme->get_snow_accumulation(T, // air temperature (input)
376  P); // precipitation rate (input-output)
377 
378  // Use degree-day factors, the number of PDDs, and the snow precipitation to get surface mass
379  // balance (and diagnostics: accumulation, melt, runoff)
380  {
381  double next_snow_depth_reset = m_next_balance_year_start;
382 
383  // make copies of firn and snow depth values at this point to avoid accessing 2D
384  // fields in the inner loop
385  double
386  ice = H(i, j),
387  firn = m_firn_depth(i, j),
388  snow = m_snow_depth(i, j);
389 
390  // accumulation, melt, runoff over this time-step
391  double
392  A = 0.0,
393  M = 0.0,
394  R = 0.0,
395  SMB = 0.0;
396 
397  for (int k = 0; k < N; ++k) {
398  if (ts[k] >= next_snow_depth_reset) {
399  snow = 0.0;
400  while (next_snow_depth_reset <= ts[k]) {
401  next_snow_depth_reset = m_grid->ctx()->time()->increment_date(next_snow_depth_reset, 1);
402  }
403  }
404 
405  const double accumulation = P[k] * dtseries;
406 
407  LocalMassBalance::Changes changes;
408  changes = m_mbscheme->step(ddf, PDDs[k],
409  ice, firn, snow, accumulation);
410 
411  // update ice thickness
412  ice += changes.smb;
413  assert(ice >= 0);
414 
415  // update firn depth
416  firn += changes.firn_depth;
417  assert(firn >= 0);
418 
419  // update snow depth
420  snow += changes.snow_depth;
421  assert(snow >= 0);
422 
423  // update total accumulation, melt, and runoff
424  {
425  A += accumulation;
426  M += changes.melt;
427  R += changes.runoff;
428  SMB += changes.smb;
429  }
430  } // end of the time-stepping loop
431 
432  // set firn and snow depths
433  m_firn_depth(i, j) = firn;
434  m_snow_depth(i, j) = snow;
435 
436  // set total accumulation, melt, and runoff, and SMB at this point, converting
437  // from "meters, ice equivalent" to "kg / m^2"
438  {
439  (*m_accumulation)(i, j) = A * ice_density;
440  (*m_melt)(i, j) = M * ice_density;
441  (*m_runoff)(i, j) = R * ice_density;
442  // m_mass_flux (unlike m_accumulation, m_melt, and m_runoff), is a
443  // rate. m * (kg / m^3) / second = kg / m^2 / second
444  m_mass_flux(i, j) = SMB * ice_density / dt;
445  }
446  }
447 
448  if (mask.ice_free_ocean(i, j)) {
449  m_firn_depth(i, j) = 0.0; // no firn in the ocean
450  m_snow_depth(i, j) = 0.0; // snow over the ocean does not stick
451  }
452  }
453  } catch (...) {
454  loop.failed();
455  }
456  loop.check();
457 
458  m_atmosphere->end_pointwise_access();
459 
460  m_next_balance_year_start = compute_next_balance_year_start(m_grid->ctx()->time()->current());
461 }

537  external_units = "Gt year-1" ;

541  m_accumulator.metadata()["units"] = accumulator_units;
542 
543  set_attrs(long_name, standard_name, internal_units, external_units, 0);

548 
549 protected:

552 
553  if (m_kind == MASS) {

555 
556  IceModelVec::AccessList list{&m_melt_mass, &melt_amount};
557 
558  for (Points p(*m_grid); p; p.next()) {
559  const int i = p.i(), j = p.j();
560  m_melt_mass(i, j) = melt_amount(i, j) * cell_area;
561  }
562  return m_melt_mass;
563  } else {
564  return melt_amount;
565  }
566  }
567 private:
568  IceModelVec2S m_melt_mass;
569  AmountKind m_kind;
570 };

692 /*!
693  * Integrate a field over the computational domain.
694  *
695  * If the input has units kg/m^2, the output will be in kg.
696  */
697 static double integrate(const IceModelVec2S &input) {
698  IceGrid::ConstPtr grid = input.grid();
699 
700  double cell_area = grid->cell_area();
701 
702  IceModelVec::AccessList list{&input};
703 
704  double result = 0.0;
705 
706  for (Points p(*grid); p; p.next()) {
707  const int i = p.i(), j = p.j();
708 
709  result += input(i, j) * cell_area;
710  }
711 
712  return GlobalSum(grid->com, result);
713 }
