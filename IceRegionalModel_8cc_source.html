93 void IceRegionalModel::model_state_setup() {
94 
95  // initialize the model state (including special fields)
96  IceModel::model_state_setup();
97 
98  InputOptions input = process_input_options(m_ctx->com(), m_config);
99 
100  // Initialize stored ice thickness and surface elevation. This goes here and not in
101  // bootstrap_2d because bed topography is not initialized at the time bootstrap_2d is
102  // called.
103  if (input.type == INIT_BOOTSTRAP) {
104  if (m_config->get_flag("regional.zero_gradient")) {
105  m_usurf_stored.set(0.0);
106  m_thk_stored.set(0.0);
107  } else {
108  m_usurf_stored.copy_from(m_geometry.ice_surface_elevation);
109  m_thk_stored.copy_from(m_geometry.ice_thickness);
110  }
111  }
112 
113  m_geometry_evolution->set_no_model_mask(m_no_model_mask);
114 
115  if (m_ch_system) {
116  const bool use_input_file = input.type == INIT_BOOTSTRAP or input.type == INIT_RESTART;
117 
118  std::unique_ptr<File> input_file;
119 
120  if (use_input_file) {
121  input_file.reset(new File(m_grid->com, input.filename, PISM_GUESS, PISM_READONLY));
122  }
123 
124  switch (input.type) {
125  case INIT_RESTART:
126  {
127  m_ch_system->restart(*input_file, input.record);
128  break;
129  }
130  case INIT_BOOTSTRAP:
131  {
132 
133  m_ch_system->bootstrap(*input_file,
134  m_geometry.ice_thickness,
135  m_surface->temperature(),
136  m_surface->mass_flux(),
137  m_btu->flux_through_top_surface());
138  break;
139  }
140  case INIT_OTHER:
141  default:
142  {
143  m_basal_melt_rate.set(m_config->get_number("bootstrapping.defaults.bmelt"));
144 
145  m_ch_system->initialize(m_basal_melt_rate,
146  m_geometry.ice_thickness,
147  m_surface->temperature(),
148  m_surface->mass_flux(),
149  m_btu->flux_through_top_surface());
150 
151  }
152  }
153  }
154 }

168 void IceRegionalModel::allocate_energy_model() {
169 
170  if (m_energy_model != NULL) {
171  return;
172  }
173 
174  m_log->message(2, "# Allocating an energy balance model...\n");
175 
176  if (m_config->get_flag("energy.enabled")) {
177  if (m_config->get_flag("energy.temperature_based")) {
178  throw RuntimeError(PISM_ERROR_LOCATION,
179  "pismr -regional does not support the '-energy cold' mode.");
180  } else {
181  m_energy_model = new energy::EnthalpyModel_Regional(m_grid, m_stress_balance.get());
182  }
183  } else {
184  m_energy_model = new energy::DummyEnergyModel(m_grid, m_stress_balance.get());
185  }
186 
187  m_submodels["energy balance model"] = m_energy_model;
188 
189  if (m_config->get_flag("energy.ch_warming.enabled") and
190  not m_ch_system) {
191 
192  m_log->message(2, "# Allocating the cryo-hydrologic warming model...\n");
193 
194  m_ch_system.reset(new energy::CHSystem(m_grid, m_stress_balance.get()));
195  m_submodels["cryo-hydrologic warming"] = m_ch_system.get();
196  }

230 //! Bootstrap a "regional" model.
231 /*!
232  * Need to initialize all the variables managed by IceModel, as well as
233  * - no_model_mask
234  * - usurf_stored
235  * - thk_stored

237 void IceRegionalModel::bootstrap_2d(const File &input_file) {
238 
239  IceModel::bootstrap_2d(input_file);
240 
241  // no_model_mask
242  {
243  if (input_file.find_variable(m_no_model_mask.metadata().get_name())) {
244  m_no_model_mask.regrid(input_file, CRITICAL);
245  } else {
246  // set using the no_model_strip parameter
247  double strip_width = m_config->get_number("regional.no_model_strip", "meters");
248  set_no_model_strip(*m_grid, strip_width, m_no_model_mask);
249  }
250 
251  // m_no_model_mask was added to m_model_state, so
252  // IceModel::regrid() will take care of regridding it, if necessary.
253  }
254 
255  if (m_config->get_flag("stress_balance.ssa.dirichlet_bc")) {
256  IceModelVec::AccessList list
257  {&m_no_model_mask, &m_velocity_bc_mask, &m_ice_thickness_bc_mask};
258 
259  for (Points p(*m_grid); p; p.next()) {
260  const int i = p.i(), j = p.j();
261 
262  if (m_no_model_mask(i, j) > 0.5) {
263  m_velocity_bc_mask(i, j) = 1;
264  m_ice_thickness_bc_mask(i, j) = 1;
265  }
266  }
267  }
268 }

288 void IceRegionalModel::energy_step() {
289 
290  if (m_ch_system) {
291  bedrock_thermal_model_step();
292 
293  energy::Inputs inputs = energy_model_inputs();
294  const IceModelVec3 *strain_heating = inputs.volumetric_heating_rate;
295  inputs.volumetric_heating_rate = m_ch_warming_flux.get();
296 
297  energy::cryo_hydrologic_warming_flux(m_config->get_number("constants.ice.thermal_conductivity"),
298  m_config->get_number("energy.ch_warming.average_channel_spacing"),
299  m_geometry.ice_thickness,
300  m_energy_model->enthalpy(),
301  m_ch_system->enthalpy(),
302  *m_ch_warming_flux);
303 
304  // Convert to the loss of energy by the CH system:
305  m_ch_warming_flux->scale(-1.0);
306 
307  m_ch_system->update(t_TempAge, dt_TempAge, inputs);
308 
309  // Add CH warming flux to the strain heating term:
310  m_ch_warming_flux->scale(-1.0);
311  m_ch_warming_flux->add(1.0, *strain_heating);
312 
313  m_energy_model->update(t_TempAge, dt_TempAge, inputs);
314 
315  m_stdout_flags = m_energy_model->stdout_flags() + m_stdout_flags;
316  } else {
317  IceModel::energy_step();
318  }
319 }

426 void IceRegionalModel::hydrology_step() {
427  hydrology::Inputs inputs;
428 
429  IceModelVec2S &sliding_speed = *m_work2d[0];
430  compute_magnitude(m_stress_balance->advective_velocity(), sliding_speed);
431 
432  inputs.no_model_mask = &m_no_model_mask;
433  inputs.geometry = &m_geometry;
434  inputs.surface_input_rate = nullptr;
435  inputs.basal_melt_rate = &m_basal_melt_rate;
436  inputs.ice_sliding_speed = &sliding_speed;
437 
438  if (m_surface_input_for_hydrology) {
439  m_surface_input_for_hydrology->update(m_time->current(), m_dt);
440  m_surface_input_for_hydrology->average(m_time->current(), m_dt);
441  inputs.surface_input_rate = m_surface_input_for_hydrology.get();
442  } else if (m_config->get_flag("hydrology.surface_input_from_runoff")) {
443  // convert [kg m-2] to [kg m-2 s-1]
444  IceModelVec2S &surface_input_rate = *m_work2d[1];
445  surface_input_rate.copy_from(m_surface->runoff());
446  surface_input_rate.scale(1.0 / m_dt);
447  inputs.surface_input_rate = &surface_input_rate;
448  }
449 
450  m_subglacial_hydrology->update(m_time->current(), m_dt, inputs);
451 }
