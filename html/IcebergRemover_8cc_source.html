55 void IcebergRemover::update_impl(const IceModelVec2Int &bc_mask,
56  IceModelVec2CellType &mask,
57  IceModelVec2S &ice_thickness) {
58  const int
59  mask_grounded_ice = 1,
60  mask_floating_ice = 2;
61 
62  // prepare the mask that will be handed to the connected component
63  // labeling code:
64  {
65  m_iceberg_mask.set(0.0);
66 
67  IceModelVec::AccessList list{&mask, &m_iceberg_mask, &bc_mask};
68 
69  for (Points p(*m_grid); p; p.next()) {
70  const int i = p.i(), j = p.j();
71 
72  if (mask.grounded_ice(i,j)) {
73  m_iceberg_mask(i,j) = mask_grounded_ice;
74  } else if (mask.floating_ice(i,j)) {
75  m_iceberg_mask(i,j) = mask_floating_ice;
76  }
77  }
78 
79  // Mark icy Dirichlet B.C. cells as "grounded" because we don't want them removed.
80  for (Points p(*m_grid); p; p.next()) {
81  const int i = p.i(), j = p.j();
82 
83  if (bc_mask(i, j) > 0.5 and mask.icy(i, j)) {
84  m_iceberg_mask(i, j) = mask_grounded_ice;
85  }
86  }
87  }
88 
89  // identify icebergs using serial code on processor 0:
90  {
91  m_iceberg_mask.put_on_proc0(*m_mask_p0);
92 
93  ParallelSection rank0(m_grid->com);
94  try {
95  if (m_grid->rank() == 0) {
96  petsc::VecArray mask_p0(*m_mask_p0);
97  label_connected_components(mask_p0.get(), m_grid->My(), m_grid->Mx(), true, mask_grounded_ice);
98  }
99  } catch (...) {
100  rank0.failed();
101  }
102  rank0.check();
103 
104  m_iceberg_mask.get_from_proc0(*m_mask_p0);
105  }
106 
107  // correct ice thickness and the cell type mask using the resulting
108  // "iceberg" mask:
109  {
110  IceModelVec::AccessList list{&ice_thickness, &mask, &m_iceberg_mask, &bc_mask};
111 
112  for (Points p(*m_grid); p; p.next()) {
113  const int i = p.i(), j = p.j();
114 
115  if (m_iceberg_mask(i,j) > 0.5 && bc_mask(i,j) < 0.5) {
116  ice_thickness(i,j) = 0.0;
117  mask(i,j) = MASK_ICE_FREE_OCEAN;
118  }
119  }
120  }
121 
122  // update ghosts of the mask and the ice thickness (then surface
123  // elevation can be updated redundantly)
124  mask.update_ghosts();
125  ice_thickness.update_ghosts();
126 }
