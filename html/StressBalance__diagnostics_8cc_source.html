96 IceModelVec::Ptr PSB_velbar::compute_impl() const {
97  // get the thickness
98  const IceModelVec2S* thickness = m_grid->variables().get_2d_scalar("land_ice_thickness");
99 
100  // Compute the vertically-integrated horizontal ice flux:
101  IceModelVec2V::Ptr result = IceModelVec2V::ToVector(PSB_flux(model).compute());
102 
103  // Override metadata set by the flux computation
104  result->metadata(0) = m_vars[0];
105  result->metadata(1) = m_vars[1];
106 
107  IceModelVec::AccessList list{thickness, result.get()};
108 
109  for (Points p(*m_grid); p; p.next()) {
110  const int i = p.i(), j = p.j();
111  double thk = (*thickness)(i,j);
112 
113  // Ice flux is masked already, but we need to check for division
114  // by zero anyway.
115  if (thk > 0.0) {
116  (*result)(i,j) /= thk;
117  } else {
118  (*result)(i,j) = 0.0;
119  }
120  }
121 
122  return result;
123 }

173 IceModelVec::Ptr PSB_flux::compute_impl() const {
174  double H_threshold = m_config->get_number("geometry.ice_free_thickness_standard");
175 
176  IceModelVec2V::Ptr result(new IceModelVec2V(m_grid, "flux", WITHOUT_GHOSTS));
177  result->metadata(0) = m_vars[0];
178  result->metadata(1) = m_vars[1];
179 
180  // get the thickness
181  const IceModelVec2S *thickness = m_grid->variables().get_2d_scalar("land_ice_thickness");
182 
183  const IceModelVec3
184  &u3 = model->velocity_u(),
185  &v3 = model->velocity_v();
186 
187  IceModelVec::AccessList list{&u3, &v3, thickness, result.get()};
188 
189  auto &z = m_grid->z();
190 
191  ParallelSection loop(m_grid->com);
192  try {
193  for (Points p(*m_grid); p; p.next()) {
194  const int i = p.i(), j = p.j();
195 
196  double H = (*thickness)(i,j);
197 
198  // an "ice-free" cell:
199  if (H < H_threshold) {
200  (*result)(i, j) = 0.0;
201  continue;
202  }
203 
204  // an icy cell:
205  {
206  auto u = u3.get_column(i, j);
207  auto v = v3.get_column(i, j);
208 
209  Vector2 Q(0.0, 0.0);
210 
211  // ks is "k just below the surface"
212  int ks = m_grid->kBelowHeight(H);
213 
214  if (ks > 0) {
215  Vector2 v0(u[0], v[0]);
216 
217  for (int k = 1; k <= ks; ++k) {
218  Vector2 v1(u[k], v[k]);
219 
220  // trapezoid rule
221  Q += (z[k] - z[k - 1]) * 0.5 * (v0 + v1);
222 
223  v0 = v1;
224  }
225  }
226 
227  // rectangle method to integrate over the last level
228  Q += (H - z[ks]) * Vector2(u[ks], v[ks]);
229 
230  (*result)(i, j) = Q;
231  }
232  }
233  } catch (...) {
234  loop.failed();
235  }
236  loop.check();
237 
238  return result;
239 }

424 IceModelVec::Ptr PSB_wvel::compute(bool zero_above_ice) const {
425  IceModelVec3::Ptr result3(new IceModelVec3(m_grid, "wvel", WITHOUT_GHOSTS, m_grid->z()));
426  result3->metadata() = m_vars[0];
427 
428  const IceModelVec2S *bed, *uplift;
429  bed = m_grid->variables().get_2d_scalar("bedrock_altitude");
430  uplift = m_grid->variables().get_2d_scalar("tendency_of_bedrock_altitude");
431 
432  const IceModelVec2S &thickness = *m_grid->variables().get_2d_scalar("land_ice_thickness");
433  const IceModelVec2CellType &mask = *m_grid->variables().get_2d_cell_type("mask");
434 
435  const IceModelVec3
436  &u3 = model->velocity_u(),
437  &v3 = model->velocity_v(),
438  &w3 = model->velocity_w();
439 
440  IceModelVec::AccessList list{&thickness, &mask, bed, &u3, &v3, &w3, uplift, result3.get()};
441 
442  const double ice_density = m_config->get_number("constants.ice.density"),
443  sea_water_density = m_config->get_number("constants.sea_water.density"),
444  R = ice_density / sea_water_density;
445 
446  ParallelSection loop(m_grid->com);
447  try {
448  for (Points p(*m_grid); p; p.next()) {
449  const int i = p.i(), j = p.j();
450 
451  const double
452  *u = u3.get_column(i, j),
453  *v = v3.get_column(i, j),
454  *w = w3.get_column(i, j);
455  double *result = result3->get_column(i, j);
456 
457  int ks = m_grid->kBelowHeight(thickness(i,j));
458 
459  // in the ice:
460  if (mask.grounded(i,j)) {
461  const double
462  bed_dx = bed->diff_x_p(i,j),
463  bed_dy = bed->diff_y_p(i,j),
464  uplift_ij = (*uplift)(i,j);
465  for (int k = 0; k <= ks ; k++) {
466  result[k] = w[k] + uplift_ij + u[k] * bed_dx + v[k] * bed_dy;
467  }
468 
469  } else { // floating
470  const double
471  z_sl = R * thickness(i,j),
472  w_sl = w3.interpolate(i, j, z_sl);
473 
474  for (int k = 0; k <= ks ; k++) {
475  result[k] = w[k] - w_sl;
476  }
477 
478  }
479 
480  // above the ice:
481  if (zero_above_ice) {
482  // set to zeros
483  for (unsigned int k = ks+1; k < m_grid->Mz() ; k++) {
484  result[k] = 0.0;
485  }
486  } else {
487  // extrapolate using the topmost value
488  for (unsigned int k = ks+1; k < m_grid->Mz() ; k++) {
489  result[k] = result[ks];
490  }
491  }
492 
493  }
494  } catch (...) {
495  loop.failed();
496  }
497  loop.check();
498 
499  return result3;
500 }

1064  const IceModelVec3 *enthalpy = m_grid->variables().get_3d_scalar("enthalpy");

1095  eigen1 = strain_rates(i, j, 0),
