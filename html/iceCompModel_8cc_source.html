59 IceCompModel::IceCompModel(IceGrid::Ptr grid, std::shared_ptr<Context> context, int test)
60  : IceModel(grid, context),
61  m_testname(test),
62  m_HexactL(m_grid, "HexactL", WITHOUT_GHOSTS),
63  m_strain_heating3_comp(m_grid, "strain_heating_comp", WITHOUT_GHOSTS, m_grid->z()),
64  m_bedrock_is_ice_forK(false)
65 {
66 
67  m_log->message(2, "starting Test %c ...\n", m_testname);
68 
69  // Override some defaults from parent class
70  m_config->set_number("stress_balance.sia.enhancement_factor", 1.0);
71  // none use bed smoothing & bed roughness parameterization
72  m_config->set_number("stress_balance.sia.bed_smoother.range", 0.0);
73 
74  // set values of flags in run()
75  m_config->set_flag("geometry.update.enabled", true);
76  m_config->set_flag("geometry.update.use_basal_melt_rate", false);
77 
78  // flow law settings
79  switch (m_testname) {
80  case 'A':
81  case 'B':
82  case 'C':
83  case 'D':
84  case 'H':
85  case 'L':
86  {
87  m_config->set_string("stress_balance.sia.flow_law", "isothermal_glen");
88  const double year = convert(m_sys, 1.0, "year", "seconds");
89  m_config->set_number("flow_law.isothermal_Glen.ice_softness", 1.0e-16 / year);
90  break;
91  }
92  case 'V':
93  {
94  m_config->set_string("stress_balance.ssa.flow_law", "isothermal_glen");
95  const double
96  hardness = 1.9e8,
97  softness = pow(hardness,
98  -m_config->get_number("stress_balance.ssa.Glen_exponent"));
99  m_config->set_number("flow_law.isothermal_Glen.ice_softness", softness);
100  break;
101  }
102  case 'F':
103  case 'G':
104  case 'K':
105  case 'O':
106  default:
107  {
108  m_config->set_string("stress_balance.sia.flow_law", "arr");
109  break;
110  }
111  }
112 
113  if (m_testname == 'H') {
114  m_config->set_string("bed_deformation.model", "iso");
115  } else {
116  m_config->set_string("bed_deformation.model", "none");
117  }
118 
119  if ((m_testname == 'F') || (m_testname == 'G') || (m_testname == 'K') || (m_testname == 'O')) {
120  m_config->set_flag("energy.enabled", true);
121  // essentially turn off run-time reporting of extremely low computed
122  // temperatures; *they will be reported as errors* anyway
123  m_config->set_number("energy.minimum_allowed_temperature", 0.0);
124  m_config->set_number("energy.max_low_temperature_count", 1000000);
125  } else {
126  m_config->set_flag("energy.enabled", false);
127  }
128 
129  // set sea level to -1e4 to ensure that all ice is grounded
130  m_config->set_number("sea_level.constant.value", -1e4);
131 
132  // special considerations for K and O wrt thermal bedrock and pressure-melting
133  if ((m_testname == 'K') || (m_testname == 'O')) {
134  m_config->set_flag("energy.allow_temperature_above_melting", false);
135  } else {
136  // note temps are generally allowed to go above pressure melting in verify
137  m_config->set_flag("energy.allow_temperature_above_melting", true);
138  }
139 
140  if (m_testname == 'V') {
141  // no sub-shelf melting
142  m_config->set_flag("geometry.update.use_basal_melt_rate", false);
143 
144  // this test is isothermal
145  m_config->set_flag("energy.enabled", false);
146 
147  // use the SSA solver
148  m_config->set_string("stress_balance_model", "ssa");
149 
150  // set sea level to 0.0
151  m_config->set_number("sea_level.constant.value", 0.0);
152 
153  m_config->set_flag("stress_balance.ssa.dirichlet_bc", true);
154  }
155 
156  m_config->set_flag("energy.temperature_based", true);
157 }

170  if (m_btu != NULL) {

174  // this switch changes Test K to make material properties for bedrock the same as for ice
175  bool biiSet = options::Bool("-bedrock_is_ice", "set bedrock properties to those of ice");
176  if (biiSet) {
177  if (m_testname == 'K') {

180  m_config->set_number("energy.bedrock_thermal.density", m_config->get_number("constants.ice.density"));
181  m_config->set_number("energy.bedrock_thermal.conductivity", m_config->get_number("constants.ice.thermal_conductivity"));
182  m_config->set_number("energy.bedrock_thermal.specific_heat_capacity", m_config->get_number("constants.ice.specific_heat_capacity"));
183  m_bedrock_is_ice_forK = true;
184  } else {
185  m_log->message(1,
186  "IceCompModel WARNING: option -bedrock_is_ice ignored; only applies to Test K\n");
187  }
188  }
189 
190  if (m_testname != 'K') {
191  // now make bedrock have same material properties as ice
192  // (note Mbz=1 also, by default, but want ice/rock interface to see
193  // pure ice from the point of view of applying geothermal boundary
194  // condition, especially in tests F and G)
195  m_config->set_number("energy.bedrock_thermal.density", m_config->get_number("constants.ice.density"));
196  m_config->set_number("energy.bedrock_thermal.conductivity", m_config->get_number("constants.ice.thermal_conductivity"));
197  m_config->set_number("energy.bedrock_thermal.specific_heat_capacity", m_config->get_number("constants.ice.specific_heat_capacity"));
198  }
199 
200  energy::BTUGrid bed_vertical_grid = energy::BTUGrid::FromOptions(m_grid->ctx());
201 
202  if (bed_vertical_grid.Mbz > 1) {
203  m_btu = new energy::BTU_Verification(m_grid, bed_vertical_grid,
204  m_testname, m_bedrock_is_ice_forK);
205  } else {
206  m_btu = new energy::BTU_Minimal(m_grid);
207  }
208 
209  m_submodels["bedrock thermal layer"] = m_btu;
210 }

306 
307  const double time = m_time->current();
308 
309  m_geometry.cell_type.set(MASK_GROUNDED);
310 
311  IceModelVec::AccessList list(m_geometry.ice_thickness);
312 
313  ParallelSection loop(m_grid->com);
314  try {
315  for (Points p(*m_grid); p; p.next()) {
316  const int i = p.i(), j = p.j();
317 
318  const double r = radius(*m_grid, i, j);
319  switch (m_testname) {
320  case 'A':

322  break;
323  case 'B':
324  m_geometry.ice_thickness(i, j) = exactB(time, r).H;
325  break;

328  break;

337  }
338  }

341  }
342  loop.check();
343

376 void IceCompModel::initTestL() {
377 
378  m_log->message(2, "* Initializing ice thickness and bed topography for test L...\n");
379 
380  // setup to evaluate test L; requires solving an ODE numerically
381  // using sorted list of radii, sorted in decreasing radius order
382  const int MM = m_grid->xm() * m_grid->ym();
383 
384  std::vector<rgrid> rrv(MM);
385  int k = 0;
386  for (Points p(*m_grid); p; p.next()) {
387  const int i = p.i(), j = p.j();
388 
389  rrv[k].i = i;
390  rrv[k].j = j;
391  rrv[k].r = radius(*m_grid, i,j);
392 
393  k += 1;
394  }
395 
396  std::sort(rrv.begin(), rrv.end(), rgridReverseSort()); // so rrv[k].r > rrv[k+1].r
397 
398  // get soln to test L at these radii; solves ODE only once (on each processor)
399  std::vector<double> rr(MM), HH(MM), bb(MM), aa(MM);
400 
401  for (k = 0; k < MM; k++) {
402  rr[k] = rrv[k].r;
403  }
404 
405  ExactLParameters L = exactL(rr);
406 
407  {
408  IceModelVec2S bed_uplift(m_grid, "uplift", WITHOUT_GHOSTS);
409 
410  IceModelVec::AccessList list{&m_geometry.ice_thickness, &m_geometry.bed_elevation};
411 
412  for (k = 0; k < MM; k++) {
413  m_geometry.ice_thickness(rrv[k].i, rrv[k].j) = L.H[k];
414  m_geometry.bed_elevation(rrv[k].i, rrv[k].j) = L.b[k];
415  }
416 
417  m_geometry.ice_thickness.update_ghosts();
418 
419  bed_uplift.set(0.0);
420 
421  m_beddef->bootstrap(m_geometry.bed_elevation, bed_uplift, m_geometry.ice_thickness,
422  m_geometry.sea_level_elevation);
423  }
424 
425  // store copy of ice_thickness for evaluating geometry errors
426  m_HexactL.copy_from(m_geometry.ice_thickness);
427 }

446 void IceCompModel::computeGeometryErrors(double &gvolexact, double &gareaexact,
447  double &gdomeHexact, double &volerr,
448  double &areaerr, double &gmaxHerr,
449  double &gavHerr, double &gmaxetaerr,
450  double &centerHerr) {
451  // compute errors in thickness, eta=thickness^{(2n+2)/n}, volume, area
452 
453  const double time = m_time->current();
454  double
455  Hexact = 0.0,
456  vol = 0.0,
457  area = 0.0,
458  domeH = 0.0,
459  volexact = 0.0,
460  areaexact = 0.0,
461  domeHexact = 0.0;
462  double
463  Herr = 0.0,
464  avHerr = 0.0,
465  etaerr = 0.0;
466 
467  IceModelVec::AccessList list(m_geometry.ice_thickness);
468  if (m_testname == 'L') {
469  list.add(m_HexactL);
470  }
471 
472  double
473  seawater_density = m_config->get_number("constants.sea_water.density"),
474  ice_density = m_config->get_number("constants.ice.density"),
475  Glen_n = m_config->get_number("stress_balance.sia.Glen_exponent"),
476  standard_gravity = m_config->get_number("constants.standard_gravity");
477 
478  // area of grid square in square km:
479  const double a = m_grid->dx() * m_grid->dy() * 1e-3 * 1e-3;
480  const double m = (2.0 * Glen_n + 2.0) / Glen_n;
481 
482  ParallelSection loop(m_grid->com);
483  try {
484  for (Points p(*m_grid); p; p.next()) {
485  const int i = p.i(), j = p.j();
486 
487  if (m_geometry.ice_thickness(i,j) > 0) {
488  area += a;
489  vol += a * m_geometry.ice_thickness(i,j) * 1e-3;
490  }
491  double xx = m_grid->x(i), r = radius(*m_grid, i,j);
492  switch (m_testname) {
493  case 'A':
494  Hexact = exactA(r).H;
495  break;
496  case 'B':
497  Hexact = exactB(time, r).H;
498  break;
499  case 'C':
500  Hexact = exactC(time, r).H;
501  break;
502  case 'D':
503  Hexact = exactD(time, r).H;
504  break;
505  case 'F':
506  if (r > m_LforFG - 1.0) { // outside of sheet
507  Hexact=0.0;
508  } else {
509  r=std::max(r,1.0);
510  std::vector<double> z(1, 0.0);
511  Hexact = exactFG(0.0, r, z, 0.0).H;
512  }
513  break;
514  case 'G':
515  if (r > m_LforFG -1.0) { // outside of sheet
516  Hexact=0.0;
517  } else {
518  r=std::max(r,1.0);
519  std::vector<double> z(1, 0.0);
520  Hexact = exactFG(time, r, z, m_ApforG).H;
521  }
522  break;
523  case 'H':
524  Hexact = exactH(m_f, time, r).H;
525  break;
526  case 'K':
527  case 'O':
528  Hexact = 3000.0;
529  break;
530  case 'L':
531  Hexact = m_HexactL(i,j);
532  break;
533  case 'V':
534  {
535  double
536  H0 = 600.0,
537  v0 = convert(m_sys, 300.0, "m year-1", "m second-1"),
538  Q0 = H0 * v0,
539  B0 = m_stress_balance->shallow()->flow_law()->hardness(0, 0),
540  C = pow(ice_density * standard_gravity * (1.0 - ice_density/seawater_density) / (4 * B0), 3);
541 
542  Hexact = pow(4 * C / Q0 * xx + 1/pow(H0, 4), -0.25);
543  }
544  break;
545  default:
546  throw RuntimeError(PISM_ERROR_LOCATION, "test must be A, B, C, D, F, G, H, K, L, or O");
547  }
548 
549  if (Hexact > 0) {
550  areaexact += a;
551  volexact += a * Hexact * 1e-3;
552  }
553  if (i == ((int)m_grid->Mx() - 1)/2 and
554  j == ((int)m_grid->My() - 1)/2) {
555  domeH = m_geometry.ice_thickness(i,j);
556  domeHexact = Hexact;
557  }
558  // compute maximum errors
559  Herr = std::max(Herr,fabs(m_geometry.ice_thickness(i,j) - Hexact));
560  etaerr = std::max(etaerr,fabs(pow(m_geometry.ice_thickness(i,j),m) - pow(Hexact,m)));
561  // add to sums for average errors
562  avHerr += fabs(m_geometry.ice_thickness(i,j) - Hexact);
563  }
564  } catch (...) {
565  loop.failed();
566  }
567  loop.check();
568 
569  // globalize (find errors over all processors)
570  double gvol, garea, gdomeH;
571  gvolexact = GlobalSum(m_grid->com, volexact);
572  gdomeHexact = GlobalMax(m_grid->com, domeHexact);
573  gareaexact = GlobalSum(m_grid->com, areaexact);
574 
575  gvol = GlobalSum(m_grid->com, vol);
576  garea = GlobalSum(m_grid->com, area);
577  volerr = fabs(gvol - gvolexact);
578  areaerr = fabs(garea - gareaexact);
579 
580  gmaxHerr = GlobalMax(m_grid->com, Herr);
581  gavHerr = GlobalSum(m_grid->com, avHerr);
582  gavHerr = gavHerr/(m_grid->Mx()*m_grid->My());
583  gmaxetaerr = GlobalMax(m_grid->com, etaerr);
584 
585  gdomeH = GlobalMax(m_grid->com, domeH);
586  centerHerr = fabs(gdomeH - gdomeHexact);
587 }

622  // -- volume error
623  // -- area error

625  // -- max T error over 3D domain of ice

631  // -- max Tb error over 3D domain of bedrock
632  // -- av Tb error over 3D domain of bedrock
633  // and strain-heating (Sigma) errors (for tests F & G):
634  // -- max Sigma error over 3D domain of ice (in 10^-3 K a^-1)
635  // -- av Sigma error over 3D domain of ice (in 10^-3 K a^-1)
636  // and basal melt rate error (for test O):
637  // -- max bmelt error over base of ice
638  // and surface velocity errors (for tests F & G):
639  // -- max |<us,vs> - <usex,vsex>| error
640  // -- av |<us,vs> - <usex,vsex>| error
641  // -- max ws error
642  // -- av ws error
643  // and basal sliding errors (for test E):
644  // -- max ub error
645  // -- max vb error
646  // -- max |<ub,vb> - <ubexact,vbexact>| error
647  // -- av |<ub,vb> - <ubexact,vbexact>| error
648 
649  bool no_report = options::Bool("-no_report", "Don't report numerical errors");
650 
651  if (no_report) {
652  return;
653  }
654 
655  double maxbmelterr, minbmelterr, volexact, areaexact, domeHexact,
656  volerr, areaerr, maxHerr, avHerr, maxetaerr, centerHerr;

672  }
673 
674  m_log->message(1,
675  "NUMERICAL ERRORS evaluated at final time (relative to exact solution):\n");
676 
677 
678  // geometry (thickness, vol) errors if appropriate; reported in m except for relmaxETA
679  if ((m_testname != 'K') && (m_testname != 'O')) {
680  computeGeometryErrors(volexact,areaexact,domeHexact,
681  volerr,areaerr,maxHerr,avHerr,maxetaerr,centerHerr);
682  m_log->message(1,

693  computeBasalTemperatureErrors(basemaxTerr, baseavTerr, basecenterTerr);

701  m_log->message(1,
