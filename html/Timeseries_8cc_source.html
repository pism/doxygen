56 //! Read timeseries data from a NetCDF file `filename`.
57 void Timeseries::read(const File &file, const Time &time_manager, const Logger &log) {
58 
59  std::string standard_name = m_variable.get_string("standard_name");
60 
61  auto var = file.find_variable(name(), standard_name);
62 
63  if (not var.exists) {
64  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "Can't find '%s' ('%s') in '%s'.\n",
65  name().c_str(), standard_name.c_str(),
66  file.filename().c_str());
67  }
68 
69  auto dims = file.dimensions(var.name);
70 
71  if (dims.size() != 1) {
72  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
73  "Variable '%s' in '%s' depends on %d dimensions,\n"
74  "but a time-series variable can only depend on 1 dimension.",
75  name().c_str(),
76  file.filename().c_str(),
77  (int)dims.size());
78  }
79 
80  auto time_name = dims[0];
81 
82  VariableMetadata time_dimension(time_name, m_unit_system);
83  time_dimension.set_string("units", time_manager.units_string());
84 
85  io::read_timeseries(file, time_dimension, time_manager, log, m_time);
86 
87  if (not is_increasing(m_time)) {
88  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
89  "dimension '%s' has to be strictly increasing (read from '%s').",
90  time_dimension.get_name().c_str(), file.filename().c_str());
91  }
92 
93  std::string time_bounds_name = file.read_text_attribute(time_name, "bounds");
94 
95  if (not time_bounds_name.empty()) {
96  m_use_bounds = true;
97 
98  VariableMetadata time_bounds(time_bounds_name, m_unit_system);
99  time_bounds.set_string("units", time_dimension.get_string("units"));
100  time_bounds.set_string("glaciological_units", time_dimension.get_string("glaciological_units"));
101 
102  io::read_time_bounds(file, time_bounds, time_manager, log, m_time_bounds);
103 
104  // Time bounds override the time dimension read from the input file.
105  //
106  // Note: we use the right end point of each interval.
107  for (unsigned int k = 0; k < m_time.size(); ++k) {
108  m_time[k] = m_time_bounds[2 * k + 1];
109  }
110  } else {
111  m_use_bounds = false;
112  }
113 
114  io::read_timeseries(file, m_variable, time_manager, log, m_values);
115 
116  if (m_time.size() != m_values.size()) {
117  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "variables %s and %s in %s have different numbers of values.",
118  time_name.c_str(),
119  m_variable.get_name().c_str(),
120  file.filename().c_str());
121  }
122 
123  report_range(log);
124 }

126 //! \brief Report the range of a time-series stored in `values`.
127 void Timeseries::report_range(const Logger &log) {
128  double min, max;
129 
130  // min_element and max_element return iterators; "*" is used to get
131  // the value corresponding to this iterator
132  min = *std::min_element(m_values.begin(), m_values.end());
133  max = *std::max_element(m_values.begin(), m_values.end());
134 
135  units::Converter c(m_unit_system,
136  m_variable.get_string("units"),
137  m_variable.get_string("glaciological_units"));
138  min = c(min);
139  max = c(max);
140 
141  std::string spacer(m_variable.get_name().size(), ' ');
142 
143  log.message(2,
144  " FOUND %s / %-60s\n"
145  " %s \\ min,max = %9.3f,%9.3f (%s)\n",
146  m_variable.get_name().c_str(),
147  m_variable.get_string("long_name").c_str(), spacer.c_str(), min, max,
148  m_variable.get_string("glaciological_units").c_str());
149 }

151 //! Get a value of timeseries at time `t`.
152 /*! Returns the first value or the last value if t is out of range on the left
153  and right, respectively.
154 
155  Uses time bounds if present (interpreting data as piecewise-constant) and
156  uses linear interpolation otherwise.
157  */
158 double Timeseries::operator()(double t) const {
159 
160  if (m_use_bounds) {
161  // piecewise-constant case
162 
163  size_t k = 0;
164  if (t < m_time[0]) {
165  k = 0;
166  } else if (t >= m_time.back()) {
167  k = m_time.size() - 1;
168  } else {
169  k = gsl_interp_bsearch(m_time.data(), t, 0, m_time.size()) + 1;
170  }
171 
172  return m_values[k];
173  } else {
174  // piecewise-linear case
175 
176  // extrapolation on the left
177  if (t < m_time[0]) {
178  return m_values[0];
179  }
180 
181  size_t k = gsl_interp_bsearch(m_time.data(), t, 0, m_time.size());
182 
183  // extrapolation on the right
184  if (k + 1 >= m_time.size()) {
185  return m_values.back();
186  }
187 
188  double alpha = (t - m_time[k]) / (m_time[k + 1] - m_time[k]);
189 
190  return m_values[k] * (1.0 - alpha) + m_values[k + 1] * alpha;
191  }
192 }

208 //! @brief Compute an average of a time-series over interval (t,t+dt) using trapezoidal
209 //! rule with N sub-intervals.
210 double Timeseries::average(double t, double dt, unsigned int N) const {
211  std::vector<double> V(N+1);
212 
213  for (unsigned int i = 0; i < N+1; ++i) {
214  double t_i = t + (dt / N) * i;
215  V[i] = (*this)(t_i);
216  }
217 
218  double sum = 0;
219  for (unsigned int i = 0; i < N; ++i) {
220  sum += V[i] + V[i+1];
221  }
222 
223  return sum / (2.0*N);
224 }
