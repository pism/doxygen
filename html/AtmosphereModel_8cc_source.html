104 /*! @brief Instantaneous near-surface air temperature. */
105 class AirTemperatureSnapshot : public Diag<AtmosphereModel> {
106 public:
107  AirTemperatureSnapshot(const AtmosphereModel *m)
108  : Diag<AtmosphereModel>(m) {
109 
110  /* set metadata: */
111  m_vars = {SpatialVariableMetadata(m_sys, "air_temp_snapshot")};
112 
113  set_attrs("instantaneous value of the near-surface air temperature",
114  "", // no standard name
115  "Kelvin", "Kelvin", 0);
116  }
117 protected:
118  IceModelVec::Ptr compute_impl() const {
119 
120  IceModelVec2S::Ptr result(new IceModelVec2S(m_grid, "air_temp_snapshot", WITHOUT_GHOSTS));
121  result->metadata(0) = m_vars[0];
122 
123  std::vector<double> current_time(1, m_grid->ctx()->time()->current());
124  std::vector<double> temperature(1, 0.0);
125 
126  model->init_timeseries(current_time);
127 
128  model->begin_pointwise_access();
129 
130  IceModelVec::AccessList list(*result);
131  ParallelSection loop(m_grid->com);
132  try {
133  for (Points p(*m_grid); p; p.next()) {
134  const int i = p.i(), j = p.j();
135 
136  model->temp_time_series(i, j, temperature);
137 
138  (*result)(i, j) = temperature[0];
139  }
140  } catch (...) {
141  loop.failed();
142  }
143  loop.check();
144 
145  model->end_pointwise_access();
146 
147  return result;
148  }
149 };
