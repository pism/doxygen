70 void Cache::update_impl(const Geometry &geometry, double t, double dt) {
71  // ignore dt and always use 1 year long time-steps when updating
72  // the input model
73  (void) dt;
74 
75  if (t >= m_next_update_time or fabs(t - m_next_update_time) < 1.0) {
76 
77  double
78  one_year_from_now = m_grid->ctx()->time()->increment_date(t, 1.0),
79  update_dt = one_year_from_now - t;
80 
81  assert(update_dt > 0.0);
82 
83  m_input_model->update(geometry, t, update_dt);
84 
85  m_next_update_time = m_grid->ctx()->time()->increment_date(m_next_update_time,
86  m_update_interval_years);
87 
88  // store outputs of the input model

100 MaxTimestep Cache::max_timestep_impl(double t) const {
101  double dt = m_next_update_time - t;
102 
103  // if we got very close to the next update time, set time step
104  // length to the interval between updates
105  if (dt < 1.0) {
106  double update_time_after_next = m_grid->ctx()->time()->increment_date(m_next_update_time,
107  m_update_interval_years);
108 
109  dt = update_time_after_next - m_next_update_time;
110  assert(dt > 0.0);
111  }
112 
113  assert(m_input_model != NULL);
114 
115  MaxTimestep cache_dt(dt, "surface cache");
116 
117  MaxTimestep input_max_timestep = m_input_model->max_timestep(t);
118  if (input_max_timestep.finite()) {
119  return std::min(input_max_timestep, cache_dt);
120  } else {
121  return cache_dt;
122  }
123 }
