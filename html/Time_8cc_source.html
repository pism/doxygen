39 std::string calendar_from_options(MPI_Comm com, const Config &config) {
40  // Set the default calendar using the config. parameter or the
41  // "-calendar" option:
42  std::string result = config.get_string("time.calendar");
43 
44  // Check if -time_file was set and override the setting above if the
45  // "calendar" attribute is found.

50  std::string time_name = config.get_string("time.dimension_name");
51  if (file.find_variable(time_name)) {
52  std::string tmp = file.read_text_attribute(time_name, "calendar");
53  if (not tmp.empty()) {

56  }

148 void Time::init_calendar(const std::string &calendar_string) {
149 
150  if (not pism_is_valid_calendar_name(calendar_string)) {
151  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "unsupported calendar: %s", calendar_string.c_str());
152  }
153 
154  m_calendar_string = calendar_string;
155 
156  double seconds_per_day = convert(m_unit_system, 1.0, "day", "seconds");
157  if (calendar_string == "360_day") {
158  m_year_length = 360 * seconds_per_day;
159  } else if (member(calendar_string, {"365_day", "noleap"})) {
160  m_year_length = 365 * seconds_per_day;
161  } else {
162  // use the ~365.2524-day year
163  m_year_length = convert(m_unit_system, 1.0, "year", "seconds");
164  }

218 std::string Time::CF_units_to_PISM_units(const std::string &input) const {
219  std::string units = input;
220  size_t n = units.find("since");
221 
222  /*!
223  \note This code finds the string "since" in the units_string and
224  terminates it on the first 's' of "since", if this sub-string was found.
225  This is done to ignore the reference date in the time units string (the
226  reference date specification always starts with this word).
227  */
228  if (n != std::string::npos) {
229  units.resize(n);
230  }
231 
232  // strip trailing spaces
233  while (ends_with(units, " ") && not units.empty()) {
234  units.resize(units.size() - 1); // this would fail on empty strings
235  }
236 
237  return units;
238 }

291 void Time::init(const Logger &log) {
292 
293  (void) log;
294 
295  double y_seconds, ys_seconds, ye_seconds;
296 
297  // At this point the calendar and the year length are set (in the
298  // constructor). The Time_Calendar class will (potentially)
299  // override all this by using settings from -time_file, so that is
300  // fine, too.
301 
302  bool y_set = process_y(y_seconds);
303  bool ys_set = process_ys(ys_seconds);
304  bool ye_set = process_ye(ye_seconds);
305 
306  if (ys_set and ye_set and y_set) {
307  throw RuntimeError(PISM_ERROR_LOCATION, "all of -y, -ys, -ye are set.");
308  }
309 
310  if (y_set and ye_set) {
311  throw RuntimeError(PISM_ERROR_LOCATION, "using -y and -ye together is not allowed.");
312  }
313 
314  // Set the start year if -ys is set, use the default otherwise.
315  if (ys_set) {
316  m_run_start = ys_seconds;
317  }
318 
319  m_time_in_seconds = m_run_start;
320 
321  if (ye_set) {
322  if (ye_seconds < m_time_in_seconds) {
323  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "-ye (%s) is less than -ys (%s) (or input file year or default).\n"
324  "PISM cannot run backward in time.",
325  date(ye_seconds).c_str(), date(m_run_start).c_str());
326  }
327  m_run_end = ye_seconds;
328  } else if (y_set) {
329  m_run_end = m_run_start + y_seconds;
330  } else {
331  m_run_end = increment_date(m_run_start, (int)m_config->get_number("time.run_length"));
332  }
333 }

355 double Time::mod(double time, unsigned int period_years) const {
356  if (period_years == 0) {
357  return time;
358  }
359 
360  double period_seconds = years_to_seconds(period_years);
361 
362  double tmp = time - floor(time / period_seconds) * period_seconds;
363 
364  if (fabs(tmp - period_seconds) < 1) {
365  tmp = 0;
366  }
367 
368  return tmp;
369 }

415 /**
416  * Parses an interval specification string.
417  *
418  * @param[in] spec specification string
419  *
420  */
421 auto Time::parse_interval_length(const std::string &spec) const -> Interval {
422 
423  // check if it is a keyword
424  if (spec == "hourly") {
425  return {3600.0, SIMPLE};
426  }
427 
428  if (spec == "daily") {
429  return {86400.0, SIMPLE};
430  }
431 
432  if (spec == "monthly") {
433  return {0.0, MONTHLY};
434  }
435 
436  if (spec == "yearly") {
437  return {0.0, YEARLY};
438  }
439 
440  try {
441  units::Unit seconds(m_unit_system, "seconds"),
442  one(m_unit_system, "1"),
443  tmp(m_unit_system, spec);
444 
445  // Check if these units are compatible with "seconds" or "1". The
446  // latter allows intervals of the form "0.5", which stands for "half
447  // of a model year". This also discards interval specs such as "days
448  // since 1-1-1", even though "days" is compatible with "seconds".
449  if (units::are_convertible(tmp, seconds)) {
450  units::Converter c(tmp, seconds);
451 
452  return {c(1.0), SIMPLE};
453  } else if (units::are_convertible(tmp, one)) {
454  units::Converter c(tmp, one);
455 
456  // convert from years to seconds without using UDUNITS-2 (this
457  // way we handle 360-day and 365-day years correctly)
458  return {years_to_seconds(c(1.0)), SIMPLE};
459  }
460  } catch (RuntimeError &e) {
461  e.add_context("processing interval length " + spec);
462  throw;
463  }
464 
465  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
466  "interval length '%s' is invalid", spec.c_str());
467 }

470 std::vector<double> Time::parse_range(const std::string &spec) const {
471  double
472  time_start = m_run_start,
473  time_end = m_run_end;
474 
475  Interval I{0.0, SIMPLE};
476 
477  if (spec == "hourly") {
478  I = {3600.0, SIMPLE};
479  } else if (spec == "daily") {
480  I = {86400.0, SIMPLE};
481  } else if (spec == "monthly") {
482  I = {0.0, MONTHLY};
483  } else if (spec == "yearly") {
484  I = {0.0, YEARLY};
485  } else {
486 
487  auto parts = pism::split(spec, ':');
488 
489  if (parts.size() == 1) {
490  I = parse_interval_length(parts[0]);
491 
492  } else if (parts.size() == 3) {
493  time_start = parse_date(parts[0]);
494  I = parse_interval_length(parts[1]);
495  time_end = parse_date(parts[2]);
496  } else {
497  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
498  "a time range must consist of exactly 3 parts separated by colons (got '%s').",
499  spec.c_str());
500  }
501  }
502 
503  std::vector<double> result;
504  compute_times(time_start, time_end, I, result);
505  return result;
506 }

510

526 /**
527  * Compute times corresponding to a "simple" time range.
528  *
529  * This is a time range with a fixed step ("hourly" is an example).
530  *
531  * @param[in] time_start beginning of the time interval, in seconds
532  * @param[in] delta step, in seconds
533  * @param[in] time_end end of the interval, in seconds
534  * @param[out] result list of model times
535  *
536  */
537 void Time::compute_times_simple(double time_start, double delta, double time_end,
538  std::vector<double> &result) const {
539  if (time_start >= time_end) {
540  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "a >= b in time range a:dt:b (got a = %s, b = %s)",
541  this->date(time_start).c_str(), this->date(time_end).c_str());
542  }
543 
544  if (delta <= 0) {
545  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "dt <= 0 in time range a:dt:b (got dt = %f seconds)", delta);
546  }
547 
548  int k = 0;
549  double t = time_start;
550 
551  result.clear();
552  do {
553  if (t >= this->start() && t <= this->end()) {
554  result.push_back(t);
555  }
556  k += 1;
557  t = time_start + k * delta;
558  } while (t <= time_end);
559 }

561 void Time::compute_times(double time_start, double time_end,
562  const Interval &interval,
563  std::vector<double> &result) const {
564  double delta = interval.dt;
565  switch (interval.type) {
566  case YEARLY:
567  delta = years_to_seconds(1.0);
568  break;
569  case MONTHLY:
570  delta = years_to_seconds(1.0/12.0);
571  break;
572  case SIMPLE:
573  break;
574  default:
575  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
576  "unknown time range type: %d",
577  interval.type);
578  }
579 
580  compute_times_simple(time_start, delta, time_end, result);
581 }
