36 static inline std::string string_strip(std::string input) {
37  if (input.empty() == true) {
38  return input;
39  }
40 
41  // strip leading spaces
42  input.erase(0, input.find_first_not_of(" \t"));
43 
44  // strip trailing spaces
45  input.substr(input.find_last_not_of(" \t"));
46 
47  return input;
48 }

72

76  e.add_context("setting time units");

81  m_run_end = increment_date(m_run_start, (int)m_config->get_number("time.run_length"));
82

141 void Time_Calendar::init_from_input_file(const File &nc,
142  const std::string &time_name,
143  const Logger &log) {
144  try {
145  // Set the calendar name from file, unless we are re-starting from a PISM run using the "none"
146  // calendar.
147  std::string new_calendar = nc.read_text_attribute(time_name, "calendar");
148  if (not new_calendar.empty() and
149  not (new_calendar == "none")) {
150  init_calendar(new_calendar);
151  }
152 
153  // Set the reference date of internal units.
154  {
155  std::string date_string = reference_date_from_file(nc, time_name);
156  m_time_units = units::Unit(m_unit_system, "seconds " + date_string);
157  }
158 
159  // Read time information from the file. (PISM output files don't have time bounds, so we don't
160  // bother checking for them.)
161  std::vector<double> time;
162  {
163  VariableMetadata time_axis(time_name, m_unit_system);
164  time_axis.set_string("units", m_time_units.format());
165 
166  io::read_timeseries(nc, time_axis, *this, log, time);
167  }
168 
169  // Set time.
170  this->set_start(time.front());
171  this->set(time.front());
172  log.message(2,
173  "* Time t = %s (calendar: %s) found in '%s'; setting current time\n",
174  this->date().c_str(),
175  this->calendar().c_str(),
176  nc.filename().c_str());
177  } catch (RuntimeError &e) {
178  e.add_context("initializing model time from \"%s\"", nc.filename().c_str());
179  throw;
180  }

205 //! \brief Sets the time from a NetCDF file with a time dimension (`-time_file`).
206 /*!
207  * This allows running PISM for the duration of the available forcing.
208  */
209 void Time_Calendar::init_from_file(const std::string &filename, const Logger &log,
210  bool set_start_time) {
211  try {
212  std::string time_name = m_config->get_string("time.dimension_name");
213 
214  File file(m_com, filename, PISM_NETCDF3, PISM_READONLY); // OK to use netcdf3
215 
216  // Set the calendar name from file.
217  std::string new_calendar = file.read_text_attribute(time_name, "calendar");
218  if (not new_calendar.empty()) {
219  init_calendar(new_calendar);
220  }
221 
222  // Set the reference date of internal units.
223  {
224  std::string date_string = reference_date_from_file(file, time_name);
225  m_time_units = units::Unit(m_unit_system, "seconds " + date_string);
226  }
227 
228  // Read time information from the file.
229  std::vector<double> time;
230  std::string time_bounds_name = file.read_text_attribute(time_name, "bounds");
231  if (not time_bounds_name.empty()) {
232  // use the time bounds
233  VariableMetadata bounds(time_bounds_name, m_unit_system);
234  bounds.set_string("units", m_time_units.format());
235 
236  io::read_time_bounds(file, bounds, *this, log, time);
237  } else {
238  // use the time axis
239  VariableMetadata time_axis(time_name, m_unit_system);
240  time_axis.set_string("units", m_time_units.format());
241 
242  io::read_timeseries(file, time_axis, *this, log, time);
243  }
244 
245  // Set time.
246  if (set_start_time) {
247  this->set_start(time.front());
248  this->set(time.front());
249  } else {
250  log.message(2, "* Using start time from an -i file to continue an interrupted run.\n");
251  }
252  this->set_end(time.back());
253  } catch (RuntimeError &e) {
254  e.add_context("initializing model time from \"%s\"", filename.c_str());
255  throw;
256  }

264 double Time_Calendar::year_fraction(double T) const {
265 
266  auto D = m_time_units.date(T, m_calendar_string);
267 
268  units::DateTime D2{D.year, 1, 1, 0, 0, 0.0};
269 
270  auto year_start = m_time_units.time(D2, m_calendar_string);
271 
272  D2.year += 1;
273  auto next_year_start = m_time_units.time(D2, m_calendar_string);
274 
275  return (T - year_start) / (next_year_start - year_start);
276 }

293 double Time_Calendar::increment_date(double T, int years) const {
294 
295  // Get the date corresponding to time T:
296  auto date = m_time_units.date(T, m_calendar_string);
297 
298  calcalcs_cal *cal = NULL;
299  int errcode, leap = 0;
300  cal = ccs_init_calendar(m_calendar_string.c_str());
301  assert(cal != NULL);
302  errcode = ccs_isleap(cal, date.year + years, &leap);
303  assert(errcode == 0);
304  ccs_free_calendar(cal);
305 
306  if (leap == 0 and date.month == 2 and date.day == 29) {
307  PetscErrorCode ierr = PetscPrintf(m_com,
308  "PISM WARNING: date %d year(s) since %d-%d-%d does not exist."
309  " Using %d-%d-%d instead of %d-%d-%d.\n",
310  years,
311  date.year, date.month, date.day,
312  date.year + years, date.month, date.day-1,
313  date.year + years, date.month, date.day);
314  PISM_CHK(ierr, "PetscPrintf");
315  date.day -= 1;
316  }
317 
318  date.year += years;
319 
320  // Return the time in seconds corresponding to the new date.
321  return m_time_units.time(date, m_calendar_string);
322 }

324 /**
325  * Parses the date string `input`.
326  *
327  * @param[in] input date string of the form "Y-M-D", where year, month,
328  * and day can use as many digits as necessary
329  * @param[out] result
330  *
331  * If `result` is NULL, then it validates the date string, but does
332  * not try to convert to seconds since the reference date. This makes
333  * it possible to validate the reference date itself.
334  *
335  */
336 double Time_Calendar::parse_date(const std::string &input) const {
337 
338  std::string spec = string_strip(input);
339 
340  if (spec.empty()) {
341  throw RuntimeError(PISM_ERROR_LOCATION, "got an empty date specification");
342  }
343 
344  // If the string starts with "-" then the year is negative. This
345  // would confuse the code below, which treats "-" as a separator, so
346  // we remember that the year is negative and remove "-".
347  bool year_is_negative = false;
348  if (spec[0] == '-') {
349  year_is_negative = true;
350  spec.substr(1);
351  }
352 
353  auto parts = split(spec, '-');
354 
355  if (parts.size() != 3) {
356  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
357  "date specification '%s' is invalid (should have 3 parts: YYYY-MM-DD, got %d)",
358  spec.c_str(), (int)parts.size());
359  }
360 
361  std::vector<int> numbers;
362  for (const auto &p : parts) {
363  // check if strtol can parse it:
364  char *endptr = NULL;
365  long int n = strtol(p.c_str(), &endptr, 10);
366  if (*endptr != '\0') {
367  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
368  "date specification '%s' is invalid ('%s' is not an integer)",
369  spec.c_str(), p.c_str());
370  }
371 
372  // FIXME: this may overflow!
373  numbers.push_back((int)n);
374  }
375 
376  if (year_is_negative) {
377  numbers[0] *= -1;
378  }
379 
380  // Validate the calendar string and the date in this calendar:
381  {
382  calcalcs_cal *cal = ccs_init_calendar(m_calendar_string.c_str());
383  if (cal == NULL) {
384  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
385  "calendar string '%s' is invalid",
386  m_calendar_string.c_str());
387  }
388 
389  int dummy = 0;
390  int errcode = ccs_date2jday(cal, numbers[0], numbers[1], numbers[2], &dummy);
391  if (errcode != 0) {
392  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
393  "date %s is invalid in the %s calendar",
394  spec.c_str(), m_calendar_string.c_str());
395  }
396  ccs_free_calendar(cal);
397  }
398 
399  units::DateTime d{numbers[0], numbers[1], numbers[2], 0, 0, 0.0};
400 
401  return m_time_units.time(d, m_calendar_string);
402 }

404 auto Time_Calendar::parse_interval_length(const std::string &spec) const -> Interval {
405  // do not allow intervals specified in terms of "fuzzy" units
406 
407  // This is called *only* if the 'spec' is *not* one of "monthly",
408  // "yearly", "daily", "hourly", so we don't need to worry about
409  // spec.find("...") finding "year" in "yearly".
410  if (spec.find("year") != std::string::npos or spec.find("month") != std::string::npos) {
411  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
412  "interval length '%s' with the calendar '%s' is not supported",
413  spec.c_str(), m_calendar_string.c_str());
414  }
415 
416  return Time::parse_interval_length(spec);
417 }

419 void Time_Calendar::compute_times_monthly(std::vector<double> &result) const {
420 
421  double time = m_run_start;
422 
423  // get the date corresponding to the current time
424  auto date = m_time_units.date(time, m_calendar_string);
425 
426  // beginning of the current month
427  units::DateTime d{date.year, date.month, 1, 0, 0, 0.0};
428 
429  result.clear();
430  while (true) {
431  // find the time corresponding to the beginning of the current month
432  time = m_time_units.time(d, m_calendar_string);
433 
434  if (time > m_run_end) {
435  break;
436  }
437 
438  if (time >= m_run_start and time <= m_run_end) {
439  result.push_back(time);
440  }
441 
442  if (d.month == 12) {
443  d.year += 1;
444  d.month = 1;
445  } else {
446  d.month += 1;
447  }
448  }
449 }
