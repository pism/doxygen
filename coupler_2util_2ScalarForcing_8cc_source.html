53 void ScalarForcing::init() {
54 
55  Config::ConstPtr config = m_ctx->config();
56 
57  auto filename = config->get_string(m_prefix + ".file");
58  int period = config->get_number(m_prefix + ".period");
59  double reference_year = config->get_number(m_prefix + ".reference_year");
60 
61  if (filename.empty()) {
62  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
63  "%s.file is required", m_prefix.c_str());
64  }
65 
66  if (period < 0.0) {
67  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
68  "invalid %s.period %d (period length cannot be negative)",
69  m_prefix.c_str(), period);
70  }
71 
72  m_period = (unsigned int)period;
73 
74  m_reference_time = units::convert(m_ctx->unit_system(),
75  reference_year, "years", "seconds");
76 
77  m_ctx->log()->message(2,
78  " reading %s data from forcing file %s...\n",
79  m_data->name().c_str(), filename.c_str());
80 
81  File file(m_ctx->com(), filename, PISM_NETCDF3, PISM_READONLY);
82  {
83  m_data->read(file, *m_ctx->time(), *m_ctx->log());
84  }
85 }
