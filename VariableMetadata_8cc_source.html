80 //! @brief Check if the range `[min, max]` is a subset of `[valid_min, valid_max]`.
81 /*! Throws an exception if this check failed.
82  */
83 void VariableMetadata::check_range(const std::string &filename, double min, double max) const {
84 
85  auto units_string = get_string("units");
86  auto name_string = get_name();
87  const char
88  *units = units_string.c_str(),
89  *name = name_string.c_str(),
90  *file = filename.c_str();
91 
92  if (has_attribute("valid_min") and has_attribute("valid_max")) {
93  double valid_min = get_number("valid_min");
94  double valid_max = get_number("valid_max");
95  if ((min < valid_min) or (max > valid_max)) {
96  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "some values of '%s' in '%s' are outside the valid range [%e, %e] (%s).\n"
97  "computed min = %e %s, computed max = %e %s",
98  name, file,
99  valid_min, valid_max, units, min, units, max, units);
100  }
101  } else if (has_attribute("valid_min")) {
102  double valid_min = get_number("valid_min");
103  if (min < valid_min) {
104  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "some values of '%s' in '%s' are less than the valid minimum (%e %s).\n"
105  "computed min = %e %s, computed max = %e %s",
106  name, file,
107  valid_min, units, min, units, max, units);
108  }
109  } else if (has_attribute("valid_max")) {
110  double valid_max = get_number("valid_max");
111  if (max > valid_max) {
112  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "some values of '%s' in '%s' are greater than the valid maximum (%e %s).\n"
113  "computed min = %e %s, computed max = %e %s",
114  name, file,
115  valid_max, units, min, units, max, units);
116  }
117  }
118 }

158 void VariableMetadata::report_range(const Logger &log, double min, double max,
159  bool found_by_standard_name) {
160 
161  // units::Converter constructor will make sure that units are compatible.
162  units::Converter c(m_unit_system,
163  this->get_string("units"),
164  this->get_string("glaciological_units"));
165  min = c(min);
166  max = c(max);
167 
168  std::string spacer(get_name().size(), ' ');
169 
170  if (has_attribute("standard_name")) {
171 
172  if (found_by_standard_name) {
173  log.message(2,
174  " %s / standard_name=%-10s\n"
175  " %s \\ min,max = %9.3f,%9.3f (%s)\n",
176  get_name().c_str(),
177  get_string("standard_name").c_str(), spacer.c_str(), min, max,
178  get_string("glaciological_units").c_str());
179  } else {
180  log.message(2,
181  " %s / WARNING! standard_name=%s is missing, found by short_name\n"
182  " %s \\ min,max = %9.3f,%9.3f (%s)\n",
183  get_name().c_str(),
184  get_string("standard_name").c_str(), spacer.c_str(), min, max,
185  get_string("glaciological_units").c_str());
186  }
187 
188  } else {
189  log.message(2,
190  " %s / %-10s\n"
191  " %s \\ min,max = %9.3f,%9.3f (%s)\n",
192  get_name().c_str(),
193  get_string("long_name").c_str(), spacer.c_str(), min, max,
194  get_string("glaciological_units").c_str());

298 //! Set a string attribute.
299 void VariableMetadata::set_string(const std::string &name, const std::string &value) {
300 
301  if (name == "units") {
302  // create a dummy object to validate the units string
303  units::Unit tmp(m_unit_system, value);
304 
305  m_strings[name] = value;
306  m_strings["glaciological_units"] = value;
307  } else if (name == "glaciological_units") {
308  m_strings[name] = value;
309 
310  units::Unit internal(m_unit_system, get_string("units"));
311  units::Unit glaciological(m_unit_system, value);
312 
313  if (not units::are_convertible(internal, glaciological)) {
314  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "units \"%s\" and \"%s\" are not compatible",
315  get_string("units").c_str(), value.c_str());
316  }
317  } else if (name == "short_name") {
318  set_name(name);
319  } else {
320  m_strings[name] = value;
321  }
322 }

341 void VariableMetadata::report_to_stdout(const Logger &log, int verbosity_threshold) const {
342 
343  const VariableMetadata::StringAttrs &strings = this->all_strings();
344  const VariableMetadata::DoubleAttrs &doubles = this->all_doubles();
345 
346  // Find the maximum name length so that we can pad output below:
347  size_t max_name_length = 0;
348  for (const auto &s : strings) {
349  max_name_length = std::max(max_name_length, s.first.size());
350  }
351  for (const auto &d : doubles) {
352  max_name_length = std::max(max_name_length, d.first.size());
353  }
354 
355  // Print text attributes:
356  for (const auto &s : strings) {
357  std::string name = s.first;
358  std::string value = s.second;
359  std::string padding(max_name_length - name.size(), ' ');
360 
361  if (value.empty()) {
362  continue;
363  }
364 
365  log.message(verbosity_threshold, " %s%s = \"%s\"\n",
366  name.c_str(), padding.c_str(), value.c_str());
367  }
368 
369  // Print double attributes:
370  for (const auto &d : doubles) {
371  std::string name = d.first;
372  std::vector<double> values = d.second;
373  std::string padding(max_name_length - name.size(), ' ');
374 
375  if (values.empty()) {
376  continue;
377  }
378 
379  const double large = 1.0e7;
380  const double small = 1.0e-4;
381  if ((std::fabs(values[0]) >= large) || (std::fabs(values[0]) <= small)) {
382  // use scientific notation if a number is big or small
383  log.message(verbosity_threshold, " %s%s = %12.3e\n",
384  name.c_str(), padding.c_str(), values[0]);
385  } else {
386  log.message(verbosity_threshold, " %s%s = %12.5f\n",
387  name.c_str(), padding.c_str(), values[0]);
388  }
389 
390  }
391 }
