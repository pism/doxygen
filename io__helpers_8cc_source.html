41 //! \brief Bi-(or tri-)linear interpolation.
42 /*!
43  * This is the interpolation code itself.
44  *
45  * Note that its inputs are (essentially)
46  * - the definition of the input grid
47  * - the definition of the output grid
48  * - input array (lic->buffer)
49  * - output array (double *output_array)
50  *
51  * The `output_array` is expected to be big enough to contain
52  * `grid.xm()*`grid.ym()*length(zlevels_out)` numbers.
53  *
54  * We should be able to switch to using an external interpolation library
55  * fairly easily...
56  */
57 static void regrid(const IceGrid& grid, const std::vector<double> &zlevels_out,
58  LocalInterpCtx *lic, double *output_array) {
59  // We'll work with the raw storage here so that the array we are filling is
60  // indexed the same way as the buffer we are pulling from (input_array)
61 
62  const int X = 1, Z = 3; // indices, just for clarity
63 
64  unsigned int nlevels = zlevels_out.size();
65  double *input_array = &(lic->buffer[0]);
66 
67  // array sizes for mapping from logical to "flat" indices
68  int
69  x_count = lic->count[X],
70  z_count = lic->count[Z];
71 
72  for (Points p(grid); p; p.next()) {
73  const int i_global = p.i(), j_global = p.j();
74 
75  const int i = i_global - grid.xs(), j = j_global - grid.ys();
76 
77  // Indices of neighboring points.
78  const int
79  X_m = lic->x->left(i),
80  X_p = lic->x->right(i),
81  Y_m = lic->y->left(j),
82  Y_p = lic->y->right(j);
83 
84  for (unsigned int k = 0; k < nlevels; k++) {
85 
86  double
87  a_mm = 0.0,
88  a_mp = 0.0,
89  a_pm = 0.0,
90  a_pp = 0.0;
91 
92  if (nlevels > 1) {
93  const int
94  Z_m = lic->z->left(k),
95  Z_p = lic->z->right(k);
96 
97  const double alpha_z = lic->z->alpha(k);
98 
99  // We pretend that there are always 8 neighbors (4 in the map plane,
100  // 2 vertical levels). And compute the indices into the input_array for
101  // those neighbors.
102  const int
103  mmm = (Y_m * x_count + X_m) * z_count + Z_m,
104  mmp = (Y_m * x_count + X_m) * z_count + Z_p,
105  mpm = (Y_m * x_count + X_p) * z_count + Z_m,
106  mpp = (Y_m * x_count + X_p) * z_count + Z_p,
107  pmm = (Y_p * x_count + X_m) * z_count + Z_m,
108  pmp = (Y_p * x_count + X_m) * z_count + Z_p,
109  ppm = (Y_p * x_count + X_p) * z_count + Z_m,
110  ppp = (Y_p * x_count + X_p) * z_count + Z_p;
111 
112  // linear interpolation in the z-direction
113  a_mm = input_array[mmm] * (1.0 - alpha_z) + input_array[mmp] * alpha_z;
114  a_mp = input_array[mpm] * (1.0 - alpha_z) + input_array[mpp] * alpha_z;
115  a_pm = input_array[pmm] * (1.0 - alpha_z) + input_array[pmp] * alpha_z;
116  a_pp = input_array[ppm] * (1.0 - alpha_z) + input_array[ppp] * alpha_z;
117  } else {
118  // we don't need to interpolate vertically for the 2-D case
119  a_mm = input_array[Y_m * x_count + X_m];
120  a_mp = input_array[Y_m * x_count + X_p];
121  a_pm = input_array[Y_p * x_count + X_m];
122  a_pp = input_array[Y_p * x_count + X_p];
123  }
124 
125  // interpolation coefficient in the x direction
126  const double x_alpha = lic->x->alpha(i);
127  // interpolation coefficient in the y direction
128  const double y_alpha = lic->y->alpha(j);
129 
130  // interpolate in x direction
131  const double a_m = a_mm * (1.0 - x_alpha) + a_mp * x_alpha;
132  const double a_p = a_pm * (1.0 - x_alpha) + a_pp * x_alpha;
133 
134  int index = (j * grid.xm() + i) * nlevels + k;
135 
136  // index into the new array and interpolate in x direction
137  output_array[index] = a_m * (1.0 - y_alpha) + a_p * y_alpha;
138  // done with the point at (x,y,z)
139  }
140  }
141 }

143 static void compute_start_and_count(const File& file,
144  units::System::Ptr unit_system,
145  const std::string &short_name,
146  unsigned int t_start, unsigned int t_count,
147  unsigned int x_start, unsigned int x_count,
148  unsigned int y_start, unsigned int y_count,
149  unsigned int z_start, unsigned int z_count,
150  std::vector<unsigned int> &start,
151  std::vector<unsigned int> &count,
152  std::vector<unsigned int> &imap) {
153  std::vector<std::string> dims = file.dimensions(short_name);
154  unsigned int ndims = dims.size();
155 
156  assert(ndims > 0);
157  if (ndims == 0) {
158  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
159  "Cannot compute start and count: variable '%s' is a scalar.",
160  short_name.c_str());
161  }
162 
163  // Resize output vectors:
164  start.resize(ndims);
165  count.resize(ndims);
166  imap.resize(ndims);
167 
168  // Assemble start, count and imap:
169  for (unsigned int j = 0; j < ndims; j++) {
170  std::string dimname = dims[j];
171 
172  AxisType dimtype = file.dimension_type(dimname, unit_system);
173 
174  switch (dimtype) {
175  case T_AXIS:
176  start[j] = t_start;
177  count[j] = t_count;
178  imap[j] = x_count * y_count * z_count;
179  break;
180  case Y_AXIS:
181  start[j] = y_start;
182  count[j] = y_count;
183  imap[j] = x_count * z_count;
184  break;
185  case X_AXIS:
186  start[j] = x_start;
187  count[j] = x_count;
188  imap[j] = z_count;
189  break;
190  default:
191  case Z_AXIS:
192  start[j] = z_start;
193  count[j] = z_count;
194  imap[j] = 1;
195  break;
196  }
197  }
198 }

276 //! \brief Define dimensions a variable depends on.
277 static void define_dimensions(const SpatialVariableMetadata& var,
278  const IceGrid& grid, const File &file) {
279 
280  // x
281  std::string x_name = var.x().get_name();
282  if (not file.find_dimension(x_name)) {
283  define_dimension(file, grid.Mx(), var.x());
284  file.write_attribute(x_name, "spacing_meters", PISM_DOUBLE,
285  {grid.x(1) - grid.x(0)});
286  file.write_attribute(x_name, "not_written", PISM_INT, {1.0});
287  }
288 
289  // y
290  std::string y_name = var.y().get_name();
291  if (not file.find_dimension(y_name)) {
292  define_dimension(file, grid.My(), var.y());
293  file.write_attribute(y_name, "spacing_meters", PISM_DOUBLE,
294  {grid.y(1) - grid.y(0)});
295  file.write_attribute(y_name, "not_written", PISM_INT, {1.0});
296  }
297 
298  // z
299  std::string z_name = var.z().get_name();
300  if (not z_name.empty()) {
301  if (not file.find_dimension(z_name)) {
302  const std::vector<double>& levels = var.levels();
303  // make sure we have at least one level
304  unsigned int nlevels = std::max(levels.size(), (size_t)1);
305  define_dimension(file, nlevels, var.z());
306  file.write_attribute(z_name, "not_written", PISM_INT, {1.0});
307 
308  bool spatial_dim = not var.z().get_string("axis").empty();
309 
310  if (nlevels > 1 and spatial_dim) {
311  double dz_max = levels[1] - levels[0];
312  double dz_min = levels.back() - levels.front();
313 
314  for (unsigned int k = 0; k < nlevels - 1; ++k) {
315  double dz = levels[k+1] - levels[k];
316  dz_max = std::max(dz_max, dz);
317  dz_min = std::min(dz_min, dz);
318  }
319 
320  file.write_attribute(z_name, "spacing_min_meters", PISM_DOUBLE,
321  {dz_min});
322  file.write_attribute(z_name, "spacing_max_meters", PISM_DOUBLE,
323  {dz_max});
324  }
325  }
326  }
327 }

360 /**
361  * Check if the storage order of a variable in the current file
362  * matches the memory storage order used by PISM.
363  *
364  * @param[in] file input file
365  * @param var_name name of the variable to check
366  * @returns false if storage orders match, true otherwise
367  */
368 static bool use_transposed_io(const File &file,
369  units::System::Ptr unit_system,
370  const std::string &var_name) {
371 
372  std::vector<std::string> dimnames = file.dimensions(var_name);
373 
374  std::vector<AxisType> storage, memory = {Y_AXIS, X_AXIS};
375 
376  for (unsigned int j = 0; j < dimnames.size(); ++j) {
377  AxisType dimtype = file.dimension_type(dimnames[j], unit_system);
378 
379  if (j == 0 && dimtype == T_AXIS) {
380  // ignore the time dimension, but only if it is the first
381  // dimension in the list
382  continue;
383  }
384 
385  if (dimtype == X_AXIS || dimtype == Y_AXIS) {
386  storage.push_back(dimtype);
387  } else if (dimtype == Z_AXIS) {
388  memory.push_back(dimtype); // now memory = {Y_AXIS, X_AXIS, Z_AXIS}
389  // assume that this variable has only one Z_AXIS in the file
390  storage.push_back(dimtype);
391  } else {
392  // an UNKNOWN_AXIS or T_AXIS at index != 0 was found, use transposed I/O
393  return true;
394  }
395  }
396 
397  // we support 2D and 3D in-memory arrays, but not 4D
398  assert(memory.size() <= 3);
399 
400  if (storage == memory) {
401  // same storage order, do not use transposed I/O
402  return false;
403  } else {
404  // different storage orders, use transposed I/O
405  return true;
406  }
407 }

440 static void regrid_vec_generic(const File &file, const IceGrid &grid,
441  const std::string &variable_name,
442  const std::vector<double> &zlevels_out,
443  unsigned int t_start,
444  bool fill_missing,
445  double default_value,
446  InterpolationType interpolation_type,
447  double *output) {
448  const int X = 1, Y = 2, Z = 3; // indices, just for clarity
449 
450  const Profiling& profiling = grid.ctx()->profiling();
451 
452  try {
453  grid_info gi(file, variable_name, grid.ctx()->unit_system(), grid.registration());
454  LocalInterpCtx lic(gi, grid, zlevels_out, interpolation_type);
455 
456  std::vector<double> &buffer = lic.buffer;
457 
458  const unsigned int t_count = 1;
459  std::vector<unsigned int> start, count, imap;
460  compute_start_and_count(file,
461  grid.ctx()->unit_system(),
462  variable_name,
463  t_start, t_count,
464  lic.start[X], lic.count[X],
465  lic.start[Y], lic.count[Y],
466  lic.start[Z], lic.count[Z],
467  start, count, imap);
468 
469  bool transposed_io = use_transposed_io(file, grid.ctx()->unit_system(), variable_name);
470  profiling.begin("io.regridding.read");
471  if (transposed_io) {
472  file.read_variable_transposed(variable_name, start, count, imap, buffer.data());
473  } else {
474  file.read_variable(variable_name, start, count, buffer.data());
475  }
476  profiling.end("io.regridding.read");
477 
478  // Replace missing values if the _FillValue attribute is present,
479  // and if we have missing values to replace.
480  {
481  auto attribute = file.read_double_attribute(variable_name, "_FillValue");
482  if (attribute.size() == 1) {
483  double
484  fill_value = attribute[0],
485  epsilon = 1e-12;
486  if (fill_missing) {
487  for (size_t i = 0; i < buffer.size(); ++i) {
488  if (fabs(buffer[i] - fill_value) < epsilon) {
489  buffer[i] = default_value;
490  }
491  }
492  } else {
493  for (size_t i = 0; i < buffer.size(); ++i) {
494  if (fabs(buffer[i] - fill_value) < epsilon) {
495  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
496  "Some values of '%s' in '%s' match the _FillValue attribute.",
497  variable_name.c_str(), file.filename().c_str());
498  }
499  }
500  }
501  }
502  }
503 
504  // interpolate
505  profiling.begin("io.regridding.interpolate");
506  regrid(grid, zlevels_out, &lic, output);
507  profiling.end("io.regridding.interpolate");
508  } catch (RuntimeError &e) {
509  e.add_context("reading variable '%s' (using linear interpolation) from '%s'",
510  variable_name.c_str(), file.filename().c_str());
511  throw;
512  }
513 }

557 //! Define a NetCDF variable corresponding to a VariableMetadata object.
558 void define_spatial_variable(const SpatialVariableMetadata &var,
559  const IceGrid &grid, const File &file,
560  IO_Type default_type) {
561  std::vector<std::string> dims;
562  std::string name = var.get_name();
563 
564  if (file.find_variable(name)) {
565  return;
566  }
567 
568  define_dimensions(var, grid, file);
569 
570  std::string
571  x = var.x().get_name(),
572  y = var.y().get_name(),
573  z = var.z().get_name();
574 
575  if (not var.get_time_independent()) {
576  dims.push_back(grid.ctx()->config()->get_string("time.dimension_name"));
577  }
578 
579  dims.push_back(y);
580  dims.push_back(x);
581 
582  if (not z.empty()) {
583  dims.push_back(z);
584  }
585 
586  assert(dims.size() > 1);
587 
588  IO_Type type = var.get_output_type();
589  if (type == PISM_NAT) {
590  type = default_type;
591  }
592  file.define_variable(name, type, dims);
593 
594  write_attributes(file, var, type);
595 
596  // add the "grid_mapping" attribute if the grid has an associated mapping. Variables lat, lon,
597  // lat_bnds, and lon_bnds should not have the grid_mapping attribute to support CDO (see issue
598  // #384).

600  if (mapping.has_attributes() and
601  not member(name, {"lat_bnds", "lon_bnds", "lat", "lon"})) {
602  file.write_attribute(var.get_name(), "grid_mapping",
603  mapping.get_name());
604  }
605 
606  if (var.get_time_independent()) {
607  // mark this variable as "not written" so that write_spatial_variable can avoid
608  // writing it more than once.
609  file.write_attribute(var.get_name(), "not_written", PISM_INT, {1.0});
610  }
611 }

633  // number of spatial dimensions.
634  {
635  // Set of spatial dimensions this field has.
636  std::set<int> axes {X_AXIS, Y_AXIS};
637  if (not variable.z().get_name().empty()) {
638  axes.insert(Z_AXIS);
639  }

642  int input_ndims = 0; // number of spatial dimensions (input file)
643  size_t matching_dim_count = 0; // number of matching dimensions

651  }
652 
653  if (axes.find(tmp) != axes.end()) {
654  ++matching_dim_count;
655  }
656  }

661  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
662  "found the %dD variable %s (%s) in '%s' while trying to read\n"
663  "'%s' ('%s'), which is %d-dimensional.",
664  input_ndims,
665  var.name.c_str(),
666  join(input_dims, ",").c_str(),
667  file.filename().c_str(),
668  variable.get_name().c_str(),
669  variable.get_string("long_name").c_str(),
670  static_cast<int>(axes.size()));
671  }
672  }
673 
674  // make sure we have at least one level
675  auto zlevels = variable.levels();
676  unsigned int nlevels = std::max(zlevels.size(), (size_t)1);
677 
678  read_distributed_array(file, grid, var.name, nlevels, time, output);
679 
680  std::string input_units = file.read_text_attribute(var.name, "units");
681  const std::string &internal_units = variable.get_string("units");
682 
683  if (input_units.empty() and not internal_units.empty()) {
684  const std::string &long_name = variable.get_string("long_name");
685  log.message(2,
686  "PISM WARNING: Variable '%s' ('%s') does not have the units attribute.\n"
687  " Assuming that it is in '%s'.\n",
688  variable.get_name().c_str(), long_name.c_str(),
689  internal_units.c_str());
690  input_units = internal_units;
691  }
692 
693  // Convert data:
694  size_t size = grid.xm() * grid.ym() * nlevels;
695

697  {
698  auto att = file.read_double_attribute(var.name, "_FillValue");
699  if (att.size() == 1) {
700  double fill_value = att[0];
701  for (size_t k = 0; k < size; ++k) {
702  if (output[k] == fill_value) {
703  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
704  "Some values of the variable '%s' in '%s' match the _FillValue attribute.",
705  var.name.c_str(), file.filename().c_str());
706  }
707  }
708  }
709  }
710 
711  units::Converter(variable.unit_system(),
712  input_units, internal_units).convert_doubles(output, size);

835 /*!
836  * Check the overlap of the input grid and the internal grid.
837  *
838  * Set `allow_extrapolation` to `true` to "extend" the vertical grid during "bootstrapping".
839  */
840 static void check_grid_overlap(const grid_info &input, const IceGrid &internal,
841  const std::vector<double> &z_internal) {
842 
843  // Grid spacing (assume that the grid is equally-spaced) and the
844  // extent of the domain. To compute the extent of the domain, assume
845  // that the grid is cell-centered, so edge of the domain is half of
846  // the grid spacing away from grid points at the edge.
847  //
848  // Note that x_min is not the same as internal.x(0).
849  const double
850  x_min = internal.x0() - internal.Lx(),
851  x_max = internal.x0() + internal.Lx(),
852  y_min = internal.y0() - internal.Ly(),
853  y_max = internal.y0() + internal.Ly(),
854  input_x_min = input.x0 - input.Lx,
855  input_x_max = input.x0 + input.Lx,
856  input_y_min = input.y0 - input.Ly,
857  input_y_max = input.y0 + input.Ly;
858 
859  // tolerance (one micron)
860  double eps = 1e-6;
861 
862  // horizontal grid extent
863  if (not (x_min >= input_x_min - eps and x_max <= input_x_max + eps and
864  y_min >= input_y_min - eps and y_max <= input_y_max + eps)) {
865  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
866  "PISM's computational domain is not a subset of the domain in '%s'\n"
867  "PISM grid: x: [%3.3f, %3.3f] y: [%3.3f, %3.3f] meters\n"
868  "input file grid: x: [%3.3f, %3.3f] y: [%3.3f, %3.3f] meters",
869  input.filename.c_str(),
870  x_min, x_max, y_min, y_max,
871  input_x_min, input_x_max, input_y_min, input_y_max);
872  }
873 
874  if (z_internal.size() == 0) {
875  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
876  "Interval vertical grid has 0 levels. This should never happen.");
877  }
878 
879  if (z_internal.size() == 1 and input.z.size() > 1) {
880  // internal field is 2D or 3D with one level, input variable is 3D with more than one
881  // vertical level
882  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
883  "trying to read in a 2D field but the input file %s contains\n"
884  "a 3D field with %d levels",
885  input.filename.c_str(),
886  static_cast<int>(input.z.size()));
887  }
888 
889  if (z_internal.size() > 1 and input.z.size() <= 1) {
890  // internal field is 3D with more than one vertical level, input variable is 2D or 3D
891  // with 1 level
892  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
893  "trying to read in a 3D field but the input file %s contains\n"
894  "a 2D field", input.filename.c_str());
895  }
896 
897  if (z_internal.size() > 1 and input.z.size() > 0) {
898  // both internal field and input variable are 3D: check vertical grid extent
899  // Note: in PISM 2D fields have one vertical level (z = 0).
900  const double
901  input_z_min = input.z.front(),
902  input_z_max = input.z.back(),
903  z_min = z_internal.front(),
904  z_max = z_internal.back();
905 
906  if (not (z_min >= input.z_min - eps and z_max <= input.z_max + eps)) {
907  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
908  "PISM's computational domain is not a subset of the domain in '%s'\n"
909  "PISM grid: z: [%3.3f, %3.3f] meters\n"
910  "input file grid: z: [%3.3f, %3.3f] meters",
911  input.filename.c_str(),
912  z_min, z_max,
913  input_z_min, input_z_max);
914  }
915  }
916 }

918 void regrid_spatial_variable(SpatialVariableMetadata &variable,
919  const IceGrid& grid, const File &file,
920  unsigned int t_start, RegriddingFlag flag,
921  bool report_range,
922  bool allow_extrapolation,
923  double default_value,
924  InterpolationType interpolation_type,
925  double *output) {
926  const Logger &log = *grid.ctx()->log();
927 
928  units::System::Ptr sys = variable.unit_system();
929  const std::vector<double>& levels = variable.levels();
930  const size_t data_size = grid.xm() * grid.ym() * levels.size();
931 
932  // Find the variable
933  auto var = file.find_variable(variable.get_name(), variable.get_string("standard_name"));
934 
935  if (var.exists) { // the variable was found successfully
936 
937  {
938  grid_info input_grid(file, var.name, sys, grid.registration());
939 
940  check_input_grid(input_grid);
941 
942  if (not allow_extrapolation) {
943  check_grid_overlap(input_grid, grid, levels);
944  }
945  }
946 
947  if (flag == OPTIONAL_FILL_MISSING or flag == CRITICAL_FILL_MISSING) {
948  log.message(2,
949  "PISM WARNING: Replacing missing values with %f [%s] in variable '%s' read from '%s'.\n",
950  default_value, variable.get_string("units").c_str(), variable.get_name().c_str(),
951  file.filename().c_str());
952 
953  regrid_vec_fill_missing(file, grid, var.name, levels,
954  t_start, default_value, interpolation_type, output);
955  } else {
956  regrid_vec(file, grid, var.name, levels, t_start, interpolation_type, output);
957  }
958 
959  // Now we need to get the units string from the file and convert
960  // the units, because check_range and report_range expect data to
961  // be in PISM (MKS) units.
962 
963  std::string input_units = file.read_text_attribute(var.name, "units");
964  std::string internal_units = variable.get_string("units");
965 
966  if (input_units.empty() and not internal_units.empty()) {
967  log.message(2,
968  "PISM WARNING: Variable '%s' ('%s') does not have the units attribute.\n"
969  " Assuming that it is in '%s'.\n",
970  variable.get_name().c_str(),
971  variable.get_string("long_name").c_str(),
972  internal_units.c_str());
973  input_units = internal_units;
974  }
975 
976  // Convert data:
977  units::Converter(sys, input_units, internal_units).convert_doubles(output, data_size);
978 
979  // Check the range and report it if necessary.
980  {
981  double min = 0.0, max = 0.0;
982  read_valid_range(file, var.name, variable);
983 
984  compute_range(grid.com, output, data_size, &min, &max);
985 
986  // Check the range and warn the user if needed:
987  variable.check_range(file.filename(), min, max);
988  if (report_range) {
989  // We can report the success, and the range now:
990  log.message(2, " FOUND ");
991 
992  variable.report_range(log, min, max, var.found_using_standard_name);
993  }
994  }
995  } else { // couldn't find the variable
996  if (flag == CRITICAL or flag == CRITICAL_FILL_MISSING) {
997  // if it's critical, print an error message and stop
998  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "Can't find '%s' in the regridding file '%s'.",
999  variable.get_name().c_str(), file.filename().c_str());
1000  }
1001 
1002  // If it is optional, fill with the provided default value.
1003  // units::Converter constructor will make sure that units are compatible.
1004  units::Converter c(sys,
1005  variable.get_string("units"),
1006  variable.get_string("glaciological_units"));
1007 
1008  std::string spacer(variable.get_name().size(), ' ');
1009  log.message(2,
1010  " absent %s / %-10s\n"
1011  " %s \\ not found; using default constant %7.2f (%s)\n",
1012  variable.get_name().c_str(),
1013  variable.get_string("long_name").c_str(),
1014  spacer.c_str(), c(default_value),
1015  variable.get_string("glaciological_units").c_str());
1016 
1017  for (size_t k = 0; k < data_size; ++k) {
1018  output[k] = default_value;
1019  }
1020  } // end of if (exists)
1021 }
