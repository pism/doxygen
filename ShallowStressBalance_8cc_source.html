143 //! \brief Compute the basal frictional heating.
144 /*!
145  Ice shelves have zero basal friction heating.
146 
147  \param[in] V *basal* sliding velocity
148  \param[in] tauc basal yield stress
149  \param[in] mask (used to determine if floating or grounded)
150  \param[out] result
151  */
152 void ShallowStressBalance::compute_basal_frictional_heating(const IceModelVec2V &V,
153  const IceModelVec2S &tauc,
154  const IceModelVec2CellType &mask,
155  IceModelVec2S &result) const {
156 
157  IceModelVec::AccessList list{&V, &result, &tauc, &mask};
158 
159  for (Points p(*m_grid); p; p.next()) {
160  const int i = p.i(), j = p.j();
161 
162  if (mask.ocean(i,j)) {
163  result(i,j) = 0.0;
164  } else {
165  const double
166  C = m_basal_sliding_law->drag(tauc(i,j), V(i,j).u, V(i,j).v),
167  basal_stress_x = - C * V(i,j).u,
168  basal_stress_y = - C * V(i,j).v;
169  result(i,j) = - basal_stress_x * V(i,j).u - basal_stress_y * V(i,j).v;
170  }
171  }
172 }

192 /*!
193  * The driving stress computed here is not used by the model, so this
194  * implementation intentionally does not use the eta-transformation or special
195  * cases at ice margins.
196  */
197 IceModelVec::Ptr SSB_taud::compute_impl() const {
198 
199  IceModelVec2V::Ptr result(new IceModelVec2V(m_grid, "result", WITHOUT_GHOSTS));
200  result->metadata(0) = m_vars[0];
201  result->metadata(1) = m_vars[1];
202 
203  const IceModelVec2S *thickness = m_grid->variables().get_2d_scalar("land_ice_thickness");
204  const IceModelVec2S *surface = m_grid->variables().get_2d_scalar("surface_altitude");
205 
206  double standard_gravity = m_config->get_number("constants.standard_gravity"),
207  ice_density = m_config->get_number("constants.ice.density");
208 
209  IceModelVec::AccessList list{surface, thickness, result.get()};
210 
211  for (Points p(*m_grid); p; p.next()) {
212  const int i = p.i(), j = p.j();
213 
214  double pressure = ice_density * standard_gravity * (*thickness)(i,j);
215  if (pressure <= 0.0) {
216  (*result)(i,j).u = 0.0;
217  (*result)(i,j).v = 0.0;
218  } else {
219  (*result)(i,j).u = - pressure * diff_x_p(*surface, i,j);
220  (*result)(i,j).v = - pressure * diff_y_p(*surface, i,j);
221  }
222  }
223 
224  return result;
225 }

266 IceModelVec::Ptr SSB_taub::compute_impl() const {
267 
268  IceModelVec2V::Ptr result(new IceModelVec2V(m_grid, "result", WITHOUT_GHOSTS));
269  result->metadata() = m_vars[0];
270  result->metadata(1) = m_vars[1];
271 
272  const IceModelVec2V &velocity = model->velocity();
273  const IceModelVec2S *tauc = m_grid->variables().get_2d_scalar("tauc");
274  const IceModelVec2CellType &mask = *m_grid->variables().get_2d_cell_type("mask");
275 
276  const IceBasalResistancePlasticLaw *basal_sliding_law = model->sliding_law();
277 
278  IceModelVec::AccessList list{tauc, &velocity, &mask, result.get()};
279  for (Points p(*m_grid); p; p.next()) {
280  const int i = p.i(), j = p.j();
281 
282  if (mask.grounded_ice(i,j)) {
283  double beta = basal_sliding_law->drag((*tauc)(i, j), velocity(i, j).u, velocity(i, j).v);
284  (*result)(i, j) = - beta * velocity(i, j);
285  } else {
286  (*result)(i, j) = 0.0;
287  }
288  }
289 
290  return result;
291 }
