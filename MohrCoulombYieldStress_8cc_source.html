35 //! \file MohrCoulombYieldStress.cc Process model which computes pseudo-plastic yield stress for the subglacial layer.
36 
37 /*! \file MohrCoulombYieldStress.cc
38 The output variable of this submodel is `tauc`, the pseudo-plastic yield stress
39 field that is used in the ShallowStressBalance objects. This quantity is
40 computed by the Mohr-Coulomb criterion [\ref SchoofTill], but using an empirical
41 relation between the amount of water in the till and the effective pressure
42 of the overlying glacier resting on the till [\ref Tulaczyketal2000].
43 
44 The "dry" strength of the till is a state variable which is private to
45 the submodel, namely `tillphi`. Its initialization is nontrivial: either the
46 `-topg_to_phi` heuristic is used or inverse modeling can be used. (In the
47 latter case `tillphi` can be read-in at the beginning of the run. Currently
48 `tillphi` does not evolve during the run.)
49 
50 The effective pressure is derived from the till (pore) water amount (the effective water
51 layer thickness). Then the effective pressure is combined with tillphi to compute an
52 updated `tauc` by the Mohr-Coulomb criterion.
53 
54 This submodel is inactive in floating areas.
55 */
56 
57 
58 /*!
59 The pseudo-plastic till basal resistance model is governed by this power law
60 equation,
61  @f[ \tau_b = - \frac{\tau_c}{|\mathbf{U}|^{1-q} U_{\mathtt{th}}^q} \mathbf{U}, @f]
62 where @f$\tau_b=(\tau_{(b)x},\tau_{(b)y})@f$ is the basal shear stress and
63 @f$U=(u,v)@f$ is the sliding velocity.
64 
65 We call the scalar field @f$\tau_c(t,x,y)@f$ the *yield stress* even when
66 the power @f$q@f$ is not zero; when that power is zero the formula describes
67 a plastic material with an actual yield stress. The constant
68 @f$U_{\mathtt{th}}@f$ is the *threshold speed*, and @f$q@f$ is the *pseudo*
69 *plasticity exponent*. The current class computes this yield stress field.
70 See also IceBasalResistancePlasticLaw::drag().
71 
72 The strength of the saturated till material, the yield stress, is modeled by a
73 Mohr-Coulomb relation [\ref Paterson, \ref SchoofStream],
74  @f[ \tau_c = c_0 + (\tan \varphi) N_{till}, @f]
75 where @f$N_{till}@f$ is the effective pressure of the glacier on the mineral
76 till.
77 
78 The determination of the till friction angle @f$\varphi(x,y)@f$ is important.
79 It is assumed in this default model to be a time-independent factor which
80 describes the strength of the unsaturated "dry" (mineral) till material. Thus
81 it is assumed to change more slowly than the till water pressure, and it follows
82 that it changes more slowly than the yield stress and the basal shear stress.
83 
84 Option `-topg_to_phi` causes call to topg_to_phi() at the beginning of the run.
85 This determines the map of @f$\varphi(x,y)@f$. If this option is note given,
86 the current method leaves `tillphi` unchanged, and thus either in its
87 read-in-from-file state or with a default constant value from the config file.
88 */

237 //! Update the till yield stress for use in the pseudo-plastic till basal stress
238 //! model. See also IceBasalResistancePlasticLaw.
239 /*!
240 Updates yield stress @f$ \tau_c @f$ based on modeled till water layer thickness
241 from a Hydrology object. We implement the Mohr-Coulomb criterion allowing
242 a (typically small) till cohesion @f$ c_0 @f$
243 and by expressing the coefficient as the tangent of a till friction angle
244  @f$ \varphi @f$ :
245  @f[ \tau_c = c_0 + (\tan \varphi) N_{till}. @f]
246 See [@ref Paterson] table 8.1 regarding values.
247 
248 The effective pressure on the till is empirically-related
249 to the amount of water in the till. We use this formula derived from
250 [@ref Tulaczyketal2000] and documented in [@ref BuelervanPeltDRAFT]:
251 
252 @f[ N_{till} = \min\left\{P_o, N_0 \left(\frac{\delta P_o}{N_0}\right)^s 10^{(e_0/C_c) (1 - s)}\right\} @f]
253 
254 where @f$ s = W_{till} / W_{till}^{max} @f$, @f$ W_{till}^{max} @f$ =`hydrology_tillwat_max`,
255 @f$ \delta @f$ =`basal_yield_stress.mohr_coulomb.till_effective_fraction_overburden`, @f$ P_o @f$ is the
256 overburden pressure, @f$ N_0 @f$ =`basal_yield_stress.mohr_coulomb.till_reference_effective_pressure` is a
257 reference effective pressure, @f$ e_0 @f$ =`basal_yield_stress.mohr_coulomb.till_reference_void_ratio` is the void ratio
258 at the reference effective pressure, and @f$ C_c @f$ =`basal_yield_stress.mohr_coulomb.till_compressibility_coefficient`
259 is the coefficient of compressibility of the till. Constants @f$ N_0, e_0, C_c @f$ are
260 found by [@ref Tulaczyketal2000] from laboratory experiments on samples of
261 till.
262 
263 If `basal_yield_stress.add_transportable_water` is yes then @f$ s @f$ in the above formula
264 becomes @f$ s = (W + W_{till}) / W_{till}^{max} @f$,
265 that is, the water amount is the sum @f$ W+W_{till} @f$.
266  */
267 void MohrCoulombYieldStress::update_impl(const YieldStressInputs &inputs,
268  double t, double dt) {
269  (void) t;
270  (void) dt;
271 
272  bool slippery_grounding_lines = m_config->get_flag("basal_yield_stress.slippery_grounding_lines"),
273  add_transportable_water = m_config->get_flag("basal_yield_stress.add_transportable_water");
274 
275  const double
276  ice_density = m_config->get_number("constants.ice.density"),
277  standard_gravity = m_config->get_number("constants.standard_gravity");
278 
279  const double high_tauc = m_config->get_number("basal_yield_stress.ice_free_bedrock"),
280  W_till_max = m_config->get_number("hydrology.tillwat_max"),
281  delta = m_config->get_number("basal_yield_stress.mohr_coulomb.till_effective_fraction_overburden"),
282  tlftw = m_config->get_number("basal_yield_stress.mohr_coulomb.till_log_factor_transportable_water");
283 
284  MohrCoulombPointwise mc(m_config);
285 
286  const IceModelVec2S
287  &W_till = *inputs.till_water_thickness,
288  &W_subglacial = *inputs.subglacial_water_thickness,
289  &ice_thickness = inputs.geometry->ice_thickness;
290 
291  const IceModelVec2CellType &cell_type = inputs.geometry->cell_type;
292  const IceModelVec2S &bed_topography = inputs.geometry->bed_elevation;
293  const IceModelVec2S &sea_level = inputs.geometry->sea_level_elevation;
294 
295  IceModelVec::AccessList list{&W_till, &m_till_phi, &m_basal_yield_stress, &cell_type,
296  &bed_topography, &sea_level, &ice_thickness};
297 
298  if (add_transportable_water) {
299  list.add(W_subglacial);
300  }
301 
302  if (m_delta) {
303  m_delta->update(t, dt);
304  m_delta->average(t, dt);
305  list.add(*m_delta);
306  }
307 
308  for (Points p(*m_grid); p; p.next()) {
309  const int i = p.i(), j = p.j();
310 
311  if (cell_type.ice_free(i, j)) {
312  m_basal_yield_stress(i, j) = high_tauc; // large yield stress if ice-free
313  } else { // grounded and there is some ice
314 
315  // user can ask that marine grounding lines get special treatment
316  double water = W_till(i,j); // usual case
317 
318  if (slippery_grounding_lines and
319  bed_topography(i, j) <= sea_level(i, j) and
320  (cell_type.next_to_floating_ice(i, j) or cell_type.next_to_ice_free_ocean(i, j))) {
321  water = W_till_max;
322  } else if (add_transportable_water) {
323  water = W_till(i, j) + tlftw * log(1.0 + W_subglacial(i, j) / tlftw);
324  }
325 
326  double P_overburden = ice_density * standard_gravity * ice_thickness(i, j);
327 
328  m_basal_yield_stress(i, j) = mc.yield_stress(m_delta ? (*m_delta)(i, j) : delta,
329  P_overburden, water, m_till_phi(i, j));
330  }
331  }
332 
333  m_basal_yield_stress.update_ghosts();
334 }

336 //! Computes the till friction angle phi as a piecewise linear function of bed elevation, according to user options.
337 /*!
338 Computes the till friction angle \f$\phi(x,y)\f$ at a location as the following
339 increasing, piecewise-linear function of the bed elevation \f$b(x,y)\f$. Let
340  \f[ M = (\phi_{\text{max}} - \phi_{\text{min}}) / (b_{\text{max}} - b_{\text{min}}) \f]
341 be the slope of the nontrivial part. Then
342  \f[ \phi(x,y) = \begin{cases}
343  \phi_{\text{min}}, & b(x,y) \le b_{\text{min}}, \\
344  \phi_{\text{min}} + (b(x,y) - b_{\text{min}}) \,M,
345  & b_{\text{min}} < b(x,y) < b_{\text{max}}, \\
346  \phi_{\text{max}}, & b_{\text{max}} \le b(x,y), \end{cases} \f]
347 where \f$\phi_{\text{min}}=\f$`phi_min`, \f$\phi_{\text{max}}=\f$`phi_max`,
348 \f$b_{\text{min}}=\f$`topg_min`, \f$b_{\text{max}}=\f$`topg_max`.
349 
350 The default values are vaguely suitable for Antarctica. See src/pism_config.cdl.
351 */
352 void MohrCoulombYieldStress::till_friction_angle(const IceModelVec2S &bed_topography,
353  IceModelVec2S &result) {
354 
355  const double
356  phi_min = m_config->get_number("basal_yield_stress.mohr_coulomb.topg_to_phi.phi_min"),
357  phi_max = m_config->get_number("basal_yield_stress.mohr_coulomb.topg_to_phi.phi_max"),
358  topg_min = m_config->get_number("basal_yield_stress.mohr_coulomb.topg_to_phi.topg_min"),
359  topg_max = m_config->get_number("basal_yield_stress.mohr_coulomb.topg_to_phi.topg_max");
360 
361  m_log->message(2,
362  " till friction angle (phi) is piecewise-linear function of bed elev (topg):\n"
363  " / %5.2f for topg < %.f\n"
364  " phi = | %5.2f + (topg - (%.f)) * (%.2f / %.f) for %.f < topg < %.f\n"
365  " \\ %5.2f for %.f < topg\n",
366  phi_min, topg_min,
367  phi_min, topg_min, phi_max - phi_min, topg_max - topg_min, topg_min, topg_max,
368  phi_max, topg_max);
369 
370  if (phi_min >= phi_max) {
371  throw RuntimeError(PISM_ERROR_LOCATION,
372  "invalid -topg_to_phi arguments: phi_min < phi_max is required");
373  }
374 
375  if (topg_min >= topg_max) {
376  throw RuntimeError(PISM_ERROR_LOCATION,
377  "invalid -topg_to_phi arguments: topg_min < topg_max is required");
378  }
379 
380  const double slope = (phi_max - phi_min) / (topg_max - topg_min);
381 
382  IceModelVec::AccessList list{&bed_topography, &result};
383 
384  for (Points p(*m_grid); p; p.next()) {
385  const int i = p.i(), j = p.j();
386  const double bed = bed_topography(i, j);
387 
388  if (bed <= topg_min) {
389  result(i, j) = phi_min;
390  } else if (bed >= topg_max) {
391  result(i, j) = phi_max;
392  } else {
393  result(i, j) = phi_min + (bed - topg_min) * slope;
394  }
395  }
396 
397  // communicate ghosts so that the tauc computation can be performed locally
398  // (including ghosts of tauc, that is)
399  result.update_ghosts();
400 }

402 /*!
403  * Compute the till friction angle in grounded areas using available basal yield stress,
404  * till water thickness, and overburden pressure.
405  *
406  * This is the inverse of the formula used by `update_impl()`.
407  */
408 void MohrCoulombYieldStress::till_friction_angle(const IceModelVec2S &basal_yield_stress,
409  const IceModelVec2S &till_water_thickness,
410  const IceModelVec2S &ice_thickness,
411  const IceModelVec2CellType &cell_type,
412  IceModelVec2S &result) {
413 
414  MohrCoulombPointwise mc(m_config);
415 
416  double
417  ice_density = m_config->get_number("constants.ice.density"),
418  standard_gravity = m_config->get_number("constants.standard_gravity");
419 
420  double
421  delta = m_config->get_number("basal_yield_stress.mohr_coulomb.till_effective_fraction_overburden");
422 
423  const IceModelVec2S
424  &W_till = till_water_thickness;
425 
426  IceModelVec::AccessList list{&cell_type, &basal_yield_stress, &W_till, &ice_thickness, &result};
427 
428  for (Points p(*m_grid); p; p.next()) {
429  const int i = p.i(), j = p.j();
430 
431  if (cell_type.ocean(i, j)) {
432  // no change
433  } else if (cell_type.ice_free(i, j)) {
434  // no change
435  } else { // grounded and there is some ice
436  double P_overburden = ice_density * standard_gravity * ice_thickness(i, j);
437 
438  result(i, j) = mc.till_friction_angle(delta, P_overburden, W_till(i, j), basal_yield_stress(i, j));
439  }
440  }
441 
442  result.update_ghosts();
443 }
