82  * Interpolation helper. Does not check if points needed for interpolation are within the current
83  * processor's sub-domain.
84  */
85 template<class F, typename T>
86 T interpolate(const F &field, double x, double y) {
87  auto grid = field.grid();
88 
89  int i_left = 0, i_right = 0, j_bottom = 0, j_top = 0;
90  grid->compute_point_neighbors(x, y, i_left, i_right, j_bottom, j_top);
91 
92  auto w = grid->compute_interp_weights(x, y);
93 
94  return (w[0] * field(i_left, j_bottom) +
95  w[1] * field(i_right, j_bottom) +
96  w[2] * field(i_right, j_top) +
97  w[3] * field(i_left, j_top));
98 }

100 //! \brief Abstract class for reading, writing, allocating, and accessing a
101 //! DA-based PETSc Vec (2D and 3D fields) from within IceModel.
102 /*!
103  @anchor icemodelvec_use
104 
105  This class represents 2D and 3D fields in PISM. Its methods common to all
106  the derived classes can be split (roughly) into six kinds:
107 
108  - memory allocation (create)
109  - point-wise access (begin_access(), end_access())
110  - arithmetic (range(), norm(), add(), shift(), scale(), set(), ...)
111  - setting or reading metadata (set_attrs(), metadata())
112  - file input/output (read, write, regrid)
113  - tracking whether a field was updated (get_state_counter(), inc_state_counter())
114 
115  ## Memory allocation
116 
117  \code
118  IceModelVec2S var(grid, "var_name", WITH_GHOSTS);
119  // var is ready to use
120  \endcode
121 
122  ("WITH_GHOSTS" means "can be used in computations using map-plane neighbors
123  of grid points.)
124 
125  It is usually a good idea to set variable metadata right after creating it.
126  The method set_attrs() is used throughout PISM to set commonly used
127  attributes.
128 
129  ## Point-wise access
130 
131  PETSc performs some pointer arithmetic magic to allow convenient indexing of
132  grid point values. Because of this one needs to surround the code using row,
133  column or level indexes with begin_access() and end_access() calls:
134 
135  \code
136  double foo;
137  int i = 0, j = 0;
138  IceModelVec2S var;
139  // assume that var was allocated
140  ierr = var.begin_access(); CHKERRQ(ierr);
141  foo = var(i,j) * 2;
142  ierr = var.end_access(); CHKERRQ(ierr);
143  \endcode
144 
145  Please see [this page](@ref computational_grid) for a discussion of the
146  organization of PISM's computational grid and examples of for-loops you will
147  probably put between begin_access() and end_access().
148 
149  To ensure that ghost values are up to date add the following call
150  before the code using ghosts:
151 
152  \code
153  ierr = var.update_ghosts(); CHKERRQ(ierr);
154  \endcode
155 
156  ## Reading and writing variables
157 
158  PISM can read variables either from files with data on a grid matching the
159  current grid (read()) or, using bilinear interpolation, from files
160  containing data on a different (but compatible) grid (regrid()).
161 
162  To write a field to a "prepared" NetCDF file, use write(). (A file is prepared
163  if it contains all the necessary dimensions, coordinate variables and global
164  metadata.)
165 
166  If you need to "prepare" a file, do:
167  \code
168  File file(grid.com, PISM_NETCDF3);
169  io::prepare_for_output(file, *grid.ctx());
170  \endcode
171 
172  A note about NetCDF write performance: due to limitations of the NetCDF
173  (classic, version 3) format, it is significantly faster to
174  \code
175  for (all variables)
176  var.define(...);
177 
178  for (all variables)
179  var.write(...);
180  \endcode
181 
182  as opposed to
183 
184  \code
185  for (all variables) {
186  var.define(...);
187  var.write(...);
188  }
189  \endcode
190 
191  IceModelVec::define() is here so that we can use the first approach.
192 
193  ## Tracking if a field changed
194 
195  It is possible to track if a certain field changed with the help of
196  get_state_counter() and inc_state_counter() methods.
197 
198  For example, PISM's SIA code re-computes the smoothed bed only if the bed
199  deformation code updated it:
200 
201  \code
202  if (bed->get_state_counter() > bed_state_counter) {
203  ierr = bed_smoother->preprocess_bed(...); CHKERRQ(ierr);
204  bed_state_counter = bed->get_state_counter();
205  }
206  \endcode
207 
208  The state counter is **not** updated automatically. For the code snippet above
209  to work, a bed deformation model has to call inc_state_counter() after an
210  update.
211 */
