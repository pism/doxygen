40 //! Allocate the PISMV (verification) context. Uses ColdEnthalpyConverter.
41 std::shared_ptr<Context> pismv_context(MPI_Comm com, const std::string &prefix) {
42  // unit system
43  units::System::Ptr sys(new units::System);
44 
45  // logger
46  Logger::Ptr logger = logger_from_options(com);
47 
48  // configuration parameters
49  Config::Ptr config = config_from_options(com, *logger, sys);
50 
51  config->set_string("grid.periodicity", "none");
52  config->set_string("grid.registration", "corner");
53 
54  set_config_from_options(sys, *config);
55 
56  print_config(*logger, 3, *config);
57 
58  Time::Ptr time = std::make_shared<Time>(com, config, *logger, sys);
59 
60  EnthalpyConverter::Ptr EC = EnthalpyConverter::Ptr(new ColdEnthalpyConverter(*config));
61 
62  return std::shared_ptr<Context>(new Context(com, sys, config, EC, time, logger, prefix));
63 }

65 GridParameters pismv_grid_defaults(Config::Ptr config,
66  char testname) {
67  // This sets the defaults for each test; command-line options can override this.
68 
69  GridParameters P;
70 
71  // use the cell corner grid registration
72  P.registration = CELL_CORNER;
73  // use the non-periodic grid:
74  P.periodicity = NOT_PERIODIC;
75  // equal spacing is the default for all the tests except K
76  P.Lx = config->get_number("grid.Lx");
77  P.Ly = config->get_number("grid.Ly");
78 
79  P.Mx = config->get_number("grid.Mx");
80  P.My = config->get_number("grid.My");
81 
82  SpacingType spacing = EQUAL;
83  double Lz = config->get_number("grid.Lz");
84  unsigned int Mz = config->get_number("grid.Mz");
85 
86  switch (testname) {
87  case 'A':
88  case 'B':
89  case 'H':
90  // use 2400km by 2400km by 4000m rectangular domain
91  P.Lx = 1200e3;
92  P.Ly = P.Lx;
93  Lz = 4000;
94  break;
95  case 'C':
96  case 'D':
97  // use 2000km by 2000km by 4000m rectangular domain
98  P.Lx = 1000e3;
99  P.Ly = P.Lx;
100  Lz = 4000;
101  break;
102  case 'F':
103  case 'G':
104  case 'L':
105  // use 1800km by 1800km by 4000m rectangular domain
106  P.Lx = 900e3;
107  P.Ly = P.Lx;
108  Lz = 4000;
109  break;
110  case 'K':
111  case 'O':
112  // use 2000km by 2000km by 4000m rectangular domain, but make truely periodic
113  config->set_number("grid.Mbz", 2);
114  config->set_number("grid.Lbz", 1000);
115  P.Lx = 1000e3;
116  P.Ly = P.Lx;
117  Lz = 4000;
118  P.periodicity = XY_PERIODIC;
119  spacing = QUADRATIC;
120  break;
121  case 'V':
122  P.My = 3; // it's a flow-line setup
123  P.Lx = 500e3; // 500 km long
124  P.periodicity = Y_PERIODIC;
125  break;
126  default:
127  throw RuntimeError(PISM_ERROR_LOCATION, "desired test not implemented\n");
128  }
129 
130  P.z = IceGrid::compute_vertical_levels(Lz, Mz, spacing,
131  config->get_number("grid.lambda"));
132  return P;
133 }

135 IceGrid::Ptr pismv_grid(std::shared_ptr<Context> ctx, char testname) {
136  auto config = ctx->config();
137 
138  auto input_file = config->get_string("input.file");
139 
140  if (config->get_flag("input.bootstrap")) {
141  throw RuntimeError(PISM_ERROR_LOCATION, "pismv does not support bootstrapping");
142  }
143 
144  if (not input_file.empty()) {
145  GridRegistration r = string_to_registration(ctx->config()->get_string("grid.registration"));
146 
147  // get grid from a PISM input file
148  return IceGrid::FromFile(ctx, input_file, {"enthalpy", "temp"}, r);
149  } else {
150  // use defaults set by pismv_grid_defaults()
151  GridParameters P = pismv_grid_defaults(ctx->config(), testname);
152  P.horizontal_size_from_options();
153  P.horizontal_extent_from_options(ctx->unit_system());
154  P.vertical_grid_from_options(ctx->config());
155  P.ownership_ranges_from_options(ctx->size());
156 
157  return IceGrid::Ptr(new IceGrid(ctx, P));
158  }
159 }

161 int main(int argc, char *argv[]) {
162  MPI_Comm com = MPI_COMM_WORLD;
163 
164  petsc::Initializer petsc(argc, argv, help);
165  com = MPI_COMM_WORLD;
166 
167  /* This explicit scoping forces destructors to be called before PetscFinalize() */
168  try {
169  std::shared_ptr<Context> ctx = pismv_context(com, "pismv");
170  Logger::Ptr log = ctx->log();
171 
172  std::string usage =
173  " pismv -test x [-no_report] [OTHER PISM & PETSc OPTIONS]\n"
174  "where:\n"
175  " -test x SIA-type verification test (x = A|B|C|D|F|G|H|K|L)\n"
176  " -no_report do not give error report at end of run\n"
177  "(see User's Manual for tests I and J).\n";
178 
179  std::vector<std::string> required(1, "-test");
180 
181  bool done = show_usage_check_req_opts(*log, "PISMV (verification mode)", required, usage);
182  if (done) {
183  return 0;
184  }
185 
186  Config::Ptr config = ctx->config();
187 
188  // determine test (and whether to report error)
189  std::string testname = options::Keyword("-test", "Specifies PISM verification test",
190  "A,B,C,D,F,G,H,K,L,V", "A");
191 
192  IceGrid::Ptr g = pismv_grid(ctx, testname[0]);
193 
194  IceCompModel m(g, ctx, testname[0]);
195 
196  m.init();
197 
198  m.run();
199  log->message(2, "done with run\n");
200 
201  if (not options::Bool("-no_report", "do not print the error report")) {
202  m.reportErrors();
203  }
204 
205  // provide a default output file name if no -o option is given.
206  m.save_results();
207 
208  print_unused_parameters(*log, 3, *config);
209  }
210  catch (...) {
211  handle_fatal_errors(com);
212  return 1;
213  }
214 
215  return 0;
216 }
