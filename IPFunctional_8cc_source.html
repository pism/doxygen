27 void gradientFD(IPFunctional<IceModelVec2S> &f, IceModelVec2S &x, IceModelVec2S &gradient) {
28  const IceGrid &grid = *x.grid();
29  double h = PETSC_SQRT_MACHINE_EPSILON;
30 
31  double F0,Fh;
32 
33  f.valueAt(x,&F0);
34 
35  IceModelVec::AccessList list(gradient);
36 
37  ParallelSection loop(grid.com);
38  try {
39  for (Points p(grid); p; p.next()) {
40  const int i = p.i(), j = p.j();
41 
42  {
43  IceModelVec::AccessList access_x(x);
44  x(i,j) += h;
45  }
46  x.update_ghosts();
47 
48  f.valueAt(x,&Fh);
49 
50  {
51  IceModelVec::AccessList access_x(x);
52  x(i,j) -= h;
53  }
54  x.update_ghosts();
55 
56  gradient(i,j) = (Fh-F0)/h;
57  }
58  } catch (...) {
59  loop.failed();
60  }
61  loop.check();
62 }
