36 //! Compute the maximum velocities for time-stepping and reporting to user.
37 /*!
38 Computes the maximum magnitude of the components \f$u,v,w\f$ of the 3D velocity.
39 
40 Under BOMBPROOF there is no CFL condition for the vertical advection.
41 The maximum vertical velocity is computed but it does not affect the output.
42  */
43 CFLData max_timestep_cfl_3d(const IceModelVec2S &ice_thickness,
44  const IceModelVec2CellType &cell_type,
45  const IceModelVec3 &u3,
46  const IceModelVec3 &v3,
47  const IceModelVec3 &w3) {
48 
49  IceGrid::ConstPtr grid = ice_thickness.grid();
50  Config::ConstPtr config = grid->ctx()->config();
51 
52  double dt_max = config->get_number("time_stepping.maximum_time_step", "seconds");
53 
54  IceModelVec::AccessList list{&ice_thickness, &u3, &v3, &w3, &cell_type};
55 
56  // update global max of abs of velocities for CFL; only velocities under surface
57  const double
58  one_over_dx = 1.0 / grid->dx(),
59  one_over_dy = 1.0 / grid->dy();
60 
61  double u_max = 0.0, v_max = 0.0, w_max = 0.0;
62  ParallelSection loop(grid->com);
63  try {
64  for (Points p(*grid); p; p.next()) {
65  const int i = p.i(), j = p.j();
66 
67  if (cell_type.icy(i, j)) {
68  const int ks = grid->kBelowHeight(ice_thickness(i, j));
69  const double
70  *u = u3.get_column(i, j),
71  *v = v3.get_column(i, j),
72  *w = w3.get_column(i, j);
73 
74  for (int k = 0; k <= ks; ++k) {
75  const double
76  u_abs = fabs(u[k]),
77  v_abs = fabs(v[k]);
78  u_max = std::max(u_max, u_abs);
79  v_max = std::max(v_max, v_abs);
80  const double denom = fabs(u_abs * one_over_dx) + fabs(v_abs * one_over_dy);
81  if (denom > 0.0) {
82  dt_max = std::min(dt_max, 1.0 / denom);
83  }
84  }
85 
86  for (int k = 0; k <= ks; ++k) {
87  w_max = std::max(w_max, fabs(w[k]));
88  }
89  }
90  }
91  } catch (...) {
92  loop.failed();
93  }
94  loop.check();
95 
96  CFLData result;
97 
98  result.u_max = GlobalMax(grid->com, u_max);
99  result.v_max = GlobalMax(grid->com, v_max);
100  result.w_max = GlobalMax(grid->com, w_max);
101  result.dt_max = MaxTimestep(GlobalMin(grid->com, dt_max));
102 
103  return result;
104 }

106 //! Compute the CFL constant associated to first-order upwinding for the sliding contribution to mass continuity.
107 /*!
108  This procedure computes the maximum horizontal speed in the icy
109  areas. In particular it computes CFL constant for the upwinding, in
110  GeometryEvolution::step(), which applies to the basal component of mass
111  flux.
112 
113  That is, because the map-plane mass continuity is advective in the
114  sliding case we have a CFL condition.
115  */
116 CFLData max_timestep_cfl_2d(const IceModelVec2S &ice_thickness,
117  const IceModelVec2CellType &cell_type,
118  const IceModelVec2V &velocity) {
119 
120  IceGrid::ConstPtr grid = ice_thickness.grid();
121  Config::ConstPtr config = grid->ctx()->config();
122 
123  double dt_max = config->get_number("time_stepping.maximum_time_step", "seconds");
124 
125  const double
126  dx = grid->dx(),
127  dy = grid->dy();
128 
129  IceModelVec::AccessList list{&velocity, &cell_type};
130 
131  double u_max = 0.0, v_max = 0.0;
132  for (Points p(*grid); p; p.next()) {
133  const int i = p.i(), j = p.j();
134 
135  if (cell_type.icy(i, j)) {
136  const double
137  u_abs = fabs(velocity(i, j).u),
138  v_abs = fabs(velocity(i, j).v);
139 
140  u_max = std::max(u_max, u_abs);
141  v_max = std::max(v_max, v_abs);
142 
143  const double denom = u_abs / dx + v_abs / dy;
144  if (denom > 0.0) {
145  dt_max = std::min(dt_max, 1.0 / denom);
146  }
147  }
148  }
149 
150  CFLData result;
151 
152  result.u_max = GlobalMax(grid->com, u_max);
153  result.v_max = GlobalMax(grid->com, v_max);
154  result.w_max = 0.0;
155  result.dt_max = MaxTimestep(GlobalMin(grid->com, dt_max));
156 
157  return result;
158 }
