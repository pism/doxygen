54 int String::process(const std::string& option,
55  const std::string& description,
56  const std::string& default_value,
57  ArgumentFlag argument_flag) {
58 
59  char string[TEMPORARY_STRING_LENGTH];
60  memset(string, 0, TEMPORARY_STRING_LENGTH);
61 
62  PetscBool flag = PETSC_FALSE;
63 
64  PetscErrorCode ierr;
65  ierr = PetscOptionsGetString(NULL, // default option database
66  NULL, // no prefix
67  option.c_str(),
68  string,
69  TEMPORARY_STRING_LENGTH,
70  &flag);
71  PISM_CHK(ierr, "PetscOptionsGetString");
72 
73  std::string result = string;
74 
75  if (flag == PETSC_TRUE) {
76  if (result.empty()) {
77  if (argument_flag == ALLOW_EMPTY) {
78  this->set("", true);
79  } else {
80  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
81  "command line option '%s'\n"
82  "(%s)\n"
83  "requires an argument.",
84  option.c_str(), description.c_str());
85  }
86  } else {
87  this->set(result, true);
88  }
89  } else {
90  this->set(default_value, false);
91  }
92 
93  return 0;
94 }

96 Keyword::Keyword(const std::string& option,
97  const std::string& description,
98  const std::string& choices,
99  const std::string& default_value) {
100 
101  if (choices.empty()) {
102  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "empty choices argument");
103  }
104 
105  std::string list = "[" + choices + "]";
106  std::string long_description = description + " Choose one of " + list;
107 
108  String input(option, long_description, default_value, DONT_ALLOW_EMPTY);
109 
110  // use the default value if the option was not set
111  if (not input.is_set()) {
112  this->set(input, input.is_set());
113  return;
114  }
115 
116  std::string word = input;
117  // find ":" and discard everything that goes after
118  size_t n = word.find(":");
119  if (n != std::string::npos) {
120  word.resize(n);
121  }
122 
123  // transform a comma-separated list of choices into a set of
124  // choices:
125  auto choices_set = set_split(choices, ',');
126 
127  // use the choice if it is valid and stop if it is not
128  if (choices_set.find(word) != choices_set.end()) {
129  this->set(word, true);
130  } else {
131  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
132  "invalid %s argument: '%s'. Please choose one of %s.\n",
133  option.c_str(), word.c_str(), list.c_str());
134  }
135 }

160 IntegerList::IntegerList(const std::string& option,
161  const std::string& description,
162  const std::vector<int> &defaults) {
163  std::vector<double> default_value;
164 
165  for (auto v : defaults) {
166  default_value.push_back(v);
167  }
168 
169  RealList input(option, description, default_value);
170  std::vector<int> result;
171 
172  for (auto v : input.value()) {
173  if (fabs(v - floor(v)) > 1e-6) {
174  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
175  "Can't process '%s': (%f is not an integer).",
176  option.c_str(), v);
177  }
178  result.push_back(static_cast<int>(v));
179  }
180 
181  this->set(result, input.is_set());
182 }

188 Real::Real(std::shared_ptr<units::System> system,
189  const std::string& option,
190  const std::string& description,
191  const std::string& units,
192  double default_value) {
193 
194  std::string buffer = pism::printf("%f", default_value);
195 
196  String input(option, description, buffer, DONT_ALLOW_EMPTY);
197 
198  if (input.is_set()) {
199  char *endptr = NULL;
200  double result = strtod(input->c_str(), &endptr);
201  if (*endptr != '\0') {
202  // assume that "input" contains units and try converting to "units":
203  try {
204  result = units::convert(system, 1.0, input.value(), units);
205  } catch (RuntimeError &e) {
206  e.add_context("trying to convert '%s' to '%s'",
207  input->c_str(), units.c_str());
208  e.add_context("processing the command-line option %s",
209  option.c_str());
210  throw;
211  }
212  }
213  this->set(result, true);
214  } else {
215  this->set(default_value, false);
216  }
217 }

220 RealList::RealList(const std::string& option,
221  const std::string& description,
222  const std::vector<double> &default_value) {
223  String input(option, description, "", DONT_ALLOW_EMPTY);
224  std::vector<double> result = default_value;
225 
226  if (input.is_set()) {
227  std::istringstream arg(input);
228  std::string tmp;
229 
230  result.clear();
231  while(getline(arg, tmp, ',')) {
232  double d;
233  char *endptr;
234 
235  d = strtod(tmp.c_str(), &endptr);
236  if (*endptr != '\0') {
237  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "Can't parse %s (%s is not a number).",
238  tmp.c_str(), tmp.c_str());
239  } else {
240  result.push_back(d);
241  }
242  }
243  }
244  this->set(result, input.is_set());
