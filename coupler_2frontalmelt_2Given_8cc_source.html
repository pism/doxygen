64 void Given::update_impl(const FrontalMeltInputs &inputs, double t, double dt) {
65 
66  const IceModelVec2CellType &cell_type = inputs.geometry->cell_type;
67 
68  // fill m_frontal_melt_rate with values read from an file
69  m_frontal_melt_rate->update(t, dt);
70  m_frontal_melt_rate->average(t, dt);
71 
72  // post-processing: keep values at grounded (or grounded and floating) margins and in
73  // the interior, filling the rest with zeros
74 
75  IceModelVec::AccessList list{&cell_type, m_frontal_melt_rate.get()};
76 
77  for (Points p(*m_grid); p; p.next()) {
78  const int i = p.i(), j = p.j();
79 
80  auto R = (*m_frontal_melt_rate)(i, j);
81 
82  if (apply(cell_type, i, j)) {
83  (*m_frontal_melt_rate)(i, j) = R;
84  } else {
85  (*m_frontal_melt_rate)(i, j) = 0.0;
86  }
87  }
88 }
