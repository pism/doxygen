106 void PicoGeometry::update(const IceModelVec2S &bed_elevation,
107  const IceModelVec2CellType &cell_type) {
108 
109  bool exclude_ice_rises = m_config->get_flag("ocean.pico.exclude_ice_rises");
110 
111  // these three could be done at the same time
112  {
113  compute_ice_rises(cell_type, exclude_ice_rises, m_ice_rises);
114 
115  compute_ocean_mask(cell_type, m_ocean_mask);
116 
117  compute_lakes(cell_type, m_lake_mask);
118 
119  }
120 
121  // these two could be optimized by trying to reduce the number of times we update ghosts
122  {
123  m_ice_rises.update_ghosts();
124  m_ocean_mask.update_ghosts();
125 
126  compute_distances_gl(m_ocean_mask, m_ice_rises, exclude_ice_rises, m_distance_gl);
127 
128  compute_distances_cf(m_ocean_mask, m_ice_rises, exclude_ice_rises, m_distance_cf);
129  }
130 
131  // computing ice_shelf_mask and box_mask could be done at the same time
132  {
133  compute_ice_shelf_mask(m_ice_rises, m_lake_mask, m_ice_shelves);
134  auto n_shelves = static_cast<int>(max(m_ice_shelves)) + 1;
135 
136  std::vector<int> cfs_in_basins_per_shelf(n_shelves*m_n_basins, 0);
137  std::vector<int> most_shelf_cells_in_basin(n_shelves, 0);
138  identify_calving_front_connection(cell_type, m_basin_mask, m_ice_shelves, n_shelves,
139  most_shelf_cells_in_basin, cfs_in_basins_per_shelf);
140 
141  split_ice_shelves(cell_type, m_basin_mask, m_basin_neighbors,
142  most_shelf_cells_in_basin, cfs_in_basins_per_shelf, n_shelves,
143  m_ice_shelves);
144 
145  double continental_shelf_depth = m_config->get_number("ocean.pico.continental_shelf_depth");
146 
147  compute_continental_shelf_mask(bed_elevation, m_ice_rises, continental_shelf_depth,
148  m_continental_shelf);
149  }
150 
151  int n_boxes = static_cast<int>(m_config->get_number("ocean.pico.number_of_boxes"));
152 
153  compute_box_mask(m_distance_gl, m_distance_cf, m_ice_shelves, n_boxes, m_boxes);
154 }

157 enum RelabelingType {BY_AREA, AREA_THRESHOLD};
158 
159 /*!
160  * Re-label components in a mask processed by label_connected_components.
161  *
162  * If type is `BY_AREA`, the biggest one gets the value of 2, all the other ones 1, the
163  * background is set to zero.
164  *
165  * If type is `AREA_THRESHOLD`, patches with areas above `threshold` get the value of 2,
166  * all the other ones 1, the background is set to zero.
167  */
168 static void relabel(RelabelingType type,
169  double threshold,
170  IceModelVec2Int &mask) {
171 
172  IceGrid::ConstPtr grid = mask.grid();
173 
174  int max_index = static_cast<int>(mask.range()[1]);
175 
176  if (max_index < 1) {
177  // No components labeled. Fill the mask with zeros and quit.
178  mask.set(0.0);
179  return;
180  }
181 
182  std::vector<double> area(max_index + 1, 0.0);
183  std::vector<double> area1(max_index + 1, 0.0);
184  {
185 
186  ParallelSection loop(grid->com);
187  try {
188  for (Points p(*grid); p; p.next()) {
189  const int i = p.i(), j = p.j();
190 
191  int index = mask.as_int(i, j);
192 
193  if (index > max_index or index < 0) {
194  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "invalid component index: %d", index);
195  }
196 
197  if (index > 0) {
198  // count areas of actual components, ignoring the background (index == 0)
199  area[index] += 1.0;
200  }
201  }
202  } catch (...) {
203  loop.failed();
204  }
205  loop.check();
206  GlobalSum(grid->com, area.data(), area1.data(), area.size());
207 
208  // copy data
209  area = area1;
210 
211  for (unsigned int k = 0; k < area.size(); ++k) {
212  area[k] = grid->cell_area() * area[k];
213  }
214  }
215 
216  if (type == BY_AREA) {
217  // find the biggest component
218  int biggest_component = 0;
219  for (unsigned int k = 0; k < area.size(); ++k) {
220  if (area[k] > area[biggest_component]) {
221  biggest_component = static_cast<int>(k);
222  }
223  }
224 
225  // re-label
226  for (Points p(*grid); p; p.next()) {
227  const int i = p.i(), j = p.j();
228 
229  int component_index = mask.as_int(i, j);
230 
231  if (component_index == biggest_component) {
232  mask(i, j) = 2.0;
233  } else if (component_index > 0) {
234  mask(i, j) = 1.0;
235  } else {
236  mask(i, j) = 0.0;
237  }
238  }
239  } else {
240  for (Points p(*grid); p; p.next()) {
241  const int i = p.i(), j = p.j();
242 
243  int component_index = mask.as_int(i, j);
244 
245  if (area[component_index] > threshold) {
246  mask(i, j) = 2.0;
247  } else if (component_index > 0) {
248  mask(i, j) = 1.0;
249  } else {
250  mask(i, j) = 0.0;
251  }
252  }
253  }
254 }

280 /*!
281  * Compute the mask identifying "subglacial lakes", i.e. floating ice areas that are not
282  * connected to the open ocean.
283  *
284  * Resulting mask contains:
285  *
286  * 0 - grounded ice
287  * 1 - floating ice not connected to the open ocean
288  * 2 - floating ice or ice-free ocean connected to the open ocean
289  */
290 void PicoGeometry::compute_lakes(const IceModelVec2CellType &cell_type, IceModelVec2Int &result) {
291  IceModelVec::AccessList list{ &cell_type, &m_tmp };
292 
293  const int
294  Mx = m_grid->Mx(),
295  My = m_grid->My();
296 
297  // assume that ocean points (i.e. floating, either icy or ice-free) at the edge of the
298  // domain belong to the "open ocean"
299  for (Points p(*m_grid); p; p.next()) {
300  const int i = p.i(), j = p.j();
301 
302  if (cell_type.ocean(i, j)) {
303  m_tmp(i, j) = 1.0;
304 
305  if (grid_edge(*m_grid, i, j)) {
306  m_tmp(i, j) = 2.0;
307  }
308  } else {
309  m_tmp(i, j) = 0.0;
310  }
311  }
312 
313  // identify "floating" areas that are not connected to the open ocean as defined above
314  {
315  m_tmp.put_on_proc0(*m_tmp_p0);
316 
317  ParallelSection rank0(m_grid->com);
318  try {
319  if (m_grid->rank() == 0) {
320  petsc::VecArray mask_p0(*m_tmp_p0);
321  label_connected_components(mask_p0.get(), My, Mx, true, 2.0);
322  }
323  } catch (...) {
324  rank0.failed();
325  }
326  rank0.check();
327 
328  m_tmp.get_from_proc0(*m_tmp_p0);
329  }
330 
331  result.copy_from(m_tmp);
332 }

334 /*!
335  * Compute the mask identifying ice rises, i.e. grounded ice areas not connected to the
336  * continental ice sheet.
337  *
338  * Resulting mask contains:
339  *
340  * 0 - ocean
341  * 1 - ice rises
342  * 2 - continental ice sheet
343  * 3 - floating ice
344  */
345 void PicoGeometry::compute_ice_rises(const IceModelVec2CellType &cell_type, bool exclude_ice_rises,
346  IceModelVec2Int &result) {
347  IceModelVec::AccessList list{ &cell_type, &m_tmp };
348 
349  // mask of zeros and ones: one if grounded ice, zero otherwise
350  for (Points p(*m_grid); p; p.next()) {
351  const int i = p.i(), j = p.j();
352 
353  if (cell_type.grounded(i, j)) {
354  m_tmp(i, j) = 2.0;
355  } else {
356  m_tmp(i, j) = 0.0;
357  }
358  }
359 
360  if (exclude_ice_rises) {
361  label_tmp();
362 
363  relabel(AREA_THRESHOLD,
364  m_config->get_number("ocean.pico.maximum_ice_rise_area", "m2"),
365  m_tmp);
366  }
367 
368  // mark floating ice areas in this mask (reduces the number of masks we need later)
369  for (Points p(*m_grid); p; p.next()) {
370  const int i = p.i(), j = p.j();
371 
372  if (m_tmp(i, j) == 0.0 and cell_type.icy(i, j)) {
373  m_tmp(i, j) = FLOATING;
374  }
375  }
376 
377  result.copy_from(m_tmp);
378 }

380 /*!
381  * Compute the continental ice shelf mask.
382  *
383  * Resulting mask contains:
384  *
385  * 0 - ocean or icy
386  * 1 - ice-free areas with bed elevation > threshold and not connected to the continental ice sheet
387  * 2 - ice-free areas with bed elevation > threshold, connected to the continental ice sheet
388  */
389 void PicoGeometry::compute_continental_shelf_mask(const IceModelVec2S &bed_elevation,
390  const IceModelVec2Int &ice_rise_mask,
391  double bed_elevation_threshold,
392  IceModelVec2Int &result) {
393  IceModelVec::AccessList list{ &bed_elevation, &ice_rise_mask, &m_tmp };
394 
395  for (Points p(*m_grid); p; p.next()) {
396  const int i = p.i(), j = p.j();
397 
398  m_tmp(i, j) = 0.0;
399 
400  if (bed_elevation(i, j) > bed_elevation_threshold) {
401  m_tmp(i, j) = 1.0;
402  }
403 
404  if (ice_rise_mask.as_int(i, j) == CONTINENTAL) {
405  m_tmp(i, j) = 2.0;
406  }
407  }
408 
409  // use "iceberg identification" to label parts *not* connected to the continental ice
410  // sheet
411  {
412  m_tmp.put_on_proc0(*m_tmp_p0);
413 
414  ParallelSection rank0(m_grid->com);
415  try {
416  if (m_grid->rank() == 0) {
417  petsc::VecArray mask_p0(*m_tmp_p0);
418  label_connected_components(mask_p0.get(), m_grid->My(), m_grid->Mx(), true, 2.0);
419  }
420  } catch (...) {
421  rank0.failed();
422  }
423  rank0.check();
424 
425  m_tmp.get_from_proc0(*m_tmp_p0);
426  }
427 
428  // At this point areas with bed > threshold are 1, everything else is zero.
429  //
430  // Now we need to mark the continental shelf itself.
431  for (Points p(*m_grid); p; p.next()) {
432  const int i = p.i(), j = p.j();
433 
434  if (m_tmp(i, j) > 0.0) {
435  continue;
436  }
437 
438  if (bed_elevation(i, j) > bed_elevation_threshold and
439  ice_rise_mask.as_int(i, j) == OCEAN) {
440  m_tmp(i, j) = 2.0;
441  }
442  }
443 
444  result.copy_from(m_tmp);
445 }

447 /*!
448  * Compute the mask identifying ice shelves.
449  *
450  * Each shelf gets an individual integer label.
451  *
452  * Two shelves connected by an ice rise are considered to be parts of the same shelf.
453  *
454  * Floating ice cells that are not connected to the ocean ("subglacial lakes") are
455  * excluded.
456  */
457 void PicoGeometry::compute_ice_shelf_mask(const IceModelVec2Int &ice_rise_mask, const IceModelVec2Int &lake_mask,
458  IceModelVec2Int &result) {
459  IceModelVec::AccessList list{ &ice_rise_mask, &lake_mask, &m_tmp };
460 
461  for (Points p(*m_grid); p; p.next()) {
462  const int i = p.i(), j = p.j();
463 
464  int M = ice_rise_mask.as_int(i, j);
465 
466  if (M == RISE or M == FLOATING) {
467  m_tmp(i, j) = 1.0;
468  } else {
469  m_tmp(i, j) = 0.0;
470  }
471  }
472 
473  label_tmp();
474 
475  // remove ice rises and lakes
476  for (Points p(*m_grid); p; p.next()) {
477  const int i = p.i(), j = p.j();
478 
479  if (ice_rise_mask.as_int(i, j) == RISE or lake_mask.as_int(i, j) == 1) {
480  m_tmp(i, j) = 0.0;
481  }
482  }
483 
484  result.copy_from(m_tmp);
485 }

518 /*!
519  * Find neighboring basins by checking for the basin boundaries on the ice free ocean.
520  *
521  * Should we identify the intersection at the coastline instead?
522  *
523  * Returns the map from the basin index to a set of indexes of neighbors.
524  */
525 std::map<int,std::set<int> > PicoGeometry::basin_neighbors(const IceModelVec2CellType &cell_type,
526  const IceModelVec2Int &basin_mask) {
527  using mask::ice_free_ocean;
528 
529  // Allocate the adjacency matrix. This uses twice the amount of storage necessary (the
530  // matrix is symmetric), but in known cases (i.e. with around 20 basins) we're wasting
531  // only ~200*sizeof(int) bytes, which is not that bad (and the code is a bit simpler).
532  std::vector<int> adjacency_matrix(m_n_basins * m_n_basins, 0);
533 
534  // short-cuts
535  auto mark_as_neighbors = [&](int b1, int b2) {
536  adjacency_matrix[b1 * m_n_basins + b2] = 1;
537  // preserve symmetry:
538  adjacency_matrix[b2 * m_n_basins + b1] = 1;
539  };
540 
541  auto adjacent = [&](int b1, int b2) {
542  return adjacency_matrix[b1 * m_n_basins + b2] > 0;
543  };
544 
545  IceModelVec::AccessList list{ &cell_type, &basin_mask };
546 
547  for (Points p(*m_grid); p; p.next()) {
548  const int i = p.i(), j = p.j();
549 
550  auto M = cell_type.star(i, j);
551  auto B = basin_mask.star(i, j);
552 
553  bool next_to_icefront = (ice_free_ocean(M.ij) and cell_type.next_to_ice(i,j));
554 
555  // skip the "dummy" basin and cells that are not in the "ice free ocean"
556  if (B.ij == 0 or not ice_free_ocean(M.ij) or not next_to_icefront) {
557  continue;
558  }
559 
560  // Zero out IDs of basins for cell neighbors that are outside the modeling domain.
561  //
562  // This prevents "wrap-around" at grid boundaries.
563  {
564  B.n *= static_cast<int>(j < (int)m_grid->My() - 1);
565  B.e *= static_cast<int>(i < (int)m_grid->Mx() - 1);
566  B.s *= static_cast<int>(j > 0);
567  B.w *= static_cast<int>(i > 0);
568  }
569 
570  if (ice_free_ocean(M.n)) {
571  mark_as_neighbors(B.ij, B.n);
572  }
573 
574  if (ice_free_ocean(M.s)) {
575  mark_as_neighbors(B.ij, B.s);
576  }
577 
578  if (ice_free_ocean(M.e)) {
579  mark_as_neighbors(B.ij, B.e);
580  }
581 
582  if (ice_free_ocean(M.w)) {
583  mark_as_neighbors(B.ij, B.w);
584  }
585  }
586 
587  // Make a copy to allreduce efficiently with IntelMPI:
588  {
589  std::vector<int> tmp(adjacency_matrix.size(), 0);
590  GlobalMax(m_grid->com, adjacency_matrix.data(), tmp.data(),
591  static_cast<int>(tmp.size()));
592  // Copy results:
593  adjacency_matrix = tmp;
594  }
595 
596  // Convert the matrix into a map "basin ID -> set of neighbors' IDs":
597  std::map<int,std::set<int> > result;
598  for (int b1 = 1; b1 < m_n_basins; ++b1) {
599  for (int b2 = b1 + 1; b2 < m_n_basins; ++b2) {
600  if (adjacent(b1, b2)) {
601  result[b1].insert(b2);
602  result[b2].insert(b1);
603  }
604  }
605  }
606 
607  return result;
608 }

610 /*!
611  * Find all basins b, in which the ice shelf s has a calving front with potential ocean water intrusion.
612  * Find the basin bmax, in which the ice shelf s has the most cells
613  */
614 void PicoGeometry::identify_calving_front_connection(const IceModelVec2CellType &cell_type,
615  const IceModelVec2Int &basin_mask,
616  const IceModelVec2Int &shelf_mask,
617  int n_shelves,
618  std::vector<int> &most_shelf_cells_in_basin,
619  std::vector<int> &cfs_in_basins_per_shelf) {
620 
621  std::vector<int> n_shelf_cells_per_basin(n_shelves * m_n_basins,0);
622  // additional vectors to allreduce efficiently with IntelMPI
623  std::vector<int> n_shelf_cells_per_basinr(n_shelves * m_n_basins,0);
624  std::vector<int> cfs_in_basins_per_shelfr(n_shelves * m_n_basins,0);
625  std::vector<int> most_shelf_cells_in_basinr(n_shelves, 0);
626 
627  IceModelVec::AccessList list{ &cell_type, &basin_mask, &shelf_mask };
628 
629  {
630  for (Points p(*m_grid); p; p.next()) {

675 void PicoGeometry::split_ice_shelves(const IceModelVec2CellType &cell_type,
676  const IceModelVec2Int &basin_mask,
677  const std::map<int, std::set<int> > &basin_neighbors,
678  const std::vector<int> &most_shelf_cells_in_basin,
679  const std::vector<int> &cfs_in_basins_per_shelf,

691  const int i = p.i(), j = p.j();

724  n_shelf_numbers_to_add += 1;

731  }

737  int b = basin_mask.as_int(i, j);

753  bool exclude_ice_rises,
754  IceModelVec2Int &result) {
755 
756  IceModelVec::AccessList list{ &ice_rises, &ocean_mask, &result };
757 
758  result.set(-1);
759 
760  // Find the grounding line and the ice front and set result to 1 if ice shelf cell is
761  // next to the grounding line, Ice holes within the shelf are treated like ice shelf
762  // cells, if exclude_ice_rises is set then ice rises are also treated as ice shelf cells.
763 
764  ParallelSection loop(m_grid->com);
765  try {
766  for (Points p(*m_grid); p; p.next()) {
767  const int i = p.i(), j = p.j();
768

770  ocean_mask.as_int(i, j) == 1 or
771  (exclude_ice_rises and ice_rises.as_int(i, j) == RISE)) {
772  // if this is an ice shelf cell (or an ice rise) or a hole in an ice shelf
773 
774  // label the shelf cells adjacent to the grounding line with 1
775  bool neighbor_to_land =
776  (ice_rises(i, j + 1) == CONTINENTAL or
777  ice_rises(i, j - 1) == CONTINENTAL or
778  ice_rises(i + 1, j) == CONTINENTAL or
779  ice_rises(i - 1, j) == CONTINENTAL or
780  ice_rises(i + 1, j + 1) == CONTINENTAL or
781  ice_rises(i + 1, j - 1) == CONTINENTAL or
782  ice_rises(i - 1, j + 1) == CONTINENTAL or
783  ice_rises(i - 1, j - 1) == CONTINENTAL);
784 
785  if (neighbor_to_land) {
786  // i.e. there is a grounded neighboring cell (which is not an ice rise!)
787  result(i, j) = 1;
788  } else {
789  result(i, j) = 0;
790  }
791  }
792  }
793  } catch (...) {
794  loop.failed();
795  }
796  loop.check();
797 
798  result.update_ghosts();
799

846 /*!
847  * Find an approximate solution of the Eikonal equation on a given domain.
848  *
849  * To specify the problem, the input field (mask) should be filled with
850  *
851  * - negative values outside the domain
852  * - zeros within the domain
853  * - ones at "wave front" locations
854  *
855  * For example, to compute distances from the grounding line within ice shelves, fill
856  * generic ice shelf locations with zeros, set neighbors of the grounding line to 1, and
857  * the rest of the grid with -1 or some other negative number.
858  *
859  * Note that this implementation updates ghosts *every* iteration. We could speed this
860  * up by checking if a point at a boundary of the processor sub-domain was updated and
861  * update ghosts in those cases only.
862  *
863  * FIXME: replace this with a better algorithm.
864  */
865 void eikonal_equation(IceModelVec2Int &mask) {
866 
867  assert(mask.stencil_width() > 0);
868 
869  IceGrid::ConstPtr grid = mask.grid();
870 
871  double current_label = 1;
872  double continue_loop = 1;
873  while (continue_loop != 0) {
874 
875  continue_loop = 0;
876 
877  for (Points p(*grid); p; p.next()) {
878  const int i = p.i(), j = p.j();
879 
880  if (mask.as_int(i, j) == 0) {
881 
882  auto R = mask.star(i, j);
883 
884  if (R.ij == 0 and
885  (R.n == current_label or R.s == current_label or
886  R.e == current_label or R.w == current_label)) {
887  // i.e. this is an shelf cell with no distance assigned yet and with a neighbor
888  // that has a distance assigned
889  mask(i, j) = current_label + 1;
890  continue_loop = 1;
891  }
892  }
893  } // loop over grid points
894 
895  current_label++;
896  mask.update_ghosts();
897 
898  continue_loop = GlobalMax(grid->com, continue_loop);
899  }
900 }

902 /*!
903  * Compute the mask identifying ice shelf "boxes" using distances to the grounding line
904  * and the calving front.
905  */
906 void PicoGeometry::compute_box_mask(const IceModelVec2Int &D_gl, const IceModelVec2Int &D_cf,
907  const IceModelVec2Int &shelf_mask, int max_number_of_boxes,
908  IceModelVec2Int &result) {
909 
910  IceModelVec::AccessList list{ &D_gl, &D_cf, &shelf_mask, &result };
911 
912  int n_shelves = static_cast<int>(shelf_mask.range()[1]) + 1;
913 
914  std::vector<double> GL_distance_max(n_shelves, 0.0);
915  std::vector<double> GL_distance_max1(n_shelves, 0.0);
916  std::vector<double> CF_distance_max(n_shelves, 0.0);
917  std::vector<double> CF_distance_max1(n_shelves, 0.0);
918 
919  for (Points p(*m_grid); p; p.next()) {
920  const int i = p.i(), j = p.j();
921 
922  int shelf_id = shelf_mask.as_int(i, j);
923  assert(shelf_id >= 0);
924  assert(shelf_id < n_shelves + 1);
925 
926  if (shelf_id == 0) {
927  // not at a shelf; skip to the next grid point
928  continue;
929  }
930 
931  if (D_gl(i, j) > GL_distance_max[shelf_id]) {
932  GL_distance_max[shelf_id] = D_gl(i, j);
933  }
934 
935  if (D_cf(i, j) > CF_distance_max[shelf_id]) {
936  CF_distance_max[shelf_id] = D_cf(i, j);
937  }
938  }
939 
940  // compute global maximums
941  GlobalMax(m_grid->com, GL_distance_max.data(), GL_distance_max1.data(), n_shelves);
942  GlobalMax(m_grid->com, CF_distance_max.data(), CF_distance_max1.data(), n_shelves);
943  // copy data
944  GL_distance_max = GL_distance_max1;
945  CF_distance_max = CF_distance_max1;
946 
947  double GL_distance_ref = *std::max_element(GL_distance_max.begin(), GL_distance_max.end());
948 
949  // compute the number of boxes in each shelf
950 
951  std::vector<int> n_boxes(n_shelves, 0);
952  const int n_min = 1;
953  const double zeta = 0.5;
954 
955  for (int k = 0; k < n_shelves; ++k) {
956  n_boxes[k] = n_min + round(pow((GL_distance_max[k] / GL_distance_ref), zeta) * (max_number_of_boxes - n_min));
957 
958  n_boxes[k] = std::min(n_boxes[k], max_number_of_boxes);
959  }
960 
961  result.set(0.0);
962 
963  for (Points p(*m_grid); p; p.next()) {
964  const int i = p.i(), j = p.j();
965 
966  int d_gl = D_gl.as_int(i, j);
967  int d_cf = D_cf.as_int(i, j);
968 
969  if (shelf_mask.as_int(i, j) > 0 and d_gl > 0.0 and d_cf > 0.0) {
970  int shelf_id = shelf_mask.as_int(i, j);
971  int n = n_boxes[shelf_id];
972 
973  // relative position on the shelf (ranges from 0 to 1), increasing towards the
974  // calving front (see equation 10 in the PICO paper)
975  double r = d_gl / (double)(d_gl + d_cf);
976 
977  double C = pow(1.0 - r, 2.0);
978  for (int k = 0; k < n; ++k) {
979  if ((n - k - 1) / (double)n <= C and C <= (n - k) / (double)n) {
980  result(i, j) = std::min(d_gl, k + 1);
981  }
982  }
983  }
984  }
985 }
