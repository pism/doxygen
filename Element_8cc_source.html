31 //! Determinant of a 3x3 matrix
32 static double det(const double a[3][3]) {
33  return (a[0][0] * (a[1][1] * a[2][2] - a[1][2] * a[2][1]) -
34  a[0][1] * (a[1][0] * a[2][2] - a[1][2] * a[2][0]) +
35  a[0][2] * (a[1][0] * a[2][1] - a[1][1] * a[2][0]));
36 }

38 //! Cross product of two 3D vectors
39 static Vector3 cross(const Vector3 &a, const Vector3 &b) {
40  return {a.y * b.z - a.z * b.y,
41  a.z * b.x - a.x * b.z,
42  a.x * b.y - a.y * b.x};
43 }

60 //! Invert a 3x3 matrix
61 static void invert(const double A[3][3], double result[3][3]) {
62  const double det_A = det(A);
63 
64  assert(det_A != 0.0);
65 
66  Vector3
67  x0 = column(A, 0),
68  x1 = column(A, 1),
69  x2 = column(A, 2),
70  a = cross(x1, x2),
71  b = cross(x2, x0),
72  c = cross(x0, x1);
73 
74  double A_cofactor[3][3] = {{a.x, a.y, a.z},
75  {b.x, b.y, b.z},
76  {c.x, c.y, c.z}};
77 
78  // inverse(A) = 1/det(A) * transpose(A_cofactor)
79  for (int i = 0; i < 3; ++i) {
80  for (int j = 0; j < 3; ++j) {
81  // note the transpose on the RHS
82  result[i][j] = A_cofactor[j][i] / det_A;
83  }
84  }
85 }

146 //! Initialize shape function values and quadrature weights of a 2D physical element.
147 /** Assumes that the Jacobian does not depend on coordinates of the current quadrature point.
148  */
149 void Element::initialize(const double J[3][3],
150  ShapeFunction f,
151  unsigned int n_chi,
152  const std::vector<QuadPoint>& points,
153  const std::vector<double>& W) {
154 
155  double J_inv[3][3];
156  invert(J, J_inv);
157 
158  for (unsigned int q = 0; q < m_Nq; q++) {
159  for (unsigned int k = 0; k < n_chi; k++) {
160  m_germs[q * m_n_chi + k] = multiply(J_inv, f(k, points[q]));
161  }
162  }
163 
164  m_weights.resize(m_Nq);
165  const double J_det = det(J);
166  for (unsigned int q = 0; q < m_Nq; q++) {
167  m_weights[q] = J_det * W[q];
168  }
169 }

301 P1Element2::P1Element2(const IceGrid &grid, const Quadrature &quadrature, int type)
302  : Element2(grid, quadrature.weights().size(), p1::n_chi, q1::n_chi) {
303 
304  double dx = grid.dx();
305  double dy = grid.dy();
306 
307  // outward pointing normals for all sides of a Q1 element with sides aligned with X and
308  // Y axes
309  const Vector2
310  n01( 0.0, -1.0), // south
311  n12( 1.0, 0.0), // east
312  n23( 0.0, 1.0), // north
313  n30(-1.0, 0.0); // west
314 
315  // "diagonal" sides
316  Vector2
317  n13( 1.0, dx / dy), // 1-3 diagonal, outward for element 0
318  n20(-1.0, dx / dy); // 2-0 diagonal, outward for element 1
319 
320  // normalize
321  n13 /= n13.magnitude();
322  n20 /= n20.magnitude();
323 
324  // coordinates of nodes of a Q1 element this P1 element is embedded in (up to
325  // translation)
326  Vector2 p[] = {{0, 0}, {dx, 0}, {dx, dy}, {0, dy}};
327  std::vector<Vector2> pts;
328 
329  switch (type) {
330  case 0:
331  m_i_offset = {0, 1, 0};
332  m_j_offset = {0, 0, 1};
333  m_normals = {n01, n13, n30};
334  pts = {p[0], p[1], p[3]};
335  break;
336  case 1:
337  m_i_offset = {1, 1, 0};
338  m_j_offset = {0, 1, 0};
339  m_normals = {n12, n20, n01};
340  pts = {p[1], p[2], p[0]};
341  break;
342  case 2:
343  m_i_offset = {1, 0, 1};
344  m_j_offset = {1, 1, 0};
345  m_normals = {n23, -1.0 * n13, n12};
346  pts = {p[2], p[3], p[1]};
347  break;
348  case 3:
349  default:
350  m_i_offset = {0, 0, 1};
351  m_j_offset = {1, 0, 1};
352  m_normals = {n30, -1.0 * n20, n23};
353  pts = {p[3], p[0], p[2]};
354  break;
355  }
356 
357  reset(0, 0);
358  // make sure add_contribution() ignores entries corresponding to the 4-th node
359  mark_row_invalid(3);
360  mark_col_invalid(3);
361 
362  m_side_lengths.resize(n_sides());
363  // compute side lengths
364  for (unsigned int k = 0; k < n_sides(); ++k) {
365  m_side_lengths[k] = (pts[k] - pts[(k + 1) % 3]).magnitude();
366  }
367 
368  // compute the Jacobian
369  double J[3][3];
370  set_to_identity(J);
371  J[0][0] = pts[1].u - pts[0].u;
372  J[0][1] = pts[1].v - pts[0].v;
373  J[1][0] = pts[2].u - pts[0].u;
374  J[1][1] = pts[2].v - pts[0].v;
375 
376  // initialize germs and quadrature weights for the quadrature on this physical element
377  initialize(J, p1::chi, p1::n_chi, quadrature.points(), quadrature.weights());
378  reset(0, 0);
379 }

447 /*! Initialize the element `i,j,k`.
448  *
449  *
450  * @param[in] i i-index of the lower left node
451  * @param[in] j j-index of the lower left node
452  * @param[in] k k-index of the lower left node
453  * @param[in] z z-coordinates of the nodes of this element
454  */
455 void Q1Element3::reset(int i, int j, int k, const double *z) {
456  // Record i,j,k corresponding to the current element:
457  m_i = i;
458  m_j = j;
459  m_k = k;
460 
461  // store z coordinates
462  for (unsigned int n = 0; n < m_n_chi; ++n) {
463  m_z_nodal[n] = z[n];
464  }
465 
466  // Set row and column info used to add contributions:
467  for (unsigned int n = 0; n < m_n_chi; ++n) {
468  m_col[n].i = k + m_k_offset[n]; // x -- z (STORAGE_ORDER)
469  m_col[n].j = i + m_i_offset[n]; // y -- x (STORAGE_ORDER)
470  m_col[n].k = j + m_j_offset[n]; // z -- y (STORAGE_ORDER)
471  m_col[n].c = 0;
472  }
473  m_row = m_col;
474 
475  // Mark rows that we don't own as invalid:
476  for (unsigned int n = 0; n < m_n_chi; n++) {
477  auto I = local_to_global(n);
478  if (I.i < m_grid.xs or m_grid.xs + m_grid.xm - 1 < I.i or
479  I.j < m_grid.ys or m_grid.ys + m_grid.ym - 1 < I.j) {
480  mark_row_invalid(n);
481  }
482  }
483 
484  // Compute J^{-1} and use it to compute m_germs and m_weights:
485  for (unsigned int q = 0; q < m_Nq; q++) {
486 
487  Vector3 dz{0.0, 0.0, 0.0};
488  for (unsigned int n = 0; n < m_n_chi; ++n) {
489  auto &chi = m_chi[q * m_n_chi + n];
490  dz.x += chi.dx * z[n];
491  dz.y += chi.dy * z[n];
492  dz.z += chi.dz * z[n];
493  }
494 
495  double J[3][3] = {{m_dx / 2.0, 0.0, dz.x},
496  { 0.0, m_dy / 2.0, dz.y},
497  { 0.0, 0.0, dz.z}};
498 
499  double J_det = J[0][0] * J[1][1] * J[2][2];
500 
501  assert(J_det != 0.0);
502 
503  double J_inv[3][3] = {{1.0 / J[0][0], 0.0, -J[0][2] / (J[0][0] * J[2][2])},
504  {0.0, 1.0 / J[1][1], -J[1][2] / (J[1][1] * J[2][2])},
505  {0.0, 0.0, 1.0 / J[2][2]}};
506 
507  m_weights[q] = J_det * m_w[q];
508 
509  for (unsigned int n = 0; n < m_n_chi; n++) {
510  auto &chi = m_chi[q * m_n_chi + n];
511  // FIXME: I should be able to use multiply() defined above, but there must be a bug
512  // there...
513  m_germs[q * m_n_chi + n] = {chi.val,
514  J_inv[0][0] * chi.dx + J_inv[0][1] * chi.dy + J_inv[0][2] * chi.dz,
515  J_inv[1][0] * chi.dx + J_inv[1][1] * chi.dy + J_inv[1][2] * chi.dz,
516  J_inv[2][0] * chi.dx + J_inv[2][1] * chi.dy + J_inv[2][2] * chi.dz};
517  }
518  }
519 }

538 void Q1Element3Face::reset(int face, const double *z) {
539 
540  // Turn coordinates of a 2D quadrature point on [-1,1]*[-1,1] into coordinates on a face
541  // of the cube [-1,1]*[-1,1]*[-1,1].
542  for (unsigned int q = 0; q < m_Nq; q++) {
543  auto pt = m_points[q];
544  QuadPoint P;
545 
546  // This expresses parameterizations of faces of the reference element: for example,
547  // face 0 is parameterized by (s, t) -> [-1, s, t].
548  switch (face) {
549  case 0:
550  P = {-1.0, pt.xi, pt.eta};
551  break;
552  case 1:
553  P = { 1.0, pt.xi, pt.eta};
554  break;
555  case 2:
556  P = {pt.xi, -1.0, pt.eta};
557  break;
558  case 3:
559  P = {pt.xi, 1.0, pt.eta};
560  break;
561  case 4:
562  P = {pt.xi, pt.eta, -1.0};
563  break;
564  case 5:
565  P = {pt.xi, pt.eta, 1.0};
566  break;
567  }
568 
569  // Compute dz/dxi, dz/deta and dz/dzeta.
570  Vector3 dz{0.0, 0.0, 0.0};
571  for (unsigned int n = 0; n < m_n_chi; ++n) {
572  // Note: chi(n, point) for a particular face does not depend on the element geometry
573  // and could be computed in advance (in the constructor). On the other hand, I
574  // expect that the number of faces in a Neumann boundary is relatively small and
575  // this optimization is not likely to pay off.
576  auto chi = q13d::chi(n, P);
577 
578  m_chi[q * m_n_chi + n] = chi.val;
579 
580  dz.x += chi.dx * z[n];
581  dz.y += chi.dy * z[n];
582  dz.z += chi.dz * z[n];
583  }
584 
585  // Use the magnitude of the normal to a face to turn quadrature weights on
586  // [-1,1]*[-1,1] into quadrature weights on a face of this physical element.
587 
588  m_weights[q] = m_w[q];
589 
590  // Sign (-1 or +1) used to set normal orientation This relies of the order of faces in
591  // fem::q13d::incident_nodes and above (see the code setting QuadPoint P).
592  double sign = 2 * (face % 2) - 1;
593 
594  switch (face) {
595  case 0:
596  case 1:
597  m_weights[q] *= 0.5 * m_dy * dz.z;
598  m_normals[q] = {sign, 0.0, 0.0};
599  break;
600  case 2:
601  case 3:
602  m_weights[q] *= 0.5 * m_dx * dz.z;
603  m_normals[q] = {0.0, sign, 0.0};
604  break;
605  case 4:
606  case 5:
607  {
608  double
609  a = 0.5 * m_dy * dz.x,
610  b = 0.5 * m_dx * dz.y,
611  c = 0.25 * m_dx * m_dy,
612  M = std::sqrt(a * a + b * b + c * c);
613  m_weights[q] *= M;
614 
615  M *= sign;
616  m_normals[q] = {a / M, b / M, c / M};
617  }
618  break;
619  }
620  } // end of the loop over quadrature points
621 }
