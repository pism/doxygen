27 /*!
28  * z coordinates of the nodes
29  *
30  * @param[in] b surface elevation of the bottom of the domain
31  * @param[in] H domain thickness
32  * @param[in] Mz number of grid points in each vertical column
33  * @param[in] k node index in the z direction
34  */
35 double grid_z(double b, double H, int Mz, int k) {
36  return b + H * k / (Mz - 1.0);

39 /* Transpose a DMDALocalInfo structure to map from PETSc's ordering to PISM's order needed
40  to ensure that vertical columns are stored contiguously in RAM.
41 
42  (What a pain.)
43 
44  Map from PETSc to PISM order:
45 
46  | PETSc | PISM |
47  |-------+------|
48  | x | z |
49  | y | x |
50  | z | y |
51 
52  Assuming that i, j, k indexes correspond to x, y, and z PETSc's indexing order is
53  [k][j][i]. After this transpose we have to use [j][i][k].
54 
55  Note that this indexing order is compatible with the PETSc-standard indexing for 2D
56  Vecs: [j][i].
57 
58  All the lines changed to implement this transpose are marked with STORAGE_ORDER: that
59  way you can use grep to find them.
60  */
61 DMDALocalInfo grid_transpose(const DMDALocalInfo &input) {
62  DMDALocalInfo result = input;
63 
64  result.mx = input.my;
65  result.my = input.mz;
66  result.mz = input.mx;
67 
68  result.xs = input.ys;
69  result.ys = input.zs;
70  result.zs = input.xs;
71 
72  result.xm = input.ym;
73  result.ym = input.zm;
74  result.zm = input.xm;
75 
76  result.gxs = input.gys;
77  result.gys = input.gzs;
78  result.gzs = input.gxs;
79 
80  result.gxm = input.gym;
81  result.gym = input.gzm;
82  result.gzm = input.gxm;
83 
84  result.bx = input.by;
85  result.by = input.bz;
86  result.bz = input.bx;
87 
88  return result;
89 }

92 /*!
93  * Set up storage for 3D data inputs (DMDAs and Vecs)
94  */
95 PetscErrorCode setup_level(DM dm, int mg_levels) {
96  PetscErrorCode ierr;
97 
98  // Create a 3D DMDA and a global Vec, then stash them in dm.
99  {
100  DM da;
101  Vec parameters;
102  int dof = 1;
103 #if PETSC_VERSION_LT(3,10,0)
104  ierr = DMDAGetReducedDMDA(dm, dof, &da); CHKERRQ(ierr);
105 #else
106  ierr = DMDACreateCompatibleDMDA(dm, dof, &da); CHKERRQ(ierr);
107 #endif
108 
109  ierr = DMSetUp(da); CHKERRQ(ierr);
110 
111  ierr = DMCreateGlobalVector(da, &parameters); CHKERRQ(ierr);
112 
113  ierr = PetscObjectCompose((PetscObject)dm, "3D_DM", (PetscObject)da); CHKERRQ(ierr);
114  ierr = PetscObjectCompose((PetscObject)dm, "3D_DM_data", (PetscObject)parameters); CHKERRQ(ierr);
115 
116  ierr = DMDestroy(&da); CHKERRQ(ierr);
117 
118  ierr = VecDestroy(&parameters); CHKERRQ(ierr);
119  }
120 
121  // get coarsening level
122  PetscInt level = 0;
123  ierr = DMGetCoarsenLevel(dm, &level); CHKERRQ(ierr);
124 
125  // report
126  {
127  MPI_Comm comm;
128  ierr = PetscObjectGetComm((PetscObject)dm, &comm); CHKERRQ(ierr);
129 
130  DMDALocalInfo info;
131  ierr = DMDAGetLocalInfo(dm, &info); CHKERRQ(ierr);
132  info = grid_transpose(info);
133 
134  int
135  Mx = info.mx,
136  My = info.my,
137  Mz = info.mz;
138  ierr = PetscPrintf(comm,
139  "Blatter grid level %D: %3d x %3d x %3d (%8d) nodes\n",
140  (mg_levels - 1) - level, Mx, My, Mz, Mx * My * Mz); CHKERRQ(ierr);
141  }
142  return 0;
143 }

145 /*! @brief Create the restriction matrix.
146  *
147  * The result of this call is attached to `dm_fine` under `mat_name`.
148  *
149  * @param[in] fine DM corresponding to the fine grid
150  * @param[in] coarse DM corresponding to the coarse grid
151  * @param[in] dm_name name of the DM for 2D or 3D parameters
152  * @param[in] mat_name name to use when attaching the restriction matrix to `fine`
153  */
154 PetscErrorCode create_restriction(DM fine, DM coarse, const char *dm_name) {
155  PetscErrorCode ierr;
156  DM da_fine, da_coarse;
157  Mat mat;
158  Vec scale;
159 
160  std::string
161  prefix = dm_name,
162  mat_name = prefix + "_restriction",
163  vec_name = prefix + "_scaling";
164 
165  /* Get the DM for parameters from the fine grid DM */
166  ierr = PetscObjectQuery((PetscObject)fine, dm_name, (PetscObject *)&da_fine); CHKERRQ(ierr);
167  if (!da_fine) {
168  SETERRQ1(PETSC_COMM_SELF, PETSC_ERR_ARG_WRONG, "No %s composed with given DMDA", dm_name); // LCOV_EXCL_LINE
169  }
170 
171  /* Get the DM for parameters from the coarse grid DM */
172  ierr = PetscObjectQuery((PetscObject)coarse, dm_name, (PetscObject *)&da_coarse); CHKERRQ(ierr);
173  if (!da_coarse) {
174  SETERRQ1(PETSC_COMM_SELF, PETSC_ERR_ARG_WRONG, "No %s composed with given DMDA", dm_name); // LCOV_EXCL_LINE
175  }
176 
177  /* call DMCreateInterpolation */
178  ierr = DMCreateInterpolation(da_coarse, da_fine, &mat, &scale); CHKERRQ(ierr);
179 
180  /* attach to the fine grid DM */
181  ierr = PetscObjectCompose((PetscObject)fine, vec_name.c_str(), (PetscObject)scale); CHKERRQ(ierr);
182  ierr = VecDestroy(&scale); CHKERRQ(ierr);
183 
184  ierr = PetscObjectCompose((PetscObject)fine, mat_name.c_str(), (PetscObject)mat); CHKERRQ(ierr);
185  ierr = MatDestroy(&mat); CHKERRQ(ierr);
186 
187  return 0;
188 }

191 /*! @brief Restrict model parameters from the `fine` grid onto the `coarse` grid.
192  *
193  * This function uses the restriction matrix created by coarsening_hook().
194  */
195 PetscErrorCode restrict_data(DM fine, DM coarse, const char *dm_name) {
196  PetscErrorCode ierr;
197  Vec X_fine, X_coarse, scaling;
198  DM da_fine, da_coarse;
199  Mat mat;
200 
201  std::string
202  prefix = dm_name,
203  mat_name = prefix + "_restriction",
204  scaling_name = prefix + "_scaling",
205  vec_name = prefix + "_data";
206 
207  /* get the restriction matrix from the fine grid DM */
208  ierr = PetscObjectQuery((PetscObject)fine, mat_name.c_str(), (PetscObject *)&mat); CHKERRQ(ierr);
209  if (!mat) {
210  SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_WRONG, "Failed to get the restriction matrix"); // LCOV_EXCL_LINE
211  }
212 
213  /* get the scaling vector from the fine grid DM */
214  ierr = PetscObjectQuery((PetscObject)fine, scaling_name.c_str(), (PetscObject *)&scaling); CHKERRQ(ierr);
215  if (!scaling) {
216  SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_WRONG, "Failed to get the scaling vector"); // LCOV_EXCL_LINE
217  }
218 
219  /* get the DMDA from the fine grid DM */
220  ierr = PetscObjectQuery((PetscObject)fine, dm_name, (PetscObject *)&da_fine); CHKERRQ(ierr);
221  if (!da_fine) {
222  SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_WRONG, "Failed to get the fine grid DM"); // LCOV_EXCL_LINE
223  }
224 
225  /* get the storage vector from the fine grid DM */
226  ierr = PetscObjectQuery((PetscObject)fine, vec_name.c_str(), (PetscObject *)&X_fine); CHKERRQ(ierr);
227  if (!X_fine) {
228  SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_WRONG, "Failed to get the fine grid Vec"); // LCOV_EXCL_LINE
229  }
230 
231  /* get the DMDA from the coarse grid DM */
232  ierr = PetscObjectQuery((PetscObject)coarse, dm_name, (PetscObject *)&da_coarse); CHKERRQ(ierr);
233  if (!da_coarse) {
234  SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_WRONG, "Failed to get the coarse grid DM"); // LCOV_EXCL_LINE
235  }
236 
237  /* get the storage vector from the coarse grid DM */
238  ierr = PetscObjectQuery((PetscObject)coarse, vec_name.c_str(), (PetscObject *)&X_coarse); CHKERRQ(ierr);
239  if (!X_coarse) {
240  SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_WRONG, "Failed to get the coarse grid Vec"); // LCOV_EXCL_LINE
241  }
242 
243  ierr = MatRestrict(mat, X_fine, X_coarse); CHKERRQ(ierr);
244 
245  ierr = VecPointwiseMult(X_coarse, X_coarse, scaling); CHKERRQ(ierr);
246 
247  return 0;
248 }
