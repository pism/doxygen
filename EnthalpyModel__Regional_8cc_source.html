71 void EnthalpyModel_Regional::update_impl(double t, double dt,
72  const Inputs &inputs) {
73 
74  unsigned int Mz = m_grid->Mz();
75 
76  EnthalpyModel::update_impl(t, dt, inputs);
77 
78  const IceModelVec2Int &no_model_mask = *inputs.no_model_mask;
79 
80  // The update_impl() call above sets m_work; ghosts are communicated
81  // later (in EnergyModel::update()).
82  IceModelVec::AccessList list{&no_model_mask, &m_work, &m_ice_enthalpy,
83  &m_basal_melt_rate, &m_basal_melt_rate_stored};
84 
85  for (Points p(*m_grid); p; p.next()) {
86  const int i = p.i(), j = p.j();
87 
88  if (no_model_mask(i, j) > 0.5) {
89  double *new_enthalpy = m_work.get_column(i, j);
90  double *old_enthalpy = m_ice_enthalpy.get_column(i, j);
91 
92  // enthalpy
93  for (unsigned int k = 0; k < Mz; ++k) {
94  new_enthalpy[k] = old_enthalpy[k];
95  }
96 
97  // basal melt rate
98  m_basal_melt_rate(i, j) = m_basal_melt_rate_stored(i, j);
99  }
100  }
101 }
