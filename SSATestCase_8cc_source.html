107 void SSATestCase::run() {
108  m_ctx->log()->message(2, "* Solving the SSA stress balance ...\n");
109 
110  m_geometry.ensure_consistency(m_config->get_number("stress_balance.ice_free_thickness_standard"));
111 
112  Inputs inputs;
113  inputs.water_column_pressure = nullptr;
114  inputs.geometry = &m_geometry;
115  inputs.enthalpy = &m_ice_enthalpy;
116  inputs.basal_yield_stress = &m_tauc;
117  inputs.bc_mask = &m_bc_mask;
118  inputs.bc_values = &m_bc_values;
119 
120  bool full_update = true;
121  m_ssa->update(inputs, full_update);
122 }

124 //! Report on the generated solution
125 void SSATestCase::report(const std::string &testname) {
126 
127  m_ctx->log()->message(3, m_ssa->stdout_report());
128 
129  double
130  maxvecerr = 0.0,
131  avvecerr = 0.0,
132  avuerr = 0.0,
133  avverr = 0.0,
134  maxuerr = 0.0,
135  maxverr = 0.0;
136  double
137  gmaxvecerr = 0.0,
138  gavvecerr = 0.0,
139  gavuerr = 0.0,
140  gavverr = 0.0,
141  gmaxuerr = 0.0,
142  gmaxverr = 0.0;
143 
144  if (m_config->get_flag("basal_resistance.pseudo_plastic.enabled") &&
145  m_config->get_number("basal_resistance.pseudo_plastic.q") != 1.0) {
146  m_ctx->log()->message(1,
147  "WARNING: numerical errors not valid for pseudo-plastic till\n");
148  }
149  m_ctx->log()->message(1,
150  "NUMERICAL ERRORS in velocity relative to exact solution:\n");
151 
152  const IceModelVec2V &vel_ssa = m_ssa->velocity();
153 
154  IceModelVec::AccessList list{&vel_ssa};
155 
156  double exactvelmax = 0, gexactvelmax = 0;
157  for (Points p(*m_grid); p; p.next()) {
158  const int i = p.i(), j = p.j();
159 
160  double uexact, vexact;
161  double myx = m_grid->x(i), myy = m_grid->y(j);
162 
163  exactSolution(i,j,myx,myy,&uexact,&vexact);
164 
165  double exactnormsq=sqrt(uexact*uexact+vexact*vexact);
166  exactvelmax = std::max(exactnormsq,exactvelmax);
167 
168  // compute maximum errors
169  const double uerr = fabs(vel_ssa(i,j).u - uexact);
170  const double verr = fabs(vel_ssa(i,j).v - vexact);
171  avuerr = avuerr + uerr;
172  avverr = avverr + verr;
173  maxuerr = std::max(maxuerr,uerr);
174  maxverr = std::max(maxverr,verr);
175  const double vecerr = sqrt(uerr * uerr + verr * verr);
176  maxvecerr = std::max(maxvecerr,vecerr);
177  avvecerr = avvecerr + vecerr;
178  }
179 
180  unsigned int N = (m_grid->Mx()*m_grid->My());
181 
182  gexactvelmax = GlobalMax(m_grid->com, exactvelmax);
183  gmaxuerr = GlobalMax(m_grid->com, maxuerr);
184  gmaxverr = GlobalMax(m_grid->com, maxverr);
185  gavuerr = GlobalSum(m_grid->com, avuerr);
186  gavuerr = gavuerr / N;
187  gavverr = GlobalSum(m_grid->com, avverr);
188  gavverr = gavverr / N;
189  gmaxvecerr = GlobalMax(m_grid->com, maxvecerr);
190  gavvecerr = GlobalSum(m_grid->com, avvecerr);
191  gavvecerr = gavvecerr / N;
192 
193  using pism::units::convert;
194 
195  m_ctx->log()->message(1,
196  "velocity : maxvector prcntavvec maxu maxv avu avv\n");
197  m_ctx->log()->message(1,
198  " %11.4f%13.5f%10.4f%10.4f%10.4f%10.4f\n",
199  convert(m_sys, gmaxvecerr, "m second-1", "m year-1"),
200  (gavvecerr/gexactvelmax)*100.0,
201  convert(m_sys, gmaxuerr, "m second-1", "m year-1"),
202  convert(m_sys, gmaxverr, "m second-1", "m year-1"),
203  convert(m_sys, gavuerr, "m second-1", "m year-1"),
204  convert(m_sys, gavverr, "m second-1", "m year-1"));
205 
206  m_ctx->log()->message(1, "NUM ERRORS DONE\n");
207 
208  report_netcdf(testname,
209  convert(m_sys, gmaxvecerr, "m second-1", "m year-1"),
210  (gavvecerr/gexactvelmax)*100.0,
211  convert(m_sys, gmaxuerr, "m second-1", "m year-1"),
212  convert(m_sys, gmaxverr, "m second-1", "m year-1"),
213  convert(m_sys, gavuerr, "m second-1", "m year-1"),
214  convert(m_sys, gavverr, "m second-1", "m year-1"));
215 }

217 void SSATestCase::report_netcdf(const std::string &testname,
218  double max_vector,
219  double rel_vector,
220  double max_u,
221  double max_v,
222  double avg_u,
223  double avg_v) {
224  VariableMetadata err("N", m_grid->ctx()->unit_system());
225  unsigned int start;
226  VariableMetadata global_attributes("PISM_GLOBAL", m_grid->ctx()->unit_system());
227 
228  options::String filename("-report_file", "NetCDF error report file");
229 
230  if (not filename.is_set()) {
231  return;
232  }
233 
234  err.set_string("units", "1");
235 
236  m_ctx->log()->message(2, "Also writing errors to '%s'...\n", filename->c_str());
237 
238  bool append = options::Bool("-append", "Append the NetCDF error report");
239 
240  IO_Mode mode = PISM_READWRITE;
241  if (not append) {
242  mode = PISM_READWRITE_MOVE;
243  }
244 
245  global_attributes.set_string("source", std::string("PISM ") + pism::revision);
246 
247  // Find the number of records in this file:
248  File file(m_grid->com, filename, PISM_NETCDF3, mode); // OK to use NetCDF3.
249  start = file.dimension_length("N");
250 
251  io::write_attributes(file, global_attributes, PISM_DOUBLE);
252 
253  // Write the dimension variable:
254  io::write_timeseries(file, err, (size_t)start, {(double)(start + 1)});
255 
256  // Always write grid parameters:
257  err.set_name("dx");
258  err.set_string("units", "meters");
259  io::write_timeseries(file, err, (size_t)start, {m_grid->dx()});
260  err.set_name("dy");
261  io::write_timeseries(file, err, (size_t)start, {m_grid->dy()});
262 
263  // Always write the test name:
264  err.clear_all_strings(); err.clear_all_doubles(); err.set_string("units", "1");
265  err.set_name("test");
266  io::write_timeseries(file, err, (size_t)start, {(double)testname[0]});
267 
268  err.clear_all_strings(); err.clear_all_doubles(); err.set_string("units", "1");
269  err.set_name("max_velocity");
270  err.set_string("units", "m year-1");
271  err.set_string("long_name", "maximum ice velocity magnitude error");
272  io::write_timeseries(file, err, (size_t)start, {max_vector});
273 
274  err.clear_all_strings(); err.clear_all_doubles(); err.set_string("units", "1");
275  err.set_name("relative_velocity");
276  err.set_string("units", "percent");
277  err.set_string("long_name", "relative ice velocity magnitude error");
278  io::write_timeseries(file, err, (size_t)start, {rel_vector});
279 
280  err.clear_all_strings(); err.clear_all_doubles(); err.set_string("units", "1");
281  err.set_name("maximum_u");
282  err.set_string("units", "m year-1");
283  err.set_string("long_name", "maximum error in the X-component of the ice velocity");
284  io::write_timeseries(file, err, (size_t)start, {max_u});
285 
286  err.clear_all_strings(); err.clear_all_doubles(); err.set_string("units", "1");
287  err.set_name("maximum_v");
288  err.set_string("units", "m year-1");
289  err.set_string("long_name", "maximum error in the Y-component of the ice velocity");
290  io::write_timeseries(file, err, (size_t)start, {max_v});
291 
292  err.clear_all_strings(); err.clear_all_doubles(); err.set_string("units", "1");
293  err.set_name("average_u");
294  err.set_string("units", "m year-1");
295  err.set_string("long_name", "average error in the X-component of the ice velocity");
296  io::write_timeseries(file, err, (size_t)start, {avg_u});
297 
298  err.clear_all_strings(); err.clear_all_doubles(); err.set_string("units", "1");
299  err.set_name("average_v");
300  err.set_string("units", "m year-1");
301  err.set_string("long_name", "average error in the Y-component of the ice velocity");
302  io::write_timeseries(file, err, (size_t)start, {avg_v});
303 
304  file.close();
305 }

313 //! Save the computation and data to a file.
314 void SSATestCase::write(const std::string &filename) {
315 
316  // Write results to an output file:
317  File file(m_grid->com, filename, PISM_NETCDF3, PISM_READWRITE_MOVE);
318  io::define_time(file, *m_grid->ctx());
319  io::append_time(file, *m_config, 0.0);
320 
321  m_geometry.ice_surface_elevation.write(file);
322  m_geometry.ice_thickness.write(file);
323  m_bc_mask.write(file);
324  m_tauc.write(file);
325  m_geometry.bed_elevation.write(file);
326  m_ice_enthalpy.write(file);
327  m_bc_values.write(file);
328 
329  const IceModelVec2V &vel_ssa = m_ssa->velocity();
330  vel_ssa.write(file);
331 
332  IceModelVec2V exact(m_grid, "_exact", WITHOUT_GHOSTS);
333  exact.set_attrs("diagnostic",
334  "X-component of the SSA exact solution",
335  "m s-1", "m year-1", "", 0);
336  exact.set_attrs("diagnostic",
337  "Y-component of the SSA exact solution",
338  "m s-1", "m year-1", "", 1);
339 
340  IceModelVec::AccessList list(exact);
341  for (Points p(*m_grid); p; p.next()) {
342  const int i = p.i(), j = p.j();
343 
344  exactSolution(i, j, m_grid->x(i), m_grid->y(j),
345  &(exact(i,j).u), &(exact(i,j).v));
346  }
347  exact.write(file);
348 
349  file.close();
350 }
