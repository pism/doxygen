274 void PNCFile::get_att_double_impl(const std::string &variable_name,
275  const std::string &att_name,
276  std::vector<double> &result) const {
277  int len, varid = get_varid(variable_name);
278  MPI_Offset attlen = 0;
279 
280  // Read the attribute length:
281  int stat = ncmpi_inq_attlen(m_file_id, varid, att_name.c_str(), &attlen);
282 
283  if (stat == NC_NOERR) {
284  len = static_cast<int>(attlen);
285  } else if (stat == NC_ENOTATT) {
286  len = 0;
287  } else {
288  check(PISM_ERROR_LOCATION, stat);
289  len = 0;
290  }
291 
292  if (len == 0) {
293  result.clear();
294  return;
295  }
296 
297  result.resize(len);
298 
299  // Now read data and broadcast stat to see if we succeeded:
300  stat = ncmpi_get_att_double(m_file_id, varid, att_name.c_str(), &result[0]); check(PISM_ERROR_LOCATION, stat);
301 
302  // On error, print a message and stop.
303  if (stat != NC_NOERR) {
304  fprintf(stderr, "Error reading the %s attribute; (varid %d, NetCDF error %s)",
305  att_name.c_str(), varid, ncmpi_strerror(stat));
306  }
307 }

311  int varid = get_varid(variable_name);
312

316  int stat = ncmpi_inq_attlen(m_file_id, varid, att_name.c_str(), &attlen);
317  int len = (stat == NC_NOERR) ? static_cast<int>(attlen) : 0;

320  if (len == 0) {

326 
327  // Now read the string and see if we succeeded:
328  stat = ncmpi_get_att_text(m_file_id, varid, att_name.c_str(), str.data());
329  check(PISM_ERROR_LOCATION, stat);

331  result = str.data();

411  nc_start[j] = start[j];
412  nc_count[j] = count[j];

416 
417  if (transposed) {
418  stat = ncmpi_get_varm_double_all(m_file_id, varid,
419  &nc_start[0], &nc_count[0], &nc_stride[0], &nc_imap[0],
420  ip);
421  check(PISM_ERROR_LOCATION, stat);

423  stat = ncmpi_get_vara_double_all(m_file_id, varid,

430 void PNCFile::init_hints() {
431 
432  for (auto hint : m_mpi_io_hints) {
433  auto words = split(hint, ':');
434 
435  if (words.size() == 2) {
436  // printf("Setting MPI I/O hint \"%s\" to \"%s\"...\n",
437  // words[0].c_str(), words[1].c_str());
438 
439  MPI_Info_set(m_mpi_info,
440  const_cast<char*>(words[0].c_str()),
441  const_cast<char*>(words[1].c_str()));
442  } else {
443  int rank = 0;
444  MPI_Comm_rank(m_com, &rank);
445  if (rank == 0) {
446  printf("PISM WARNING: invalid MPI I/O hint: %s. Ignoring it...\n",
447  hint.c_str());
448  }
449  }
450  }
