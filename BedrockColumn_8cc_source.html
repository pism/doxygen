43 /*!
44  * Advance the heat equation in time.
45  *
46  * @param[in] dt time step length
47  * @param[in] Q_bottom heat flux into the column through the bottom surface
48  * @param[in] T_top temperature at the top surface
49  * @param[in] T_old current temperature in the column
50  * @param[out] T_new output
51  *
52  * Note: T_old and T_new may point to the same location.
53  */
54 void BedrockColumn::solve(double dt, double Q_bottom, double T_top,
55  const double *T_old, double *T_new) {
56 
57  double R = m_D * dt / (m_dz * m_dz);
58  double G = -Q_bottom / m_k;
59 
60  m_system.L(0) = 0.0; // not used
61  m_system.D(0) = 1.0 + 2.0 * R;
62  m_system.U(0) = -2.0 * R;
63  m_system.RHS(0) = T_old[0] - 2.0 * G * m_dz * R;
64 
65  unsigned int N = m_M - 1;
66 
67  for (unsigned int k = 1; k < N; ++k) {
68  m_system.L(k) = -R;
69  m_system.D(k) = 1.0 + 2.0 * R;
70  m_system.U(k) = -R;
71  m_system.RHS(k) = T_old[k];
72  }
73 
74  m_system.L(N) = 0.0;
75  m_system.D(N) = 1.0;
76  m_system.U(N) = 0.0; // not used
77  m_system.RHS(N) = T_top;
78 
79  m_system.solve(m_M, T_new);
80 }
