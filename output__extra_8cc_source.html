35 //! Computes the maximum time-step we can take and still hit all `-extra_times`.
36 MaxTimestep IceModel::extras_max_timestep(double my_t) {
37 
38  if ((not m_save_extra) or
39  (not m_config->get_flag("time_stepping.hit_extra_times"))) {
40  return MaxTimestep("reporting (-extra_times)");
41  }
42 
43  double eps = m_config->get_number("time_stepping.resolution");
44 
45  return reporting_max_timestep(m_extra_times, my_t, eps,
46  "reporting (-extra_times)");
47 }

117 //! Initialize the code saving spatially-variable diagnostic quantities.
118 void IceModel::init_extras() {
119 
120  m_last_extra = 0; // will be set in write_extras()
121  m_next_extra = 0;
122 
123  m_extra_filename = m_config->get_string("output.extra.file");
124  std::string times = m_config->get_string("output.extra.times");
125  std::string vars = m_config->get_string("output.extra.vars");
126  bool split = m_config->get_flag("output.extra.split");
127  bool append = m_config->get_flag("output.extra.append");
128 
129  bool extra_file_set = not m_extra_filename.empty();
130  bool times_set = not times.empty();
131 
132  if (extra_file_set ^ times_set) {
133  throw RuntimeError(PISM_ERROR_LOCATION,
134  "you need to set both output.extra.file and output.extra.times"
135  " to save spatial time-series.");
136  }
137 
138  if (not extra_file_set and not times_set) {
139  m_save_extra = false;
140  return;
141  }
142 
143  try {
144  m_extra_times = m_time->parse_times(times);
145  } catch (RuntimeError &e) {
146  e.add_context("parsing the output.extra.times argument %s", times.c_str());
147  throw;
148  }
149 
150  if (m_extra_times.size() == 0) {
151  throw RuntimeError(PISM_ERROR_LOCATION, "output.extra.times cannot be empty");
152  }
153 
154  if (append and split) {
155  throw RuntimeError(PISM_ERROR_LOCATION,
156  "both output.extra.split and output.extra.append are set.");
157  }
158 
159  if (append) {
160  File file(m_grid->com, m_extra_filename, PISM_NETCDF3, PISM_READONLY);
161 
162  std::string time_name = m_config->get_string("time.dimension_name");
163  if (file.find_variable(time_name)) {
164  double time_max = vector_max(file.read_dimension(time_name));
165 
166  while (m_next_extra + 1 < m_extra_times.size() && m_extra_times[m_next_extra + 1] < time_max) {
167  m_next_extra++;
168  }
169 
170  if (m_next_extra > 0) {
171  m_log->message(2,
172  "skipping times before the last record in %s (at %s)\n",
173  m_extra_filename.c_str(), m_time->date(time_max).c_str());
174  }
175 
176  // discard requested times before the beginning of the run
177  std::vector<double> tmp(m_extra_times.size() - m_next_extra);
178  for (unsigned int k = 0; k < tmp.size(); ++k) {
179  tmp[k] = m_extra_times[m_next_extra + k];
180  }
181 
182  m_extra_times = tmp;
183  m_next_extra = 0;
184  }
185  }
186 
187  m_save_extra = true;
188  m_extra_file_is_ready = false;
189  m_split_extra = false;
190 
191  if (split) {
192  m_split_extra = true;
193  m_log->message(2, "saving spatial time-series to '%s+year.nc'; ",
194  m_extra_filename.c_str());
195  } else {
196  if (not ends_with(m_extra_filename, ".nc")) {
197  m_log->message(2,
198  "PISM WARNING: spatial time-series file name '%s' does not have the '.nc' suffix!\n",
199  m_extra_filename.c_str());
200  }
201  m_log->message(2, "saving spatial time-series to '%s'; ",
202  m_extra_filename.c_str());
203  }
204 
205  m_log->message(2, "times requested: %s\n", times.c_str());
206 
207  if (m_extra_times.size() > 500) {
208  m_log->message(2,
209  "PISM WARNING: more than 500 times requested. This might fill your hard-drive!\n");
210  }
211 
212 #ifdef NC_HAVE_META_H
213  {
214  if (100 * NC_VERSION_MAJOR + 10 * NC_VERSION_MINOR + NC_VERSION_PATCH < 473) {
215  if (m_extra_times.size() > 5000 and m_config->get_string("output.format") == "netcdf4_parallel") {
216  throw RuntimeError(PISM_ERROR_LOCATION,
217  "more than 5000 times requested."
218  "Please use -extra_split to avoid a crash caused by a bug in NetCDF versions older than 4.7.3.\n"
219  "Alternatively\n"
220  "- split this simulation into several runs and then concatenate results\n"
221  "- select a different output.format value\n"
222  "- upgrade NetCDF to 4.7.3");
223  }
224  }
225  }
226 #endif
227 
228  if (not vars.empty()) {
229  m_extra_vars = process_extra_shortcuts(*m_config, set_split(vars, ','));
230  m_log->message(2, "variables requested: %s\n", vars.c_str());
231  } else {
232  m_log->message(2,
233  "PISM WARNING: output.extra.vars was not set. Writing the model state...\n");
234  } // end of the else clause after "if (extra_vars_set)"
235 }

237 //! Write spatially-variable diagnostic quantities.
238 void IceModel::write_extras() {
239  double saving_after = -1.0e30; // initialize to avoid compiler warning; this
240  // value is never used, because saving_after
241  // is only used if save_now == true, and in
242  // this case saving_after is guaranteed to be
243  // initialized. See the code below.
244  std::string filename;
245  unsigned int current_extra;
246  // determine if the user set the -save_at and -save_to options
247  if (not m_save_extra) {
248  return;
249  }
250 
251  const double time_resolution = m_config->get_number("time_stepping.resolution");
252  double current_time = m_time->current();
253 
254  // do we need to save *now*?
255  if (m_next_extra < m_extra_times.size() and
256  (current_time >= m_extra_times[m_next_extra] or
257  fabs(current_time - m_extra_times[m_next_extra]) < time_resolution)) {
258  // the condition above is "true" if we passed a requested time or got to
259  // within time_resolution seconds from it
260 
261  current_extra = m_next_extra;
262 
263  // update next_extra
264  while (m_next_extra < m_extra_times.size() and
265  (m_extra_times[m_next_extra] <= current_time or
266  fabs(current_time - m_extra_times[m_next_extra]) < time_resolution)) {
267  m_next_extra++;
268  }
269 
270  saving_after = m_extra_times[current_extra];
271  } else {
272  return;
273  }
274 
275  if (current_extra == 0) {
276  // The first time defines the left end-point of the first reporting interval; we don't write a
277  // report at this time.
278 
279  // Re-initialize last_extra (the correct value is not known at the time init_extras() is
280  // called).
281  m_last_extra = current_time;
282 
283  // ISMIP6 runs need to save diagnostics at the beginning of the run
284  if (not m_config->get_flag("output.ISMIP6")) {
285  return;
286  }
287  }
288 
289  if (saving_after < m_time->start()) {
290  // Suppose a user tells PISM to write data at times 0:1000:10000. Suppose
291  // also that PISM writes a backup file at year 2500 and gets stopped.
292  //
293  // When restarted, PISM will decide that it's time to write data for time
294  // 2000, but
295  // * that record was written already and
296  // * PISM will end up writing at year 2500, producing a file containing one
297  // more record than necessary.
298  //
299  // This check makes sure that this never happens.
300  return;
301  }
302 
303  if (m_split_extra) {
304  m_extra_file_is_ready = false; // each time-series record is written to a separate file
305  auto date_without_spaces = std::regex_replace(m_time->date(m_time->current()),
306  std::regex(" "), "_");
307  filename = pism::printf("%s_%s.nc",
308  m_extra_filename.c_str(),
309  date_without_spaces.c_str());
310  } else {
311  filename = m_extra_filename;
312  }
313 
314  m_log->message(3,
315  "saving spatial time-series to %s at %s\n",
316  filename.c_str(), m_time->date(m_time->current()).c_str());
317 
318  // default behavior is to move the file aside if it exists already; option allows appending
319  bool append = m_config->get_flag("output.extra.append");
320  IO_Mode mode = m_extra_file_is_ready or append ? PISM_READWRITE : PISM_READWRITE_MOVE;
321 
322  const Profiling &profiling = m_ctx->profiling();
323  profiling.begin("io.extra_file");
324  {
325  if (not m_extra_file) {
326  m_extra_file.reset(new File(m_grid->com,
327  filename,
328  string_to_backend(m_config->get_string("output.format")),
329  mode,
330  m_ctx->pio_iosys_id()));
331  }
332 
333  std::string time_name = m_config->get_string("time.dimension_name");
334 
335  if (not m_extra_file_is_ready) {
336  // Prepare the file:
337  io::define_time(*m_extra_file, *m_ctx);
338  m_extra_file->write_attribute(time_name, "bounds", "time_bounds");
339 
340  io::define_time_bounds(m_extra_bounds,
341  time_name, "nv", *m_extra_file);
342 
343  write_metadata(*m_extra_file, WRITE_MAPPING, PREPEND_HISTORY);
344 
345  m_extra_file_is_ready = true;
346  }
347 
348  write_run_stats(*m_extra_file);
349 
350  save_variables(*m_extra_file,
351  m_extra_vars.empty() ? INCLUDE_MODEL_STATE : JUST_DIAGNOSTICS,
352  m_extra_vars,
353  0.5 * (m_last_extra + current_time), // use the mid-point of the
354  // current reporting interval
355  PISM_FLOAT);
356 
357  // Get the length of the time dimension *after* it is appended to.
358  unsigned int time_length = m_extra_file->dimension_length(time_name);
359  size_t time_start = time_length > 0 ? static_cast<size_t>(time_length - 1) : 0;
360 
361  io::write_time_bounds(*m_extra_file, m_extra_bounds,
362  time_start, {m_last_extra, current_time});
363  // make sure all changes are written
364  m_extra_file->sync();
365  }
366  profiling.end("io.extra_file");
367 
368  flush_timeseries();
369 
370  if (m_split_extra) {
371  // each record is saved to a new file, so we can close this one
372  m_extra_file.reset(nullptr);
373  }
374 
375  m_last_extra = current_time;
376 
377  // reset accumulators in diagnostics that compute time averaged quantities
378  reset_diagnostics();
379 }
