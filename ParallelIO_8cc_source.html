246 void ParallelIO::write_darray_impl(const std::string &variable_name,
247  const IceGrid &grid,
248  unsigned int z_count,
249  unsigned int record,
250  const double *input) {
251 
252  int stat = 0, varid;
253 
254  stat = PIOc_inq_varid(m_file_id, variable_name.c_str(), &varid);
255  check(PISM_ERROR_LOCATION, stat);
256 
257  int type = 0;
258  stat = PIOc_inq_vartype(m_file_id, varid, &type);
259 
260  int decompid = grid.pio_io_decomposition(z_count, type);
261 
262  size_t length = grid.xm() * grid.ym() * z_count;
263 
264  switch (type) {
265  case PIO_DOUBLE:
266  // no conversion necessary
267  stat = PIOc_put_vard_double(m_file_id, varid, decompid, (PIO_Offset)record, input);
268  check(PISM_ERROR_LOCATION, stat);
269  break;
270  case PIO_FLOAT:
271  {
272  auto buffer = convert_data<float>(input, length);
273  stat = PIOc_put_vard_float(m_file_id, varid, decompid, (PIO_Offset)record, buffer.data());
274  check(PISM_ERROR_LOCATION, stat);
275  break;
276  }
277  case PIO_INT:
278  {
279  auto buffer = convert_data<int>(input, length);
280  stat = PIOc_put_vard_int(m_file_id, varid, decompid, (PIO_Offset)record, buffer.data());
281  check(PISM_ERROR_LOCATION, stat);
282  break;
283  }
284  default:
285  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
286  "ParallelIO: type conversion is not supported");
287  }
288 }

369  const std::string &att_name,
370  std::vector<double> &result) const {

377  int len = 0;

385 
386  if (len == 0) {
387  result.clear();
388  return;
389  }
390 
391  result.resize(len);
392 
393  stat = PIOc_get_att_double(m_file_id, varid, att_name.c_str(), result.data());
394  check(PISM_ERROR_LOCATION, stat);
