46 /*!
47  * Compute the modified mask to avoid "wrapping around" of front retreat at domain
48  * boundaries.
49  */
50 void FrontRetreat::compute_modified_mask(const IceModelVec2CellType &input,
51  IceModelVec2CellType &output) const {
52 
53  IceModelVec::AccessList list{&input, &output};
54 
55  const int Mx = m_grid->Mx();
56  const int My = m_grid->My();
57 
58  ParallelSection loop(m_grid->com);
59  try {
60  for (PointsWithGhosts p(*m_grid); p; p.next()) {
61  const int i = p.i(), j = p.j();
62 
63  if (i < 0 or i >= Mx or j < 0 or j >= My) {
64  output(i, j) = MASK_ICE_FREE_OCEAN;
65  } else {
66  output(i, j) = input(i, j);
67  }
68  }
69  } catch (...) {
70  loop.failed();
71  }
72  loop.check();
73 }

75 /*!
76  * Compute the maximum time step length provided a horizontal retreat rate.
77  */
78 MaxTimestep FrontRetreat::max_timestep(const IceModelVec2CellType &cell_type,
79  const IceModelVec2Int &bc_mask,
80  const IceModelVec2S &retreat_rate) const {
81 
82  IceGrid::ConstPtr grid = retreat_rate.grid();
83  units::System::Ptr sys = grid->ctx()->unit_system();
84 
85  using units::convert;
86 
87  // About 9 hours which corresponds to 10000 km year-1 on a 10 km grid
88  double dt_min = convert(sys, 0.001, "years", "seconds");
89 
90  double
91  retreat_rate_max = 0.0,
92  retreat_rate_mean = 0.0;
93  int N_cells = 0;
94 
95  IceModelVec::AccessList list{&cell_type, &bc_mask, &retreat_rate};
96 
97  for (Points pt(*grid); pt; pt.next()) {
98  const int i = pt.i(), j = pt.j();
99 
100  if (cell_type.ice_free_ocean(i, j) and
101  cell_type.next_to_ice(i, j) and
102  bc_mask(i, j) < 0.5) {
103  // NB: this condition has to match the one in update_geometry()
104 
105  const double C = retreat_rate(i, j);
106 
107  N_cells += 1;
108  retreat_rate_mean += C;
109  retreat_rate_max = std::max(C, retreat_rate_max);
110  }
111  }
112 
113  N_cells = GlobalSum(grid->com, N_cells);
114  retreat_rate_mean = GlobalSum(grid->com, retreat_rate_mean);
115  retreat_rate_max = GlobalMax(grid->com, retreat_rate_max);
116 
117  if (N_cells > 0.0) {
118  retreat_rate_mean /= N_cells;
119  } else {
120  retreat_rate_mean = 0.0;
121  }
122 
123  double denom = retreat_rate_max / grid->dx();
124  const double epsilon = convert(sys, 0.001 / (grid->dx() + grid->dy()), "seconds", "years");
125 
126  double dt = 1.0 / (denom + epsilon);
127 
128  m_log->message(3,
129  " frontal retreat: maximum rate = %.2f m/year gives dt=%.5f years\n"
130  " mean rate = %.2f m/year over %d cells\n",
131  convert(m_sys, retreat_rate_max, "m second-1", "m year-1"),
132  convert(m_sys, dt, "seconds", "years"),
133  convert(m_sys, retreat_rate_mean, "m second-1", "m year-1"),
134  N_cells);
135 
136  return MaxTimestep(std::max(dt, dt_min), "front_retreat");
137 }

139 /*!
140  * Update ice geometry by applying a horizontal retreat rate.
141  *
142  * This code applies a horizontal retreat rate at "partially-filled" cells that are next
143  * to icy cells.
144  *
145  * Models providing the "retreat rate" should set this field to zero in areas where a
146  * particular parameterization does not apply. (For example: some calving models apply at
147  * shelf calving fronts, others may apply at grounded termini but not at ice shelves,
148  * etc).
149  */
150 void FrontRetreat::update_geometry(double dt,
151  const Geometry &geometry,
152  const IceModelVec2Int &bc_mask,
153  const IceModelVec2S &retreat_rate,
154  IceModelVec2S &Href,
155  IceModelVec2S &ice_thickness) {
156 
157  const IceModelVec2S &bed = geometry.bed_elevation;
158  const IceModelVec2S &sea_level = geometry.sea_level_elevation;
159  const IceModelVec2S &surface_elevation = geometry.ice_surface_elevation;
160 
161  if (m_config->get_flag("geometry.front_retreat.wrap_around")) {
162  m_cell_type.copy_from(geometry.cell_type);
163  } else {
164  // compute the modified mask needed to prevent front retreat from "wrapping around"
165  // the computational domain
166  compute_modified_mask(geometry.cell_type, m_cell_type);
167  }
168 
169  const double dx = m_grid->dx();
170 
171  m_tmp.set(0.0);
172 
173  IceModelVec::AccessList list{&ice_thickness, &bc_mask,
174  &bed, &sea_level, &m_cell_type, &Href, &m_tmp, &retreat_rate,
175  &surface_elevation};
176 
177  // Prepare to loop over neighbors: directions
178  const Direction dirs[] = {North, East, South, West};
179 
180  // Step 1: Apply the computed horizontal retreat rate:
181  for (Points pt(*m_grid); pt; pt.next()) {
182  const int i = pt.i(), j = pt.j();
183 
184  // apply retreat rate at the margin (i.e. to partially-filled cells) only
185  if (m_cell_type.ice_free_ocean(i, j) and
186  m_cell_type.next_to_ice(i, j) and
187  bc_mask(i, j) < 0.5) {
188  // NB: this condition has to match the one in max_timestep()
189 
190  const double
191  rate = retreat_rate(i, j),
192  Href_old = Href(i, j);
193 
194  // Compute the number of floating neighbors and the neighbor-averaged ice thickness:
195  double H_threshold = part_grid_threshold_thickness(m_cell_type.star(i, j),
196  ice_thickness.star(i, j),
197  surface_elevation.star(i, j),
198  bed(i, j));
199 
200  // Calculate mass loss with respect to the associated ice thickness and the grid size:
201  const double Href_change = -dt * rate * H_threshold / dx; // in m
202 
203  if (Href_old + Href_change >= 0.0) {
204  // Href is high enough to absorb the mass loss
205  Href(i, j) = Href_old + Href_change;
206  } else {
207  Href(i, j) = 0.0;
208  // Href + Href_change is negative: need to distribute mass loss to neighboring points
209 
210  // Find the number of neighbors to distribute to.
211  //
212  // We consider floating cells and grounded cells with the base below sea level. In other
213  // words, additional mass losses are distributed to shelf calving fronts and grounded marine
214  // termini.
215  int N = 0;
216  {
217  auto
218  M = m_cell_type.star(i, j),
219  BC = bc_mask.star(i, j);
220 
221  auto
222  b = bed.star(i, j),
223  sl = sea_level.star(i, j);
224 
225  for (int n = 0; n < 4; ++n) {
226  Direction direction = dirs[n];
227  int m = M[direction];
228  int bc = BC[direction];
229 
230  // note: this is where the modified cell type mask is used to prevent
231  // wrap-around
232  if (bc == 0 and // distribute to regular (*not* Dirichlet B.C.) neighbors only
233  (mask::floating_ice(m) or
234  (mask::grounded_ice(m) and b[direction] < sl[direction]))) {
235  N += 1;
236  }
237  }
238  }
239 
240  if (N > 0) {
241  m_tmp(i, j) = (Href_old + Href_change) / (double)N;
242  } else {
243  // No shelf calving front of grounded terminus to distribute to: retreat stops here.
244  m_tmp(i, j) = 0.0;
245  }
246  }
247 
248  } // end of "if ice free ocean next to ice and not a BC location "
249  } // end of loop over grid points
250 
251  // Step 2: update ice thickness and Href in neighboring cells if we need to propagate mass losses.
252  m_tmp.update_ghosts();
253 
254  for (Points p(*m_grid); p; p.next()) {
255  const int i = p.i(), j = p.j();
256 
257  // Note: this condition has to match the one in step 1 above.
258  if (bc_mask.as_int(i, j) == 0 and
259  (m_cell_type.floating_ice(i, j) or
260  (m_cell_type.grounded_ice(i, j) and bed(i, j) < sea_level(i, j)))) {
261 
262  const double delta_H = (m_tmp(i + 1, j) + m_tmp(i - 1, j) +
263  m_tmp(i, j + 1) + m_tmp(i, j - 1));
264 
265  if (delta_H < 0.0) {
266  Href(i, j) = ice_thickness(i, j) + delta_H; // in m
267  ice_thickness(i, j) = 0.0;
268  }
269 
270  // Stop retreat if the current cell does not have enough ice to absorb the loss.
271  if (Href(i, j) < 0.0) {
272  Href(i, j) = 0.0;
273  }
274  }
275  }
276 }
