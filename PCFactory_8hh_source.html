49  //! Creates a boundary model.
50  virtual std::shared_ptr<Model> create(const std::string &type) {
51 
52  std::vector<std::string> choices = split(type, ',');
53 
54  // the first element has to be an *actual* model (not a modifier)
55  auto j = choices.begin();
56 
57  auto result = model(*j);
58 
59  ++j;
60 
61  // process remaining arguments:
62  for (;j != choices.end(); ++j) {
63  result = modifier(*j, result);
64  }
65 
66  return result;
67  }

93  std::shared_ptr<Model> model(const std::string &type) {
94  if (m_models.find(type) == m_models.end()) {
