225 void TSSnapshotDiagnostic::evaluate(double t0, double t1, double v) {
226 
227  // skip times before the beginning of this time step
228  while (m_current_time < m_requested_times->size() and (*m_requested_times)[m_current_time] < t0) {
229  m_current_time += 1;
230  }
231 
232  while (m_current_time < m_requested_times->size() and (*m_requested_times)[m_current_time] <= t1) {
233  const unsigned int k = m_current_time;
234  m_current_time += 1;
235 
236  // skip the first time: it defines the beginning of a reporting interval
237  if (k == 0) {
238  continue;
239  }
240 
241  const double t_s = (*m_requested_times)[k - 1];
242  const double t_e = (*m_requested_times)[k];
243 
244  // store computed data in the buffer
245  {
246  m_time.push_back(t_e);
247  m_values.push_back(v);
248  m_bounds.push_back(t_s);
249  m_bounds.push_back(t_e);
250  }
251  }
252 }

254 void TSRateDiagnostic::evaluate(double t0, double t1, double change) {
255  static const double epsilon = 1e-4; // seconds
256  assert(t1 > t0);
257 
258  // skip times before and including the beginning of this time step
259  while (m_current_time < m_requested_times->size() and (*m_requested_times)[m_current_time] <= t0) {
260  m_current_time += 1;
261  }
262 
263  // number of requested times in this time step
264  unsigned int N = 0;
265 
266  // loop through requested times that are within this time step
267  while (m_current_time < m_requested_times->size() and (*m_requested_times)[m_current_time] <= t1) {
268  const unsigned int k = m_current_time;
269  m_current_time += 1;
270 
271  N += 1;
272 
273  // skip the first time: it defines the beginning of a reporting interval
274  if (k == 0) {
275  continue;
276  }
277 
278  const double t_s = (*m_requested_times)[k - 1];
279  const double t_e = (*m_requested_times)[k];
280 
281  double rate = 0.0;
282  if (N == 1) {
283  // it is the right end-point of the first reporting interval in this time step: count the
284  // contribution from the last time step plus the one from the beginning of this time step
285  const double
286  total_change = m_accumulator + change * (t_e - t0) / (t1 - t0);
287  const double dt = t_e - t_s;
288 
289  rate = total_change / dt;
290 
291  } else {
292  // this reporting interval is completely contained within the time step, so the rate of change
293  // does not depend on its length
294  rate = change / (t1 - t0);
295  }
296 
297  // store computed data in the buffer
298  {
299  m_time.push_back(t_e);
300  m_values.push_back(rate);
301  m_bounds.push_back(t_s);
302  m_bounds.push_back(t_e);
303  }
304 
305  m_accumulator = 0.0;
306  }
307 
308  if (N == 0) {
309  // if this time step contained no requested times we need to add the whole change to the
310  // accumulator
311  m_accumulator += change;
312  } else {
313  // if this time step contained some requested times we need to add the change since the last one
314  // to the accumulator
315  const double dt = t1 - (*m_requested_times)[m_current_time - 1];
316  if (dt > epsilon) {
317  m_accumulator += change * (dt / (t1 - t0));
318  }
319  }
320 }

368 void TSDiagnostic::flush() {
369 
370  if (m_time.empty()) {
371  return;
372  }
373 
374  std::string dimension_name = m_dimension.get_name();
375 
376  File file(m_grid->com, m_output_filename, PISM_NETCDF3, PISM_READWRITE); // OK to use netcdf3
377 
378  unsigned int len = file.dimension_length(dimension_name);
379 
380  if (len > 0) {
381  double last_time = vector_max(file.read_dimension(dimension_name));
382 
383  if (last_time < m_time.front()) {
384  m_start = len;
385  }
386  }
387 
388  auto time_name = m_config->get_string("time.dimension_name");
389 
390  if (len == m_start) {
391  if (not file.find_variable(m_dimension.get_name())) {
392  io::define_timeseries(m_dimension, time_name, file, PISM_DOUBLE);
393  }
394  io::write_timeseries(file, m_dimension, m_start, m_time);
395 
396  if (not file.find_variable(m_time_bounds.get_name())) {
397  io::define_time_bounds(m_time_bounds, time_name, "nv", file, PISM_DOUBLE);
398  }
399  io::write_time_bounds(file, m_time_bounds, m_start, m_bounds);
400  }
401 
402  if (not file.find_variable(m_variable.get_name())) {
403  io::define_timeseries(m_variable, time_name, file, PISM_DOUBLE);
404  }
405  io::write_timeseries(file, m_variable, m_start, m_values);
406 
407  m_start += m_time.size();
408 
409  {
410  m_time.clear();
411  m_bounds.clear();
412  m_values.clear();
413  }
414 }
