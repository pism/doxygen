171 /*!
172  * The default implementation computes start and count and calls put_vara_double()
173  */
174 void NCFile::write_darray_impl(const std::string &variable_name,
175  const IceGrid &grid,
176  unsigned int z_count,
177  unsigned int record,
178  const double *input) {
179  std::vector<std::string> dims;
180  this->inq_vardimid(variable_name, dims);
181 
182  unsigned int ndims = dims.size();
183 
184  bool time_dependent = ((z_count > 1 and ndims == 4) or
185  (z_count == 1 and ndims == 3));
186 
187  std::vector<unsigned int> start, count;
188 
189  // time
190  if (time_dependent) {
191  start.push_back(record);
192  count.push_back(1);
193  }
194 
195  // y
196  start.push_back(grid.ys());
197  count.push_back(grid.ym());
198 
199  // x
200  start.push_back(grid.xs());
201  count.push_back(grid.xm());
202 
203  // z (these are not used when writing 2D fields)
204  start.push_back(0);
205  count.push_back(z_count);
206 
207  this->put_vara_double(variable_name, start, count, input);
208 }
