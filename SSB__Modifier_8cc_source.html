92 //! \brief Distribute the input velocity throughout the column.
93 /*!
94  * Things to update:
95  * - 3D-distributed horizontal velocity
96  * - maximum horizontal velocity
97  * - diffusive ice flux
98  * - maximum diffusivity
99  * - strain heating (strain_heating)
100  */
101 void ConstantInColumn::update(const IceModelVec2V &sliding_velocity,
102  const Inputs &inputs,
103  bool full_update) {
104 
105  (void) inputs;
106 
107  if (not full_update) {
108  return;
109  }
110 
111  // horizontal velocity and its maximum:
112  IceModelVec::AccessList list{&m_u, &m_v, &sliding_velocity};
113 
114  for (Points p(*m_grid); p; p.next()) {
115  const int i = p.i(), j = p.j();
116 
117  m_u.set_column(i,j, sliding_velocity(i,j).u);
118  m_v.set_column(i,j, sliding_velocity(i,j).v);
119  }
120 
121  // Communicate to get ghosts (needed to compute w):
122  m_u.update_ghosts();
123  m_v.update_ghosts();
124 
125  // diffusive flux and maximum diffusivity
126  m_diffusive_flux.set(0.0);
127  m_D_max = 0.0;
128 }
