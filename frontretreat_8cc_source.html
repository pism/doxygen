44 void IceModel::front_retreat_step() {
45  // compute retreat rates due to eigencalving, von Mises calving, Hayhurst calving,
46  // and frontal melt.
47  // We do this first to make sure that all three mechanisms use the same ice geometry.
48  {
49  if (m_eigen_calving) {
50  m_eigen_calving->update(m_geometry.cell_type,
51  m_stress_balance->shallow()->velocity());
52  }
53 
54  if (m_hayhurst_calving) {
55  m_hayhurst_calving->update(m_geometry.cell_type,
56  m_geometry.ice_thickness,
57  m_geometry.sea_level_elevation,
58  m_geometry.bed_elevation);
59  }
60 
61  if (m_vonmises_calving) {
62  // FIXME: consider computing vertically-averaged hardness here and providing that
63  // instead of using ice thickness and enthalpy.
64  m_vonmises_calving->update(m_geometry.cell_type,
65  m_geometry.ice_thickness,
66  m_stress_balance->shallow()->velocity(),
67  m_energy_model->enthalpy());
68  }
69 
70  if (m_frontal_melt) {
71  IceModelVec2S &flux_magnitude = *m_work2d[0];
72 
73  compute_magnitude(m_subglacial_hydrology->flux(), flux_magnitude);
74 
75  FrontalMeltInputs inputs;
76 
77  inputs.geometry = &m_geometry;
78  inputs.subglacial_water_flux = &flux_magnitude;
79 
80  m_frontal_melt->update(inputs, m_time->current(), m_dt);
81  }
82  }
83 
84  IceModelVec2S
85  &old_H = *m_work2d[0],
86  &old_Href = *m_work2d[1];
87 
88  // frontal melt
89  if (m_frontal_melt) {
90  assert(m_front_retreat);
91 
92  old_H.copy_from(m_geometry.ice_thickness);
93  old_Href.copy_from(m_geometry.ice_area_specific_volume);
94 
95  // apply frontal melt rate
96  m_front_retreat->update_geometry(m_dt, m_geometry, m_ice_thickness_bc_mask,
97  m_frontal_melt->retreat_rate(),
98  m_geometry.ice_area_specific_volume,
99  m_geometry.ice_thickness);
100  bool add_values = false;
101  compute_geometry_change(m_geometry.ice_thickness,
102  m_geometry.ice_area_specific_volume,
103  old_H, old_Href,
104  add_values,
105  m_thickness_change.frontal_melt);
106  } else {
107  m_thickness_change.frontal_melt.set(0.0);
108  }
109 
110  // calving
111  if (m_eigen_calving or m_vonmises_calving or m_hayhurst_calving or
112  m_float_kill_calving or m_thickness_threshold_calving) {
113 
114  old_H.copy_from(m_geometry.ice_thickness);
115  old_Href.copy_from(m_geometry.ice_area_specific_volume);
116 
117  if (m_eigen_calving or m_vonmises_calving or m_hayhurst_calving) {
118  assert(m_front_retreat);
119 
120  IceModelVec2S &retreat_rate = *m_work2d[2];
121  retreat_rate.set(0.0);
122 
123  if (m_eigen_calving) {
124  retreat_rate.add(1.0, m_eigen_calving->calving_rate());
125  }
126 
127  if (m_hayhurst_calving) {
128  retreat_rate.add(1.0, m_hayhurst_calving->calving_rate());
129  }
130 
131  if (m_vonmises_calving) {
132  retreat_rate.add(1.0, m_vonmises_calving->calving_rate());
133  }
134 
135  if (m_calving_rate_factor) {
136  double T = m_time->current() + 0.5 * m_dt;
137  retreat_rate.scale(m_calving_rate_factor->value(T));
138  }
139 
140  m_front_retreat->update_geometry(m_dt, m_geometry, m_ice_thickness_bc_mask,
141  retreat_rate,
142  m_geometry.ice_area_specific_volume,
143  m_geometry.ice_thickness);
144 
145  auto thickness_threshold = m_config->get_number("stress_balance.ice_free_thickness_standard");
146 
147  m_geometry.ensure_consistency(thickness_threshold);
148 
149  if (m_eigen_calving or m_vonmises_calving or m_hayhurst_calving) {
150  remove_narrow_tongues(m_geometry, m_geometry.ice_thickness);
151 
152  m_geometry.ensure_consistency(thickness_threshold);
153  }
154  }
155 
156  if (m_float_kill_calving) {
157  m_float_kill_calving->update(m_geometry.cell_type, m_geometry.ice_thickness);
158  }
159 
160  if (m_thickness_threshold_calving) {
161  m_thickness_threshold_calving->update(m_time->current(), m_dt,
162  m_geometry.cell_type,
163  m_geometry.ice_thickness);
164  }
165 
166  bool add_values = false;
167  compute_geometry_change(m_geometry.ice_thickness,
168  m_geometry.ice_area_specific_volume,
169  old_H, old_Href,
170  add_values,
171  m_thickness_change.calving);
172  } else {
173  m_thickness_change.calving.set(0.0);
174  }
175 
176  // prescribed retreat
177 
178  if (m_prescribed_retreat) {
179  old_H.copy_from(m_geometry.ice_thickness);
180  old_Href.copy_from(m_geometry.ice_area_specific_volume);
181 
182  m_prescribed_retreat->update(m_time->current(), m_dt,
183  m_geometry.ice_thickness,
184  m_geometry.ice_area_specific_volume);
185 
186  bool add_values = false;
187  compute_geometry_change(m_geometry.ice_thickness,
188  m_geometry.ice_area_specific_volume,
189  old_H, old_Href,
190  add_values,
191  m_thickness_change.forced_retreat);
192 
193  } else {
194  m_thickness_change.forced_retreat.set(0.0);
195  }
196 
197 
198  // Changes above may create icebergs; here we remove them and account for additional
199  // mass losses.
200  {
201  old_H.copy_from(m_geometry.ice_thickness);
202  old_Href.copy_from(m_geometry.ice_area_specific_volume);
203 
204  enforce_consistency_of_geometry(REMOVE_ICEBERGS);
205 
206  bool add_values = true;
207  compute_geometry_change(m_geometry.ice_thickness,
208  m_geometry.ice_area_specific_volume,
209  old_H, old_Href,
210  add_values,
211  m_thickness_change.calving);
212  }
213 }

215 /**
216  * Compute the change in ice geometry from "old" to "current".
217  *
218  * Units: ice equivalent meters.
219  *
220  * @param thickness current ice thickness
221  * @param Href current "reference ice thickness"
222  * @param thickness_old old ice thickness
223  * @param Href_old old "reference ice thickness"
224  * @param[in] add_values if `true`, add computed values to `output`, otherwise
225  * overwrite them
226  * @param[in,out] output computed change
227  */
228 void IceModel::compute_geometry_change(const IceModelVec2S &thickness,
229  const IceModelVec2S &Href,
230  const IceModelVec2S &thickness_old,
231  const IceModelVec2S &Href_old,
232  bool add_values,
233  IceModelVec2S &output) {
234 
235  IceModelVec::AccessList list{&thickness, &thickness_old,
236  &Href, &Href_old, &output};
237 
238  for (Points p(*m_grid); p; p.next()) {
239  const int i = p.i(), j = p.j();
240 
241  const double
242  H_old = thickness_old(i, j) + Href_old(i, j),
243  H_new = thickness(i, j) + Href(i, j),
244  change = H_new - H_old;
245 
246  if (add_values) {
247  output(i, j) += change;
248  } else {
249  output(i, j) = change;
250  }
251  }
252 }
