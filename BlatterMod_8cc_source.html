74 /*!
75  * Copy ice velocity from the sigma vertical grid onto PISM's vertical grid.
76  *
77  * Uses constant extrapolation above the ice surface.
78  */
79 void BlatterMod::transfer(const IceModelVec2S &ice_thickness) {
80 
81  auto u_sigma = m_solver->velocity_u_sigma();
82  auto v_sigma = m_solver->velocity_v_sigma();
83 
84  const auto &zlevels = m_u.levels();
85  int Mz = zlevels.size();
86 
87  IceModelVec::AccessList list{&m_u, &m_v, u_sigma.get(), v_sigma.get(), &ice_thickness};
88 
89  for (Points p(*m_grid); p; p.next()) {
90  const int i = p.i(), j = p.j();
91 
92  auto u = m_u.get_column(i, j);
93  auto v = m_v.get_column(i, j);
94 
95  double H = ice_thickness(i, j);
96 
97  if (H > 0.0) {
98  for (int k = 0; k < Mz; ++k) {
99  double sigma = std::min(zlevels[k] / H, 1.0);
100 
101  u[k] = u_sigma->interpolate(i, j, sigma);
102  v[k] = v_sigma->interpolate(i, j, sigma);
103  }
104  } else {
105  m_u.set_column(i, j, 0.0);
106  m_v.set_column(i, j, 0.0);
107  }
108  }
109 
110  m_u.update_ghosts();
111  m_v.update_ghosts();
112 }

114 /*!
115  * Estimate max SIA-type diffusivity assuming that `Q = -D \nabla h`.
116  */
117 void BlatterMod::compute_max_diffusivity(const IceModelVec2V &velocity,
118  const IceModelVec2S &ice_thickness,
119  const IceModelVec2S &surface) {
120  double eps = 1e-3;
121  double
122  dx = m_grid->dx(),
123  dy = m_grid->dy();
124 
125  IceModelVec::AccessList list{&velocity, &ice_thickness, &surface};
126 
127  m_D_max = 0.0;
128  for (Points p(*m_grid); p; p.next()) {
129  const int i = p.i(), j = p.j();
130 
131  auto h = surface.star(i, j);
132  auto H = ice_thickness(i, j);
133 
134  if (H > 0.0) {
135  Vector2 grad_h = {(h.e - h.w) / (2.0 * dx),
136  (h.n - h.s) / (2.0 * dy)};
137 
138  double D = H * velocity(i, j).magnitude() / (grad_h.magnitude() + eps);
139 
140  m_D_max = std::max(D, m_D_max);
141  }
142  }
143 
144  m_D_max = GlobalMax(m_grid->com, m_D_max);
145 }
