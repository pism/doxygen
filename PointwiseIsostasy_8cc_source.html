69 //! Updates the pointwise isostasy model.
70 void PointwiseIsostasy::update_impl(const IceModelVec2S &ice_thickness,
71  const IceModelVec2S &sea_level_elevation,
72  double t, double dt) {
73  (void) t;
74 
75  const double
76  mantle_density = m_config->get_number("bed_deformation.mantle_density"),
77  load_density = m_config->get_number("constants.ice.density"),
78  ocean_density = m_config->get_number("constants.sea_water.density"),
79  ice_density = m_config->get_number("constants.ice.density"),
80  f = load_density / mantle_density;
81 
82  //! Our goal: topg = topg_last - f*(load - load_last)
83 
84  IceModelVec::AccessList list{&m_topg, &m_topg_last,
85  &ice_thickness, &sea_level_elevation, &m_load_last};
86 
87  ParallelSection loop(m_grid->com);
88  try {
89  for (Points p(*m_grid); p; p.next()) {
90  const int i = p.i(), j = p.j();
91 
92  double load = compute_load(m_topg(i, j),
93  ice_thickness(i, j),
94  sea_level_elevation(i, j),
95  ice_density, ocean_density);
96 
97  m_topg(i, j) = m_topg_last(i, j) - f * (load - m_load_last(i, j));
98  m_load_last(i, j) = load;
99  }
100  } catch (...) {
101  loop.failed();
102  }
103  loop.check();
104 
105  //! Finally, we need to update bed uplift, topg_last and load_last.
106  compute_uplift(m_topg, m_topg_last, dt, m_uplift);
107 
108  m_topg_last.copy_from(m_topg);
109 
110  //! Increment the topg state counter. SIAFD relies on this!
111  m_topg.inc_state_counter();
112 }
