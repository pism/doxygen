42 //! Compute the maximum time step allowed by the diffusive SIA.
43 /*!
44 If maximum diffusivity is positive (i.e. if there is diffusion going on) then
45 updates dt.
46 
47 Note adapt_ratio * 2 is multiplied by dx^2/(2*maxD) so dt <= adapt_ratio *
48 dx^2/maxD (if dx=dy).
49 
50 Reference: [\ref MortonMayers] pp 62--63.
51  */
52 MaxTimestep IceModel::max_timestep_diffusivity() {
53  double D_max = m_stress_balance->max_diffusivity();
54 
55  if (D_max > 0.0) {
56  const double
57  dx = m_grid->dx(),
58  dy = m_grid->dy(),
59  adaptive_timestepping_ratio = m_config->get_number("time_stepping.adaptive_ratio"),
60  grid_factor = 1.0 / (dx*dx) + 1.0 / (dy*dy);
61 
62  return MaxTimestep(adaptive_timestepping_ratio * 2.0 / (D_max * grid_factor),
63  "diffusivity");
64  } else {
65  return MaxTimestep(m_config->get_number("time_stepping.maximum_time_step", "seconds"),
66  "max time step");
67  }
68 }

70 /** @brief Compute the skip counter using "long" (usually determined
71  * using the CFL stability criterion) and "short" (typically
72  * determined using the diffusivity-based stability criterion) time
73  * step lengths.
74  *
75  *
76  * @param[in] input_dt long time-step
77  * @param[in] input_dt_diffusivity short time-step
78  *
79  * @return new skip counter
80  */
81 unsigned int IceModel::skip_counter(double input_dt, double input_dt_diffusivity) {
82 
83  if (not m_config->get_flag("time_stepping.skip.enabled")) {
84  return 0;
85  }
86 
87  const unsigned int skip_max = static_cast<int>(m_config->get_number("time_stepping.skip.max"));
88 
89  if (input_dt_diffusivity > 0.0) {
90  const double conservativeFactor = 0.95;
91  const double counter = floor(conservativeFactor * (input_dt / input_dt_diffusivity));
92  const unsigned int result = static_cast<unsigned int>(counter);
93  return std::min(result, skip_max);
94  } else {
95  return skip_max;
96  }
97 
98  return 0;
99 }

101 //! Use various stability criteria to determine the time step for an evolution run.
102 /*!
103 The main loop in run() approximates many physical processes. Several of these approximations,
104 including the mass continuity and temperature equations in particular, involve stability
105 criteria. This procedure builds the length of the next time step by using these criteria and
106 by incorporating choices made by options (e.g. `-max_dt`) and by derived classes.
107 
108 @param[in] counter current time-step skipping counter
109  */
110 IceModel::TimesteppingInfo IceModel::max_timestep(unsigned int counter) {
111 
112  const double current_time = m_time->current();
113 
114  std::vector<MaxTimestep> restrictions;
115 
116  // get time-stepping restrictions from sub-models
117  for (auto m : m_submodels) {
118  restrictions.push_back(m.second->max_timestep(current_time));
119  }
120 
121  // mechanisms that use a retreat rate
122  bool front_retreat = (m_eigen_calving or m_vonmises_calving or
123  m_hayhurst_calving or m_frontal_melt);
124  if (front_retreat and m_config->get_flag("geometry.front_retreat.use_cfl")) {
125  // at least one of front retreat mechanisms is active *and* PISM is told to use a CFL
126  // restriction
127 
128  IceModelVec2S &retreat_rate = *m_work2d[0];
129  retreat_rate.set(0.0);
130 
131  if (m_eigen_calving) {
132  retreat_rate.add(1.0, m_eigen_calving->calving_rate());
133  }
134 
135  if (m_hayhurst_calving) {
136  retreat_rate.add(1.0, m_hayhurst_calving->calving_rate());
137  }
138 
139  if (m_vonmises_calving) {
140  retreat_rate.add(1.0, m_vonmises_calving->calving_rate());
141  }
142 
143  if (m_frontal_melt) {
144  retreat_rate.add(1.0, m_frontal_melt->retreat_rate());
145  }
146 
147  assert(m_front_retreat);
148 
149  restrictions.push_back(m_front_retreat->max_timestep(m_geometry.cell_type,
150  m_ssa_dirichlet_bc_mask,
151  retreat_rate));
152  }
153 
154  const char* end = "end of the run";
155  const char* max = "max";
156 
157  // Always consider the maximum allowed time-step length.
158  double max_timestep = m_config->get_number("time_stepping.maximum_time_step", "seconds");
159  if (max_timestep > 0.0) {
160  restrictions.push_back(MaxTimestep(max_timestep, max));
161  }
162 
163  // Never go past the end of a run.
164  const double time_to_end = m_time->end() - current_time;
165  if (time_to_end > 0.0) {
166  restrictions.push_back(MaxTimestep(time_to_end, end));
167  }
168 
169  // reporting
170  {
171  restrictions.push_back(ts_max_timestep(current_time));
172  restrictions.push_back(extras_max_timestep(current_time));
173  restrictions.push_back(save_max_timestep(current_time));
174  }
175 
176  // mass continuity stability criteria
177  if (m_config->get_flag("geometry.update.enabled")) {
178  CFLData cfl = m_stress_balance->max_timestep_cfl_2d();
179 
180  restrictions.push_back(MaxTimestep(cfl.dt_max.value(), "2D CFL"));
181  restrictions.push_back(max_timestep_diffusivity());
182  }
183 
184  // sort time step restrictions to find the strictest one
185  std::sort(restrictions.begin(), restrictions.end());
186 
187  // note that restrictions has at least 2 elements
188  // the first element is the max time step we can take
189  MaxTimestep dt_max = restrictions[0];
190  MaxTimestep dt_other = restrictions[1];
191 
192  TimesteppingInfo result;
193  result.dt = dt_max.value();
194  result.reason = (dt_max.description() + " (overrides " + dt_other.description() + ")");
195  result.skip_counter = 0;
196 
197  // Hit multiples of X years, if requested.
198  {
199  int timestep_hit_multiples = static_cast<int>(m_config->get_number("time_stepping.hit_multiples"));
200  if (timestep_hit_multiples > 0) {
201  double
202  epsilon = 1.0, // 1 second tolerance
203  next_time = m_timestep_hit_multiples_last_time;
204 
205  while (m_time->increment_date(next_time, timestep_hit_multiples) <= current_time + result.dt + epsilon) {
206  next_time = m_time->increment_date(next_time, timestep_hit_multiples);
207  }
208 
209  if (next_time > current_time and next_time <= current_time + result.dt + epsilon) {
210  result.dt = next_time - current_time;
211  result.reason = pism::printf("hit multiples of %d years (overrides %s)",
212  timestep_hit_multiples, dt_max.description().c_str());
213 
214  m_timestep_hit_multiples_last_time = next_time;
215  }
216  }
217  }
218 
219  // the "skipping" mechanism
220  {
221  if (dt_max.description() == "diffusivity" and counter == 0) {
222  result.skip_counter = skip_counter(dt_other.value(), dt_max.value());
223  }
224 
225  // "max" and "end of the run" limit the "big" time-step (in
226  // the context of the "skipping" mechanism), so we might need to
227  // reset the skip_counter_result to 1.
228  if (member(dt_max.description(), {max, end}) and counter > 1) {
229  result.skip_counter = 1;
230  }
231  }
232 
233  double resolution = m_config->get_number("time_stepping.resolution");
234 
235  if (resolution > 0.0) {
236  double dt = std::floor(result.dt * resolution) / resolution;
237 
238  // Ensure that the resulting time step is never zero. This may happen if the length of
239  // the run is not an integer multiple of "resolution".
240  if (dt >= resolution) {
241  result.dt = dt;
242  }
243  }
244 
245  return result;
246 }
