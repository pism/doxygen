35 /*!
36  * Model of the energy content in the cryo-hydrologic system.
37  *
38  * This model can be used to include the influence of cryo-hydrologic warming on the
39  * internal energy of the ice.
40  *
41  * This model is based on
42  *
43  * @article{PhillipsRajaramSteffan2010,
44  * author = {Phillips, T. and Rajaram, H. and Steffen, K.},
45  * doi = {10.1029/2010GL044397},
46  * journal = {Geophy. Res. Lett.},
47  * number = {20},
48  * pages = {1--5},
49  * title = {{Cryo-hydrologic warming: A potential mechanism for rapid thermal response of ice sheets}},
50  * volume = {37},
51  * year = {2010}
52  * }
53  *
54  * We assume that during the melt season the enthalpy in the cryo-hydrologic system is
55  * equal to the enthalpy of the ice with a fixed water fraction corresponding to the
56  * amount of water remaining in the system once all of the moving water drains out (0.5%
57  * by default). During the winter the CH system is allowed to cool.
58 */

119 void CHSystem::update_impl(double t, double dt, const Inputs &inputs) {
120  // current time does not matter here
121  (void) t;
122 
123  EnthalpyConverter::Ptr EC = m_grid->ctx()->enthalpy_converter();
124 
125  inputs.check();
126 
127  // give them names that are a bit shorter...
128  const IceModelVec3
129  &volumetric_heat = *inputs.volumetric_heating_rate,
130  &u3 = *inputs.u3,
131  &v3 = *inputs.v3,
132  &w3 = *inputs.w3;
133 
134  const IceModelVec2CellType &cell_type = *inputs.cell_type;
135 
136  const IceModelVec2S
137  &basal_frictional_heating = *inputs.basal_frictional_heating,
138  &basal_heat_flux = *inputs.basal_heat_flux,
139  &ice_thickness = *inputs.ice_thickness,
140  &surface_liquid_fraction = *inputs.surface_liquid_fraction,
141  &shelf_base_temp = *inputs.shelf_base_temp,
142  &ice_surface_temp = *inputs.surface_temp;
143 
144  energy::enthSystemCtx system(m_grid->z(), "energy.ch_warming", m_grid->dx(), m_grid->dy(), dt,
145  *m_config, m_ice_enthalpy, u3, v3, w3, volumetric_heat, EC);
146 
147  const size_t Mz_fine = system.z().size();
148  const double dz = system.dz();
149  std::vector<double> Enthnew(Mz_fine); // new enthalpy in column
150 
151  IceModelVec::AccessList list{&ice_surface_temp, &shelf_base_temp, &surface_liquid_fraction,
152  &ice_thickness, &basal_frictional_heating, &basal_heat_flux,
153  &cell_type, &u3, &v3, &w3, &volumetric_heat, &m_ice_enthalpy,
154  &m_work};
155 
156  double
157  margin_threshold = m_config->get_number("energy.margin_ice_thickness_limit"),
158  T_pm = m_config->get_number("constants.fresh_water.melting_point_temperature"),
159  residual_water_fraction = m_config->get_number("energy.ch_warming.residual_water_fraction");
160 
161  const std::vector<double> &z = m_grid->z();
162  const unsigned int Mz = m_grid->Mz();
163 
164  ParallelSection loop(m_grid->com);
165  try {
166  for (Points pt(*m_grid); pt; pt.next()) {
167  const int i = pt.i(), j = pt.j();
168 
169  const double H = ice_thickness(i, j);
170 
171  if (ice_surface_temp(i, j) >= T_pm) {
172  // We use surface temperature to determine if we're in a melt season or not. It
173  // probably makes sense to use the surface mass balance instead.
174 
175  double *column = m_work.get_column(i, j);
176  for (unsigned int k = 0; k < Mz; ++k) {
177  double
178  depth = std::max(H - z[k], 0.0),
179  P = EC->pressure(depth);
180  column[k] = EC->enthalpy(EC->melting_temperature(P),
181  residual_water_fraction,
182  P);
183  }
184  continue;
185  }
186 
187  // enthalpy and pressures at top of ice
188  const double
189  depth_ks = H - system.ks() * dz,
190  p_ks = EC->pressure(depth_ks); // FIXME issue #15
191 
192  const double Enth_ks = EC->enthalpy_permissive(ice_surface_temp(i, j),
193  surface_liquid_fraction(i, j), p_ks);
194 
195  system.init(i, j,
196  marginal(ice_thickness, i, j, margin_threshold),
197  H);
198 
199  const bool ice_free_column = (system.ks() == 0);
200 
201  // deal completely with columns with no ice
202  if (ice_free_column) {
203  m_work.set_column(i, j, Enth_ks);
204  continue;
205  } // end of if (ice_free_column)
206 
207  if (system.lambda() < 1.0) {
208  m_stats.reduced_accuracy_counter += 1; // count columns with lambda < 1
209  }
210 
211  // set boundary conditions and update enthalpy
212  {
213  system.set_surface_dirichlet_bc(Enth_ks);
214 
215  if (cell_type.ocean(i, j)) {
216  // floating base: Dirichlet application of known temperature from ocean coupler;
217  // assumes base of ice shelf has zero liquid fraction
218  double Enth0 = EC->enthalpy_permissive(shelf_base_temp(i, j), 0.0, EC->pressure(H));
219 
220  system.set_basal_dirichlet_bc(Enth0);
221  } else {
222  // grounded
223  system.set_basal_heat_flux(basal_heat_flux(i, j) + basal_frictional_heating(i, j));
224  }
225  // solve the system
226  system.solve(Enthnew);
227  }
228 
229  system.fine_to_coarse(Enthnew, i, j, m_work);
230  }
231  } catch (...) {
232  loop.failed();
233  }
234  loop.check();
235 }

245 /*!
246  * Compute the heat flux corresponding to the cryo-hydrologic warming.
247  *
248  * `Q = (k / R**2) * (T_ch - T_ice),`
249  *
250  * where `k` is the thermal conductivity of ice and `R` us the average spacing of
251  * channels in the cryo-hydrologic system.
252  */
253 void cryo_hydrologic_warming_flux(double k,
254  double R,
255  const IceModelVec2S &ice_thickness,
256  const IceModelVec3 &ice_enthalpy,
257  const IceModelVec3 &ch_enthalpy,
258  IceModelVec3 &result) {
259 
260  auto grid = result.grid();
261 
262  const auto &z = grid->z();
263  auto Mz = grid->Mz();
264 
265  auto EC = grid->ctx()->enthalpy_converter();
266 
267  IceModelVec::AccessList access{&ice_thickness, &ice_enthalpy, &ch_enthalpy, &result};
268 
269  double C = k / (R * R);
270 
271  ParallelSection loop(grid->com);
272  try {
273  for (Points p(*grid); p; p.next()) {
274  const int i = p.i(), j = p.j();
275 
276  const double
277  *E_ch = ch_enthalpy.get_column(i, j),
278  *E_ice = ice_enthalpy.get_column(i, j);
279  double *Q = result.get_column(i, j);
280 
281  for (unsigned int m = 0; m < Mz; ++m) {
282  double
283  depth = ice_thickness(i, j) - z[m];
284 
285  if (depth > 0.0) {
286  double P = EC->pressure(depth);
287  Q[m] = std::max(C * (EC->temperature(E_ch[m], P) - EC->temperature(E_ice[m], P)),
288  0.0);
289  } else {
290  Q[m] = 0.0;
291  }
292  }
293  }
294  } catch (...) {
295  loop.failed();
296  }
297  loop.check();
298 }
