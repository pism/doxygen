45 MaxTimestep reporting_max_timestep(const std::vector<double> &times, double t,
46  double resolution,
47  const std::string &description) {
48 
49  const size_t N = times.size();
50  if (t >= times.back()) {
51  return MaxTimestep();
52  }
53 
54  size_t j = 0;
55  double dt = 0.0;
56  if (t < times[0]) {
57  j = -1;
58  } else {
59  j = gsl_interp_bsearch(&times[0], t, 0, N - 1);
60  }
61 
62  dt = times[j + 1] - t;
63 
64  // now make sure that we don't end up taking a time-step of less than "resolution"
65  // second long
66  if (dt < resolution) {
67  if (j + 2 < N) {
68  return MaxTimestep(times[j + 2] - t, description);
69  } else {
70  return MaxTimestep(description);
71  }
72  } else {
73  return MaxTimestep(dt, description);
74  }
75 }

104 void IceModel::save_results() {
105  {
106  update_run_stats();
107 
108  auto str = pism::printf(
109  "PISM done. Performance stats: %.4f wall clock hours, %.4f proc.-hours, %.4f model years per proc.-hour.",
110  (double)m_run_stats["wall_clock_hours"],
111  (double)m_run_stats["processor_hours"],
112  (double)m_run_stats["model_years_per_processor_hour"]);
113 
114  prepend_history(str);
115  }
116 
117  std::string filename = m_config->get_string("output.file_name");
118 
119  if (filename.empty()) {
120  m_log->message(2, "WARNING: output.file_name is empty. Using unnamed.nc instead.\n");
121  filename = "unnamed.nc";
122  }
123 
124  if (not ends_with(filename, ".nc")) {
125  m_log->message(2,
126  "PISM WARNING: output file name does not have the '.nc' suffix!\n");
127  }
128 
129  const Profiling &profiling = m_ctx->profiling();
130 
131  profiling.begin("io.model_state");
132  if (m_config->get_string("output.size") != "none") {
133  m_log->message(2, "Writing model state to file `%s'...\n", filename.c_str());
134  File file(m_grid->com,
135  filename,
136  string_to_backend(m_config->get_string("output.format")),
137  PISM_READWRITE_MOVE,
138  m_ctx->pio_iosys_id());
139 
140  write_metadata(file, WRITE_MAPPING, PREPEND_HISTORY);
141 
142  write_run_stats(file);
143 
144  save_variables(file, INCLUDE_MODEL_STATE, m_output_vars,
145  m_time->current());
146  }
147  profiling.end("io.model_state");
148 }

150 void IceModel::write_mapping(const File &file) {
151  // only write mapping if it is set.
152  const VariableMetadata &mapping = m_grid->get_mapping_info().mapping;
153  std::string name = mapping.get_name();
154  if (mapping.has_attributes()) {
155  if (not file.find_variable(name)) {
156  file.define_variable(name, PISM_DOUBLE, {});
157  }
158  io::write_attributes(file, mapping, PISM_DOUBLE);
159 
160  // Write the PROJ string to mapping:proj_params (for CDO).
161  std::string proj = m_grid->get_mapping_info().proj;
162  if (not proj.empty()) {
163  file.write_attribute(name, "proj_params", proj);
164  }
165  }

176 void IceModel::save_variables(const File &file,
177  OutputKind kind,
178  const std::set<std::string> &variables,
179  double time,
180  IO_Type default_diagnostics_type) {
181 
182  // Compress 2D and 3D variables if output.compression_level > 0 and the output.format
183  // supports it.
184  file.set_compression_level(m_config->get_number("output.compression_level"));
185 
186  // define the time dimension if necessary (no-op if it is already defined)
187  io::define_time(file, *m_grid->ctx());
188  // define the "timestamp" (wall clock time since the beginning of the run)
189  // Note: it is time-dependent, so we need to define time first.
190  io::define_timeseries(m_timestamp,
191  m_config->get_string("time.dimension_name"),
192  file, PISM_FLOAT);
193  // append to the time dimension
194  io::append_time(file, *m_config, time);
195 
196  // Write metadata *before* everything else:
197  //
198  // FIXME: we should write this to variables instead of attributes because NetCDF-4 crashes after
199  // about 2^16 attribute modifications per variable. :-(
200  write_run_stats(file);
201 
202  if (kind == INCLUDE_MODEL_STATE) {
203  define_model_state(file);
204  }
205  define_diagnostics(file, variables, default_diagnostics_type);
206 
207  // Done defining variables
208 
209  {
210  // Note: we don't use "variables" (an argument of this method) here because it
211  // contains PISM's names of diagnostic quantities which (in some cases) map to more
212  // than one NetCDF variable. Moreover, here we're concerned with file contents, not
213  // the list of requested variables.
214  std::set<std::string> var_names;
215  unsigned int n_vars = file.nvariables();
216  for (unsigned int k = 0; k < n_vars; ++k) {
217  var_names.insert(file.variable_name(k));
218  }
219 
220  // If this output file contains variables lat and lon...
221  if (member("lat", var_names) and member("lon", var_names)) {
222 
223  // add the coordinates attribute to all variables that use x and y dimensions
224  for (auto v : var_names) {
225  std::set<std::string> dims;
226  for (auto d : file.dimensions(v)) {
227  dims.insert(d);
228  }
229 
230  if (not member(v, {"lat", "lon", "lat_bnds", "lon_bnds"}) and
231  member("x", dims) and member("y", dims)) {
232  file.write_attribute(v, "coordinates", "lat lon");
233  }
234  }
235 
236  // and if it also contains lat_bnds and lon_bnds, add the bounds attribute to lat
237  // and lon.
238  if (member("lat_bnds", var_names) and member("lon_bnds", var_names)) {
239  file.write_attribute("lat", "bounds", "lat_bnds");
240  file.write_attribute("lon", "bounds", "lon_bnds");
241  }
242  }
243  }
244 
245  if (kind == INCLUDE_MODEL_STATE) {
246  write_model_state(file);
247  }
248  write_diagnostics(file, variables);
249 
250  // find out how much time passed since the beginning of the run and save it to the output file
251  {
252  unsigned int time_length = file.dimension_length(m_config->get_string("time.dimension_name"));
253  size_t start = time_length > 0 ? static_cast<size_t>(time_length - 1) : 0;
254  io::write_timeseries(file, m_timestamp, start,
255  {wall_clock_hours(m_grid->com, m_start_time)});
256  }
257 }
