58 /*!
59  * Solve the Poisson equation on the domain defined by `mask == 1` with Dirichlet BC
60  * provided in `bc` (used only where `mask == 0`, possibly redundant away from the domain)
61  * with the constant right hand side `rhs`.
62  *
63  * Set the mask to 2 to use zero Neumann BC.
64  */
65 int Poisson::solve(const IceModelVec2Int& mask, const IceModelVec2S& bc, double rhs,
66  bool reuse_matrix) {
67 
68  PetscErrorCode ierr;
69 
70  // make a ghosted copy of the mask
71  m_mask.copy_from(mask);
72 
73  if (reuse_matrix) {
74  // Use non-zero initial guess. I assume that re-using the matrix means that the BC and
75  // RHS provided are close to the ones used in the previous call and the solution
76  // stored in m_x is a good initial guess for the current problem.
77  ierr = KSPSetInitialGuessNonzero(m_KSP, PETSC_TRUE);
78  PISM_CHK(ierr, "KSPSetInitialGuessNonzero");
79  } else {
80  ierr = KSPSetInitialGuessNonzero(m_KSP, PETSC_FALSE);
81  PISM_CHK(ierr, "KSPSetInitialGuessNonzero");
82 
83  assemble_matrix(m_mask, m_A);
84  }
85 
86  assemble_rhs(rhs, m_mask, bc, m_b);
87 
88  // Call PETSc to solve linear system by iterative method.
89  ierr = KSPSetOperators(m_KSP, m_A, m_A);
90  PISM_CHK(ierr, "KSPSetOperator");
91 
92  ierr = KSPSolve(m_KSP, m_b.vec(), m_x.vec());
93  PISM_CHK(ierr, "KSPSolve");
94 
95  // Check if diverged
96  KSPConvergedReason reason;
97  ierr = KSPGetConvergedReason(m_KSP, &reason);
98  PISM_CHK(ierr, "KSPGetConvergedReason");
99 
100  if (reason < 0) {
101  // KSP diverged
102  m_log->message(1,
103  "PISM ERROR: KSP iteration failed while solving the Poisson equation\n"
104  " reason = %d = '%s'\n",
105  reason, KSPConvergedReasons[reason]);
106 
107  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "KSP iteration failed: %s",
108  KSPConvergedReasons[reason]);
109  }
110 
111  // report on KSP success
112  PetscInt ksp_iterations = 0;
113  ierr = KSPGetIterationNumber(m_KSP, &ksp_iterations);
114  PISM_CHK(ierr, "KSPGetIterationNumber");
115 
116  return ksp_iterations;
117 }

123 // Maxima code deriving the discretization
124 //
125 // /* Shift in x and y directions. */
126 // shift(expr, dx, dy) := op(expr)[args(expr)[1] + dx, args(expr)[2] + dy]$
127 //
128 // d_px(var) := (shift(var, 1, 0) - var)$
129 // d_mx(var) := (var - shift(var, -1, 0))$
130 //
131 // d_py(var) := (shift(var, 0, 1) - var)$
132 // d_my(var) := (var - shift(var, 0, -1))$
133 //
134 // constants : [C_x = 1 / (dx^2), C_y = 1 / (dy^2)]$
135 //
136 // /* discretization of -\nabla \dot (D \nabla W) */
137 // L: (E * d_px(W[i, j]) - W * d_mx(W[i, j])) / dx^2 +
138 // (N * d_py(W[i, j]) - S * d_my(W[i, j])) / dy^2$
139 //
140 // /* discretization of the Poisson equation */
141 // eq: - L = f;
142 //
143 // /* cleaned up equation */
144 // s : map(lambda([x,y], solve(x, y)[1]), constants, [dx^2, dy^2]);
145 // eq2 : at(eq, s);
146 //
147 // /* compute matrix coefficients */
148 // for m: -1 thru 1 do (for n: -1 thru 1 do
149 // (c[2 - n, m + 2] : factor(ratcoef(lhs(eq2), W[i + m, j + n]))))$
150 //
151 // /* print results */
152 // A : genmatrix(c, 3, 3);
153 //
154 // b : rhs(eq2);
155 //
156 // print(''out)$
157 void Poisson::assemble_matrix(const IceModelVec2Int &mask, Mat A) {
158  PetscErrorCode ierr = 0;
159 
160  const double
161  dx = m_grid->dx(),
162  dy = m_grid->dy(),
163  C_x = 1.0 / (dx * dx),
164  C_y = 1.0 / (dy * dy);
165 
166  const int
167  nrow = 1,
168  ncol = 5,
169  Mx = m_grid->Mx(),
170  My = m_grid->My();
171 
172  ierr = MatZeroEntries(A); PISM_CHK(ierr, "MatZeroEntries");
173 
174  IceModelVec::AccessList list{&mask};
175 
176  /* matrix assembly loop */
177  ParallelSection loop(m_grid->com);
178  try {
179  MatStencil row, col[ncol];
180  row.c = 0;
181 
182  for (int m = 0; m < ncol; m++) {
183  col[m].c = 0;
184  }
185 
186  for (Points p(*m_grid); p; p.next()) {
187  const int i = p.i(), j = p.j();
188 
189  /* i indices */
190  const int I[] = {i, i - 1, i, i + 1, i};
191 
192  /* j indices */
193  const int J[] = {j + 1, j, j, j, j - 1};
194 
195  row.i = i;
196  row.j = j;
197 
198  for (int m = 0; m < ncol; m++) {
199  col[m].i = I[m];
200  col[m].j = J[m];
201  }
202 
203  auto M = mask.star(i, j);
204 
205  if (M.ij == 1) {
206  // Regular location: use coefficients of the discretization of the Laplacian
207 
208  // Use zero Neumann BC if a neighbor is marked as a Neumann boundary
209  double
210  N = M.n == 2 ? 0.0 : 1.0,
211  E = M.e == 2 ? 0.0 : 1.0,
212  W = M.w == 2 ? 0.0 : 1.0,
213  S = M.s == 2 ? 0.0 : 1.0;
214 
215  // Use zero Neumann BC at edges of the computational domain
216  {
217  N = j == My - 1 ? 0.0 : N;
218  E = i == Mx - 1 ? 0.0 : E;
219  W = i == 0 ? 0.0 : W;
220  S = j == 0 ? 0.0 : S;
221  }
222 
223  // discretization of the Laplacian
224  double L[ncol] = {- N * C_y,
225  - W * C_x, (W + E) * C_x + (N + S) * C_y, - E * C_x,
226  - S * C_y};
227 
228  ierr = MatSetValuesStencil(A, nrow, &row, ncol, col, L, INSERT_VALUES);
229  PISM_CHK(ierr, "MatSetValuesStencil");
230  } else {
231  // Boundary or outside the domain: assemble trivial equations (1 on the diagonal,
232  // 0 elsewhere)
233  double D[ncol] = {0.0,
234  0.0, 1.0, 0.0,
235  0.0};
236 
237  ierr = MatSetValuesStencil(A, nrow, &row, ncol, col, D, INSERT_VALUES);
238  PISM_CHK(ierr, "MatSetValuesStencil");
239  }
240 
241  } // i,j-loop
242  } catch (...) {
243  loop.failed();
244  }
245  loop.check();
246 
247  ierr = MatAssemblyBegin(A, MAT_FINAL_ASSEMBLY); PISM_CHK(ierr, "MatAssemblyBegin");
248  ierr = MatAssemblyEnd(A, MAT_FINAL_ASSEMBLY); PISM_CHK(ierr, "MatAssemblyif");
249 
250 #if (Pism_DEBUG==1)
251  ierr = MatSetOption(A, MAT_NEW_NONZERO_LOCATION_ERR, PETSC_TRUE);
252  PISM_CHK(ierr, "MatSetOption");
253 #endif
254 
255 }

259  const IceModelVec2S &bc,

264  const int i = p.i(), j = p.j();
265 
266  if (mask.as_int(i, j) == 1) {
267  // inside the domain
