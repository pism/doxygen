37 //! @brief Class representing diagnostic computations in PISM.
38 /*!
39  * The main goal of this abstraction is to allow accessing metadata
40  * corresponding to a diagnostic quantity *before* it is computed.
41  *
42  * Another goal is to create an interface for computing diagnostics *without*
43  * knowing which PISM module is responsible for the computation.
44  *
45  * Technical note: to compute some diagnostic quantities we need access to
46  * protected members of classes. C++ forbids obtaining pointers to non-static
47  * methods of a class, but it is possible to define a (friend) function
48  *
49  * @code
50  * IceModelVec::Ptr compute_bar(Foo* model, ...);
51  * @endcode
52  *
53  * which is the same as creating a method `Foo::compute_bar()`, but you *can*
54  * get a pointer to it.
55  *
56  * Diagnostic creates a common interface for all these compute_bar
57  * functions.
58  */

61  Diagnostic(IceGrid::ConstPtr g);
62  virtual ~Diagnostic();

119 /*!
120  * Helper template wrapping quantities with dedicated storage in diagnostic classes.
121  *
122  * Note: Make sure that that created diagnostics don't outlast fields that they wrap (or you'll have
123  * dangling pointers).
124  */
125 template<class T>
126 class DiagWithDedicatedStorage : public Diagnostic {
127 public:
128  DiagWithDedicatedStorage(const T &input)
129  : Diagnostic(input.grid()),
130  m_input(input)
131  {
132  for (unsigned int j = 0; j < input.ndof(); ++j) {
133  m_vars.emplace_back(input.metadata(j));
134  }
135  }
136 protected:
137 
138  IceModelVec::Ptr compute_impl() const {
139  auto result = m_input.allocate_copy();
140 
141  result->set_name(m_input.get_name());
142  for (unsigned int k = 0; k < m_vars.size(); ++k) {
143  result->metadata(k) = m_vars[k];
144  }
145 
146  result->copy_from(m_input);
147 
148  return result;
149  }
150 
151  const T &m_input;
152 };

154 template<typename T>
155 Diagnostic::Ptr Diagnostic::wrap(const T &input) {
156  return Ptr(new DiagWithDedicatedStorage<T>(input));

169 /*!
170  * Report a time-averaged rate of change of a quantity by accumulating changes over several time
171  * steps.
172  */
173 template<class M>
174 class DiagAverageRate : public Diag<M>
175 {
176 public:
177 
178  enum InputKind {TOTAL_CHANGE = 0, RATE = 1};
179 
180  DiagAverageRate(const M *m, const std::string &name, InputKind kind)
181  : Diag<M>(m),
182  m_factor(1.0),
183  m_input_kind(kind),
184  m_accumulator(Diagnostic::m_grid, name + "_accumulator", WITHOUT_GHOSTS),
185  m_interval_length(0.0),
186  m_time_since_reset(name + "_time_since_reset", Diagnostic::m_sys) {
187 
188  m_time_since_reset.set_string("units", "seconds");
189  m_time_since_reset.set_string("long_name",
190  "time since " + m_accumulator.get_name() +
191  " was reset to 0");
192 
193  m_accumulator.metadata().set_string("long_name",
194  "accumulator for the " + name + " diagnostic");
195 
196  m_accumulator.set(0.0);
197  }
198 protected:
199  void init_impl(const File &input, unsigned int time) {
200  if (input.find_variable(m_accumulator.get_name())) {
201  m_accumulator.read(input, time);
202  } else {
203  m_accumulator.set(0.0);
204  }
205 
206  if (input.find_variable(m_time_since_reset.get_name())) {
207  input.read_variable(m_time_since_reset.get_name(),
208  {time}, {1}, // start, count
209  &m_interval_length);
210  } else {
211  m_interval_length = 0.0;
212  }
213  }
214 
215  void define_state_impl(const File &output) const {
216  m_accumulator.define(output);
217  io::define_timeseries(m_time_since_reset,
218  Diagnostic::m_config->get_string("time.dimension_name"),
219  output, PISM_DOUBLE);
220  }
221 
222  void write_state_impl(const File &output) const {
223  m_accumulator.write(output);
224 
225  auto time_name = Diagnostic::m_config->get_string("time.dimension_name");
226 
227  unsigned int time_length = output.dimension_length(time_name);
228  unsigned int t_start = time_length > 0 ? time_length - 1 : 0;
229  io::write_timeseries(output, m_time_since_reset, t_start, {m_interval_length});
230  }
231 
232  virtual void update_impl(double dt) {
233  // Here the "factor" is used to convert units (from m to kg m-2, for example) and (possibly)
234  // integrate over the time integral using the rectangle method.
235 
236  double factor = m_factor * (m_input_kind == TOTAL_CHANGE ? 1.0 : dt);
237 
238  m_accumulator.add(factor, this->model_input());
239 
240  m_interval_length += dt;
241  }
242 
243  virtual void reset_impl() {
244  m_accumulator.set(0.0);
245  m_interval_length = 0.0;
246  }
247 
248  virtual IceModelVec::Ptr compute_impl() const {
249  IceModelVec2S::Ptr result(new IceModelVec2S(Diagnostic::m_grid,
250  "diagnostic", WITHOUT_GHOSTS));
251  result->metadata(0) = Diagnostic::m_vars.at(0);
252 
253  if (m_interval_length > 0.0) {
254  result->copy_from(m_accumulator);
255  result->scale(1.0 / m_interval_length);
256  } else {
257  result->set(Diagnostic::to_internal(Diagnostic::m_fill_value));
258  }
259 
260  return result;
261  }
262 
263  // constants initialized in the constructor
264  double m_factor;
265  InputKind m_input_kind;
266  // the state (read from and written to files)
267  IceModelVec2S m_accumulator;
268  // length of the reporting interval, accumulated along with the cumulative quantity
269  double m_interval_length;
270  VariableMetadata m_time_since_reset;
271 
272  // it should be enough to implement the constructor and this method
273  virtual const IceModelVec2S& model_input() {
274  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "no default implementation");
275  }
276 };

382 //! Scalar diagnostic reporting a "flux".
383 /*!
384  * The flux is averaged over reporting intervals.
385  *
386  * The method compute() should return the change due to a flux over a time step.
387  *
388  * Fluxes can be computed using TSRateDiagnostic, but that would require keeping track of the total
389  * change due to a flux. It is possible for the magnitude of the total change to grow indefinitely,
390  * leading to the loss of precision; this is why we use changes over individual time steps instead.
391  *
392  * (The total change due to a flux can grow in magnitude even it the amount does not change. For
393  * example: if calving removes as much ice as we have added due to the SMB, the total mass is
394  * constant, but total SMB will grow.)
395  */
396 class TSFluxDiagnostic : public TSRateDiagnostic {
397 public:
398  TSFluxDiagnostic(IceGrid::ConstPtr g, const std::string &name);
399 private:
400  void update_impl(double t0, double t1);
401 };
