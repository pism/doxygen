30 /* FIXMEs
31  *
32  * - At some later date I might want to support water input rates coming from the model.
33  * In that case I will have to accumulate input at every time step and then use the
34  * average (in time) when it's time to update the model.
35  *
36  * - IceModel::step() updates ice geometry before calling IceModel::hydrology_step(). This
37  * means that a hydrology model has to be able to provide its outputs before the first
38  * update() call. We save subglacial water flux to ensure that this is true for
39  * re-started runs, but in runs started using bootstrapping the first time step will see
40  * "bootstrapped" hydrology outputs (in this case: zero flux).
41  *
42  * - In this context (i.e. computing water flux using piecewise-in-time forcing data) it
43  * makes sense to update the flux at the beginning of an "update interval". In the case
44  * described above (water input coming from the model) we would want to update at the
45  * *end* of an interval.
46  *
47  */

83 void SteadyState::update_impl(double t, double dt, const Inputs& inputs) {
84  NullTransport::update_impl(t, dt, inputs);
85 
86  double t_next = m_t_last + max_timestep(m_t_last).value();
87 
88  if (t >= t_next or std::abs(t_next - t) < m_t_eps or
89  m_bootstrap) {
90 
91  m_log->message(3, " Updating the steady-state subglacial water flux...\n");
92 
93  m_grid->ctx()->profiling().begin("steady_emptying");
94 
95  m_emptying_problem->update(*inputs.geometry,
96  inputs.no_model_mask,
97  m_surface_input_rate);
98 
99  m_grid->ctx()->profiling().end("steady_emptying");
100  m_Q.copy_from(m_emptying_problem->flux());
101 
102  m_t_last = t;
103  m_bootstrap = false;
104  }
105 }

113 MaxTimestep SteadyState::max_timestep_impl(double t) const {
114 
115  // compute the maximum time step coming from the forcing (water input rate)
116  double dt_forcing = 0.0;
117  if (m_time.size() > 0) {
118 
119  // the right end point of the last time interval in the forcing file
120  double t_last = m_time_bounds.back();
121 
122  double t_next = 0.0;
123  if (t < m_time[0]) {
124  // Allow stepping until the left end point of the first interval.
125  t_next = m_time[0];
126  } else if (t >= t_last) {
127  // Went past the right end point of the last forcing intervals: no time step
128  // restriction from forcing.
129  t_next = t + m_update_interval;
130  } else {
131  // find the index k such that m_time[k] <= t < m_time[k + 1]
132  size_t k = gsl_interp_bsearch(m_time.data(), t, 0, m_time.size());
133 
134  assert(m_time[k] <= t);
135  assert(k + 1 == m_time.size() or t < m_time[k + 1]);
136 
137  t_next = m_time_bounds[2 * k + 1];
138 
139  if (std::abs(t_next - t) < m_t_eps) {
140  // reached t_next; use the right end point of the next interval
141  if (k + 1 < m_time.size()) {
142  t_next = m_time_bounds[2 * (k + 1) + 1];
143  } else {
144  // No time step restriction from forcing. We set dt_forcing to m_update_interval
145  // because dt_interval below will not exceed this, effectively selecting
146  // dt_interval.
147  t_next = t + m_update_interval;
148  }
149  }
150  }
151 
152  dt_forcing = t_next - t;
153 
154  assert(dt_forcing > 0.0);
155  } else {
156  dt_forcing = m_update_interval;
157  }
158 
159  // compute the maximum time step using the update interval
160  double dt_interval = 0.0;
161  {
162  if (t < m_t_last) {
163  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
164  "time %f is less than the previous time %f",
165  t, m_t_last);
166  }
167 
168  // Find the smallest time of the form m_t_last + k * m_update_interval that is greater
169  // than t
170  double k = ceil((t - m_t_last) / m_update_interval);
171 
172  double
173  t_next = m_t_last + k * m_update_interval;
174 
175  dt_interval = t_next - t;
176 
177  if (dt_interval < m_t_eps) {
178  dt_interval = m_update_interval;
179  }
180  }
181 
182  double dt = std::min(dt_forcing, dt_interval);
183 
184  auto dt_null = NullTransport::max_timestep_impl(t);
185  if (dt_null.finite()) {
186  dt = std::min(dt, dt_null.value());
187  }
188 
189  return MaxTimestep(dt, "hydrology 'steady'");
190 }

233 void SteadyState::bootstrap_impl(const File &input_file,
234  const IceModelVec2S &ice_thickness) {
235  NullTransport::bootstrap_impl(input_file, ice_thickness);
236 
237  init_time(m_config->get_string("hydrology.surface_input.file"));
238 
239  // Read m_t_last
240  {
241  if (input_file.find_variable(m_time_name)) {
242  input_file.read_variable(m_time_name, {0}, {1}, &m_t_last);
243  } else {
244  m_t_last = m_grid->ctx()->time()->current();
245  }
246  }
247 
248  // Read water flux
249  if (input_file.find_variable(m_Q.metadata().get_name())) {
250  // Regrid from the input file.
251  m_Q.regrid(input_file, CRITICAL);
252 
253  // Allow regridding from a different file.
254  regrid("hydrology 'steady'", m_Q, REGRID_WITHOUT_REGRID_VARS);
255  } else {
256  // No water flux in the input file; try regridding from a different file.
257  auto n = m_Q.state_counter();
258 
259  regrid("hydrology 'steady'", m_Q, REGRID_WITHOUT_REGRID_VARS);
260 
261  if (n == m_Q.state_counter()) {
262  // Regridding did not fill m_Q: we need to bootstrap during the first update_impl()
263  // call.
264  m_bootstrap = true;
265  }
266  }
267 }

280 /*!
281  * Read time bounds corresponding to the water input rate in the forcing file.
282  *
283  * These times are used to compute the maximum time step the model can take while still
284  * capturing temporal variability of the forcing.
285  */
286 void SteadyState::init_time(const std::string &input_file) {
287 
288  std::string variable_name = "water_input_rate";
289 
290  File file(m_grid->com, input_file, PISM_GUESS, PISM_READONLY);
291 
292  auto time_name = io::time_dimension(m_grid->ctx()->unit_system(),
293  file, variable_name);
294 
295  if (time_name.empty()) {
296  // Water input rate is time-independent. m_time and m_time_bounds are left empty.
297  return;
298  }
299 
300  std::string bounds_name = file.read_text_attribute(time_name, "bounds");
301 
302  if (bounds_name.empty()) {
303  // no time bounds attribute
304  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
305  "Variable '%s' does not have the time_bounds attribute.\n"
306  "Cannot use time-dependent water input rate without time bounds.",
307  time_name.c_str());
308  }
309 
310  // read time bounds data from a file
311  VariableMetadata tb(bounds_name, m_grid->ctx()->unit_system());
312  tb["units"] = m_grid->ctx()->time()->units_string();
313 
314  io::read_time_bounds(file, tb, *m_log, m_time_bounds);
315 
316  // time bounds data overrides the time variable: we make t[j] be the
317  // left end-point of the j-th interval
318  m_time.resize(m_time_bounds.size() / 2);
319  for (unsigned int k = 0; k < m_time.size(); ++k) {
320  m_time[k] = m_time_bounds[2*k];
321  }
322 
323  if (not is_increasing(m_time)) {
324  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
325  "time bounds in %s are invalid", input_file.c_str());
326  }
327 }
