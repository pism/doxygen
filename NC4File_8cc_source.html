120 // var
121 void NC4File::def_var_impl(const std::string &name,
122  IO_Type nctype,
123  const std::vector<std::string> &dims) const {
124  std::vector<int> dimids;
125  int stat = 0, varid = -1;
126 
127  for (auto d : dims) {
128  int dimid = -1;
129  stat = nc_inq_dimid(m_file_id, d.c_str(), &dimid); check(PISM_ERROR_LOCATION, stat);
130  dimids.push_back(dimid);
131  }
132 
133  stat = nc_def_var(m_file_id, name.c_str(), pism_type_to_nc_type(nctype),
134  static_cast<int>(dims.size()), &dimids[0], &varid);
135  check(PISM_ERROR_LOCATION, stat);
136 
137  // Compress 2D and 3D variables
138  if (m_compression_level > 0 and dims.size() > 1) {
139  stat = nc_inq_varid(m_file_id, name.c_str(), &varid); check(PISM_ERROR_LOCATION, stat);
140  stat = nc_def_var_deflate(m_file_id, varid, 0, 1, m_compression_level);
141 
142  // The NetCDF version used by PISM may not support compression.
143  if (stat == NC_EINVAL) {
144  stat = NC_NOERR;
145  }
146 
147  check(PISM_ERROR_LOCATION, stat);
148  }
149 }

262 void NC4File::get_att_double_impl(const std::string &variable_name,
263  const std::string &att_name,
264  std::vector<double> &result) const {
265  int varid = get_varid(variable_name);
266 
267  // Read the attribute length:
268  size_t attlen = 0;
269  int stat = nc_inq_attlen(m_file_id, varid, att_name.c_str(), &attlen);
270 
271  int len = 0;
272  if (stat == NC_NOERR) {
273  len = static_cast<int>(attlen);
274  } else if (stat == NC_ENOTATT) {
275  len = 0;
276  } else {
277  check(PISM_ERROR_LOCATION, stat);
278  len = 0;
279  }
280 
281  if (len == 0) {
282  result.clear();
283  return;
284  }
285 
286  result.resize(len);
287 
288  // Now read data and see if we succeeded:
289  stat = nc_get_att_double(m_file_id, varid, att_name.c_str(), &result[0]);
290  check(PISM_ERROR_LOCATION, stat);
291 }

293 // Get a text (character array) attribute on rank 0.
294 static void get_att_text(int ncid, int varid, const std::string &att_name,
295  std::string &result) {
296  size_t attlen = 0;
297  int stat = nc_inq_attlen(ncid, varid, att_name.c_str(), &attlen);
298  if (stat != NC_NOERR) {
299  result = "";
300  return;
301  }
302

304  stat = nc_get_att_text(ncid, varid, att_name.c_str(), buffer.data());
305 
306  result = (stat == NC_NOERR) ? buffer.data() : "";

309 // Get a string attribute on rank 0. In "string array" attributes array elements are
310 // concatenated using "," as the separator.
311 static void get_att_string(int ncid, int varid, const std::string &att_name,
312  std::string &result) {
313  size_t attlen = 0;
314  int stat = nc_inq_attlen(ncid, varid, att_name.c_str(), &attlen);
315  if (stat != NC_NOERR) {
316  result = "";
317  return;
318  }
319 
320  std::vector<char*> buffer(attlen + 1, 0);
321  stat = nc_get_att_string(ncid, varid, att_name.c_str(), buffer.data());
322  if (stat == NC_NOERR) {
323  std::vector<std::string> strings(attlen);
324  for (size_t k = 0; k < attlen; ++k) {
325  strings[k] = buffer[k];
326  }
327  result = join(strings, ",");
328  } else {
329  result = "";
330  }
331  stat = nc_free_string(attlen, buffer.data());
332  check(PISM_ERROR_LOCATION, stat);
333 }

335 void NC4File::get_att_text_impl(const std::string &variable_name,
336  const std::string &att_name, std::string &result) const {
337  int varid = get_varid(variable_name);
338 
339  nc_type nctype;
340  int stat = nc_inq_atttype(m_file_id, varid, att_name.c_str(), &nctype);
341 
342  if (stat == NC_NOERR) {
343  if (nctype == NC_CHAR) {
344  pism::io::get_att_text(m_file_id, varid, att_name, result);
345  } else if (nctype == NC_STRING) {
346  pism::io::get_att_string(m_file_id, varid, att_name, result);
347  } else {
348  result = "";
349  }
350  } else if (stat == NC_ENOTATT) {
351  result = "";
352  } else {
353  check(PISM_ERROR_LOCATION, stat);
354  }
355 }

414 void NC4File::get_put_var_double(const std::string &variable_name,
415  const std::vector<unsigned int> &start,
416  const std::vector<unsigned int> &count,
417  const std::vector<unsigned int> &imap_input,
418  double *op,
419  bool get,
420  bool transposed) const {
421  int stat, varid, ndims = static_cast<int>(start.size());
422  std::vector<unsigned int> imap = imap_input;
423 
424  if (not transposed) {
425  imap.resize(ndims);
426  }
427 
428  std::vector<size_t> nc_start(ndims), nc_count(ndims);
429  std::vector<ptrdiff_t> nc_imap(ndims), nc_stride(ndims);
430 
431  stat = nc_inq_varid(m_file_id, variable_name.c_str(), &varid); check(PISM_ERROR_LOCATION, stat);
432 
433  for (int j = 0; j < ndims; ++j) {
434  nc_start[j] = start[j];
435  nc_count[j] = count[j];
436  nc_imap[j] = imap[j];
437  nc_stride[j] = 1;
438  }
439 
440  if (transposed) {
441 
442  set_access_mode(varid, transposed);
443 
444  if (get) {
445  stat = nc_get_varm_double(m_file_id, varid,
446  &nc_start[0], &nc_count[0], &nc_stride[0], &nc_imap[0],
447  op); check(PISM_ERROR_LOCATION, stat);
448  } else {
449  stat = nc_put_varm_double(m_file_id, varid,
450  &nc_start[0], &nc_count[0], &nc_stride[0], &nc_imap[0],
451  op); check(PISM_ERROR_LOCATION, stat);
452  }
453  } else {
454 
455  set_access_mode(varid, transposed);
456 
457  if (get) {
458  stat = nc_get_vara_double(m_file_id, varid,
459  &nc_start[0], &nc_count[0],
460  op); check(PISM_ERROR_LOCATION, stat);
461  } else {
462  stat = nc_put_vara_double(m_file_id, varid,
463  &nc_start[0], &nc_count[0],
464  op); check(PISM_ERROR_LOCATION, stat);
465  }
466  }
467 }
