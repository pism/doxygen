69 void Given::init_impl(const Geometry &geometry) {
70  m_log->message(2,
71  "* Initializing the atmosphere model reading near-surface air temperature\n"
72  " and ice-equivalent precipitation from a file...\n");
73 
74  ForcingOptions opt(*m_grid->ctx(), "atmosphere.given");
75 
76  m_air_temp->init(opt.filename, opt.periodic);

79  // read time-independent data right away:
