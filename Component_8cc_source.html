42 /*! Process command-line options -i and -bootstrap.
43  *
44  */
45 InputOptions process_input_options(MPI_Comm com, Config::ConstPtr config) {
46  InitializationType type = INIT_OTHER;
47  unsigned int record = 0;
48 
49  std::string input_filename = config->get_string("input.file");
50 
51  bool bootstrap = config->get_flag("input.bootstrap") and (not input_filename.empty());
52  bool restart = (not config->get_flag("input.bootstrap")) and (not input_filename.empty());
53 
54  if (restart) {
55  // re-start a run by initializing from an input file
56  type = INIT_RESTART;
57  } else if (bootstrap) {
58  // initialize from an input file using bootstrapping heuristics
59  type = INIT_BOOTSTRAP;
60  } else {
61  // other types of initialization (usually from formulas)
62  type = INIT_OTHER;
63  }
64 
65  // get the index of the last record in the input file
66  if (not input_filename.empty()) {
67  File input_file(com, input_filename, PISM_NETCDF3, PISM_READONLY);
68 
69  // Find the index of the last record in the input file.
70  unsigned int last_record = input_file.nrecords();
71  if (last_record > 0) {
72  last_record -= 1;
73  }
74 
75  record = last_record;
76  } else {
77  record = 0;
78  }
79 
80  return InputOptions(type, input_filename, record);
81 }

140 /**
141  * Regrid a variable by processing -regrid_file and -regrid_vars.
142  *
143  * @param[in] module_name Module name, used to annotate options when run with -help.
144  *
145  * @param[out] variable pointer to an IceModelVec; @c variable has to
146  * have metadata set for this to work.
147  *
148  * @param[in] flag Regridding flag. If set to
149  * REGRID_WITHOUT_REGRID_VARS, regrid this variable by
150  * default, if `-regrid_vars` was not set. Otherwise a
151  * variable is only regridded if both `-regrid_file` and
152  * `-regrid_vars` are set *and* the name of the variable is
153  * found in the set of names given with `-regrid_vars`.
154  */
155 void Component::regrid(const std::string &module_name, IceModelVec &variable,
156  RegriddingFlag flag) {
157 
158  auto regrid_file = m_config->get_string("input.regrid.file");
159  auto regrid_vars = set_split(m_config->get_string("input.regrid.vars"), ',');
160 
161  if (regrid_file.empty()) {
162  return;
163  }
164 
165  SpatialVariableMetadata &m = variable.metadata();
166 
167  if (((not regrid_vars.empty()) and member(m.get_string("short_name"), regrid_vars)) or
168  (regrid_vars.empty() and flag == REGRID_WITHOUT_REGRID_VARS)) {
169 
170  m_log->message(2,
171  " %s: regridding '%s' from file '%s' ...\n",
172  module_name.c_str(),
173  m.get_string("short_name").c_str(), regrid_file.c_str());
174 
175  variable.regrid(regrid_file, CRITICAL);
176  }
177 }
