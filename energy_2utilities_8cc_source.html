36 //! Compute ice enthalpy from temperature temperature by assuming the ice has zero liquid fraction.
37 /*!
38 First this method makes sure the temperatures is at most the pressure-melting
39 value, before computing the enthalpy for that temperature, using zero liquid
40 fraction.
41 
42 Because of how EnthalpyConverter::pressure() works, the energy
43 content in the air is set to the value that ice would have if it a chunk of it
44 occupied the air; the atmosphere actually has much lower energy content. It is
45 done this way for regularity (i.e. dEnth/dz computations).
46 */
47 void compute_enthalpy_cold(const IceModelVec3 &temperature,
48  const IceModelVec2S &ice_thickness,
49  IceModelVec3 &result) {
50 
51  IceGrid::ConstPtr grid = result.grid();
52  EnthalpyConverter::Ptr EC = grid->ctx()->enthalpy_converter();
53 
54  IceModelVec::AccessList list{&temperature, &result, &ice_thickness};
55 
56  const unsigned int Mz = grid->Mz();
57  const std::vector<double> &z = grid->z();
58 
59  for (Points p(*grid); p; p.next()) {
60  const int i = p.i(), j = p.j();
61 
62  const double *Tij = temperature.get_column(i,j);
63  double *Enthij = result.get_column(i,j);
64 
65  for (unsigned int k = 0; k < Mz; ++k) {
66  const double depth = ice_thickness(i, j) - z[k]; // FIXME issue #15
67  Enthij[k] = EC->enthalpy_permissive(Tij[k], 0.0, EC->pressure(depth));
68  }
69  }
70 
71  result.inc_state_counter();
72 
73  result.update_ghosts();
74 }

84 
85  const unsigned int Mz = grid->Mz();

103 
104  result.update_ghosts();

217 //! Computes the total ice enthalpy in J.
218 /*!
219  Units of the specific enthalpy field are J kg-1. We integrate
220  \f$E(t,x,y,z)\f$ over the entire ice fluid region \f$\Omega(t)\f$, multiplying
221  by the density to get units of energy:
222  \f[ E_{\text{total}}(t) = \int_{\Omega(t)} E(t,x,y,z) \rho_i \,dx\,dy\,dz. \f]
223 */
224 double total_ice_enthalpy(double thickness_threshold,
225  const IceModelVec3 &ice_enthalpy,
226  const IceModelVec2S &ice_thickness) {
227  double enthalpy_sum = 0.0;
228 
229  IceGrid::ConstPtr grid = ice_enthalpy.grid();
230  Config::ConstPtr config = grid->ctx()->config();
231 
232  auto cell_area = grid->cell_area();
233 
234  const std::vector<double> &z = grid->z();
235 
236  IceModelVec::AccessList list{&ice_enthalpy, &ice_thickness};
237  ParallelSection loop(grid->com);
238  try {
239  for (Points p(*grid); p; p.next()) {
240  const int i = p.i(), j = p.j();
241 
242  const double H = ice_thickness(i, j);
243 
244  if (H >= thickness_threshold) {
245  const int ks = grid->kBelowHeight(H);
246 
247  const double
248  *E = ice_enthalpy.get_column(i, j);
249 
250  for (int k = 0; k < ks; ++k) {
251  enthalpy_sum += cell_area * E[k] * (z[k+1] - z[k]);
252  }
253  enthalpy_sum += cell_area * E[ks] * (H - z[ks]);
254  }
255  }
256  } catch (...) {
257  loop.failed();
258  }
259  loop.check();
260 
261  enthalpy_sum *= config->get_number("constants.ice.density");
262 
263  return GlobalSum(grid->com, enthalpy_sum);
264 }

266 //! Create a temperature field within the ice from provided ice thickness, surface temperature, surface mass balance, and geothermal flux.
267 /*!
268 In bootstrapping we need to determine initial values for the temperature within
269 the ice (and the bedrock). There are various data available at bootstrapping,
270 but not the 3D temperature field needed as initial values for the temperature. Here
271 we take a "guess" based on an assumption of steady state and a simple model of
272 the vertical velocity in the column. The rule is certainly heuristic but it
273 seems to work well anyway.
274 
275 The result is *not* the temperature field which is in steady state with the ice
276 dynamics. Spinup is most-definitely needed in many applications. Such spinup
277 usually starts from the temperature field computed by this procedure and then
278 runs for a long time (e.g. \f$10^4\f$ to \f$10^6\f$ years), possibly with fixed
279 geometry, to get closer to thermomechanically-coupled equilibrium.
280 
281 Consider a horizontal grid point. Suppose the surface temperature
282 \f$T_s\f$, surface mass balance \f$m\f$, and geothermal flux \f$g\f$ are given at that location.
283 Within the column denote the temperature by \f$T(z)\f$ at height \f$z\f$ above
284 the base of the ice. Suppose the column of ice has height \f$H\f$, the ice
285 thickness.
286 
287 There are two alternative bootstrap methods determined by the configuration parameter
288 `config.get_number("bootstrapping.temperature_heuristic"))`. Allowed values are `"smb"` and
289 `"quartic_guess"`.
290 
291 1. If the `smb` method is chosen, which is the default, and if \f$m>0\f$,
292 then the method sets the ice
293 temperature to the solution of the steady problem [\ref Paterson]
294  \f[\rho_i c w \frac{\partial T}{\partial z} = k_i \frac{\partial^2 T}{\partial z^2} \qquad \text{with boundary conditions} \qquad T(H) = T_s \quad \text{and} \quad \frac{\partial T}{\partial z}(0) = - \frac{g}{k_i}, \f]
295 where the vertical velocity is linear between the surface value \f$w=-m\f$ and
296 a velocity of zero at the base:
297  \f[w(z) = - m z / H.\f]
298 (Note that because \f$m>0\f$, this vertical velocity is downward.)
299 This is a two-point boundary value problem for a linear ODE. In fact, if
300 \f$K = k_i / (\rho_i c)\f$ then we can write the ODE as
301  \f[K T'' + \frac{m z}{H} T' = 0.\f]
302 Then let
303  \f[C_0 = \frac{g \sqrt{\pi H K}}{k_i \sqrt{2 m}}, \qquad \gamma_0 = \sqrt{\frac{mH}{2K}}.\f]
304 (Note \f$\gamma_0\f$ is, up to a constant, the square root of the Peclet number
305 [\ref Paterson]; compare [\ref vanderWeletal2013].) The solution to the
306 two-point boundary value problem is then
307  \f[T(z) = T_s + C_0 \left(\operatorname{erf}(\gamma_0) - \operatorname{erf}\left(\gamma_0 \frac{z}{H}\right)\right).\f]
308 If `usesmb` is true and \f$m \le 0\f$, then the velocity in the column, relative
309 to the base, is taken to be zero. Thus the solution is
310  \f[ T(z) = \frac{g}{k_i} \left( H - z \right) + T_s, \f]
311 a straight line whose slope is determined by the geothermal flux and whose value
312 at the ice surface is the surface temperature, \f$T(H) = T_s\f$.
313 2. If the `quartic_guess` method is chosen, the "quartic guess" formula which was in older
314 versions of PISM is used. Namely, within the ice we set
315 \f[T(z) = T_s + \alpha (H-z)^2 + \beta (H-z)^4\f]
316 where \f$\alpha,\beta\f$ are chosen so that
317 \f[\frac{\partial T}{\partial z}\Big|_{z=0} = - \frac{g}{k_i} \qquad \text{and} \qquad \frac{\partial T}{\partial z}\Big|_{z=H/4} = - \frac{g}{2 k_i}.\f]
318 The purpose of the second condition is that when ice is advecting downward then
319 the temperature gradient is much larger in roughly the bottom quarter of the
320 ice column. However, without the surface mass balance, much less the solution
321 of the stress balance equations, we cannot estimate the vertical velocity, so
322 we make such a rough guess.
323 
324 In either case the temperature within the ice is not allowed to exceed the
325 pressure-melting temperature.
326 
327 We set \f$T(z)=T_s\f$ above the top of the ice.
328 
329 This method determines \f$T(0)\f$, the ice temperature at the ice base. This
330 temperature is used by BedThermalUnit::bootstrap() to determine a
331 bootstrap temperature profile in the bedrock.
332 */
333 void bootstrap_ice_temperature(const IceModelVec2S &ice_thickness,
334  const IceModelVec2S &ice_surface_temp,
335  const IceModelVec2S &surface_mass_balance,
336  const IceModelVec2S &basal_heat_flux,
337  IceModelVec3 &result) {
338 
339  auto grid = result.grid();
340  auto ctx = grid->ctx();
341  auto config = ctx->config();
342  auto log = ctx->log();
343  auto EC = ctx->enthalpy_converter();
344 
345  auto use_smb = config->get_string("bootstrapping.temperature_heuristic") == "smb";
346 
347  if (use_smb) {
348  log->message(2,
349  " - Filling 3D ice temperatures using surface temperature"
350  " (and mass balance for velocity estimate)...\n");
351 
352  } else {
353  log->message(2,
354  " - Filling 3D ice temperatures using surface temperature"
355  " (and a quartic guess without SMB)...\n");
356  }
357 
358  const double
359  ice_k = config->get_number("constants.ice.thermal_conductivity"),
360  ice_density = config->get_number("constants.ice.density"),
361  ice_c = config->get_number("constants.ice.specific_heat_capacity"),
362  K = ice_k / (ice_density * ice_c),
363  T_min = config->get_number("energy.minimum_allowed_temperature"),
364  T_melting = config->get_number("constants.fresh_water.melting_point_temperature",
365  "Kelvin");
366 
367  IceModelVec::AccessList list{&ice_surface_temp, &surface_mass_balance,
368  &ice_thickness, &basal_heat_flux, &result};
369 
370  ParallelSection loop(grid->com);
371  try {
372  for (Points p(*grid); p; p.next()) {
373  const int i = p.i(), j = p.j();
374 
375  const double
376  T_surface = std::min(ice_surface_temp(i, j), T_melting),
377  H = ice_thickness(i, j),
378  G = basal_heat_flux(i, j);
379 
380  const unsigned int ks = grid->kBelowHeight(H);
381 
382  if (G < 0.0 and ks > 0) {
383  auto units = basal_heat_flux.metadata().get_string("units");
384  int Mbz = config->get_number("grid.Mbz");
385  const char *quantity = (Mbz > 0 ?
386  "temperature of the bedrock thermal layer" :
387  "geothermal flux");
388  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
389  "Negative upward heat flux (%f %s) through the bottom of the ice column\n"
390  "is not allowed by PISM's ice temperature bootstrapping method.\n"
391  "Please check the %s at i=%d, j=%d.",
392  G, units.c_str(), quantity, i, j);
393  }
394 
395  if (T_surface < T_min) {
396  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
397  "T_surface(%d,%d) = %f < T_min = %f Kelvin",
398  i, j, T_surface, T_min);
399  }
400 
401  double *T = result.get_column(i, j);
402 
403  // within ice
404  if (use_smb) { // method 1: includes surface mass balance in estimate
405 
406  // Convert SMB from "kg m-2 s-1" to "m second-1".
407  const double SMB = surface_mass_balance(i, j) / ice_density;
408 
409  for (unsigned int k = 0; k < ks; k++) {
410  const double z = grid->z(k);
411  T[k] = ice_temperature_guess_smb(EC, H, z, T_surface, G, ice_k, K, SMB);
412  }
413 
414  } else { // method 2: a quartic guess; does not use SMB
415 
416  for (unsigned int k = 0; k < ks; k++) {
417  const double z = grid->z(k);
418  T[k] = ice_temperature_guess(EC, H, z, T_surface, G, ice_k);
419  }
420 
421  }
422 
423  // Make sure that resulting temperatures are not too low.
424  for (unsigned int k = 0; k < ks; k++) {
425  if (T[k] < T_min) {
426  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
427  "T(%d,%d,%d) = %f < T_min = %f Kelvin",
428  i, j, k, T[k], T_min);
429  }
430  }
431 
432  // above ice
433  for (unsigned int k = ks; k < grid->Mz(); k++) {
434  T[k] = T_surface;
435  }
436  }
437  } catch (...) {
438  loop.failed();
439  }
440  loop.check();
441 
442  result.update_ghosts();
443 }
