59 static int weight(int M_ij, int M_n, double h_ij, double h_n) {
60  // fjord walls, nunataks, headwalls
61  if ((mask::icy(M_ij) and mask::ice_free(M_n) and h_n > h_ij) or
62  (mask::ice_free(M_ij) and mask::icy(M_n) and h_ij > h_n)) {
63  return 0;
64  }
65 
66  return 1;
67 }

69 void SSAFD_Regional::compute_driving_stress(const IceModelVec2S &ice_thickness,
70  const IceModelVec2S &surface_elevation,
71  const IceModelVec2CellType &cell_type,
72  const IceModelVec2Int *no_model_mask,
73  IceModelVec2V &result) const {
74 
75  SSAFD::compute_driving_stress(ice_thickness, surface_elevation, cell_type, no_model_mask, result);
76 
77  double
78  dx = m_grid->dx(),
79  dy = m_grid->dy();
80 
81  int
82  Mx = m_grid->Mx(),
83  My = m_grid->My();
84 
85  IceModelVec::AccessList list{&result, &cell_type, no_model_mask, m_h_stored, m_H_stored};
86 
87  for (Points p(*m_grid); p; p.next()) {
88  const int i = p.i(), j = p.j();
89 
90  auto M = no_model_mask->star(i, j);
91 
92  if (M.ij == 0) {
93  // this grid point is in the modeled area so we don't need to modify the driving
94  // stress
95  continue;
96  }
97 
98  double pressure = m_EC->pressure((*m_H_stored)(i, j));
99  if (pressure <= 0) {
100  result(i, j) = 0.0;
101  continue;
102  }
103 
104  auto h = m_h_stored->star(i, j);
105  auto CT = cell_type.star(i, j);
106 
107  // x-derivative
108  double h_x = 0.0;
109  {
110  double
111  west = M.w == 1 and i > 0,
112  east = M.e == 1 and i < Mx - 1;
113 
114  // don't use differences spanning "cliffs"
115  west *= weight(CT.ij, CT.w, h.ij, h.w);
116  east *= weight(CT.ij, CT.e, h.ij, h.e);
117 
118  if (east + west > 0) {
119  h_x = 1.0 / ((west + east) * dx) * (west * (h.ij - h.w) + east * (h.e - h.ij));
120  } else {
121  h_x = 0.0;
122  }
123  }
124 
125  // y-derivative
126  double h_y = 0.0;
127  {
128  double
129  south = M.s == 1 and j > 0,
130  north = M.n == 1 and j < My - 1;
131 
132  // don't use differences spanning "cliffs"
133  south *= weight(CT.ij, CT.s, h.ij, h.s);
134  north *= weight(CT.ij, CT.n, h.ij, h.n);
135 
136  if (north + south > 0) {
137  h_y = 1.0 / ((south + north) * dy) * (south * (h.ij - h.s) + north * (h.n - h.ij));
138  } else {
139  h_y = 0.0;
140  }
141  }
142 
143  result(i, j) = - pressure * Vector2(h_x, h_y);
144  } // end of the loop over grid points
145 }
