351 /*!
352  * Update geometry by applying changes due to surface and basal mass fluxes.
353  *
354  * Note: This method performs these changes in the same order as the code ensuring
355  * non-negativity. This is important.
356  */
357 void GeometryEvolution::apply_mass_fluxes(Geometry &geometry) const {
358 
359  const IceModelVec2S
360  &dH_SMB = top_surface_mass_balance(),
361  &dH_BMB = bottom_surface_mass_balance();
362  IceModelVec2S &H = geometry.ice_thickness;
363 
364  IceModelVec::AccessList list{&H, &dH_SMB, &dH_BMB};
365  ParallelSection loop(m_grid->com);
366  try {
367  for (Points p(*m_grid); p; p.next()) {
368  const int i = p.i(), j = p.j();
369 
370  // To preserve non-negativity of thickness we need to apply changes in this exact order.
371  // (Recall that floating-point arithmetic is not associative.)
372  const double H_new = (H(i, j) + dH_SMB(i, j)) + dH_BMB(i, j);
373 
374 #if (Pism_DEBUG==1)
375  if (H_new < 0.0) {
376  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "H = %f (negative) at i=%d, j=%d",
377  H_new, i, j);
378  }
379 #endif
380 
381  H(i, j) = H_new;
382  }
383  } catch (...) {
384  loop.failed();
385  }
386  loop.check();
387 }

390 /*!
391  * Prevent advective ice flow from floating ice to ice-free land, as well as in the
392  * ice-free areas.
393  */
394 static double limit_advective_flux(int current, int neighbor, double input) {
395 
396  // Case 1: Flow between grounded_ice and grounded_ice.
397  if (grounded_ice(current) and grounded_ice(neighbor)) {
398  return input;
399  }
400 
401  // Cases 2 and 3: Flow between grounded_ice and floating_ice.
402  if ((grounded_ice(current) and floating_ice(neighbor)) or
403  (floating_ice(current) and grounded_ice(neighbor))) {
404  return input;
405  }
406 
407  // Cases 4 and 5: Flow between grounded_ice and ice_free_land.
408  if ((grounded_ice(current) and ice_free_land(neighbor)) or
409  (ice_free_land(current) and grounded_ice(neighbor))) {
410  return input;
411  }
412 
413  // Cases 6 and 7: Flow between grounded_ice and ice_free_ocean.
414  if ((grounded_ice(current) and ice_free_ocean(neighbor)) or
415  (ice_free_ocean(current) and grounded_ice(neighbor))) {
416  return input;
417  }
418 
419  // Case 8: Flow between floating_ice and floating_ice.
420  if (floating_ice(current) and floating_ice(neighbor)) {
421  return input;
422  }
423 
424  // Cases 9 and 10: Flow between floating_ice and ice_free_land.
425  if ((floating_ice(current) and ice_free_land(neighbor)) or
426  (ice_free_land(current) and floating_ice(neighbor))) {
427  // Disable all flow. This ensures that an ice shelf does not climb up a cliff.
428  return 0.0;
429  }
430 
431  // Cases 11 and 12: Flow between floating_ice and ice_free_ocean.
432  if ((floating_ice(current) and ice_free_ocean(neighbor)) or
433  (ice_free_ocean(current) and floating_ice(neighbor))) {
434  return input;
435  }
436 
437  // Case 13: Flow between ice_free_land and ice_free_land.
438  if (ice_free_land(current) and ice_free_land(neighbor)) {
439  return 0.0;
440  }
441 
442  // Cases 14 and 15: Flow between ice_free_land and ice_free_ocean.
443  if ((ice_free_land(current) and ice_free_ocean(neighbor)) or
444  (ice_free_ocean(current) and ice_free_land(neighbor))) {
445  return 0.0;
446  }
447 
448  // Case 16: Flow between ice_free_ocean and ice_free_ocean.
449  if (ice_free_ocean(current) and ice_free_ocean(neighbor)) {
450  return 0.0;
451  }
452 
453  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
454  "cannot handle the case current=%d, neighbor=%d",
455  current, neighbor);
456 }

526 /*!
527  * Combine advective velocity and the diffusive flux on the staggered grid with the ice thickness to
528  * compute the total flux through cell interfaces.
529  *
530  * Uses first-order upwinding to compute the advective flux.
531  *
532  * Limits the diffusive flux to prevent SIA-driven flow in the ocean and ice-free areas.
533  */
534 void GeometryEvolution::compute_interface_fluxes(const IceModelVec2CellType &cell_type,
535  const IceModelVec2S &ice_thickness,
536  const IceModelVec2V &velocity,
537  const IceModelVec2Stag &diffusive_flux,
538  IceModelVec2Stag &output) {
539 
540  IceModelVec::AccessList list{&cell_type, &velocity, &ice_thickness,
541  &diffusive_flux, &output};
542 
543  ParallelSection loop(m_grid->com);
544  try {
545  // compute advective fluxes and put them in output
546  for (Points p(*m_grid); p; p.next()) {
547  const int
548  i = p.i(),
549  j = p.j(),
550  M = cell_type(i, j);
551 
552  const double H = ice_thickness(i, j);
553  const Vector2 V = velocity(i, j);
554 
555  for (int n = 0; n < 2; ++n) {
556  const int
557  oi = 1 - n, // offset in the i direction
558  oj = n, // offset in the j direction
559  i_n = i + oi, // i index of a neighbor
560  j_n = j + oj; // j index of a neighbor
561 
562  const int M_n = cell_type(i_n, j_n);
563 
564  // advective velocity at the current interface
565  double v = 0.0;
566  {
567  Vector2 V_n = velocity(i_n, j_n);
568  int
569  W = icy(M),
570  W_n = icy(M_n);
571 
572  auto v_staggered = (W * V + W_n * V_n) / std::max(W + W_n, 1);
573  v = n == 0 ? v_staggered.u : v_staggered.v;
574  }
575 
576  // advective flux
577  const double
578  H_n = ice_thickness(i_n, j_n),
579  Q_advective = v * (v > 0.0 ? H : H_n); // first order upwinding
580 
581  output(i, j, n) = Q_advective;
582  } // end of the loop over neighbors (n)
583  }
584 
585  // limit the advective flux and add the diffusive flux to it to get the total
586  for (Points p(*m_grid); p; p.next()) {
587  const int
588  i = p.i(),
589  j = p.j(),
590  M = cell_type(i, j);
591 
592  for (int n = 0; n < 2; ++n) {
593  const int
594  oi = 1 - n, // offset in the i direction
595  oj = n, // offset in the j direction
596  i_n = i + oi, // i index of a neighbor
597  j_n = j + oj; // j index of a neighbor
598 
599  const int M_n = cell_type(i_n, j_n);
600 
601  // diffusive flux
602  const double
603  Q_diffusive = limit_diffusive_flux(M, M_n, diffusive_flux(i, j, n)),

618  *
619  * The flux divergence at *ice thickness* Dirichlet B.C. locations is set to zero.
620  */
621 void GeometryEvolution::compute_flux_divergence(double dt,
622  const IceModelVec2Stag &flux,
623  const IceModelVec2Int &thickness_bc_mask,
624  IceModelVec2S &conservation_error,
625  IceModelVec2S &output) {

628  dy = m_grid->dy();

635  const int i = p.i(), j = p.j();

656 /*!
657  * Update ice thickness and area_specific_volume *in place*.
658  *
659  * It would be better to compute the change in ice thickness and area_specific_volume and then apply
660  * them, but it would require re-writing all the part-grid code from scratch. So, I make copies of
661  * ice thickness and area_specific_volume, use this old code, then compute differences to get changes.
662  * Compute ice thickness changes due to the flow of the ice.
663  *
664  * @param[in] dt time step, seconds
665  * @param[in] bed_elevation bed elevation, meters
666  * @param[in] sea_level sea level elevation
667  * @param[in] ice_thickness ice thickness
668  * @param[in] area_specific_volume area-specific volume (m3/m2)
669  * @param[in] flux_divergence flux divergence
670  * @param[out] thickness_change ice thickness change due to flow
671  * @param[out] area_specific_volume_change area specific volume change due to flow
672  */
673 void GeometryEvolution::update_in_place(double dt,
674  const IceModelVec2S &bed_topography,
675  const IceModelVec2S &sea_level,
676  const IceModelVec2S &flux_divergence,
677  IceModelVec2S &ice_thickness,
678  IceModelVec2S &area_specific_volume) {
679 
680  m_impl->gc.compute(sea_level, bed_topography, ice_thickness,
681  m_impl->cell_type, m_impl->surface_elevation);
682 
683  IceModelVec::AccessList list{&ice_thickness, &flux_divergence};
684 
685  if (m_impl->use_part_grid) {
686  m_impl->residual.set(0.0);
687 
688  // Store ice thickness. We need this copy to make sure that modifying ice_thickness in the loop
689  // below does not affect the computation of the threshold thickness. (Note that
690  // part_grid_threshold_thickness uses neighboring values of the mask, ice thickness, and surface
691  // elevation.)
692  m_impl->thickness.copy_from(ice_thickness);
693 
694  list.add({&area_specific_volume, &m_impl->residual, &m_impl->thickness,
695  &m_impl->surface_elevation, &bed_topography, &m_impl->cell_type});
696  }
697 
698 #if (Pism_DEBUG==1)
699  const double Lz = m_grid->Lz();
700 #endif
701 
702  ParallelSection loop(m_grid->com);
703  try {
704  for (Points p(*m_grid); p; p.next()) {
705  const int i = p.i(), j = p.j();
706 
707  double divQ = flux_divergence(i, j);
708 
709  if (m_impl->use_part_grid) {
710  if (m_impl->cell_type.ice_free_ocean(i, j) and m_impl->cell_type.next_to_ice(i, j)) {
711  // Add the flow contribution to this partially filled cell.
712  area_specific_volume(i, j) += -divQ * dt;
713 
714  double threshold = part_grid_threshold_thickness(m_impl->cell_type.star(i, j),
715  m_impl->thickness.star(i, j),
716  m_impl->surface_elevation.star(i, j),
717  bed_topography(i, j));
718 
719  // if threshold is zero, turn all the area specific volume into ice thickness, with zero
720  // residual
721  if (threshold == 0.0) {
722  threshold = area_specific_volume(i, j);
723  }
724 
725  if (area_specific_volume(i, j) >= threshold) {
726  ice_thickness(i, j) += threshold;
727  m_impl->residual(i, j) = area_specific_volume(i, j) - threshold;
728  area_specific_volume(i, j) = 0.0;
729  }
730 
731  // In this case the flux goes into the area_specific_volume variable and does not directly
732  // contribute to ice thickness at this location.
733  divQ = 0.0;
734  }
735  } // end of if (use_part_grid)
736 
737  ice_thickness(i, j) += - dt * divQ;
738 
739 #if (Pism_DEBUG==1)
740  if (ice_thickness(i, j) > Lz) {
741  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "ice thickness exceeds Lz at i=%d, j=%d (H=%f, Lz=%f)",
742  i, j, ice_thickness(i, j), Lz);
743  }
744 #endif
745  }
746  } catch (...) {
747  loop.failed();
748  }
749  loop.check();
750 
751  ice_thickness.update_ghosts();
752 
753  // Compute the mask corresponding to the new thickness.
754  m_impl->gc.compute_mask(sea_level, bed_topography, ice_thickness, m_impl->cell_type);
755 
756  /*
757  Redistribute residual ice mass from subgrid-scale parameterization.
758 
759  See [@ref Albrechtetal2011].
760  */
761  if (m_impl->use_part_grid) {
762  const int max_n_iterations = m_config->get_number("geometry.part_grid.max_iterations");
763 
764  bool done = false;
765  for (int i = 0; i < max_n_iterations and not done; ++i) {
766  m_log->message(4, "redistribution iteration %d\n", i);
767 
768  // this call may set done to true
769  residual_redistribution_iteration(bed_topography,
770  sea_level,
771  m_impl->surface_elevation,
772  ice_thickness,
773  m_impl->cell_type,
774  area_specific_volume,
775  m_impl->residual,
776  done);
777  }
778 
779  if (not done) {
780  m_log->message(2,
781  "WARNING: not done redistributing mass after %d iterations, remaining residual: %f m^3.\n",
782  max_n_iterations, sum(m_impl->residual) * m_grid->cell_area());
783 
784  // Add residual to ice thickness, preserving total ice mass. (This is not great, but
785  // better than losing mass.)
786  ice_thickness.add(1.0, m_impl->residual);
787  m_impl->residual.set(0.0);
788  }
789  }
790 }

792 //! @brief Perform one iteration of the residual mass redistribution.
793 /*!
794  @param[in] bed_topography bed elevation
795  @param[in] sea_level sea level elevation
796  @param[in,out] ice_surface_elevation surface elevation; used as temp. storage
797  @param[in,out] ice_thickness ice thickness; updated
798  @param[in,out] cell_type cell type mask; used as temp. storage
799  @param[in,out] area_specific_volume area specific volume; updated
800  @param[in,out] residual ice volume that still needs to be distributed; updated
801  @param[in,out] done result flag: true if this iteration should be the last one
802  */
803 void GeometryEvolution::residual_redistribution_iteration(const IceModelVec2S &bed_topography,
804  const IceModelVec2S &sea_level,
805  IceModelVec2S &ice_surface_elevation,
806  IceModelVec2S &ice_thickness,
807  IceModelVec2CellType &cell_type,
808  IceModelVec2S &area_specific_volume,
809  IceModelVec2S &residual,
810  bool &done) {
811 
812  m_impl->gc.compute_mask(sea_level, bed_topography, ice_thickness, cell_type);
813 
814  const Direction directions[4] = {North, East, South, West};
815 
816  // First step: distribute residual mass
817  {
818  // will be destroyed at the end of the block
819  IceModelVec::AccessList list{&cell_type, &ice_thickness, &area_specific_volume, &residual};
820 
821  for (Points p(*m_grid); p; p.next()) {
822  const int i = p.i(), j = p.j();
823 
824  if (residual(i, j) <= 0.0) {
825  continue;
826  }
827 
828  auto m = cell_type.star(i, j);
829 
830  int N = 0; // number of empty or partially filled neighbors
831  for (unsigned int n = 0; n < 4; ++n) {
832  const Direction direction = directions[n];
833  if (ice_free_ocean(m[direction])) {
834  N++;
835  }
836  }
837 
838  if (N > 0) {
839  // Remaining ice mass will be redistributed equally among all adjacent
840  // ice-free-ocean cells (is there a more physical way?)
841  residual(i, j) /= N;
842  } else {
843  // Conserve mass, but (possibly) create a "ridge" at the shelf
844  // front
845  ice_thickness(i, j) += residual(i, j);
846  residual(i, j) = 0.0;
847  }
848  }
849 
850  residual.update_ghosts();
851 
852  // update area_specific_volume using adjusted residuals
853  for (Points p(*m_grid); p; p.next()) {
854  const int i = p.i(), j = p.j();
855 
856  if (cell_type.ice_free_ocean(i, j)) {
857  area_specific_volume(i, j) += (residual(i + 1, j) +
858  residual(i - 1, j) +
859  residual(i, j + 1) +
860  residual(i, j - 1));
861  }
862 
863  }
864 
865  residual.set(0.0);
866  }
867 
868  ice_thickness.update_ghosts();
869 
870  // Store ice thickness. We need this copy to make sure that modifying ice_thickness in the loop
871  // below does not affect the computation of the threshold thickness. (Note that
872  // part_grid_threshold_thickness uses neighboring values of the mask, ice thickness, and surface
873  // elevation.)
874  m_impl->thickness.copy_from(ice_thickness);
875 
876  // The loop above updated ice_thickness, so we need to re-calculate the mask and the
877  // surface elevation:
878  m_impl->gc.compute(sea_level, bed_topography, ice_thickness, cell_type, ice_surface_elevation);
879 
880  double remaining_residual = 0.0;
881 
882  // Second step: we need to redistribute residual ice volume if
883  // neighbors which gained redistributed ice also become full.
884  {
885  // will be destroyed at the end of the block
886  IceModelVec::AccessList list{&m_impl->thickness, &ice_thickness,
887  &ice_surface_elevation, &bed_topography, &cell_type};
888 
889  for (Points p(*m_grid); p; p.next()) {
890  const int i = p.i(), j = p.j();
891 
892  if (area_specific_volume(i, j) <= 0.0) {
893  continue;
894  }
895 
896  double threshold = part_grid_threshold_thickness(cell_type.star(i, j),
897  m_impl->thickness.star(i, j),
898  ice_surface_elevation.star(i, j),
899  bed_topography(i, j));
900 
901  // if threshold is zero, turn all the area specific volume into ice thickness, with zero
902  // residual
903  if (threshold == 0.0) {
904  threshold = area_specific_volume(i, j);
905  }
906 
907  if (area_specific_volume(i, j) >= threshold) {
908  ice_thickness(i, j) += threshold;
909  residual(i, j) = area_specific_volume(i, j) - threshold;
910  area_specific_volume(i, j) = 0.0;
911 
912  remaining_residual += residual(i, j);
913  }
914  }
915  }
916 
917  // check if redistribution should be run once more
918  remaining_residual = GlobalSum(m_grid->com, remaining_residual);
919 
920  if (remaining_residual > 0.0) {
921  done = false;
922  } else {
923  done = true;
924  }
925 
926  ice_thickness.update_ghosts();
927 }

929 /*!
930  * Correct `thickness_change` and `area_specific_volume_change` so that applying them will not
931  * result in negative `ice_thickness` and `area_specific_volume`.
932  *
933  * Compute the the amount of ice that is added to preserve non-negativity of ice thickness.
934  *
935  * @param[in] ice_thickness ice thickness (m)
936  * @param[in] area_specific_volume area-specific volume (m3/m2)
937  * @param[in,out] thickness_change "proposed" thickness change (m)
938  * @param[in,out] area_specific_volume_change "proposed" area-specific volume change (m3/m2)
939  * @param[in,out] conservation_error computed conservation error (m)
940  *
941  * This computation is purely local.
942  */
943 void GeometryEvolution::ensure_nonnegativity(const IceModelVec2S &ice_thickness,
944  const IceModelVec2S &area_specific_volume,
945  IceModelVec2S &thickness_change,
946  IceModelVec2S &area_specific_volume_change,
947  IceModelVec2S &conservation_error) {
948 
949  IceModelVec::AccessList list{&ice_thickness, &area_specific_volume, &thickness_change,
950  &area_specific_volume_change, &conservation_error};
951 
952  ParallelSection loop(m_grid->com);
953  try {
954  for (Points p(*m_grid); p; p.next()) {
955  const int i = p.i(), j = p.j();
956 
957  const double
958  H = ice_thickness(i, j),
959  dH = thickness_change(i, j);
960 
961  // applying thickness_change will lead to negative thickness
962  if (H + dH < 0.0) {
963  thickness_change(i, j) = H;
964  conservation_error(i, j) += - (H + dH);
965  }
966

968  V = area_specific_volume(i, j),
969  dV = area_specific_volume_change(i, j);
970 
971  if (V + dV < 0.0) {
972  area_specific_volume_change(i, j) = V;
973  conservation_error(i, j) += - (V + dV);
974  }
975  }
976  } catch (...) {
977  loop.failed();
978  }
979  loop.check();
980 }

1006  *
