61 void PrescribedRetreat::update(double t,
62  double dt,
63  IceModelVec2S& ice_thickness,
64  IceModelVec2S& ice_area_specific_volume) {
65  m_retreat_mask->update(t, dt);
66  m_retreat_mask->average(t, dt);
67 
68  IceModelVec::AccessList list{m_retreat_mask.get(), &ice_thickness, &ice_area_specific_volume};
69 
70  for (Points p(*m_grid); p; p.next()) {
71  const int i = p.i(), j = p.j();
72 
73  double f = (*m_retreat_mask)(i, j);
74 
75  if (f <= 0.0) {
76  ice_area_specific_volume(i, j) = 0.0;
77  ice_thickness(i, j) = 0.0;
78  } else if (f < 1.0) {
79  ice_area_specific_volume(i, j) = ice_thickness(i, j) * f;
80  ice_thickness(i, j) = 0.0;
81  } else {
82  // M == 1.0: do nothing
83  }
84  }
85 }
