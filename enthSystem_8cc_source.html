114 void enthSystemCtx::init(int i, int j, bool marginal, double ice_thickness) {
115  m_ice_thickness = ice_thickness;
116 
117  m_marginal = marginal;
118 
119  init_column(i, j, m_ice_thickness);
120 
121  if (m_ks == 0) {
122  return;
123  }
124 
125  coarse_to_fine(m_u3, m_i, m_j, &m_u[0]);
126  coarse_to_fine(m_v3, m_i, m_j, &m_v[0]);
127 
128  if (m_marginal and m_exclude_vertical_advection) {
129  for (unsigned int k = 0; k < m_w.size(); ++k) {
130  m_w[k] = 0.0;
131  }
132  } else {
133  coarse_to_fine(m_w3, m_i, m_j, &m_w[0]);
134  }
135 
136  coarse_to_fine(m_strain_heating3, m_i, m_j, &m_strain_heating[0]);
137  coarse_to_fine(m_Enth3, m_i, m_j, &m_Enth[0]);
138 
139  coarse_to_fine(m_Enth3, m_i, m_j+1, &m_E_n[0]);
140  coarse_to_fine(m_Enth3, m_i+1, m_j, &m_E_e[0]);
141  coarse_to_fine(m_Enth3, m_i, m_j-1, &m_E_s[0]);
142  coarse_to_fine(m_Enth3, m_i-1, m_j, &m_E_w[0]);
143 
144  compute_enthalpy_CTS();
145 
146  m_lambda = compute_lambda();
147 
148  assemble_R();
149 }

151 //! Compute the CTS value of enthalpy in an ice column.
152 /*! m_Enth_s is set to the enthalpy value for the pressure-melting
153  temperature with zero water fraction at the corresponding z level.
154  */
155 void enthSystemCtx::compute_enthalpy_CTS() {
156 
157  for (unsigned int k = 0; k <= m_ks; k++) {
158  const double
159  depth = m_ice_thickness - k * m_dz,
160  p = m_EC->pressure(depth); // FIXME issue #15
161  m_Enth_s[k] = m_EC->enthalpy_cts(p);
162  }
163 
164  const double Es_air = m_EC->enthalpy_cts(m_p_air);
165  for (unsigned int k = m_ks+1; k < m_Enth_s.size(); k++) {
166  m_Enth_s[k] = Es_air;
167  }
168 }

170 //! Compute the lambda for BOMBPROOF.
171 /*!
172 See page \ref bombproofenth.
173  */
174 double enthSystemCtx::compute_lambda() {
175  double result = 1.0; // start with centered implicit for more accuracy
176  const double epsilon = 1e-6 / 3.15569259747e7;
177 
178  for (unsigned int k = 0; k <= m_ks; k++) {
179  if (m_Enth[k] > m_Enth_s[k]) { // lambda = 0 if temperate ice present in column
180  result = 0.0;
181  } else {
182  const double denom = (fabs(m_w[k]) + epsilon) * m_ice_density * m_ice_c * m_dz;
183  result = std::min(result, 2.0 * m_ice_k / denom);
184  }
185  }
186  return result;
187 }

222 //! Set enthalpy flux at the surface.
223 /*! This method should probably be used for debugging only. Its purpose is to allow setting the
224  enthalpy flux even if K == 0, i.e. in a "pure advection" setup.
225  */
226 void enthSystemCtx::set_surface_neumann_bc(double G) {
227  const bool include_horizontal_advection = not (m_marginal and m_exclude_horizontal_advection);
228  const bool include_strain_heating = not (m_marginal and m_exclude_strain_heat);
229 
230  const double
231  Rminus = 0.5 * (m_R[m_ks - 1] + m_R[m_ks]), // R_{ks-1/2}
232  Rplus = m_R[m_ks], // R_{ks+1/2}
233  mu_w = 0.5 * m_nu * m_w[m_ks];
234 
235  const double A_l = m_w[m_ks] < 0.0 ? 1.0 - m_lambda : m_lambda - 1.0;
236  const double A_d = m_w[m_ks] < 0.0 ? m_lambda - 1.0 : 1.0 - m_lambda;
237  const double A_b = m_w[m_ks] < 0.0 ? m_lambda - 2.0 : -m_lambda;
238 
239  // modified lower-diagonal entry:
240  m_L_ks = - Rminus - Rplus + 2.0 * mu_w * A_l;
241  // diagonal entry
242  m_D_ks = 1.0 + Rminus + Rplus + 2.0 * mu_w * A_d;
243  // upper-diagonal entry (not used)
244  m_U_ks = 0.0;
245  // m_Enth[m_ks] (below) is there due to the fully-implicit discretization in time, the second term is
246  // the modification of the right-hand side implementing the Neumann B.C. (similar to
247  // set_basal_heat_flux(); see that method for details)
248  m_B_ks = m_Enth[m_ks] + 2.0 * G * m_dz * (Rplus + mu_w * A_b);
249 
250  // treat horizontal velocity using first-order upwinding:
251  double upwind_u = 0.0;
252  double upwind_v = 0.0;
253  if (include_horizontal_advection) {
254  upwind_u = upwind(m_u[m_ks], m_E_w[m_ks], m_Enth[m_ks], m_E_e[m_ks], 1.0 / m_dx);
255  upwind_v = upwind(m_v[m_ks], m_E_s[m_ks], m_Enth[m_ks], m_E_n[m_ks], 1.0 / m_dy);
256  }
257  double Sigma = 0.0;
258  if (include_strain_heating) {
259  Sigma = m_strain_heating[m_ks];
260  }
261 
262  m_B_ks += m_dt * ((Sigma / m_ice_density) - upwind_u - upwind_v); // = rhs[m_ks]
263 }

294 //! Set coefficients in discrete equation for Neumann condition at base of ice.
295 /*!
296 This method generates the Neumann boundary condition for the linear system.
297 
298 The Neumann boundary condition is
299  @f[ \frac{\partial E}{\partial z} = - \frac{\phi}{K} @f]
300 where \f$\phi\f$ is the heat flux. Here \f$K\f$ is allowed to vary, and takes
301 its value from the value computed in assemble_R().
302 
303 The boundary condition is combined with the partial differential equation by the
304 technique of introducing an imaginary point at \f$z=-\Delta z\f$ and then
305 eliminating it.
306 
307 In other words, we combine the centered finite difference approximation
308 @f[ \frac{ E_{1} - E_{-1} }{2\dz} = -\frac{\phi}{K} @f]
309 with
310 
311 @f[ -R_{k-\frac12} E_{k-1} + \left( 1 + R_{k-\frac12} + R_{k+\frac12} \right) E_{k} - R_{k+\frac12} E_{k+1} + \text{advective terms} = \dots @f]
312 
313 to get
314 
315 @f{align*}{
316  \frac{E_{1}-E_{-1}}{2\,\Delta z} & = -\frac{\phi}{K_{0}}, \\
317  E_{1}-E_{-1} & = -\frac{2\,\Delta z\,\phi}{K_{0}}, \\
318  E_{-1}\,R_{-\frac12}-R_{-\frac12}\,E_{1} & = \frac{2\,R_{-\frac12}\,\Delta z\,\phi}{K_{0}}, \\
319  -R_{\frac12}\,E_{1}+E_{0}\,\left(R_{\frac12}+R_{-\frac12}+1\right)-E_{-1}\,R_{-\frac12} + \text{advective terms} & = \dots, \\
320  \left(-R_{\frac12}-R_{-\frac12}\right)\,E_{1}+E_{0}\,\left(R_{\frac12}+R_{-\frac12}+1\right) + \text{advective terms} & = \frac{2\,R_{-\frac12}\,\Delta z\,\phi}{K_{0}}+\dots.
321 @f}
322 
323 The error in the pure conductive and smooth conductivity case is @f$ O(\dz^2) @f$.
324 
325 This method should only be called if everything but the basal boundary condition
326 is already set.
327 
328 @param[in] heat_flux prescribed heat flux (positive means flux into the ice)
329 
330  */
331 void enthSystemCtx::set_basal_heat_flux(double heat_flux) {
332  using std::pow;
333  // extract K from R[0], so this code works even if K=K(T)
334  // recall: R = (ice_K / ice_density) * dt / PetscSqr(dz)
335  const double
336  K = (m_ice_density * pow(m_dz, 2) * m_R[0]) / m_dt,
337  G = - heat_flux / K;
338 
339  this->set_basal_neumann_bc(G);
340 }

342 //! Set enthalpy flux at the base.
343 /*! This method should probably be used for debugging only. Its purpose is to allow setting the
344  enthalpy flux even if K == 0, i.e. in a "pure advection" setup.
345  */
346 void enthSystemCtx::set_basal_neumann_bc(double G) {
347  const bool include_horizontal_advection = not (m_marginal and m_exclude_horizontal_advection);
348  const bool include_strain_heating = not (m_marginal and m_exclude_strain_heat);
349 
350  const double
351  Rminus = m_R[0], // R_{-1/2}
352  Rplus = 0.5 * (m_R[0] + m_R[1]), // R_{+1/2}
353  mu_w = 0.5 * m_nu * m_w[0];
354 
355  const double A_d = m_w[0] < 0.0 ? m_lambda - 1.0 : 1.0 - m_lambda;
356  const double A_u = m_w[0] < 0.0 ? 1.0 - m_lambda : m_lambda - 1.0;
357  const double A_b = m_w[0] < 0.0 ? -m_lambda : m_lambda - 2.0;
358 
359  // diagonal entry
360  m_D0 = 1.0 + Rminus + Rplus + 2.0 * mu_w * A_d;
361  // upper-diagonal entry
362  m_U0 = - Rminus - Rplus + 2.0 * mu_w * A_u;
363  // right-hand side, excluding the strain heating term and the horizontal advection
364  m_B0 = m_Enth[0] + 2.0 * G * m_dz * (-Rminus + mu_w * A_b);
365 
366  // treat horizontal velocity using first-order upwinding:
367  double upwind_u = 0.0;
368  double upwind_v = 0.0;
369  if (include_horizontal_advection) {
370  upwind_u = upwind(m_u[0], m_E_w[0], m_Enth[0], m_E_e[0], 1.0 / m_dx);
371  upwind_v = upwind(m_v[0], m_E_s[0], m_Enth[0], m_E_n[0], 1.0 / m_dy);
372  }
373  double Sigma = 0.0;
374  if (include_strain_heating) {
375  Sigma = m_strain_heating[0];
376  }
377 
378  m_B0 += m_dt * ((Sigma / m_ice_density) - upwind_u - upwind_v); // = rhs[m_ks]
379 }

382 //! \brief Assemble the R array. The R value switches at the CTS.
383 /*! In a simple abstract diffusion
384  \f[ \frac{\partial u}{\partial t} = D \frac{\partial^2 u}{\partial z^2}, \f]
385 with time steps \f$\Delta t\f$ and spatial steps \f$\Delta z\f$ we define
386  \f[ R = \frac{D \Delta t}{\Delta z^2}. \f]
387 This is used in an implicit method to write each line in the linear system, for
388 example [\ref MortonMayers]:
389  \f[ -R U_{j-1}^{n+1} + (1+2R) U_j^{n+1} - R U_{j+1}^{n+1} = U_j^n. \f]
390 
391 In the case of conservation of energy [\ref AschwandenBuelerKhroulevBlatter],
392  \f[ u=E \qquad \text{ and } \qquad D = \frac{K}{\rho} \qquad \text{ and } \qquad K = \frac{k}{c}. \f]
393 Thus
394  \f[ R = \frac{k \Delta t}{\rho c \Delta z^2}. \f]
395  */
396 void enthSystemCtx::assemble_R() {
397  if (not m_k_depends_on_T) {
398 
399  for (unsigned int k = 1; k <= m_ks; k++) {
400  m_R[k] = (m_Enth[k] < m_Enth_s[k]) ? m_R_cold : m_R_temp;
401  }
402  //still the cold ice value, if no temperate layer above
403  m_R[0] = (m_Enth[1] < m_Enth_s[1]) ? m_R_cold : m_R_temp;
404 
405  } else {
406 
407  for (unsigned int k = 1; k <= m_ks; k++) {
408  if (m_Enth[k] < m_Enth_s[k]) {
409  // cold case
410  const double depth = m_ice_thickness - k * m_dz;
411  double T = m_EC->temperature(m_Enth[k],
412  m_EC->pressure(depth)); // FIXME: issue #15
413 
414  m_R[k] = ((m_k_depends_on_T ? k_from_T(T) : m_ice_k) / m_EC->c()) * m_R_factor;
415  } else {
416  // temperate case
417  m_R[k] = m_R_temp;
418  }
419  }
420  // still the cold ice value, if no temperate layer above
421  if (m_Enth[1] < m_Enth_s[1]) {
422  double T = m_EC->temperature(m_Enth[0],
423  m_EC->pressure(m_ice_thickness)); // FIXME: issue #15
424  m_R[0] = ((m_k_depends_on_T ? k_from_T(T) : m_ice_k) / m_EC->c()) * m_R_factor;
425  } else {
426  // temperate layer case
427  m_R[0] = m_R_temp;
428  }
429 
430  }
431 
432  // R[k] for k > m_ks are never used
433 #if (Pism_DEBUG==1)
434  for (unsigned int k = m_ks + 1; k < m_R.size(); ++k) {
435  m_R[k] = GSL_NAN;
436  }
437 #endif
438 }

441 /*! \brief Solve the tridiagonal system, in a single column, which
442  * determines the new values of the ice enthalpy.
443  *
444  * We are solving a convection-diffusion equation, treating the @f$ z @f$ direction implicitly and
445  * @f$ x, y @f$ directions explicitly. See @ref bombproofenth for the documentation of the treatment
446  * of convection terms. The notes below document the way we treat diffusion using a simplified PDE.
447  *
448  * To discretize
449  * @f[ \diff{}{z} \left( K(E) \diff{E}{z}\right) = \diff{E}{t} @f]
450  *
451  * at a location @f$ k @f$ of the vertical grid, we use centered finite differences in space,
452  * backward differences in time, and evaluate @f$ K(E) @f$ at staggered-grid locations:
453  *
454  * @f[ \frac{K_{k+\frac12}\frac{E_{k+1} - E_{k}}{\dz} - K_{k-\frac12}\frac{E_{k} - E_{k-1}}{\dz}}{\dz} = \frac{E_{k} - E^{n-1}_{k}}{\dt}, @f]
455  *
456  * where @f$ E = E^{n} @f$ for clarity and @f$ K_{k\pm \frac12} = K(E^{n-1}_{k\pm \frac12}) @f$,
457  * %i.e. we linearize this PDE by evaluating @f$ K(E) @f$ at the _previous_ time-step.
458  *
459  * We define
460  *
461  * @f[ R_i = \frac{\dt\, K_i}{\dz^2}, @f]
462  *
463  * and the discretization takes form
464  *
465  * @f[ -R_{k-\frac12} E_{k-1} + \left( 1 + R_{k-\frac12} + R_{k+\frac12} \right) E_{k} - R_{k+\frac12} E_{k+1} = E^{n-1}_{k}. @f]
466  *
467  * In the assembly of the tridiagonal system this corresponds to
468  *
469  * @f{align*}{
470  * L_i &= - \frac12 (R_{i} + R_{i-1}),\\
471  * D_i &= 1 + \frac12 (R_{i} + R_{i-1}) + \frac12 (R_{i} + R_{i+1}),\\
472  * U_i &= - \frac12 (R_{i} + R_{i+1}),\\
473  * b_i &= E^{n-1}_{i},
474  * @f}
475  *
476  * where @f$ L_i, D_i, U_i @f$ are lower-diagonal, diagonal, and upper-diagonal entries
477  * corresponding to an equation @f$ i @f$ and @f$ b_i @f$ is the corresponding right-hand side.
478  * (Staggered-grid values are approximated by interpolating from the regular grid).
479  *
480  * This method is _unconditionally stable_ and has a maximum principle (see [@ref MortonMayers,
481  * section 2.11]).
482  */
483 void enthSystemCtx::solve(std::vector<double> &x) {
484 
485  TridiagonalSystem &S = *m_solver;
486 
487 #if (Pism_DEBUG==1)
488  checkReadyToSolve();
489  if (gsl_isnan(m_D0) || gsl_isnan(m_U0) || gsl_isnan(m_B0)) {
490  throw RuntimeError(PISM_ERROR_LOCATION,
491  "solveThisColumn() should only be called after\n"
492  " setting basal boundary condition in enthSystemCtx");
493  }
494 #endif
495 
496  // k=0 equation is already established
497  // L[0] = 0.0; // not used
498  S.D(0) = m_D0;
499  S.U(0) = m_U0;
500  S.RHS(0) = m_B0;
501 
502  const double
503  one_over_rho = 1.0 / m_ice_density,
504  Dx = 1.0 / m_dx,
505  Dy = 1.0 / m_dy;
506 
507  const bool include_horizontal_advection = not (m_marginal and m_exclude_horizontal_advection);
508  const bool include_strain_heating = not (m_marginal and m_exclude_strain_heat);
509 
510  // generic ice segment in k location (if any; only runs if m_ks >= 2)
511  for (unsigned int k = 1; k < m_ks; k++) {
512  const double
513  Rminus = 0.5 * (m_R[k-1] + m_R[k]), // R_{k-1/2}
514  Rplus = 0.5 * (m_R[k] + m_R[k+1]), // R_{k+1/2}
515  nu_w = m_nu * m_w[k];
516 
517  const double
518  A_l = m_w[k] >= 0.0 ? 0.5 * m_lambda - 1.0 : -0.5 * m_lambda,
519  A_d = m_w[k] >= 0.0 ? 1.0 - m_lambda : m_lambda - 1.0,
520  A_u = m_w[k] >= 0.0 ? 0.5 * m_lambda : 1.0 - 0.5 * m_lambda;
521 
522  S.L(k) = - Rminus + nu_w * A_l;
523  S.D(k) = 1.0 + Rminus + Rplus + nu_w * A_d;
524  S.U(k) = - Rplus + nu_w * A_u;
525 
526  // horizontal velocity and strain heating
527  double upwind_u = 0.0;
528  double upwind_v = 0.0;
529  if (include_horizontal_advection) {
530  upwind_u = upwind(m_u[k], m_E_w[k], m_Enth[k], m_E_e[k], Dx);
531  upwind_v = upwind(m_v[k], m_E_s[k], m_Enth[k], m_E_n[k], Dy);
532  }
533  double Sigma = 0.0;
534  if (include_strain_heating) {
535  Sigma = m_strain_heating[k];
536  }
537 
538  S.RHS(k) = m_Enth[k] + m_dt * (one_over_rho * Sigma - upwind_u - upwind_v);
539  }
540 
541  // Assemble the top surface equation. Values m_{L,D,U,B}_ks are set using set_surface_dirichlet()
542  // or set_surface_heat_flux().
543  if (m_ks > 0) {
544  S.L(m_ks) = m_L_ks;
545  }
546  S.D(m_ks) = m_D_ks;
547  if (m_ks < m_z.size() - 1) {
548  S.U(m_ks) = m_U_ks;
549  }
550  S.RHS(m_ks) = m_B_ks;
551 
552  // Solve it; note drainage is not addressed yet and post-processing may occur
553  try {
554  S.solve(m_ks + 1, x);
555  }
556  catch (RuntimeError &e) {
557  e.add_context("solving the tri-diagonal system (enthSystemCtx) at (%d,%d)\n"
558  "saving system to m-file... ", m_i, m_j);
559  reportColumnZeroPivotErrorMFile(m_ks + 1);
560  throw;
561  }
562 
563  // air above
564  for (unsigned int k = m_ks+1; k < x.size(); k++) {
565  x[k] = m_B_ks;
566  }
567 
568 #if (Pism_DEBUG==1)
569  // if success, mark column as done by making scheme params and b.c. coeffs invalid
570  m_lambda = -1.0;
571  m_D0 = GSL_NAN;
572  m_U0 = GSL_NAN;
573  m_B0 = GSL_NAN;
574  m_L_ks = GSL_NAN;
575  m_D_ks = GSL_NAN;
576  m_U_ks = GSL_NAN;
577  m_B_ks = GSL_NAN;
578 #endif
579 }
