27 //! Determine the normalization constant for the functional.
28 /*! Sets the normalization constant \f$c_N\f$ so that
29 \f[
30 J(x)=1
31 \f]
32 if \f$|x| = \mathtt{scale}\f$ everywhere.
33 */
34 void IPLogRelativeFunctional::normalize(double scale) {
35 
36  // The local value of the weights
37  double value = 0;
38 
39  double scale_sq = scale*scale;
40 
41  double w = 1.;
42 
43  IceModelVec::AccessList list(m_u_observed);
44 
45  if (m_weights) {
46  list.add(*m_weights);
47  }
48 
49  for (Points p(*m_grid); p; p.next()) {
50  const int i = p.i(), j = p.j();
51 
52  Vector2 &u_obs_ij = m_u_observed(i, j);
53  if (m_weights) {
54  w = (*m_weights)(i, j);
55  }
56  double obsMagSq = (u_obs_ij.u*u_obs_ij.u + u_obs_ij.v*u_obs_ij.v) + m_eps*m_eps;
57  value += log(1 + w*scale_sq/obsMagSq);
58  }
59 
60  m_normalization = GlobalSum(m_grid->com, value);
61 }

63 void IPLogRelativeFunctional::valueAt(IceModelVec2V &x, double *OUTPUT) {
64

76  const int i = p.i(), j = p.j();
77 
78  Vector2 &x_ij = x(i, j);
79  Vector2 &u_obs_ij = m_u_observed(i, j);
80  if (m_weights) {

82  }
83  double obsMagSq = (u_obs_ij.u*u_obs_ij.u + u_obs_ij.v*u_obs_ij.v) + m_eps*m_eps;
84  value += log(1 + w*(x_ij.u*x_ij.u + x_ij.v*x_ij.v)/obsMagSq);

92 void IPLogRelativeFunctional::gradientAt(IceModelVec2V &x, IceModelVec2V &gradient) {
93  gradient.set(0);
94 
95  double w = 1;
96 
97  IceModelVec::AccessList list{&x, &gradient, &m_u_observed};
98  if (m_weights) {
99  list.add(*m_weights);
100  }
101 
102  for (Points p(*m_grid); p; p.next()) {
103  const int i = p.i(), j = p.j();
104 
105  Vector2 &x_ij = x(i, j);
106  Vector2 &u_obs_ij = m_u_observed(i, j);
107  if (m_weights) {
108  w = (*m_weights)(i, j);
109  }
110  double obsMagSq = u_obs_ij.u*u_obs_ij.u + u_obs_ij.v*u_obs_ij.v + m_eps*m_eps;
111  double dJdxsq = w/(obsMagSq + w*(x_ij.u*x_ij.u + x_ij.v*x_ij.v));
112 
113  gradient(i, j).u = dJdxsq*2*x_ij.u/m_normalization;
114  gradient(i, j).v = dJdxsq*2*x_ij.v/m_normalization;
115  }
