109  m_element.nodal_values(x.array(), x_e);

113  }
114  m_element.evaluate(x_e, x_q, dxdx_q, dxdy_q);
115 
116  // Zero out the element-local residual in preparation for updating it.
117  for (unsigned int k = 0; k < Nk; k++) {
118  gradient_e[k] = 0;
119  }
120

122  auto W = m_element.weight(q);
123  const double &dxdx_qq = dxdx_q[q], &dxdy_qq = dxdy_q[q];
124  for (unsigned int k = 0; k < Nk; k++) {

126  *(dxdx_qq*m_element.chi(q, k).dx + dxdy_qq*m_element.chi(q, k).dy);
127  } // k
128  } // q
