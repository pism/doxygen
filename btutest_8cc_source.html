45 //! Allocate the PISMV (verification) context. Uses ColdEnthalpyConverter.
46 std::shared_ptr<pism::Context> btutest_context(MPI_Comm com, const std::string &prefix) {
47  using namespace pism;
48 
49  // unit system
50  units::System::Ptr sys(new units::System);
51 
52  // logger
53  Logger::Ptr logger = logger_from_options(com);
54 
55  // configuration parameters
56  Config::Ptr config = config_from_options(com, *logger, sys);
57 
58  // default vertical grid parameters
59  config->set_number("grid.Mbz", 11);
60  config->set_number("grid.Lbz", 1000);
61 
62  // when IceGrid constructor is called, these settings are used
63  config->set_string("time.start", "0s");
64  config->set_number("time.run_length", 1.0);
65 
66  set_config_from_options(sys, *config);
67 
68  print_config(*logger, 3, *config);
69 
70  Time::Ptr time = std::make_shared<Time>(com, config, *logger, sys);
71 
72  EnthalpyConverter::Ptr EC = EnthalpyConverter::Ptr(new ColdEnthalpyConverter(*config));
73 
74  return std::shared_ptr<Context>(new Context(com, sys, config, EC, time, logger, prefix));
75 }

77 int main(int argc, char *argv[]) {
78 
79  using namespace pism;
80 
81  MPI_Comm com = MPI_COMM_WORLD;
82  petsc::Initializer petsc(argc, argv, help);
83 
84  com = MPI_COMM_WORLD;
85 
86  try {
87  std::shared_ptr<Context> ctx = btutest_context(com, "btutest");
88  Logger::Ptr log = ctx->log();
89 
90  std::string usage =
91  " btutest -Mbz NN -Lbz 1000.0 [-o OUT.nc -ys A -ye B -dt C -Mz D -Lz E]\n"
92  "where these are required because they are used in BedThermalUnit:\n"
93  " -Mbz number of bedrock thermal layer levels to use\n"
94  " -Lbz 1000.0 depth of bedrock thermal layer (required; Lbz=1000.0 m in Test K)\n"
95  "and these are allowed:\n"
96  " -o output file name; NetCDF format\n"
97  " -ys start year in using Test K\n"
98  " -ye end year in using Test K\n"
99  " -dt time step B (= positive float) in years\n";
100 
101  bool done = show_usage_check_req_opts(*log, "BTUTEST %s (test program for BedThermalUnit)",
102  {"-Mbz"}, usage);
103  if (done) {
104  return 0;
105  }
106 
107  log->message(2,
108  " initializing IceGrid from options ...\n");
109 
110  Config::Ptr config = ctx->config();
111 
112  GridParameters P(config);
113  P.Mx = 3;
114  P.My = P.Mx;
115  P.Lx = 1500e3;
116  P.Ly = P.Lx;
117 
118  P.vertical_grid_from_options(config);
119  P.ownership_ranges_from_options(ctx->size());
120 
121  // create grid and set defaults
122  IceGrid::Ptr grid(new IceGrid(ctx, P));
123 
124  auto outname = config->get_string("output.file_name");
125 
126  options::Real dt_years(ctx->unit_system(),
127  "-dt", "Time-step, in years", "years", 1.0);
128 
129  // allocate tools and IceModelVecs
130  IceModelVec2S bedtoptemp(grid, "bedtoptemp", WITHOUT_GHOSTS);
131  bedtoptemp.set_attrs("",
132  "temperature at top of bedrock thermal layer",
133  "K", "K", "", 0);
134 
135  IceModelVec2S heat_flux_at_ice_base(grid, "upward_heat_flux_at_ice_base", WITHOUT_GHOSTS);
136  heat_flux_at_ice_base.set_attrs("",
137  "upward geothermal flux at bedrock thermal layer base",
138  "W m-2", "mW m-2", "", 0);
139 
140  // initialize BTU object:
141  energy::BTUGrid bedrock_grid = energy::BTUGrid::FromOptions(ctx);
142 
143  energy::BedThermalUnit::Ptr btu;
144 
145  if (bedrock_grid.Mbz > 1) {
146  btu.reset(new energy::BTU_Verification(grid, bedrock_grid, 'K', false));
147  } else {
148  btu.reset(new energy::BTU_Minimal(grid));
149  }
150 
151  InputOptions opts = process_input_options(com, config);
152  btu->init(opts);
153 
154  double dt_seconds = units::convert(ctx->unit_system(), dt_years, "years", "seconds");
155 
156  // worry about time step
157  int N = (int)ceil((ctx->time()->end() - ctx->time()->start()) / dt_seconds);
158  dt_seconds = (ctx->time()->end() - ctx->time()->start()) / (double)N;
159  log->message(2,
160  " user set timestep of %.4f years ...\n"
161  " reset to %.4f years to get integer number of steps ... \n",
162  dt_years.value(), units::convert(ctx->unit_system(), dt_seconds, "seconds", "years"));
163  MaxTimestep max_dt = btu->max_timestep(0.0);
164  log->message(2,
165  " BedThermalUnit reports max timestep of %.4f years ...\n",
166  units::convert(ctx->unit_system(), max_dt.value(), "seconds", "years"));
167 
168  // actually do the time-stepping
169  log->message(2," running ...\n");
170  for (int n = 0; n < N; n++) {
171  // time at start of time-step
172  const double time = ctx->time()->start() + dt_seconds * (double)n;
173 
174  // compute exact ice temperature at z=0 at time y
175  {
176  IceModelVec::AccessList list(bedtoptemp);
177  for (Points p(*grid); p; p.next()) {
178  const int i = p.i(), j = p.j();
179 
180  bedtoptemp(i,j) = exactK(time, 0.0, 0).T;
181  }
182  }
183  // no need to update ghost values
184 
185  // update the temperature inside the thermal layer using bedtoptemp
186  btu->update(bedtoptemp, time, dt_seconds);
187  log->message(2,".");
188  }
189 
190  log->message(2, "\n done ...\n");
191 
192  // compute final output heat flux G_0 at z=0
193  heat_flux_at_ice_base.copy_from(btu->flux_through_top_surface());
194 
195  auto time = ctx->time();
196 
197  // get, and tell stdout, the correct answer from Test K
198  const double FF = exactK(time->end(), 0.0, 0).F;
199  log->message(2,
200  " exact Test K reports upward heat flux at z=0, at end time %s, as G_0 = %.7f W m-2;\n",
201  time->date(time->end()).c_str(), FF);
202 
203  // compute numerical error
204  heat_flux_at_ice_base.shift(-FF);
205  double max_error = heat_flux_at_ice_base.norm(NORM_INFINITY)[0];
206  double avg_error = heat_flux_at_ice_base.norm(NORM_1)[0];
207  heat_flux_at_ice_base.shift(+FF); // shift it back for writing
208  avg_error /= (grid->Mx() * grid->My());
209  log->message(2,
210  "case dt = %.5f:\n", dt_years.value());
211  log->message(1,
212  "NUMERICAL ERRORS in upward heat flux at z=0 relative to exact solution:\n");
213  log->message(1,
214  "bheatflx0 : max prcntmax av\n");
215  log->message(1,
216  " %11.7f %11.7f %11.7f\n",
217  max_error, 100.0*max_error/FF, avg_error);
218  log->message(1, "NUM ERRORS DONE\n");
219 
220  File file(grid->com,
221  outname,
222  string_to_backend(config->get_string("output.format")),
223  PISM_READWRITE_MOVE,
224  ctx->pio_iosys_id());
225 
226  io::define_time(file, *ctx);
227  io::append_time(file, *ctx->config(), ctx->time()->current());
228 
229  btu->write_model_state(file);
230 
231  bedtoptemp.write(file);
232  heat_flux_at_ice_base.write(file);
233 
234  log->message(2, "done.\n");
235  }
236  catch (...) {
237  handle_fatal_errors(com);
238  return 1;
239  }
240 
241  return 0;
242 }
