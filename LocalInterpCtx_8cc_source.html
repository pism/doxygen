37 //! Compute start and count for getting a subset of x.
38 /*! Given a grid `x` we find `x_start` and `x_count` so that `[subset_x_min, subset_x_max]` is in
39  * `[x[x_start], x[x_start + x_count]]`.
40  *
41  * `x_start` and `x_count` define the smallest subset of `x` with this property.
42  *
43  * Note that `x_start + x_count <= x.size()` as long as `x` is strictly increasing.
44  *
45  * @param[in] x input grid (defined interpolation domain)
46  * @param[in] subset_x_min minimum `x` of a subset (interpolation range)
47  * @param[in] subset_x_max maxumum `x` of a subset (interpolation range)
48  * @param[out] x_start starting index
49  * @param[out] x_count number of elements required
50  */
51 static void subset_start_and_count(const std::vector<double> &x,
52  double subset_x_min, double subset_x_max,
53  unsigned int &x_start, unsigned int &x_count) {
54  unsigned int x_size = x.size();
55 
56  x_start = gsl_interp_bsearch(&x[0], subset_x_min, 0, x_size - 1);
57 
58  unsigned int x_end = gsl_interp_bsearch(&x[0], subset_x_max, 0, x_size - 1) + 1;
59 
60  x_end = std::min(x_size - 1, x_end);
61 
62  x_count = x_end - x_start + 1;
63 }

65 //! Construct a local interpolation context.
66 /*!
67  The essential quantities to compute are where each processor should start within the NetCDF file grid
68  (`start[]`) and how many grid points, from the starting place, the processor has. The resulting
69  portion of the grid is stored in array `a` (a field of the `LocalInterpCtx`).
70 
71  We make conservative choices about `start[]` and `count[]`. In particular, the portions owned by
72  processors \e must overlap at one point in the NetCDF file grid, but they \e may overlap more than that
73  (as computed here).
74 
75  Note this constructor doesn't extract new information from the NetCDF file or
76  do communication. The information from the NetCDF file must already be
77  extracted, validly stored in a grid_info structure `input`.
78 
79  The `IceGrid` is used to determine what ranges of the target arrays (i.e. \c
80  Vecs into which NetCDF information will be interpolated) are owned by each
81  processor.
82 */
83 LocalInterpCtx::LocalInterpCtx(const grid_info &input, const IceGrid &grid,
84  const std::vector<double> &z_output,
85  InterpolationType type) {
86  const int T = 0, X = 1, Y = 2, Z = 3; // indices, just for clarity
87 
88  grid.ctx()->log()->message(4, "\nRegridding file grid info:\n");
89  input.report(*grid.ctx()->log(), 4, grid.ctx()->unit_system());
90 
91  // limits of the processor's part of the target computational domain
92  const double
93  x_min_proc = grid.x(grid.xs()),
94  x_max_proc = grid.x(grid.xs() + grid.xm() - 1),
95  y_min_proc = grid.y(grid.ys()),
96  y_max_proc = grid.y(grid.ys() + grid.ym() - 1);
97 
98  // T
99  start[T] = input.t_len - 1; // use the latest time.
100  count[T] = 1; // read only one record
101 
102  // X
103  subset_start_and_count(input.x, x_min_proc, x_max_proc, start[X], count[X]);
104 
105  // Y
106  subset_start_and_count(input.y, y_min_proc, y_max_proc, start[Y], count[Y]);
107 
108  // Z
109  start[Z] = 0; // always start at the base
110  count[Z] = std::max(input.z_len, 1u); // read at least one level
111 
112  // We need a buffer for the local data, but node 0 needs to have as much
113  // storage as the node with the largest block (which may be anywhere), hence
114  // we perform a reduce so that node 0 has the maximum value.
115  unsigned int buffer_size = count[X] * count[Y] * std::max(count[Z], 1u);
116  unsigned int proc0_buffer_size = buffer_size;
117  MPI_Reduce(&buffer_size, &proc0_buffer_size, 1, MPI_UNSIGNED, MPI_MAX, 0, grid.com);
118 
119  ParallelSection allocation(grid.com);
120  try {
121  if (grid.rank() == 0) {
122  buffer.resize(proc0_buffer_size);
123  } else {
124  buffer.resize(buffer_size);
125  }
126  } catch (...) {
127  allocation.failed();
128  }
129  allocation.check();
130 
131  if (type == LINEAR or type == NEAREST) {
132  x.reset(new Interpolation(type, &input.x[start[X]], count[X],
133  &grid.x()[grid.xs()], grid.xm()));
134 
135  y.reset(new Interpolation(type, &input.y[start[Y]], count[Y],
136  &grid.y()[grid.ys()], grid.ym()));
137 
138  z.reset(new Interpolation(type, input.z, z_output));
139  } else {
140  throw RuntimeError(PISM_ERROR_LOCATION, "invalid interpolation type in LocalInterpCtx");
141  }
142 }
