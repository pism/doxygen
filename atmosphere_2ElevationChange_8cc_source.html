103 void ElevationChange::update_impl(const Geometry &geometry, double t, double dt) {
104 
105  m_input_model->update(geometry, t, dt);
106 
107  m_reference_surface->update(t, dt);
108  m_reference_surface->interp(t + 0.5*dt);
109 
110  // make a copy of the surface elevation so that it is available in methods computing
111  // temperature and precipitation time series
112  m_surface.copy_from(geometry.ice_surface_elevation);
113 
114  const auto &reference_surface = *m_reference_surface;
115 
116  // temperature
117  {
118  m_temperature->copy_from(m_input_model->mean_annual_temp());
119 
120  lapse_rate_correction(m_surface, reference_surface,
121  m_temp_lapse_rate, *m_temperature);
122  }
123 
124  // precipitation
125  {
126  m_precipitation->copy_from(m_input_model->mean_precipitation());
127 
128  switch (m_precip_method) {
129  case SCALE:
130  {
131  IceModelVec::AccessList list{&m_surface, &reference_surface, m_precipitation.get()};
132 
133  for (Points p(*m_grid); p; p.next()) {
134  const int i = p.i(), j = p.j();
135 
136  double dT = -m_precip_temp_lapse_rate * (m_surface(i, j) - reference_surface(i, j));
137 
138  (*m_precipitation)(i, j) *= std::exp(m_precip_exp_factor * dT);
139  }
140 
141  }
142  break;
143  case SHIFT:
144  default:
145  {
146  lapse_rate_correction(m_surface, *m_reference_surface,
147  m_precip_lapse_rate, *m_precipitation);
148  }
149  break;
150  }
151  }
152 }

182 void ElevationChange::temp_time_series_impl(int i, int j, std::vector<double> &result) const {
183  std::vector<double> reference_surface(m_ts_times.size());
184 
185  m_input_model->temp_time_series(i, j, result);

187  m_reference_surface->interp(i, j, reference_surface);

190  result[m] -= m_temp_lapse_rate * (m_surface(i, j) - reference_surface[m]);

194 void ElevationChange::precip_time_series_impl(int i, int j, std::vector<double> &result) const {
195  auto N = m_ts_times.size();

197 
198  m_input_model->precip_time_series(i, j, result);
199 
200  m_reference_surface->interp(i, j, reference_surface);
201 
202  switch (m_precip_method) {
203  case SCALE:
204  {
205  for (unsigned int m = 0; m < N; ++m) {
206  double dT = -m_precip_temp_lapse_rate * (m_surface(i, j) - reference_surface[m]);
207  result[m] *= std::exp(m_precip_exp_factor * dT);
208  }
209  }
210  break;
211  case SHIFT:
212  for (unsigned int m = 0; m < N; ++m) {
213  result[m] -= m_precip_lapse_rate * (m_surface(i, j) - reference_surface[m]);
214  }
215  break;
