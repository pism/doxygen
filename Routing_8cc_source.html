191 //! Compute the hydraulic potential.
192 /*!
193  Computes \f$\psi = P + \rho_w g (b + W)\f$.
194 */
195 void hydraulic_potential(const IceModelVec2S &W,
196  const IceModelVec2S &P,
197  const IceModelVec2S &sea_level,
198  const IceModelVec2S &bed,
199  const IceModelVec2S &ice_thickness,
200  IceModelVec2S &result) {
201 
202  IceGrid::ConstPtr grid = result.grid();
203 
204  Config::ConstPtr config = grid->ctx()->config();
205 
206  double
207  ice_density = config->get_number("constants.ice.density"),
208  sea_water_density = config->get_number("constants.sea_water.density"),
209  C = ice_density / sea_water_density,
210  rg = (config->get_number("constants.fresh_water.density") *
211  config->get_number("constants.standard_gravity"));
212 
213  IceModelVec::AccessList list{&P, &W, &sea_level, &ice_thickness, &bed, &result};
214 
215  for (Points p(*grid); p; p.next()) {
216  const int i = p.i(), j = p.j();
217 
218  double b = std::max(bed(i, j), sea_level(i, j) - C * ice_thickness(i, j));
219 
220  result(i, j) = P(i, j) + rg * (b + W(i, j));
221  }
222 }

395 //! Average the regular grid water thickness to values at the center of cell edges.
396 /*! Uses mask values to avoid averaging using water thickness values from
397  either ice-free or floating areas. */
398 void Routing::water_thickness_staggered(const IceModelVec2S &W,
399  const IceModelVec2CellType &mask,
400  IceModelVec2Stag &result) {
401 
402  bool include_floating = m_config->get_flag("hydrology.routing.include_floating_ice");
403 
404  IceModelVec::AccessList list{ &mask, &W, &result };
405 
406  for (Points p(*m_grid); p; p.next()) {
407  const int i = p.i(), j = p.j();
408 
409  if (include_floating) {
410  // east
411  if (mask.icy(i, j)) {
412  result(i, j, 0) = mask.icy(i + 1, j) ? 0.5 * (W(i, j) + W(i + 1, j)) : W(i, j);
413  } else {
414  result(i, j, 0) = mask.icy(i + 1, j) ? W(i + 1, j) : 0.0;
415  }
416  // north
417  if (mask.icy(i, j)) {
418  result(i, j, 1) = mask.icy(i, j + 1) ? 0.5 * (W(i, j) + W(i, j + 1)) : W(i, j);
419  } else {
420  result(i, j, 1) = mask.icy(i, j + 1) ? W(i, j + 1) : 0.0;
421  }
422  } else {
423  // east
424  if (mask.grounded_ice(i, j)) {
425  result(i, j, 0) = mask.grounded_ice(i + 1, j) ? 0.5 * (W(i, j) + W(i + 1, j)) : W(i, j);
426  } else {
427  result(i, j, 0) = mask.grounded_ice(i + 1, j) ? W(i + 1, j) : 0.0;
428  }
429  // north
430  if (mask.grounded_ice(i, j)) {
431  result(i, j, 1) = mask.grounded_ice(i, j + 1) ? 0.5 * (W(i, j) + W(i, j + 1)) : W(i, j);
432  } else {
433  result(i, j, 1) = mask.grounded_ice(i, j + 1) ? W(i, j + 1) : 0.0;
434  }
435  }
436  }
437 
438  result.update_ghosts();
439 }

442 //! Compute the nonlinear conductivity at the center of cell edges.
443 /*!
444  Computes
445 
446  \f[ K = K(W, \nabla P, \nabla b) = k W^{\alpha-1} |\nabla R|^{\beta-2} \f]
447 
448  on the staggered grid, where \f$R = P+\rho_w g b\f$. We denote
449 
450  \f[ \Pi = |\nabla R|^2 \f]
451 
452  internally; this is computed on a staggered grid by a Mahaffy-like ([@ref Mahaffy])
453  scheme. This requires \f$R\f$ to be defined on a box stencil of width 1.
454 
455  Also returns the maximum over all staggered points of \f$ K W \f$.
456 */
457 void Routing::compute_conductivity(const IceModelVec2Stag &W,
458  const IceModelVec2S &P,
459  const IceModelVec2S &bed_elevation,
460  IceModelVec2Stag &result,
461  double &KW_max) const {
462  const double
463  k = m_config->get_number("hydrology.hydraulic_conductivity"),
464  alpha = m_config->get_number("hydrology.thickness_power_in_flux"),
465  beta = m_config->get_number("hydrology.gradient_power_in_flux"),
466  betapow = (beta - 2.0) / 2.0;
467 
468  IceModelVec::AccessList list({&result, &W});
469 
470  KW_max = 0.0;
471 
472  if (beta != 2.0) {
473  // Put the squared norm of the gradient of the simplified hydrolic potential (Pi) in
474  // "result"
475  //
476  // FIXME: we don't need to re-compute this during every hydrology time step: the
477  // simplified hydrolic potential does not depend on the water amount and can be
478  // computed *once* in update_impl(), before entering the time-stepping loop
479  {
480  // R <-- P + rhow g b
481  P.add(m_rg, bed_elevation, m_R); // yes, it updates ghosts
482 
483  list.add(m_R);
484  for (Points p(*m_grid); p; p.next()) {
485  const int i = p.i(), j = p.j();
486 
487  double dRdx, dRdy;
488  dRdx = (m_R(i + 1, j) - m_R(i, j)) / m_dx;
489  dRdy = (m_R(i + 1, j + 1) + m_R(i, j + 1) - m_R(i + 1, j - 1) - m_R(i, j - 1)) / (4.0 * m_dy);
490  result(i, j, 0) = dRdx * dRdx + dRdy * dRdy;
491 
492  dRdx = (m_R(i + 1, j + 1) + m_R(i + 1, j) - m_R(i - 1, j + 1) - m_R(i - 1, j)) / (4.0 * m_dx);
493  dRdy = (m_R(i, j + 1) - m_R(i, j)) / m_dy;
494  result(i, j, 1) = dRdx * dRdx + dRdy * dRdy;
495  }
496  }
497 
498  // We regularize negative power |\grad psi|^{beta-2} by adding eps because large
499  // head gradient might be 10^7 Pa per 10^4 m or 10^3 Pa/m.
500  const double eps = beta < 2.0 ? 1.0 : 0.0;
501 
502  for (Points p(*m_grid); p; p.next()) {
503  const int i = p.i(), j = p.j();
504 
505  for (int o = 0; o < 2; ++o) {
506  const double Pi = result(i, j, o);
507 
508  // FIXME: same as Pi above: we don't need to re-compute this each time we make a
509  // short hydrology time step
510  double B = pow(Pi + eps * eps, betapow);
511 
512  result(i, j, o) = k * pow(W(i, j, o), alpha - 1.0) * B;
513 
514  KW_max = std::max(KW_max, result(i, j, o) * W(i, j, o));
515  }
516  }
517  } else {
518  for (Points p(*m_grid); p; p.next()) {
519  const int i = p.i(), j = p.j();
520 
521  for (int o = 0; o < 2; ++o) {
522  result(i, j, o) = k * pow(W(i, j, o), alpha - 1.0);
523 
524  KW_max = std::max(KW_max, result(i, j, o) * W(i, j, o));
525  }
526  }
527  }
528 
529  KW_max = GlobalMax(m_grid->com, KW_max);
530 
531  result.update_ghosts();
532 }

535 //! Compute the wall melt rate which comes from (turbulent) dissipation of flow energy.
536 /*!
537  This code fills `result` with
538  \f[ \frac{m_{wall}}{\rho_w} = - \frac{1}{L \rho_w} \mathbf{q} \cdot \nabla \psi = \left(\frac{k}{L \rho_w}\right) W^\alpha |\nabla R|^\beta \f]
539  where \f$R = P+\rho_w g b\f$.
540 
541  Note that conductivity_staggered() computes the related quantity
542  \f$K = k W^{\alpha-1} |\nabla R|^{\beta-2}\f$ on the staggered grid, but
543  contriving to reuse that code would be inefficient because of the
544  staggered-versus-regular change.
545 
546  At the current state of the code, this is a diagnostic calculation only.
547 */
548 void wall_melt(const Routing &model,
549  const IceModelVec2S &bed_elevation,
550  IceModelVec2S &result) {
551 
552  IceGrid::ConstPtr grid = result.grid();
553 
554  Config::ConstPtr config = grid->ctx()->config();
555 
556  const double
557  k = config->get_number("hydrology.hydraulic_conductivity"),
558  L = config->get_number("constants.fresh_water.latent_heat_of_fusion"),
559  alpha = config->get_number("hydrology.thickness_power_in_flux"),
560  beta = config->get_number("hydrology.gradient_power_in_flux"),
561  g = config->get_number("constants.standard_gravity"),
562  rhow = config->get_number("constants.fresh_water.density"),
563  rg = rhow * g,
564  CC = k / (L * rhow);
565 
566  // FIXME: could be scaled with overall factor hydrology_coefficient_wall_melt ?
567  if (alpha < 1.0) {
568  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
569  "alpha = %f < 1 which is not allowed", alpha);
570  }
571 
572  IceModelVec2S R(grid, "R", WITH_GHOSTS);
573 
574  // R <-- P + rhow g b
575  model.subglacial_water_pressure().add(rg, bed_elevation, R);
576  // yes, it updates ghosts
577 
578  IceModelVec2S W(grid, "W", WITH_GHOSTS);
579  W.copy_from(model.subglacial_water_thickness());
580 
581  IceModelVec::AccessList list{&R, &W, &result};
582 
583  double dx = grid->dx();
584  double dy = grid->dy();
585 
586  for (Points p(*grid); p; p.next()) {
587  const int i = p.i(), j = p.j();
588  double dRdx, dRdy;
589 
590  if (W(i, j) > 0.0) {
591  dRdx = 0.0;

613 //! Get the advection velocity V at the center of cell edges.
614 /*!
615  Computes the advection velocity @f$\mathbf{V}@f$ on the staggered
616  (edge-centered) grid. If V = (u, v) in components then we have
617  <code> result(i, j, 0) = u(i+1/2, j) </code> and
618  <code> result(i, j, 1) = v(i, j+1/2) </code>
619 
620  The advection velocity is given by the formula
621 
622  @f[ \mathbf{V} = - K \left(\nabla P + \rho_w g \nabla b\right) @f]
623 
624  where @f$\mathbf{V}@f$ is the water velocity, @f$P@f$ is the water
625  pressure, and @f$b@f$ is the bedrock elevation.
626 
627  If the corresponding staggered grid value of the water thickness is zero then that
628  component of V is set to zero. This does not change the flux value (which would be zero
629  anyway) but it does provide the correct max velocity in the CFL calculation. We assume
630  bed has valid ghosts.
631 */
632 void Routing::compute_velocity(const IceModelVec2Stag &W,
633  const IceModelVec2S &pressure,
634  const IceModelVec2S &bed,
635  const IceModelVec2Stag &K,
636  const IceModelVec2Int *no_model_mask,
637  IceModelVec2Stag &result) const {

639  P.copy_from(pressure); // yes, it updates ghosts
640 
641  IceModelVec::AccessList list{&P, &W, &K, &bed, &result};
642 
643  for (Points p(*m_grid); p; p.next()) {
644  const int i = p.i(), j = p.j();
645 
646  if (W(i, j, 0) > 0.0) {
647  double
648  P_x = (P(i + 1, j) - P(i, j)) / m_dx,
649  b_x = (bed(i + 1, j) - bed(i, j)) / m_dx;
650  result(i, j, 0) = - K(i, j, 0) * (P_x + m_rg * b_x);
651  } else {
652  result(i, j, 0) = 0.0;
653  }
654 
655  if (W(i, j, 1) > 0.0) {
656  double
657  P_y = (P(i, j + 1) - P(i, j)) / m_dy,
658  b_y = (bed(i, j + 1) - bed(i, j)) / m_dy;
659  result(i, j, 1) = - K(i, j, 1) * (P_y + m_rg * b_y);
660  } else {
661  result(i, j, 1) = 0.0;
662  }
663  }
664 
665  if (no_model_mask) {
666  list.add(*no_model_mask);
667 
668  for (Points p(*m_grid); p; p.next()) {
669  const int i = p.i(), j = p.j();
670 
671  auto M = no_model_mask->int_star(i, j);
672 
673  if (M.ij or M.e) {
674  result(i, j, 0) = 0.0;
675  }
676 
677  if (M.ij or M.n) {
678  result(i, j, 1) = 0.0;
679  }

733 /*!

735  \f[ \frac{\partial W_{till}}{\partial t} = \frac{m}{\rho_w} - C\f]
736

739  `hydrology_tillwat_max`. Here \f$m/\rho_w\f$ is `total_input`.

744

746  2. uses time steps determined by the rest of the hydrology::Routing model;
747  3. does not check mask because the enforce_bounds() call addresses that.

751 void Routing::update_Wtill(double dt,
752  const IceModelVec2S &Wtill,
753  const IceModelVec2S &surface_input_rate,
754  const IceModelVec2S &basal_melt_rate,
755  IceModelVec2S &Wtill_new) {
756  const double
757  tillwat_max = m_config->get_number("hydrology.tillwat_max"),

759 
760  IceModelVec::AccessList list{&Wtill, &Wtill_new, &basal_melt_rate};
761 
762  bool add_surface_input = m_config->get_flag("hydrology.add_water_input_to_till_storage");
763  if (add_surface_input) {
764  list.add(surface_input_rate);
765  }
766 
767  for (Points p(*m_grid); p; p.next()) {
768  const int i = p.i(), j = p.j();
769 
770  double input_rate = basal_melt_rate(i, j);
771  if (add_surface_input) {
772  input_rate += surface_input_rate(i, j);
773  }
774

780 void Routing::W_change_due_to_flow(double dt,
781  const IceModelVec2S &W,
782  const IceModelVec2Stag &Wstag,
783  const IceModelVec2Stag &K,
784  const IceModelVec2Stag &Q,
785  IceModelVec2S &result) {
786  const double
787  wux = 1.0 / (m_dx * m_dx),
788  wuy = 1.0 / (m_dy * m_dy);
789 
790  IceModelVec::AccessList list{&W, &Wstag, &K, &Q, &result};
791 
792  for (Points p(*m_grid); p; p.next()) {
793  const int i = p.i(), j = p.j();
794 
795  auto q = Q.star(i, j);
796  const double divQ = (q.e - q.w) / m_dx + (q.n - q.s) / m_dy;
797 
798  auto k = K.star(i, j);
799  auto ws = Wstag.star(i, j);
800 
801  const double
802  De = m_rg * k.e * ws.e,
803  Dw = m_rg * k.w * ws.w,
804  Dn = m_rg * k.n * ws.n,
805  Ds = m_rg * k.s * ws.s;
806 
807  auto w = W.star(i, j);
808  const double diffW = (wux * (De * (w.e - w.ij) - Dw * (w.ij - w.w)) +
809  wuy * (Dn * (w.n - w.ij) - Ds * (w.ij - w.s)));
810 
811  result(i, j) = dt * (- divQ + diffW);
812  }
813 }

847 //! Update the model state variables W and Wtill by applying the subglacial hydrology model equations.
848 /*!
849  Runs the hydrology model from time t to time t + dt. Here [t, dt]
850  is generally on the order of months to years. This hydrology model will take its
851  own shorter time steps, perhaps hours to weeks.
852 
853  To update W = `bwat` we call update_W(), and to update Wtill = `tillwat` we
854  call update_Wtill().
855 */
856 void Routing::update_impl(double t, double dt, const Inputs& inputs) {
857 
858  ice_bottom_surface(*inputs.geometry, m_bottom_surface);
859 
860  double
861  ht = t,
862  hdt = 0.0;
863 
864  const double
865  t_final = t + dt,
866  dt_max = m_config->get_number("hydrology.maximum_time_step", "seconds");
867 
868  m_Qstag_average.set(0.0);
869 
870  // make sure W has valid ghosts before starting hydrology steps
871  m_W.update_ghosts();
872 
873  unsigned int step_counter = 0;
874  for (; ht < t_final; ht += hdt) {
875  step_counter++;
876 
877 #if (Pism_DEBUG==1)
878  double huge_number = 1e6;
879  check_bounds(m_W, huge_number);
880 
881  check_bounds(m_Wtill, m_config->get_number("hydrology.tillwat_max"));
882 #endif
883 
884  // updates ghosts of m_Wstag
885  water_thickness_staggered(m_W,
886  inputs.geometry->cell_type,
887  m_Wstag);
888 
889  double maxKW = 0.0;
890  // updates ghosts of m_Kstag
891  m_grid->ctx()->profiling().begin("routing_conductivity");
892  compute_conductivity(m_Wstag,
893  subglacial_water_pressure(),
894  m_bottom_surface,
895  m_Kstag, maxKW);
896  m_grid->ctx()->profiling().end("routing_conductivity");
897 
898  // ghosts of m_Vstag are not updated
899  m_grid->ctx()->profiling().begin("routing_velocity");
900  compute_velocity(m_Wstag,
901  subglacial_water_pressure(),
902  m_bottom_surface,
903  m_Kstag,
904  inputs.no_model_mask,
905  m_Vstag);
906  m_grid->ctx()->profiling().end("routing_velocity");
907 
908  // to get Q, W needs valid ghosts (ghosts of m_Vstag are not used)
909  // updates ghosts of m_Qstag
910  m_grid->ctx()->profiling().begin("routing_flux");
911  advective_fluxes(m_Vstag, m_W, m_Qstag);
912  m_grid->ctx()->profiling().end("routing_flux");
913 
914  m_Qstag_average.add(hdt, m_Qstag);
915 
916  {
917  const double
918  dt_cfl = max_timestep_W_cfl(),
919  dt_diff_w = max_timestep_W_diff(maxKW);
920 
921  hdt = std::min(t_final - ht, dt_max);
922  hdt = std::min(hdt, dt_cfl);
923  hdt = std::min(hdt, dt_diff_w);
924  }
925 
926  m_log->message(3, " hydrology step %05d, dt = %f s\n", step_counter, hdt);
927 
928  // update Wtillnew from Wtill and input_rate
929  {
930  m_grid->ctx()->profiling().begin("routing_Wtill");
931  update_Wtill(hdt,
932  m_Wtill,
933  m_surface_input_rate,
934  m_basal_melt_rate,
935  m_Wtillnew);
936  // remove water in ice-free areas and account for changes
937  enforce_bounds(inputs.geometry->cell_type,
938  inputs.no_model_mask,
939  0.0, // do not limit maximum thickness
940  m_Wtillnew,
941  m_grounded_margin_change,
942  m_grounding_line_change,
943  m_conservation_error_change,
944  m_no_model_mask_change);
945  m_grid->ctx()->profiling().end("routing_Wtill");
946  }
947 
948  // update Wnew from W, Wtill, Wtillnew, Wstag, Q, input_rate
949  // uses ghosts of m_W, m_Wstag, m_Qstag, m_Kstag
950  {
951  m_grid->ctx()->profiling().begin("routing_W");
952  update_W(hdt,
953  m_surface_input_rate,
954  m_basal_melt_rate,
955  m_W, m_Wstag,
956  m_Wtill, m_Wtillnew,
957  m_Kstag, m_Qstag,
958  m_Wnew);
959  // remove water in ice-free areas and account for changes
960  enforce_bounds(inputs.geometry->cell_type,
961  inputs.no_model_mask,
962  0.0, // do not limit maximum thickness
963  m_Wnew,
964  m_grounded_margin_change,
965  m_grounding_line_change,
966  m_conservation_error_change,
967  m_no_model_mask_change);
968 
969  // transfer new into old (updates ghosts of m_W)
970  m_W.copy_from(m_Wnew);
971  m_grid->ctx()->profiling().end("routing_W");
972  }
973 
974  // m_Wtill has no ghosts
975  m_Wtill.copy_from(m_Wtillnew);
976  } // end of the time-stepping loop
977 
978  staggered_to_regular(inputs.geometry->cell_type, m_Qstag_average,
979  m_config->get_flag("hydrology.routing.include_floating_ice"),
980  m_Q);
981  m_Q.scale(1.0 / dt);
982 
983  m_log->message(2,
984  " took %d hydrology sub-steps with average dt = %.6f years (%.3f s or %.3f hours)\n",
985  step_counter,
986  units::convert(m_sys, dt / step_counter, "seconds", "years"),
987  dt / step_counter,
988  (dt / step_counter) / 3600.0);
989 }
