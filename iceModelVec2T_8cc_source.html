94 /*!
95  * Allocate an instance that will be used to load and use a forcing field from a file.
96  *
97  * Checks the number of records in a file and allocates storage accordingly.
98  *
99  * If `periodic` is true, allocate enough storage to hold all the records, otherwise
100  * allocate storage for at most `max_buffer_size` records.
101  *
102  * @param[in] grid computational grid
103  * @param[in] file input file
104  * @param[in] short_name variable name in `file`
105  * @param[in] standard_name standard name (if available); leave blank to ignore
106  * @param[in] max_buffer_size maximum buffer size for non-periodic fields
107  * @param[in] periodic true if this forcing field should be interpreted as periodic
108  */
109 std::shared_ptr<IceModelVec2T> IceModelVec2T::ForcingField(IceGrid::ConstPtr grid,
110  const File &file,
111  const std::string &short_name,
112  const std::string &standard_name,
113  int max_buffer_size,
114  bool periodic,
115  InterpolationType interpolation_type) {
116 
117  int n_records = file.nrecords(short_name, standard_name,
118  grid->ctx()->unit_system());
119 
120  int buffer_size = 0;
121  if (periodic) {
122  // In the periodic case we try to keep all the records in RAM.
123  buffer_size = n_records;
124 
125  if (interpolation_type == LINEAR) {
126  // add two more records at the beginning and the end of the period to simplify
127  // interpolation in time
128  buffer_size += 2;
129  }
130  } else {
131  buffer_size = std::min(n_records, max_buffer_size);
132  }
133 
134  // Allocate storage for one record if the variable was not found. This is needed to be
135  // able to cheaply allocate and then discard an "-atmosphere given" model
136  // (atmosphere::Given) when "-surface given" (Given) is selected.
137  buffer_size = std::max(buffer_size, 1);
138 
139  return std::make_shared<IceModelVec2T>(grid, short_name, buffer_size,
140  interpolation_type);
141 }

234 void IceModelVec2T::init(const std::string &filename, bool periodic) {
235  try {
236  auto ctx = m_impl->grid->ctx();
237  auto time = ctx->time();
238 
239  bool extrapolate = ctx->config()->get_flag("input.forcing.time_extrapolation");
240 
241  m_data->filename = filename;
242 
243  File file(m_impl->grid->com, m_data->filename, PISM_GUESS, PISM_READONLY);
244  auto var = file.find_variable(m_impl->metadata[0].get_name(),
245  m_impl->metadata[0].get_string("standard_name"));
246  if (not var.exists) {
247  throw RuntimeError(PISM_ERROR_LOCATION, "variable not found");
248  }
249 
250  auto time_name = io::time_dimension(ctx->unit_system(), file, var.name);
251 
252  // dimension_length() will return 0 if a dimension is missing
253  bool one_record = file.dimension_length(time_name) < 2;
254 
255  if (not one_record) {
256  std::vector<double> times{};
257  std::vector<double> bounds{};
258  io::read_time_info(*ctx->log(), ctx->unit_system(),
259  file, time_name, time->units_string(),
260  times, bounds);
261 
262  if (periodic) {
263  m_data->period_start = bounds.front();
264  m_data->period = bounds.back() - bounds.front();
265  }
266 
267  if (m_data->interp_type == PIECEWISE_CONSTANT) {
268  // Time bounds data overrides the time variable: we make t[j] be the
269  // left end-point of the j-th interval
270  for (unsigned int k = 0; k < times.size(); ++k) {
271  times[k] = bounds[2*k + 0];
272  }
273  }
274 
275  m_data->time = times;
276  m_data->time_range = {bounds.front(), bounds.back()};
277 
278  if (not (extrapolate or periodic)) {
279  check_forcing_duration(*time, bounds.front(), bounds.back());
280  }
281 
282  } else {
283  // Only one time record or no time dimension at all: set fake time bounds assuming
284  // that the user wants to use constant-in-time forcing for the whole simulation
285  extrapolate = true;
286 
287  // this value does not matter
288  m_data->time = {0.0};
289 
290  // set fake time bounds:
291  double eps = 0.5 * m_data->dt_min;
292  m_data->time_range = {-eps, eps};
293 
294  // note that in this case all data is periodic and constant in time
295  m_data->period = 0.0;
296  m_data->period_start = 0.0;
297  }
298 
299  if (periodic) {
300  // read periodic data right away (we need to hold it all in memory anyway)
301  init_periodic_data(file);
302  }
303  } catch (RuntimeError &e) {
304  e.add_context("reading %s (%s) from '%s'",
305  m_impl->metadata[0].get_string("long_name").c_str(),
306  m_impl->metadata[0].get_name().c_str(),
307  m_data->filename.c_str());
308  throw;
309  }
310 }

312 /*!
313  * Read all periodic data from the file and add two more records to simplify
314  * interpolation.
315  */
316 void IceModelVec2T::init_periodic_data(const File &file) {
317 
318  auto ctx = grid()->ctx();
319 
320  auto name = get_name();
321  auto n_records = file.nrecords(name, metadata().get_string("standard_name"),
322  ctx->unit_system());
323 
324  auto buffer_required = n_records + 2 * (m_data->interp_type == LINEAR);
325 
326  if (m_data->buffer_size < buffer_required) {
327  throw RuntimeError(PISM_ERROR_LOCATION,
328  "the buffer is too small to contain periodic data");
329  }
330 
331  bool allow_extrapolation = ctx->config()->get_flag("grid.allow_extrapolation");
332 
333  int offset = m_data->interp_type == LINEAR ? 1 : 0;
334 
335  // Read all the records and store them. The index offset leaves room for an extra record
336  // needed to simplify interpolation
337  for (unsigned int j = 0; j < n_records; ++j) {
338  {
339  petsc::VecArray tmp_array(vec());
340  io::regrid_spatial_variable(m_impl->metadata[0], *grid(), file, j, CRITICAL,
341  m_impl->report_range, allow_extrapolation,
342  0.0, m_impl->interpolation_type, tmp_array.get());
343  }
344 
345  auto t = ctx->time();
346  auto log = ctx->log();
347  log->message(5, " %s: reading entry #%02d, time %s...\n",
348  name.c_str(),
349  j,
350  t->date(m_data->time[j]).c_str());
351 
352  set_record(offset + j);
353  }
354 
355  m_data->n_records = buffer_required;
356  m_data->first = 0;
357 
358  if (m_data->interp_type == PIECEWISE_CONSTANT) {
359  return;
360  }
361 
362  double t0 = m_data->time_range[0];
363  double t1 = m_data->time_range[1];
364 
365  // compute the interpolation factor used to find the value at the beginning of the
366  // period
367  double alpha = 0.0;
368  {
369  double dt1 = m_data->time.front() - t0;
370  double dt2 = t1 - m_data->time.back();
371 
372  alpha = dt1 + dt2 > 0 ? dt2 / (dt1 + dt2) : 0.0;
373  }
374 
375  // indexes used to access the first and the last entry in the buffer
376  int first = 1;
377  int last = buffer_required - 2;
378 
379  IceModelVec::AccessList list{this};
380 
381  // compute values at the beginning (and so at the end) of the period
382  double **a2 = array();
383  double ***a3 = array3();
384  for (Points p(*grid()); p; p.next()) {
385  const int i = p.i(), j = p.j();
386 
387  a2[j][i] = (1.0 - alpha) * a3[j][i][last] + alpha * a3[j][i][first];
388  }
389 
390  set_record(0);
391  set_record(m_data->buffer_size - 1);
392 
393  // add two more records to m_data->time
394  {
395  m_data->time.insert(m_data->time.begin(), t0);
396  m_data->time.push_back(t1);
397  }
398 }

400 //! Read some data to make sure that the interval (t, t + dt) is covered.
401 void IceModelVec2T::update(double t, double dt) {
402 
403  if (m_data->filename.empty()) {
404  // We are not reading data from a file.
405  return;
406  }
407 
408  if (m_data->period > 0.0) {
409  // we read all data in IceModelVec2T::init() (see above)
410  return;
411  }
412 
413  if (m_data->n_records > 0) {
414  // in-file index of the last record currently in memory
415  unsigned int last = m_data->first + (m_data->n_records - 1);
416 
417  double t0{}, t1{};
418  if (m_data->interp_type == LINEAR) {
419  t0 = m_data->time[m_data->first];
420  t1 = m_data->time[last];
421  } else {
422  // piece-wise constant
423  t0 = m_data->time[m_data->first];
424  if (last + 1 < m_data->time.size()) {
425  t1 = m_data->time[last + 1];
426  } else {
427  t1 = m_data->time_range[1];
428  }
429  }
430 
431  // just return if we have all the data we need:
432  if (t >= t0 and t + dt <= t1) {
433  return;
434  }
435  }
436 
437  Interpolation I(m_data->interp_type, m_data->time, {t, t + dt});
438 
439  unsigned int
440  first = I.left(0),
441  last = I.right(1),
442  N = last - first + 1;
443 
444  // check if all the records necessary to cover this interval fit in the
445  // buffer:
446  if (N > m_data->buffer_size) {
447  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
448  "cannot read %d records of %s (buffer size: %d)",
449  N, m_impl->name.c_str(), m_data->buffer_size);
450  }
451 
452  update(first);
453 }

455 //! Update by reading at most buffer_size records from the file.
456 void IceModelVec2T::update(unsigned int start) {
457 
458  unsigned int time_size = (int)m_data->time.size();
459 
460  if (start >= time_size) {
461  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
462  "IceModelVec2T::update(int start): start = %d is invalid", start);
463  }
464 
465  unsigned int missing = std::min(m_data->buffer_size, time_size - start);
466 
467  if (start == static_cast<unsigned int>(m_data->first)) {
468  // nothing to do
469  return;
470  }
471 
472  int kept = 0;
473  if (m_data->first >= 0) {
474  unsigned int last = m_data->first + (m_data->n_records - 1);
475  if ((m_data->n_records > 0) && (start >= (unsigned int)m_data->first) && (start <= last)) {
476  int discarded = start - m_data->first;
477  kept = last - start + 1;
478  discard(discarded);
479  missing -= kept;
480  start += kept;
481  m_data->first += discarded;
482  } else {
483  m_data->first = start;
484  }
485  } else {
486  m_data->first = start;
487  }
488 
489  if (missing <= 0) {
490  return;
491  }
492 
493  m_data->n_records = kept + missing;
494 
495  auto t = m_impl->grid->ctx()->time();
496 
497  auto log = m_impl->grid->ctx()->log();
498  if (this->buffer_size() > 1) {
499  log->message(4,
500  " reading \"%s\" into buffer\n"
501  " (short_name = %s): %d records, time %s through %s...\n",
502  metadata().get_string("long_name").c_str(), m_impl->name.c_str(), missing,
503  t->date(m_data->time[start]).c_str(),
504  t->date(m_data->time[start + missing - 1]).c_str());
505  m_impl->report_range = false;
506  } else {
507  m_impl->report_range = true;
508  }
509 
510  File file(m_impl->grid->com, m_data->filename, PISM_GUESS, PISM_READONLY);
511 
512  const bool allow_extrapolation = m_impl->grid->ctx()->config()->get_flag("grid.allow_extrapolation");
513 
514  for (unsigned int j = 0; j < missing; ++j) {
515  {
516  petsc::VecArray tmp_array(vec());
517  io::regrid_spatial_variable(m_impl->metadata[0], *m_impl->grid, file, start + j, CRITICAL,
518  m_impl->report_range, allow_extrapolation,
519  0.0, m_impl->interpolation_type, tmp_array.get());
520  }
521 
522  log->message(5, " %s: reading entry #%02d, year %s...\n",
523  m_impl->name.c_str(),
524  start + j,
525  t->date(m_data->time[start + j]).c_str());
526 
527  set_record(kept + j);
528  }
529 }

532 void IceModelVec2T::discard(int number) {
533 
534  if (number == 0) {
535  return;

537 
538  m_data->n_records -= number;

565 //! @brief Given the time t determines the maximum possible time-step this IceModelVec2T
566 //! allows.
567 MaxTimestep IceModelVec2T::max_timestep(double t) const {
568  auto time_size = m_data->time.size();
569 
570  if (m_data->period > 0.0) {
571  // all periodic data is stored in RAM and there is no time step restriction
572  return {};
573  }
574 
575  if (t >= m_data->time.back()) {
576  // Reached the end of forcing: no time step restriction. We will need only one record
577  // to use constant extrapolation and it will surely fit in the buffer.
578  return {};
579  }
580 
581  // find the index k such that m_data->time[k] <= x < m_data->time[k + 1]
582  // Note: `L` below will be strictly less than `time_size - 1`.
583  size_t L = gsl_interp_bsearch(m_data->time.data(), t, 0, time_size - 1);
584 
585  // find the index of the last record we could read in given the size of the buffer
586  size_t R = L + m_data->buffer_size - 1;
587 
588  if (R >= time_size - 1) {
589  // We can read all the remaining records: no time step restriction from now on
590  return {};
591  }
592 
593  if (m_data->interp_type == PIECEWISE_CONSTANT) {
594  // in the piece-wise constant case we can go all the way to the *next* record
595  R = std::min(R + 1, time_size - 1);
596  return m_data->time[R] - t;
597  } else {
598  return m_data->time[R] - t;
599  }
600 
601  return {};
602 }

604 /*
605  * @brief Initialize IceModelVec2T with the value at time `t`.
606  *
607  * @note This method does not check if an update() call is necessary!
608  *
609  * @param[in] t requested time
610  *
611  */
612 void IceModelVec2T::interp(double t) {
613 
614  init_interpolation({t});
615 
616  // There is only one point to interpolate at ("t" above). Here we get interpolation
617  // indexes and the corresponding weight. Here L == R for the piecewise-constant
618  // interpolation.
619  int L = m_data->interp->left(0);
620  int R = m_data->interp->right(0);
621  double alpha = m_data->interp->alpha(0);
622 
623  AccessList l{this};
624  double ***a3 = array3();
625  double **a2 = array();
626 
627  for (Points p(*m_impl->grid); p; p.next()) {
628  const int i = p.i(), j = p.j();
629  auto column = a3[j][i];
630  // result (LHS) is a weighted average of two values.
631  a2[j][i] = column[L] + alpha * (column[R] - column[L]);
632  }
633 }

636 /**
637  * Compute the average value over the time interval `[t, t + dt]`.
638  *
639  * @param t start of the time interval, in seconds
640  * @param dt length of the time interval, in seconds
641  *
642  */
643 void IceModelVec2T::average(double t, double dt) {
644 
645  // if only one record, nothing to do
646  if (m_data->time.size() == 1 or
647  m_data->n_records == 1 or
648  dt == 0.0) {
649  interp(t);
650  return;
651  }
652 
653  const double *data = &m_data->time[m_data->first];
654  size_t data_size = m_data->n_records;
655  auto type = m_data->interp_type;
656 
657  std::map<size_t, double> weights{};
658 
659  if (m_data->period > 0.0) {
660  double a = t;
661  double b = t + dt;
662  double t0 = m_data->period_start;
663  double P = m_data->period;
664 
665  double N = std::floor((a - t0) / P);
666  double M = std::floor((b - t0) / P);
667  double delta = a - t0 - P * N;
668  double gamma = b - t0 - P * M;
669 
670  double N_periods = M - (N + 1);
671 
672  if (N_periods >= 0.0) {
673  auto W1 = integration_weights(data, data_size, type, t0 + delta, t0 + P);
674  auto W2 = integration_weights(data, data_size, type, t0, t0 + P);
675  auto W3 = integration_weights(data, data_size, type, t0, t0 + gamma);
676 
677  // before the first complete period:
678  weights = W1;
679  // an integer number of complete periods:
680  for (const auto &w : W2) {
681  weights[w.first] += N_periods * w.second;
682  }
683  // after the last complete period:
684  for (const auto &w : W3) {
685  weights[w.first] += w.second;
686  }
687  } else {
688  weights = integration_weights(data, data_size, type, t0 + delta, t0 + gamma);
689  }
690  } else {
691  weights = integration_weights(data, data_size, type, t, t + dt);
692  }
693 
694  AccessList l{this};
695  double **a2 = array();
696  double ***a3 = array3();
697 
698  for (Points p(*m_impl->grid); p; p.next()) {
699  const int i = p.i(), j = p.j();
700 
701  a2[j][i] = 0.0;
702  for (const auto &weight : weights) {
703  size_t k = weight.first;
704  double w = weight.second;
705  a2[j][i] += w * a3[j][i][k];
706  }
707  a2[j][i] /= dt;
708  }
709 }

713  * This is used *both* for time-series and "snapshots".
714  *

716  *
717  */
718 void IceModelVec2T::init_interpolation(const std::vector<double> &ts) {
719 
720  assert(m_data->first >= 0);
721 
722  // Compute "periodized" times if necessary.
723  std::vector<double> times_requested(ts.size());
724  if (m_data->period > 0.0) {
725  double P = m_data->period;
726  double T0 = m_data->period_start;
727 
728  for (unsigned int k = 0; k < ts.size(); ++k) {
729  double t = ts[k] - T0;
730 
731  t -= std::floor(t / P) * P;
732 
733  times_requested[k] = T0 + t;
734  }
735  } else {
736  times_requested = ts;
737  }
738 
739  m_data->interp.reset(new Interpolation(m_data->interp_type,
740  &m_data->time[m_data->first],
741  m_data->n_records,
742  times_requested.data(),
743  times_requested.size()));
744 }
