251 void OceanModel::compute_average_water_column_pressure(const Geometry &geometry,
252  double ice_density,
253  double water_density,
254  double g,
255  IceModelVec2S &result) {
256 
257  auto grid = result.grid();
258 
259  const IceModelVec2S
260  &bed = geometry.bed_elevation,
261  &H = geometry.ice_thickness,
262  &z_s = geometry.sea_level_elevation;
263 
264  IceModelVec::AccessList l{&bed, &H, &z_s, &result};
265 
266  ParallelSection loop(grid->com);
267  try {
268  for (Points p(*grid); p; p.next()) {
269  const int i = p.i(), j = p.j();
270 
271  result(i, j) = pism::average_water_column_pressure(H(i, j), bed(i, j), z_s(i, j),
272  ice_density, water_density, g);
273  }
274  } catch (...) {
275  loop.failed();
276  }
277  loop.check();
278 
279 
280 }
