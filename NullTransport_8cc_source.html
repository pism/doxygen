71 MaxTimestep NullTransport::max_timestep_impl(double t) const {
72  (void) t;
73  if (m_diffuse_tillwat) {
74  const double
75  dx2 = m_grid->dx() * m_grid->dx(),
76  dy2 = m_grid->dy() * m_grid->dy(),
77  L = m_diffusion_distance,
78  T = m_diffusion_time,
79  K = L * L / (2.0 * T);
80 
81  return MaxTimestep(dx2 * dy2 / (2.0 * K * (dx2 + dy2)), "null-transport hydrology");
82  } else {
83  return MaxTimestep("null-transport hydrology");
84  }
85 }

87 //! Update the till water thickness by simply integrating the melt input.
88 /*!
89  Does a step of the trivial integration
90 
91  \f[ \frac{\partial W_{till}}{\partial t} = \frac{m}{\rho_w} - C\f]
92 
93  where \f$C=\f$`hydrology_tillwat_decay_rate`. Enforces bounds
94  \f$0 \le W_{till} \le W_{till}^{max}\f$ where the upper bound is
95  `hydrology_tillwat_max`. Here \f$m/\rho_w\f$ is `total_input`.
96 
97  Uses the current mass-continuity timestep `dt`. (Compare
98  hydrology::Routing::update_Wtill() which will generally be taking time steps
99  determined by the evolving transportable water layer in that model.)
100 
101  There is no attempt to report on conservation errors because this
102  hydrology::NullTransport model does not conserve water.
103 
104  There is no tranportable water thickness variable and no interaction with it.
105 */
106 void NullTransport::update_impl(double t, double dt, const Inputs& inputs) {
107  (void) t;
108 
109  bool add_surface_input = m_config->get_flag("hydrology.add_water_input_to_till_storage");
110 
111  // no transportable basal water
112  m_W.set(0.0);
113 
114  m_input_change.add(dt, m_basal_melt_rate);
115  if (add_surface_input) {
116  m_input_change.add(dt, m_surface_input_rate);
117  }
118 
119  const double
120  water_density = m_config->get_number("constants.fresh_water.density"),
121  kg_per_m = m_grid->cell_area() * water_density; // kg m-1
122 
123  const IceModelVec2CellType &cell_type = inputs.geometry->cell_type;
124 
125  IceModelVec::AccessList list{&cell_type, &m_Wtill, &m_basal_melt_rate,
126  &m_conservation_error_change};
127 
128  if (add_surface_input) {
129  list.add(m_surface_input_rate);
130  }
131 
132  for (Points p(*m_grid); p; p.next()) {
133  const int i = p.i(), j = p.j();
134 
135  double
136  W_old = m_Wtill(i, j),
137  dW_input = dt * m_basal_melt_rate(i, j);
138 
139  if (add_surface_input) {
140  dW_input += dt * m_surface_input_rate(i, j);
141  }
142 
143  if (W_old < 0.0) {
144  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
145  "negative subglacial water thickness of %f m at (%d, %d)",
146  W_old, i, j);
147  }
148 
149  // decay rate in areas under grounded ice
150  double dW_decay = dt * (- m_tillwat_decay_rate);
151 
152  if (not cell_type.grounded_ice(i, j)) {
153  dW_decay = 0.0;
154  }
155 
156  m_Wtill(i, j) = (W_old + dW_input) + dW_decay;
157 
158  // dW_decay is a "conservation error"
159  m_conservation_error_change(i, j) += dW_decay * kg_per_m;
160  }
161 
162  if (m_diffuse_tillwat) {
163  diffuse_till_water(dt);
164  }
165 
166  // remove water in ice-free areas and account for changes
167  enforce_bounds(inputs.geometry->cell_type,
168  inputs.no_model_mask,
169  m_tillwat_max, // global maximum till water thickness
170  m_tillwat_max, // till water thickness under the ocean
171  m_Wtill,
172  m_grounded_margin_change,
173  m_grounding_line_change,
174  m_conservation_error_change,
175  m_no_model_mask_change);
176 }

178 void NullTransport::diffuse_till_water(double dt) {
179  // note: this call updates ghosts of m_Wtill_old
180  m_Wtill_old.copy_from(m_Wtill);
181 
182  const double
183  dx = m_grid->dx(),
184  dy = m_grid->dy(),
185  L = m_diffusion_distance,
186  T = m_diffusion_time,
187  K = L * L / (2.0 * T),
188  Rx = K * dt / (dx * dx),
189  Ry = K * dt / (dy * dy);
190 
191  IceModelVec::AccessList list{&m_Wtill, &m_Wtill_old, &m_flow_change};
192  for (Points p(*m_grid); p; p.next()) {
193  const int i = p.i(), j = p.j();
194 
195  auto W = m_Wtill_old.star(i, j);
196 
197  double dWtill = (Rx * (W.w - 2.0 * W.ij + W.e) + Ry * (W.s - 2.0 * W.ij + W.n));
198 
199  m_Wtill(i, j) = W.ij + dWtill;
200 
201  m_flow_change(i, j) = dWtill;
202  }
203 }
