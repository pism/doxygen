30 MaxTimestep IceModel::save_max_timestep(double my_t) {
31 
32  if ((not m_save_snapshots) or

43 //! Initializes the snapshot-saving mechanism.
44 void IceModel::init_snapshots() {
45  m_current_snapshot = 0;
46 
47  m_snapshots_filename = m_config->get_string("output.snapshot.file");
48  bool filename_set = not m_snapshots_filename.empty();
49 
50  auto save_times = m_config->get_string("output.snapshot.times");
51  bool times_set = not save_times.empty();
52 
53  bool split = m_config->get_flag("output.snapshot.split");
54 
55  m_snapshot_vars = output_variables(m_config->get_string("output.snapshot.size"));
56 
57  if (filename_set ^ times_set) {
58  throw RuntimeError(PISM_ERROR_LOCATION,
59  "you need to set both output.snapshot.file and output.snapshot.times"
60  " to save snapshots.");
61  }
62 
63  if (not filename_set and not times_set) {
64  m_save_snapshots = false;
65  return;
66  }
67 
68  try {
69  // parse
70  std::vector<double> times = m_time->parse_times(save_times);
71 
72  // discard times before the beginning and after the end of the run
73  m_snapshot_times.clear();
74  for (const auto &t : times) {
75  if (t >= m_time->start() and t <= m_time->end()) {
76  m_snapshot_times.push_back(t);
77  }
78  }
79  } catch (RuntimeError &e) {
80  e.add_context("processing output.snapshot.times");
81  throw;
82  }
83 
84  if (m_snapshot_times.size() == 0) {
85  throw RuntimeError(PISM_ERROR_LOCATION,
86  "output.snapshot.times was set, but all requested times"
87  " are outside of the modeled time interval");
88  }
89 
90  m_save_snapshots = true;
91  m_snapshots_file_is_ready = false;
92  m_split_snapshots = false;
93 
94  if (split) {
95  m_split_snapshots = true;
96  } else if (not ends_with(m_snapshots_filename, ".nc")) {
97  m_log->message(2,
98  "PISM WARNING: snapshots file name does not have the '.nc' suffix!\n");
99  }
100 
101  if (split) {
102  m_log->message(2, "saving snapshots to '%s+year.nc'; ",
103  m_snapshots_filename.c_str());
104  } else {
105  m_log->message(2, "saving snapshots to '%s'; ",
106  m_snapshots_filename.c_str());
107  }
108 
109  m_log->message(2, "times requested: %s\n", save_times.c_str());
110 }

112  //! Writes a snapshot of the model state (if necessary)
113 void IceModel::write_snapshot() {
114  double saving_after = -1.0e30; // initialize to avoid compiler warning; this
115  // value is never used, because saving_after
116  // is only used if save_now == true, and in
117  // this case saving_after is guaranteed to be
118  // initialized. See the code below.
119  std::string filename;
120 
121  // determine if the user set the -save_times and -save_file options
122  if (not m_save_snapshots) {
123  return;
124  }
125 
126  // do we need to save *now*?
127  if ((m_time->current() >= m_snapshot_times[m_current_snapshot]) and
128  (m_current_snapshot < m_snapshot_times.size())) {
129  saving_after = m_snapshot_times[m_current_snapshot];
130 
131  while ((m_current_snapshot < m_snapshot_times.size()) and
132  (m_snapshot_times[m_current_snapshot] <= m_time->current())) {
133  m_current_snapshot++;
134  }
135  } else {
136  // we don't need to save now, so just return
137  return;
138  }
139 
140  // flush time-series buffers
141  flush_timeseries();
142 
143  if (m_split_snapshots) {
144  m_snapshots_file_is_ready = false; // each snapshot is written to a separate file
145  auto date_without_spaces = std::regex_replace(m_time->date(saving_after),
146  std::regex(" "), "_");
147  filename = pism::printf("%s_%s.nc",
148  m_snapshots_filename.c_str(),
149  date_without_spaces.c_str());
150  } else {
151  filename = m_snapshots_filename.c_str();
152  }
153 
154  m_log->message(2,
155  "saving snapshot to %s at %s, for time-step goal %s\n",
156  filename.c_str(), m_time->date(m_time->current()).c_str(),
157  m_time->date(saving_after).c_str());
158 
159  const Profiling &profiling = m_ctx->profiling();
160 
161  profiling.begin("io.snapshots");
162  IO_Mode mode = m_snapshots_file_is_ready ? PISM_READWRITE : PISM_READWRITE_MOVE;
163  {
164  File file(m_grid->com,
165  filename,
166  string_to_backend(m_config->get_string("output.format")),
167  mode,
168  m_ctx->pio_iosys_id());
169 
170  if (not m_snapshots_file_is_ready) {
171  write_metadata(file, WRITE_MAPPING, PREPEND_HISTORY);
172 
173  m_snapshots_file_is_ready = true;
174  }
175 
176  write_run_stats(file);
177 
178  save_variables(file, INCLUDE_MODEL_STATE, m_snapshot_vars, m_time->current());
179  }
180  profiling.end("io.snapshots");
181 }
