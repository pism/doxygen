83 void PIK::melting_point_temperature(const IceModelVec2S &depth,
84  IceModelVec2S &result) const {
85  const double
86  T0 = m_config->get_number("constants.fresh_water.melting_point_temperature"), // K
87  beta_CC = m_config->get_number("constants.ice.beta_Clausius_Clapeyron"),
88  g = m_config->get_number("constants.standard_gravity"),
89  ice_density = m_config->get_number("constants.ice.density");
90 
91  IceModelVec::AccessList list{&depth, &result};
92 
93  for (Points p(*m_grid); p; p.next()) {
94  const int i = p.i(), j = p.j();
95  const double pressure = ice_density * g * depth(i,j); // FIXME task #7297
96  // result is set to melting point at depth
97  result(i,j) = T0 - beta_CC * pressure;
98  }

101 //! \brief Computes mass flux in [kg m-2 s-1].
102 /*!
103  * Assumes that mass flux is proportional to the shelf-base heat flux.
104  */
105 void PIK::mass_flux(const IceModelVec2S &ice_thickness, IceModelVec2S &result) const {
106  const double
107  melt_factor = m_config->get_number("ocean.pik_melt_factor"),
108  L = m_config->get_number("constants.fresh_water.latent_heat_of_fusion"),
109  sea_water_density = m_config->get_number("constants.sea_water.density"),
110  ice_density = m_config->get_number("constants.ice.density"),
111  c_p_ocean = 3974.0, // J/(K*kg), specific heat capacity of ocean mixed layer
112  gamma_T = 1e-4, // m/s, thermal exchange velocity
113  ocean_salinity = 35.0, // g/kg
114  T_ocean = units::convert(m_sys, -1.7, "Celsius", "Kelvin"); //Default in PISM-PIK
115 
116  //FIXME: gamma_T should be a function of the friction velocity, not a const
117 
118  IceModelVec::AccessList list{&ice_thickness, &result};
119 
120  for (Points p(*m_grid); p; p.next()) {
121  const int i = p.i(), j = p.j();
122 
123  // compute T_f(i, j) according to beckmann_goosse03, which has the
124  // meaning of the freezing temperature of the ocean water directly
125  // under the shelf, (of salinity 35psu) [this is related to the
126  // Pressure Melting Temperature, see beckmann_goosse03 eq. 2 for
127  // details]
128  double
129  shelfbaseelev = - (ice_density / sea_water_density) * ice_thickness(i,j),
130  T_f = 273.15 + (0.0939 -0.057 * ocean_salinity + 7.64e-4 * shelfbaseelev);
131  // add 273.15 to convert from Celsius to Kelvin
132 
133  // compute ocean_heat_flux according to beckmann_goosse03
134  // positive, if T_oc > T_ice ==> heat flux FROM ocean TO ice
135  double ocean_heat_flux = melt_factor * sea_water_density * c_p_ocean * gamma_T * (T_ocean - T_f); // in W/m^2
136 
137  // TODO: T_ocean -> field!
138 
139  // shelfbmassflux is positive if ice is freezing on; here it is always negative:
140  // same sign as ocean_heat_flux (positive if massflux FROM ice TO ocean)
141  result(i,j) = ocean_heat_flux / (L * ice_density); // m s-1
142 
143  // convert from [m s-1] to [kg m-2 s-1]:
144  result(i,j) *= ice_density;
145  }
146 }
