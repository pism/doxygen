129 void FractureDensity::update(double dt,
130  const Geometry &geometry,
131  const IceModelVec2V &velocity,
132  const IceModelVec2S &hardness,
133  const IceModelVec2S &bc_mask) {
134  using std::pow;
135 
136  const double
137  dx = m_grid->dx(),
138  dy = m_grid->dy();
139  const int
140  Mx = m_grid->Mx(),
141  My = m_grid->My();
142 
143  IceModelVec2S
144  &D = m_density,
145  &D_new = m_density_new,
146  &A = m_age,
147  &A_new = m_age_new;
148 
149  m_velocity.copy_from(velocity);
150 
151  stressbalance::compute_2D_principal_strain_rates(m_velocity,
152  geometry.cell_type,
153  m_strain_rates);
154 
155  stressbalance::compute_2D_stresses(*m_flow_law,
156  m_velocity,
157  hardness,
158  geometry.cell_type,
159  m_deviatoric_stresses);
160 
161  IceModelVec::AccessList list{&m_velocity, &m_strain_rates, &m_deviatoric_stresses,
162  &D, &D_new, &geometry.cell_type, &bc_mask, &A, &A_new,
163  &m_growth_rate, &m_healing_rate, &m_flow_enhancement,
164  &m_toughness, &hardness, &geometry.ice_thickness};
165 
166  D_new.copy_from(D);
167 
168  //options
169  /////////////////////////////////////////////////////////
170  double soft_residual = m_config->get_number("fracture_density.softening_lower_limit");
171  // assume linear response function: E_fr = (1-(1-soft_residual)*phi) -> 1-phi
172  //
173  // See the following article for more:
174  //
175  // Albrecht, T. / Levermann, A.
176  // Fracture-induced softening for large-scale ice dynamics
177  // 2014-04
178  //
179  // The Cryosphere , Vol. 8, No. 2
180  // Copernicus GmbH
181  // p. 587-605
182  //
183  // doi:10.5194/tc-8-587-2014
184  //
185  // get four options for calculation of fracture density.
186  // 1st: fracture growth constant gamma
187  // 2nd: fracture initiation stress threshold sigma_cr
188  // 3rd: healing rate constant gamma_h
189  // 4th: healing strain rate threshold
190  //
191  // more: T. Albrecht, A. Levermann; Fracture field for large-scale
192  // ice dynamics; (2012), Journal of Glaciology, Vol. 58, No. 207,
193  // 165-176, DOI: 10.3189/2012JoG11J191.
194 
195  double gamma = m_config->get_number("fracture_density.gamma");
196  double initThreshold = m_config->get_number("fracture_density.initiation_threshold");
197  double gammaheal = m_config->get_number("fracture_density.gamma_h");
198  double healThreshold = m_config->get_number("fracture_density.healing_threshold");
199 
200  m_log->message(3, "PISM-PIK INFO: fracture density is found with parameters:\n"
201  " gamma=%.2f, sigma_cr=%.2f, gammah=%.2f, healing_cr=%.1e and soft_res=%f \n",
202  gamma, initThreshold, gammaheal, healThreshold, soft_residual);
203 
204  bool do_fracground = m_config->get_flag("fracture_density.include_grounded_ice");
205 
206  double fdBoundaryValue = m_config->get_number("fracture_density.phi0");
207 
208  bool constant_healing = m_config->get_flag("fracture_density.constant_healing");
209 
210  bool fracture_weighted_healing = m_config->get_flag("fracture_density.fracture_weighted_healing");
211 
212  bool max_shear_stress = m_config->get_flag("fracture_density.max_shear_stress");
213 
214  bool lefm = m_config->get_flag("fracture_density.lefm");
215 
216  bool constant_fd = m_config->get_flag("fracture_density.constant_fd");
217 
218  bool fd2d_scheme = m_config->get_flag("fracture_density.fd2d_scheme");
219 
220  double glen_exponent = m_flow_law->exponent();
221 
222  bool borstad_limit = m_config->get_flag("fracture_density.borstad_limit");
223 
224  double minH = m_config->get_number("stress_balance.ice_free_thickness_standard");
225 
226  for (Points p(*m_grid); p; p.next()) {
227  const int i = p.i(), j = p.j();
228 
229  double tempFD = 0.0;
230 
231  double u = m_velocity(i, j).u;
232  double v = m_velocity(i, j).v;
233 
234  if (fd2d_scheme) {
235  if (u >= dx * v / dy and v >= 0.0) { //1
236  tempFD = u * (D(i, j) - D(i - 1, j)) / dx + v * (D(i - 1, j) - D(i - 1, j - 1)) / dy;
237  } else if (u <= dx * v / dy and u >= 0.0) { //2
238  tempFD = u * (D(i, j - 1) - D(i - 1, j - 1)) / dx + v * (D(i, j) - D(i, j - 1)) / dy;
239  } else if (u >= -dx * v / dy and u <= 0.0) { //3
240  tempFD = -u * (D(i, j - 1) - D(i + 1, j - 1)) / dx + v * (D(i, j) - D(i, j - 1)) / dy;
241  } else if (u <= -dx * v / dy and v >= 0.0) { //4
242  tempFD = -u * (D(i, j) - D(i + 1, j)) / dx + v * (D(i + 1, j) - D(i + 1, j - 1)) / dy;
243  } else if (u <= dx * v / dy and v <= 0.0) { //5
244  tempFD = -u * (D(i, j) - D(i + 1, j)) / dx - v * (D(i + 1, j) - D(i + 1, j + 1)) / dy;
245  } else if (u >= dx * v / dy and u <= 0.0) { //6
246  tempFD = -u * (D(i, j + 1) - D(i + 1, j + 1)) / dx - v * (D(i, j) - D(i, j + 1)) / dy;
247  } else if (u <= -dx * v / dy and u >= 0.0) { //7
248  tempFD = u * (D(i, j + 1) - D(i - 1, j + 1)) / dx - v * (D(i, j) - D(i, j + 1)) / dy;
249  } else if (u >= -dx * v / dy and v <= 0.0) { //8
250  tempFD = u * (D(i, j) - D(i - 1, j)) / dx - v * (D(i - 1, j) - D(i - 1, j + 1)) / dy;
251  } else {
252  m_log->message(3, "######### missing case of angle %f of %f and %f at %d, %d \n",
253  atan(v / u) / M_PI * 180., u * 3e7, v * 3e7, i, j);
254  }
255  } else {
256  tempFD += u * (u < 0 ? D(i + 1, j) - D(i, j) : D(i, j) - D(i - 1, j)) / dx;
257  tempFD += v * (v < 0 ? D(i, j + 1) - D(i, j) : D(i, j) - D(i, j - 1)) / dy;
258  }
259 
260  D_new(i, j) -= tempFD * dt;
261 
262  //sources /////////////////////////////////////////////////////////////////
263  ///von mises criterion
264 
265  double
266  txx = m_deviatoric_stresses(i, j, 0),
267  tyy = m_deviatoric_stresses(i, j, 1),
268  txy = m_deviatoric_stresses(i, j, 2),
269  T1 = 0.5 * (txx + tyy) + sqrt(0.25 * pow(txx - tyy, 2) + pow(txy, 2)), //Pa
270  T2 = 0.5 * (txx + tyy) - sqrt(0.25 * pow(txx - tyy, 2) + pow(txy, 2)), //Pa
271  sigmat = sqrt(pow(T1, 2) + pow(T2, 2) - T1 * T2);
272 
273 
274  ///max shear stress criterion (more stringent than von mises)
275  if (max_shear_stress) {
276  double maxshear = fabs(T1);
277  maxshear = std::max(maxshear, fabs(T2));
278  maxshear = std::max(maxshear, fabs(T1 - T2));
279 
280  sigmat = maxshear;
281  }
282 
283  ///lefm mixed-mode criterion
284  if (lefm) {
285  double sigmamu = 0.1; //friction coefficient between crack faces
286 
287  double sigmac = 0.64 / M_PI; //initial crack depth 20cm
288 
289  double sigmabetatest, sigmanor, sigmatau, Kone, Ktwo, KSI, KSImax = 0.0, sigmatetanull;
290 
291  for (int l = 46; l <= 90; ++l) { //optimize for various precursor angles beta
292  sigmabetatest = l * M_PI / 180.0;
293 
294  //rist_sammonds99
295  sigmanor = 0.5 * (T1 + T2) - (T1 - T2) * cos(2 * sigmabetatest);
296  sigmatau = 0.5 * (T1 - T2) * sin(2 * sigmabetatest);
297  //shayam_wu90
298  if (sigmamu * sigmanor < 0.0) { //compressive case
299  if (fabs(sigmatau) <= fabs(sigmamu * sigmanor)) {
300  sigmatau = 0.0;
301  } else {
302  if (sigmatau > 0) { //coulomb friction opposing sliding
303  sigmatau += (sigmamu * sigmanor);
304  } else {
305  sigmatau -= (sigmamu * sigmanor);
306  }
307  }
308  }
309 
310  //stress intensity factors
311  Kone = sigmanor * sqrt(M_PI * sigmac); //normal
312  Ktwo = sigmatau * sqrt(M_PI * sigmac); //shear
313 
314  if (Ktwo == 0.0) {
315  sigmatetanull = 0.0;
316  } else { //eq15 in hulbe_ledoux10 or eq15 shayam_wu90
317  sigmatetanull = -2.0 * atan((sqrt(pow(Kone, 2) + 8.0 * pow(Ktwo, 2)) - Kone) / (4.0 * Ktwo));
318  }
319 
320  KSI = cos(0.5 * sigmatetanull) *
321  (Kone * cos(0.5 * sigmatetanull) * cos(0.5 * sigmatetanull) - 0.5 * 3.0 * Ktwo * sin(sigmatetanull));
322  // mode I stress intensity
323 
324  KSImax = std::max(KSI, KSImax);
325  }
326  sigmat = KSImax;
327  }
328 
329  //////////////////////////////////////////////////////////////////////////////
330 
331  // fracture density
332  double fdnew = 0.0;
333  if (borstad_limit) {
334  if (geometry.ice_thickness(i, j) > minH) {
335  // mean parameters from paper
336  double t0 = initThreshold;
337  double kappa = 2.8;
338 
339  // effective strain rate
340  double e1 = m_strain_rates(i, j, 0);
341  double e2 = m_strain_rates(i, j, 1);
342  double ee = sqrt(pow(e1, 2.0) + pow(e2, 2.0) - e1 * e2);
343 
344  // threshold for unfractured ice
345  double e0 = pow((t0 / hardness(i, j)), glen_exponent);
346 
347  // threshold for fractured ice (exponential law)
348  double ex = exp((e0 - ee) / (e0 * (kappa - 1)));
349 
350  // stress threshold for fractures ice
351  double te = t0 * ex;
352 
353  // actual effective stress
354  double ts = hardness(i, j) * pow(ee, 1.0 / glen_exponent) * (1 - D_new(i, j));
355 
356  // fracture formation if threshold is hit
357  if (ts > te and ee > e0) {
358  // new fracture density:
359  fdnew = 1.0 - (ex * pow((ee / e0), -1 / glen_exponent));
360  D_new(i, j) = fdnew;
361  }
362  }
363  } else {
364  fdnew = gamma * (m_strain_rates(i, j, 0) - 0.0) * (1 - D_new(i, j));
365  if (sigmat > initThreshold) {
366  D_new(i, j) += fdnew * dt;
367  }
368  }
369 
370  //healing
371  double fdheal = gammaheal * std::min(0.0,(m_strain_rates(i, j, 0) - healThreshold));
372  if (geometry.cell_type.icy(i, j)) {
373  if (constant_healing) {
374  fdheal = gammaheal * (-healThreshold);
375  if (fracture_weighted_healing) {
376  D_new(i, j) += fdheal * dt * (1 - D(i, j));
377  } else {
378  D_new(i, j) += fdheal * dt;
379  }
380  } else if (m_strain_rates(i, j, 0) < healThreshold) {
381  if (fracture_weighted_healing) {
382  D_new(i, j) += fdheal * dt * (1 - D(i, j));
383  } else {
384  D_new(i, j) += fdheal * dt;
385  }
386  }
387  }
388 
389  // bounding
390  D_new(i, j) = pism::clip(D_new(i, j), 0.0, 1.0);
391 
392  if (geometry.cell_type.icy(i, j)) {
393  //fracture toughness
394  m_toughness(i, j) = sigmat;
395 
396  // fracture growth rate
397  if (sigmat > initThreshold) {
398  m_growth_rate(i, j) = fdnew;
399  //m_growth_rate(i,j)=gamma*(vPrinStrain1(i,j)-0.0)*(1-D_new(i,j));
400  } else {
401  m_growth_rate(i, j) = 0.0;
402  }
403 
404  // fracture healing rate
405  if (geometry.cell_type.icy(i, j)) {
406  if (constant_healing or (m_strain_rates(i, j, 0) < healThreshold)) {
407  if (fracture_weighted_healing) {
408  m_healing_rate(i, j) = fdheal * (1 - D(i, j));
409  } else {
410  m_healing_rate(i, j) = fdheal;
411  }
412  } else {
413  m_healing_rate(i, j) = 0.0;
414  }
415  }
416 
417  // fracture age since fracturing occurred
418  {
419  auto a = A.star(i, j);
420  A_new(i, j) -= dt * u * (u < 0 ? a.e - a.ij : a.ij - a.w) / dx;
421  A_new(i, j) -= dt * v * (v < 0 ? a.n - a.ij : a.ij - a.s) / dy;
422  A_new(i, j) += dt;
423  if (sigmat > initThreshold) {
424  A_new(i, j) = 0.0;
425  }
426  }
427 
428  // additional flow enhancement due to fracture softening
429  double softening = pow((1.0 - (1.0 - soft_residual) * D_new(i, j)), -glen_exponent);
430  if (geometry.cell_type.icy(i, j)) {
431  m_flow_enhancement(i, j) = 1.0 / pow(softening, 1 / glen_exponent);
432  } else {
433  m_flow_enhancement(i, j) = 1.0;
434  }
435  }
436 
437  // boundary condition
438  if (geometry.cell_type.grounded(i, j) and not do_fracground) {
439 
440  if (bc_mask(i, j) > 0.5) {
441  D_new(i, j) = fdBoundaryValue;
442 
443  {
444  A_new(i, j) = 0.0;
445  m_growth_rate(i, j) = 0.0;
446  m_healing_rate(i, j) = 0.0;
447  m_flow_enhancement(i, j) = 1.0;
448  m_toughness(i, j) = 0.0;
449  }
450  }
451  }
452 
453  // ice free regions and boundary of computational domain
454  if (geometry.cell_type.ice_free(i, j) or i == 0 or j == 0 or i == Mx - 1 or j == My - 1) {
455 
456  D_new(i, j) = 0.0;
457 
458  {
459  A_new(i, j) = 0.0;
460  m_growth_rate(i, j) = 0.0;
461  m_healing_rate(i, j) = 0.0;
462  m_flow_enhancement(i, j) = 1.0;
463  m_toughness(i, j) = 0.0;
464  }
465  }
466 
467  if (constant_fd) { // no fd evolution
468  D_new(i, j) = D(i, j);
469  }
470  }
471 
472  A.copy_from(A_new);
473  D.copy_from(D_new);
474 }
