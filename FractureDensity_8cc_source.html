134 void FractureDensity::update(double dt,
135  const Geometry &geometry,
136  const IceModelVec2V &velocity,
137  const IceModelVec2S &hardness,
138  const IceModelVec2S &bc_mask) {
139  using std::pow;
140 
141  const double
142  dx = m_grid->dx(),
143  dy = m_grid->dy();
144  const int
145  Mx = m_grid->Mx(),
146  My = m_grid->My();
147 
148  IceModelVec2S
149  &D = m_density,
150  &D_new = m_density_new,
151  &A = m_age,
152  &A_new = m_age_new;
153 
154  m_velocity.copy_from(velocity);
155 
156  stressbalance::compute_2D_principal_strain_rates(m_velocity,
157  geometry.cell_type,
158  m_strain_rates);
159 
160  stressbalance::compute_2D_stresses(*m_flow_law,
161  m_velocity,
162  hardness,
163  geometry.cell_type,
164  m_deviatoric_stresses);
165 
166  IceModelVec::AccessList list{&m_velocity, &m_strain_rates, &m_deviatoric_stresses,
167  &D, &D_new, &geometry.cell_type, &bc_mask, &A, &A_new,
168  &m_growth_rate, &m_healing_rate, &m_flow_enhancement,
169  &m_toughness};
170 
171  D_new.copy_from(D);
172 
173  //options
174  /////////////////////////////////////////////////////////
175  double soft_residual = options::Real(m_sys, "-fracture_softening",
176  "soft_residual", "1", 1.0);
177  // assume linear response function: E_fr = (1-(1-soft_residual)*phi) -> 1-phi
178  //
179  // more: T. Albrecht, A. Levermann; Fracture-induced softening for
180  // large-scale ice dynamics; (2013), The Cryosphere Discussions 7;
181  // 4501-4544; DOI:10.5194/tcd-7-4501-2013
182 
183  // get four options for calculation of fracture density.
184  // 1st: fracture growth constant gamma
185  // 2nd: fracture initiation stress threshold sigma_cr
186  // 3rd: healing rate constant gamma_h
187  // 4th: healing strain rate threshold
188  // more: T. Albrecht, A. Levermann; Fracture field for large-scale
189  // ice dynamics; (2012), Journal of Glaciology, Vol. 58, No. 207,
190  // 165-176, DOI: 10.3189/2012JoG11J191.
191 
192  double gamma = 1.0, initThreshold = 7.0e4, gammaheal = 0.0, healThreshold = 2.0e-10;
193  {
194  options::RealList fractures("-fracture_parameters",
195  "gamma, initThreshold, gammaheal, healThreshold",
196  {gamma, initThreshold, gammaheal, healThreshold});
197  if (fractures->size() != 4) {
198  throw RuntimeError(PISM_ERROR_LOCATION, "option -fracture_parameters requires exactly 4 arguments");
199  }
200  gamma = fractures[0];
201  initThreshold = fractures[1];
202  gammaheal = fractures[2];
203  healThreshold = fractures[3];
204  }
205 
206  m_log->message(3, "PISM-PIK INFO: fracture density is found with parameters:\n"
207  " gamma=%.2f, sigma_cr=%.2f, gammah=%.2f, healing_cr=%.1e and soft_res=%f \n",
208  gamma, initThreshold, gammaheal, healThreshold, soft_residual);
209 
210  bool do_fracground = m_config->get_flag("fracture_density.include_grounded_ice");
211 
212  double fdBoundaryValue = m_config->get_number("fracture_density.phi0");
213 
214  bool constant_healing = m_config->get_flag("fracture_density.constant_healing");
215 
216  bool fracture_weighted_healing = m_config->get_flag("fracture_density.fracture_weighted_healing");
217 
218  bool max_shear_stress = m_config->get_flag("fracture_density.max_shear_stress");
219 
220  bool lefm = m_config->get_flag("fracture_density.lefm");
221 
222  bool constant_fd = m_config->get_flag("fracture_density.constant_fd");
223 
224  bool fd2d_scheme = m_config->get_flag("fracture_density.fd2d_scheme");
225 
226  for (Points p(*m_grid); p; p.next()) {
227  const int i = p.i(), j = p.j();
228 
229  double tempFD = 0.0;
230 
231  double u = m_velocity(i, j).u;
232  double v = m_velocity(i, j).v;
233 
234  if (fd2d_scheme) {
235  if (u >= dx * v / dy and v >= 0.0) { //1
236  tempFD = u * (D(i, j) - D(i - 1, j)) / dx + v * (D(i - 1, j) - D(i - 1, j - 1)) / dy;
237  } else if (u <= dx * v / dy and u >= 0.0) { //2
238  tempFD = u * (D(i, j - 1) - D(i - 1, j - 1)) / dx + v * (D(i, j) - D(i, j - 1)) / dy;
239  } else if (u >= -dx * v / dy and u <= 0.0) { //3
240  tempFD = -u * (D(i, j - 1) - D(i + 1, j - 1)) / dx + v * (D(i, j) - D(i, j - 1)) / dy;
241  } else if (u <= -dx * v / dy and v >= 0.0) { //4
242  tempFD = -u * (D(i, j) - D(i + 1, j)) / dx + v * (D(i + 1, j) - D(i + 1, j - 1)) / dy;
243  } else if (u <= dx * v / dy and v <= 0.0) { //5
244  tempFD = -u * (D(i, j) - D(i + 1, j)) / dx - v * (D(i + 1, j) - D(i + 1, j + 1)) / dy;
245  } else if (u >= dx * v / dy and u <= 0.0) { //6
246  tempFD = -u * (D(i, j + 1) - D(i + 1, j + 1)) / dx - v * (D(i, j) - D(i, j + 1)) / dy;
247  } else if (u <= -dx * v / dy and u >= 0.0) { //7
248  tempFD = u * (D(i, j + 1) - D(i - 1, j + 1)) / dx - v * (D(i, j) - D(i, j + 1)) / dy;
249  } else if (u >= -dx * v / dy and v <= 0.0) { //8
250  tempFD = u * (D(i, j) - D(i - 1, j)) / dx - v * (D(i - 1, j) - D(i - 1, j + 1)) / dy;
251  } else {
252  m_log->message(3, "######### missing case of angle %f of %f and %f at %d, %d \n",
253  atan(v / u) / M_PI * 180., u * 3e7, v * 3e7, i, j);
254  }
255  } else {
256  tempFD += u * (u < 0 ? D(i + 1, j) - D(i, j) : D(i, j) - D(i - 1, j)) / dx;
257  tempFD += v * (v < 0 ? D(i, j + 1) - D(i, j) : D(i, j) - D(i, j - 1)) / dy;
258  }
259 
260  D_new(i, j) -= tempFD * dt;
261 
262  //sources /////////////////////////////////////////////////////////////////
263  ///von mises criterion
264 
265  double
266  txx = m_deviatoric_stresses(i, j, 0),
267  tyy = m_deviatoric_stresses(i, j, 1),
268  txy = m_deviatoric_stresses(i, j, 2),
269  T1 = 0.5 * (txx + tyy) + sqrt(0.25 * pow(txx - tyy, 2) + pow(txy, 2)), //Pa
270  T2 = 0.5 * (txx + tyy) - sqrt(0.25 * pow(txx - tyy, 2) + pow(txy, 2)), //Pa
271  sigmat = sqrt(pow(T1, 2) + pow(T2, 2) - T1 * T2);
272 
273 
274  ///max shear stress criterion (more stringent than von mises)
275  if (max_shear_stress) {
276  double maxshear = fabs(T1);
277  maxshear = std::max(maxshear, fabs(T2));
278  maxshear = std::max(maxshear, fabs(T1 - T2));
279 
280  sigmat = maxshear;
281  }
282 
283  ///lefm mixed-mode criterion
284  if (lefm) {
285  double sigmamu = 0.1; //friction coefficient between crack faces
286 
287  double sigmac = 0.64 / M_PI; //initial crack depth 20cm
288 
289  double sigmabetatest, sigmanor, sigmatau, Kone, Ktwo, KSI, KSImax = 0.0, sigmatetanull;
290 
291  for (int l = 46; l <= 90; ++l) { //optimize for various precursor angles beta
292  sigmabetatest = l * M_PI / 180.0;
293 
294  //rist_sammonds99
295  sigmanor = 0.5 * (T1 + T2) - (T1 - T2) * cos(2 * sigmabetatest);
296  sigmatau = 0.5 * (T1 - T2) * sin(2 * sigmabetatest);
297  //shayam_wu90
298  if (sigmamu * sigmanor < 0.0) { //compressive case
299  if (fabs(sigmatau) <= fabs(sigmamu * sigmanor)) {
300  sigmatau = 0.0;
301  } else {
302  if (sigmatau > 0) { //coulomb friction opposing sliding
303  sigmatau += (sigmamu * sigmanor);
304  } else {
305  sigmatau -= (sigmamu * sigmanor);
306  }
307  }
308  }
309 
310  //stress intensity factors
311  Kone = sigmanor * sqrt(M_PI * sigmac); //normal
312  Ktwo = sigmatau * sqrt(M_PI * sigmac); //shear
313 
314  if (Ktwo == 0.0) {
315  sigmatetanull = 0.0;
316  } else { //eq15 in hulbe_ledoux10 or eq15 shayam_wu90
317  sigmatetanull = -2.0 * atan((sqrt(pow(Kone, 2) + 8.0 * pow(Ktwo, 2)) - Kone) / (4.0 * Ktwo));
318  }
319 
320  KSI = cos(0.5 * sigmatetanull) *
321  (Kone * cos(0.5 * sigmatetanull) * cos(0.5 * sigmatetanull) - 0.5 * 3.0 * Ktwo * sin(sigmatetanull));
322  // mode I stress intensity
323 
324  KSImax = std::max(KSI, KSImax);
325  }
326  sigmat = KSImax;
327  }
328 
329  //////////////////////////////////////////////////////////////////////////////
330 
331  //fracture density
332  double fdnew = gamma * (m_strain_rates(i, j, 0) - 0.0) * (1 - D_new(i, j));
333  if (sigmat > initThreshold) {
334  D_new(i, j) += fdnew * dt;
335  }
336 
337  //healing
338  double fdheal = gammaheal * (m_strain_rates(i, j, 0) - healThreshold);
339  if (geometry.cell_type.icy(i, j)) {
340  if (constant_healing) {
341  fdheal = gammaheal * (-healThreshold);
342  if (fracture_weighted_healing) {
343  D_new(i, j) += fdheal * dt * (1 - D(i, j));
344  } else {
345  D_new(i, j) += fdheal * dt;
346  }
347  } else if (m_strain_rates(i, j, 0) < healThreshold) {
348  if (fracture_weighted_healing) {
349  D_new(i, j) += fdheal * dt * (1 - D(i, j));
350  } else {
351  D_new(i, j) += fdheal * dt;
352  }
353  }
354  }
355 
356  // bounding
357  D_new(i, j) = pism::clip(D_new(i, j), 0.0, 1.0);
358 
359  if (geometry.cell_type.icy(i, j)) {
360  //fracture toughness
361  m_toughness(i, j) = sigmat;
362 
363  // fracture growth rate
364  if (sigmat > initThreshold) {
365  m_growth_rate(i, j) = fdnew;
366  //m_growth_rate(i,j)=gamma*(vPrinStrain1(i,j)-0.0)*(1-D_new(i,j));
367  } else {
368  m_growth_rate(i, j) = 0.0;
369  }
370 
371  // fracture healing rate
372  if (geometry.cell_type.icy(i, j)) {
373  if (constant_healing or (m_strain_rates(i, j, 0) < healThreshold)) {
374  if (fracture_weighted_healing) {
375  m_healing_rate(i, j) = fdheal * (1 - D(i, j));
376  } else {
377  m_healing_rate(i, j) = fdheal;
378  }
379  } else {
380  m_healing_rate(i, j) = 0.0;
381  }
382  }
383 
384  // fracture age since fracturing occurred
385  {
386  auto a = A.star(i, j);
387  A_new(i, j) -= dt * u * (u < 0 ? a.e - a.ij : a.ij - a.w) / dx;
388  A_new(i, j) -= dt * v * (v < 0 ? a.n - a.ij : a.ij - a.s) / dy;
389  A_new(i, j) += dt;
390  if (sigmat > initThreshold) {
391  A_new(i, j) = 0.0;
392  }
393  }
394 
395  // additional flow enhancement due to fracture softening
396  double phi_exp = 3.0; //flow_law->exponent();
397  double softening = pow((1.0 - (1.0 - soft_residual) * D_new(i, j)), -phi_exp);
398  if (geometry.cell_type.icy(i, j)) {
399  m_flow_enhancement(i, j) = 1.0 / pow(softening, 1 / 3.0);
400  } else {
401  m_flow_enhancement(i, j) = 1.0;
402  }
403  }
404 
405  // boundary condition
406  if (geometry.cell_type.grounded(i, j) and not do_fracground) {
407 
408  if (bc_mask(i, j) > 0.5) {
409  D_new(i, j) = fdBoundaryValue;
410 
411  {
412  A_new(i, j) = 0.0;
413  m_growth_rate(i, j) = 0.0;
414  m_healing_rate(i, j) = 0.0;
415  m_flow_enhancement(i, j) = 1.0;
416  m_toughness(i, j) = 0.0;
417  }
418  }
419  }
420 
421  // ice free regions and boundary of computational domain
422  if (geometry.cell_type.ice_free(i, j) or i == 0 or j == 0 or i == Mx - 1 or j == My - 1) {
423 
424  D_new(i, j) = 0.0;
425 
426  {
427  A_new(i, j) = 0.0;
428  m_growth_rate(i, j) = 0.0;
429  m_healing_rate(i, j) = 0.0;
430  m_flow_enhancement(i, j) = 1.0;
431  m_toughness(i, j) = 0.0;
432  }
433  }
434 
435  if (constant_fd) { // no fd evolution
436  D_new(i, j) = D(i, j);
437  }
438  }
439 
440  A_new.update_ghosts(A);
441  D_new.update_ghosts(D);
442 }
