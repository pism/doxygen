101 //! Return value of scalar quantity at level z (m) above base of ice (by linear interpolation).
102 double IceModelVec3::interpolate(int i, int j, double z) const {
103  auto zs = levels();
104 
105 #if (Pism_DEBUG==1)
106  assert(m_array != NULL);
107  check_array_indices(i, j, 0);
108 
109  if (not legal_level(zs, z)) {
110  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
111  "IceModelVec3 interpolate(): level %f is not legal; name = %s",
112  z, m_impl->name.c_str());
113  }
114 #endif
115 
116  const auto* column = get_column(i, j);
117 
118  if (z >= zs.back()) {
119  unsigned int nlevels = zs.size();
120  return column[nlevels - 1];
121  }
122 
123  if (z <= zs.front()) {
124  return column[0];
125  }
126 
127  auto mcurr = gsl_interp_accel_find(m_impl->bsearch_accel,
128  &zs[0], zs.size(), z);
129 
130  const double incr = (z - zs[mcurr]) / (zs[mcurr+1] - zs[mcurr]);
131  const double valm = column[mcurr];
132  return valm + incr * (column[mcurr + 1] - valm);
133 }

182 /** Sum a 3-D vector in the Z direction to create a 2-D vector.
183 
184 Note that this sums up all the values in a column, including ones
185 above the ice. This may or may not be what you need. Also, take a look
186 at IceModel::compute_ice_enthalpy(PetscScalar &result) in iMreport.cc.
187 
188 As for the difference between IceModelVec2 and IceModelVec2S, the
189 former can store fields with more than 1 "degree of freedom" per grid
190 point (such as 2D fields on the "staggered" grid, with the first
191 degree of freedom corresponding to the i-offset and second to
192 j-offset).
193 
194 IceModelVec2S is just IceModelVec2 with "dof == 1", and
195 IceModelVec2V is IceModelVec2 with "dof == 2". (Plus some extra
196 methods, of course.)
197 
198 Either one of IceModelVec2 and IceModelVec2S would work in this
199 case.
200 
201 Computes output = A*output + B*sum_columns(input) + C
202 
203 @see https://github.com/pism/pism/issues/229 */
204 void sum_columns(const IceModelVec3 &data, double A, double B, IceModelVec2S &output) {
205  const unsigned int Mz = data.grid()->Mz();
206 
207  AccessList access{&data, &output};
208 
209  for (Points p(*data.grid()); p; p.next()) {
210  const int i = p.i(), j = p.j();
211 
212  const double *column = data.get_column(i,j);
213 
214  double scalar_sum = 0.0;
215  for (unsigned int k = 0; k < Mz; ++k) {
216  scalar_sum += column[k];
217  }
218  output(i,j) = A * output(i,j) + B * scalar_sum;
219  }
220 }

222 void IceModelVec3::copy_from(const IceModelVec3 &input) {
223  IceModelVec::AccessList list {this, &input};
224 
225  assert(levels().size() == input.levels().size());
226  assert(ndof() == input.ndof());
227 
228  // 3D arrays have more than one level and ndof() of 1, collections of fields have one
229  // level and ndof() > 1
230  auto N = std::max((size_t)ndof(), levels().size());
231 
232  ParallelSection loop(m_impl->grid->com);
233  try {
234  for (Points p(*m_impl->grid); p; p.next()) {
235  const int i = p.i(), j = p.j();
236 
237 #if PETSC_VERSION_LT(3, 12, 0)
238  PetscMemmove(this->get_column(i, j),
239  const_cast<double*>(input.get_column(i, j)), N*sizeof(double));
240 #else
241  PetscArraymove(this->get_column(i, j), input.get_column(i, j), N);
242 #endif
243  }
244  } catch (...) {
245  loop.failed();
246  }
247  loop.check();
248 
249  update_ghosts();
250 
251  inc_state_counter();
252 }
