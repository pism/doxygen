161 //! Return value of scalar quantity at level z (m) above base of ice (by linear interpolation).
162 double IceModelVec3::interpolate(int i, int j, double z) const {
163 #if (Pism_DEBUG==1)
164  assert(m_array != NULL);
165  check_array_indices(i, j, 0);
166 
167  if (not legal_level(z)) {
168  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
169  "IceModelVec3 interpolate(): level %f is not legal; name = %s",
170  z, m_impl->name.c_str());
171  }
172 #endif
173 
174  double ***arr = (double***) m_array;
175  if (z >= m_impl->zlevels.back()) {
176  unsigned int nlevels = m_impl->zlevels.size();
177  return arr[j][i][nlevels - 1];
178  } else if (z <= m_impl->zlevels.front()) {
179  return arr[j][i][0];
180  }
181 
182  unsigned int mcurr = gsl_interp_accel_find(m_impl->bsearch_accel,
183  &m_impl->zlevels[0], m_impl->zlevels.size(), z);
184 
185  const double incr = (z - m_impl->zlevels[mcurr]) / (m_impl->zlevels[mcurr+1] - m_impl->zlevels[mcurr]);
186  const double valm = arr[j][i][mcurr];
187  return valm + incr * (arr[j][i][mcurr+1] - valm);
188 }

246 /** Sum a 3-D vector in the Z direction to create a 2-D vector.
247 
248 Note that this sums up all the values in a column, including ones
249 above the ice. This may or may not be what you need. Also, take a look
250 at IceModel::compute_ice_enthalpy(PetscScalar &result) in iMreport.cc.
251 
252 As for the difference between IceModelVec2 and IceModelVec2S, the
253 former can store fields with more than 1 "degree of freedom" per grid
254 point (such as 2D fields on the "staggered" grid, with the first
255 degree of freedom corresponding to the i-offset and second to
256 j-offset).
257 
258 IceModelVec2S is just IceModelVec2 with "dof == 1", and
259 IceModelVec2V is IceModelVec2 with "dof == 2". (Plus some extra
260 methods, of course.)
261 
262 Either one of IceModelVec2 and IceModelVec2S would work in this
263 case.
264 
265 Computes output = A*output + B*sum_columns(input) + C
266 
267 @see https://github.com/pism/pism/issues/229 */
268 void IceModelVec3::sum_columns(double A, double B, IceModelVec2S &output) const {
269  const unsigned int Mz = m_impl->grid->Mz();
270 
271  AccessList access{this, &output};
272 
273  for (Points p(*m_impl->grid); p; p.next()) {
274  const int i = p.i(), j = p.j();
275 
276  const double *column = this->get_column(i,j);
277 
278  double scalar_sum = 0.0;
279  for (unsigned int k = 0; k < Mz; ++k) {
280  scalar_sum += column[k];
281  }
282  output(i,j) = A * output(i,j) + B * scalar_sum;
283  }
284 }
