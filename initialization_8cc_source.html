81  bool use_calendar = m_config->get_flag("output.runtime.time_use_calendar");

106 //! Sets the starting values of model state variables.
107 /*!
108  There are two cases:
109 
110  1) Initializing from a PISM output file.
111 
112  2) Setting the values using command-line options only (verification and
113  simplified geometry runs, for example) or from a bootstrapping file, using
114  heuristics to fill in missing and 3D fields.
115 
116  Calls IceModel::regrid().
117 
118  This function is called after all the memory allocation is done and all the
119  physical parameters are set.
120 
121  Calling this method should be all one needs to set model state variables.
122  Please avoid modifying them in other parts of the initialization sequence.
123 
124  Also, please avoid operations that would make it unsafe to call this more
125  than once (memory allocation is one example).
126  */
127 void IceModel::model_state_setup() {
128 
129  // Check if we are initializing from a PISM output file:
130  InputOptions input = process_input_options(m_ctx->com(), m_config);
131 
132  const bool use_input_file = input.type == INIT_BOOTSTRAP or input.type == INIT_RESTART;
133 
134  std::unique_ptr<File> input_file;
135 
136  if (use_input_file) {
137  input_file.reset(new File(m_grid->com, input.filename, PISM_GUESS, PISM_READONLY));
138  }
139 
140  // Initialize 2D fields owned by IceModel (ice geometry, etc)
141  {
142  switch (input.type) {
143  case INIT_RESTART:
144  restart_2d(*input_file, input.record);
145  break;
146  case INIT_BOOTSTRAP:
147  bootstrap_2d(*input_file);
148  break;
149  case INIT_OTHER:
150  default:
151  initialize_2d();
152  }
153 
154  regrid();
155  }
156 
157  // Get projection information and compute latitudes and longitudes *before* a component
158  // decides to use them...
159  {
160  if (use_input_file) {
161  std::string mapping_name = m_grid->get_mapping_info().mapping.get_name();
162  MappingInfo info = get_projection_info(*input_file, mapping_name,
163  m_ctx->unit_system());
164 
165  if (not info.proj.empty()) {
166  m_log->message(2, "* Got projection parameters \"%s\" from \"%s\".\n",
167  info.proj.c_str(), input.filename.c_str());
168  }
169 
170  m_output_global_attributes["proj"] = info.proj;
171  m_grid->set_mapping_info(info);
172 
173  std::string history = input_file->read_text_attribute("PISM_GLOBAL", "history");
174  m_output_global_attributes["history"] = history + m_output_global_attributes.get_string("history");
175 
176  }
177 
178  compute_lat_lon();
179  }
180 
181  m_sea_level->init(m_geometry);
182 
183  // Initialize a bed deformation model.
184  if (m_beddef) {
185  m_beddef->init(input, m_geometry.ice_thickness, m_sea_level->elevation());
186  m_grid->variables().add(m_beddef->bed_elevation());
187  m_grid->variables().add(m_beddef->uplift());
188  }
189 
190  // Now ice thickness, bed elevation, and sea level are available, so we can compute the ice
191  // surface elevation and the cell type mask. This also ensures consistency of ice geometry.
192  //
193  // The stress balance code is executed near the beginning of a step and ice geometry is
194  // updated near the end, so during the second time step the stress balance code is
195  // guaranteed not to see "icebergs". Here we make sure that the first time step is OK
196  // too.
197  enforce_consistency_of_geometry(REMOVE_ICEBERGS);
198 
199  // By now ice geometry is set (including regridding) and so we can initialize the ocean model,
200  // which may need ice thickness, bed topography, and the cell type mask.
201  {
202  m_ocean->init(m_geometry);
203  }
204 
205  // Now surface elevation is initialized, so we can initialize surface models (some use
206  // elevation-based parameterizations of surface temperature and/or mass balance).
207  m_surface->init(m_geometry);
208 
209  if (m_subglacial_hydrology) {
210 
211  switch (input.type) {
212  case INIT_RESTART:
213  m_subglacial_hydrology->restart(*input_file, input.record);
214  break;
215  case INIT_BOOTSTRAP:
216  m_subglacial_hydrology->bootstrap(*input_file,
217  m_geometry.ice_thickness);
218  break;
219  case INIT_OTHER:
220  {
221  IceModelVec2S
222  &W_till = *m_work2d[0],
223  &W = *m_work2d[1],
224  &P = *m_work2d[2];
225 
226  W_till.set(m_config->get_number("bootstrapping.defaults.tillwat"));
227  W.set(m_config->get_number("bootstrapping.defaults.bwat"));
228  P.set(m_config->get_number("bootstrapping.defaults.bwp"));
229 
230  m_subglacial_hydrology->init(W_till, W, P);
231  break;
232  }
233  }
234  }
235 
236  // basal_yield_stress_model->init() needs till water thickness so this must happen
237  // after subglacial_hydrology->init()
238  if (m_basal_yield_stress_model) {
239  auto inputs = yield_stress_inputs();
240 
241  switch (input.type) {
242  case INIT_RESTART:
243  m_basal_yield_stress_model->restart(*input_file, input.record);
244  break;
245  case INIT_BOOTSTRAP:
246  m_basal_yield_stress_model->bootstrap(*input_file, inputs);
247  break;
248  default:
249  case INIT_OTHER:
250  m_basal_yield_stress_model->init(inputs);
251  }
252  m_basal_yield_stress.copy_from(m_basal_yield_stress_model->basal_material_yield_stress());
253  }
254 
255  // Initialize the bedrock thermal layer model.
256  //
257  // If
258  // - PISM is bootstrapping and
259  // - we are using a non-zero-thickness thermal layer
260  //
261  // initialization of m_btu requires the temperature at the top of the bedrock. This is a problem
262  // because get_bed_top_temp() uses the enthalpy field that is not initialized until later and
263  // bootstrapping enthalpy uses the flux through the bottom surface of the ice (top surface of the
264  // bedrock) provided by m_btu.
265  //
266  // We get out of this by using the fact that the full state of m_btu is not needed and
267  // bootstrapping of the temperature field can be delayed.
268  //
269  // Note that to bootstrap m_btu we use the steady state solution of the heat equation in columns
270  // of the bedrock (a straight line at each column), so the flux through the top surface of the
271  // bedrock after bootstrapping is the same as the time-independent geothermal flux applied at the
272  // BOTTOM surface of the bedrock layer.
273  //
274  // The code then delays bootstrapping of the thickness field until the first time step.
275  if (m_btu) {
276  m_btu->init(input);
277  }
278 
279  if (m_age_model) {
280  m_age_model->init(input);
281  m_grid->variables().add(m_age_model->age());
282  }
283 
284  // Initialize the energy balance sub-model.
285  {
286  switch (input.type) {
287  case INIT_RESTART:
288  {
289  m_energy_model->restart(*input_file, input.record);
290  break;
291  }
292  case INIT_BOOTSTRAP:
293  {
294 
295  m_energy_model->bootstrap(*input_file,
296  m_geometry.ice_thickness,
297  m_surface->temperature(),
298  m_surface->mass_flux(),
299  m_btu->flux_through_top_surface());
300  break;
301  }
302  case INIT_OTHER:
303  default:
304  {
305  m_basal_melt_rate.set(m_config->get_number("bootstrapping.defaults.bmelt"));
306 
307  m_energy_model->initialize(m_basal_melt_rate,
308  m_geometry.ice_thickness,
309  m_surface->temperature(),
310  m_surface->mass_flux(),
311  m_btu->flux_through_top_surface());
312 
313  }
314  }
315  m_grid->variables().add(m_energy_model->enthalpy());
316  }
317 
318  // this has to go after we add enthalpy to m_grid->variables()
319  if (m_stress_balance) {
320  m_stress_balance->init();
321  }
322 
323  // we keep ice thickness fixed at all the locations where the sliding (SSA) velocity is
324  // prescribed
325  {
326  IceModelVec::AccessList list{&m_ice_thickness_bc_mask, &m_velocity_bc_mask};
327 
328  for (Points p(*m_grid); p; p.next()) {
329  const int i = p.i(), j = p.j();
330 
331  if (m_velocity_bc_mask.as_int(i, j)) {
332  m_ice_thickness_bc_mask(i, j) = 1.0;
333  }
334  }
335  }
336 
337  // miscellaneous steps
338  {
339  reset_counters();
340 
341  auto startstr = pism::printf("PISM (%s) started on %d procs.",
342  pism::revision, (int)m_grid->size());
343  prepend_history(startstr + args_string());
344  }
345 }

362 void IceModel::bootstrap_2d(const File &input_file) {
363 
364  m_log->message(2, "bootstrapping from file '%s'...\n", input_file.filename().c_str());
365 
366  auto usurf = input_file.find_variable("usurf", "surface_altitude");
367 
368  bool mask_found = input_file.find_variable("mask");
369 
370  // now work through all the 2d variables, regridding if present and otherwise
371  // setting to default values appropriately
372 
373  if (mask_found) {
374  m_log->message(2, " WARNING: 'mask' found; IGNORING IT!\n");
375  }
376 
377  if (usurf.exists) {
378  m_log->message(2, " WARNING: surface elevation 'usurf' found; IGNORING IT!\n");
379  }
380 
381  m_log->message(2, " reading 2D model state variables by regridding ...\n");
382 
383  // longitude
384  {
385  m_geometry.longitude.regrid(input_file, OPTIONAL);
386 
387  auto lon = input_file.find_variable("lon", "longitude");
388 
389  if (not lon.exists) {
390  m_geometry.longitude.metadata()["missing_at_bootstrap"] = "true";
391  }
392  }
393 
394  // latitude
395  {
396  m_geometry.latitude.regrid(input_file, OPTIONAL);
397 
398  auto lat = input_file.find_variable("lat", "latitude");
399 
400  if (not lat.exists) {
401  m_geometry.latitude.metadata()["missing_at_bootstrap"] = "true";
402  }
403  }
404 
405  m_geometry.ice_thickness.regrid(input_file, OPTIONAL,
406  m_config->get_number("bootstrapping.defaults.ice_thickness"));
407  // check the range of the ice thickness
408  {
409  auto thk_range = m_geometry.ice_thickness.range();
410 
411  if (thk_range[1] >= m_grid->Lz() + 1e-6) {
412  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "Maximum ice thickness (%f meters)\n"
413  "exceeds the height of the computational domain (%f meters).",
414  thk_range[1], m_grid->Lz());
415  }
416  }
417 
418  if (m_config->get_flag("geometry.part_grid.enabled")) {
419  // Read the Href field from an input file. This field is
420  // grid-dependent, so interpolating it from one grid to a
421  // different one does not make sense in general.
422  // (IceModel::Href_cleanup() will take care of the side effects of
423  // such interpolation, though.)
424  //
425  // On the other hand, we need to read it in to be able to re-start
426  // from a PISM output file using the -bootstrap option.
427  m_geometry.ice_area_specific_volume.regrid(input_file, OPTIONAL, 0.0);
428  }
429 
430  if (m_config->get_flag("stress_balance.ssa.dirichlet_bc")) {
431  // Do not use Dirichlet B.C. anywhere if bc_mask is not present.
432  m_velocity_bc_mask.regrid(input_file, OPTIONAL, 0.0);
433  // In absence of u_bc and v_bc in the file the only B.C. that make sense are the
434  // zero Dirichlet B.C.
435  m_velocity_bc_values.regrid(input_file, OPTIONAL, 0.0);
436  } else {
437  m_velocity_bc_mask.set(0.0);
438  m_velocity_bc_values.set(0.0);
439  }
440 
441  m_ice_thickness_bc_mask.regrid(input_file, OPTIONAL, 0.0);
442 
443  // check if Lz is valid
444  auto thk_range = m_geometry.ice_thickness.range();
445 
446  if (thk_range[1] > m_grid->Lz()) {
447  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "Max. ice thickness (%3.3f m)\n"
448  "exceeds the height of the computational domain (%3.3f m).",
449  thk_range[1], m_grid->Lz());
450  }
451 }

459 //! Manage regridding based on user options.
460 void IceModel::regrid() {
461 
462  auto filename = m_config->get_string("input.regrid.file");
463  auto regrid_vars = set_split(m_config->get_string("input.regrid.vars"), ',');
464 
465  // Return if no regridding is requested:
466  if (filename.empty()) {
467  return;
468  }
469 
470  m_log->message(2, "regridding from file %s ...\n", filename.c_str());
471 
472  {
473  File regrid_file(m_grid->com, filename, PISM_GUESS, PISM_READONLY);
474  for (auto v : m_model_state) {
475  if (regrid_vars.find(v->get_name()) != regrid_vars.end()) {
476  v->regrid(regrid_file, CRITICAL);
477  }
478  }
479 
480  // Check the range of the ice thickness.
481  {
482  double
483  max_thickness = m_geometry.ice_thickness.range()[1],
484  Lz = m_grid->Lz();
485 
486  if (max_thickness >= Lz + 1e-6) {
487  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "Maximum ice thickness (%f meters)\n"
488  "exceeds the height of the computational domain (%f meters).",
489  max_thickness, Lz);
490  }
491  }
492  }

604 //! \brief Decide which subglacial hydrology model to use.
605 void IceModel::allocate_subglacial_hydrology() {
606 
607  using namespace pism::hydrology;
608 
609  std::string hydrology_model = m_config->get_string("hydrology.model");
610 
611  if (m_subglacial_hydrology) { // indicates it has already been allocated
612  return;
613  }
614 
615  m_log->message(2, "# Allocating a subglacial hydrology model...\n");
616 
617  if (hydrology_model == "null") {
618  m_subglacial_hydrology.reset(new NullTransport(m_grid));

736 //! Miscellaneous initialization tasks plus tasks that need the fields that can come from regridding.
737 void IceModel::misc_setup() {

741 
742  if (not (opts.type == INIT_OTHER)) {
743  // initializing from a file
744  File file(m_grid->com, opts.filename, PISM_GUESS, PISM_READONLY);
745 
746  std::string source = file.read_text_attribute("PISM_GLOBAL", "source");
747

749  // If it's missing, print a warning
750  if (source.empty()) {
751  m_log->message(1,
752  "PISM WARNING: file '%s' does not have the 'source' global attribute.\n"
753  " If '%s' is a PISM output file, please run the following to get rid of this warning:\n"
754  " ncatted -a source,global,c,c,PISM %s\n",
755  opts.filename.c_str(), opts.filename.c_str(), opts.filename.c_str());
756  } else if (source.find("PISM") == std::string::npos) {
757  // If the 'source' attribute does not contain the string "PISM", then print
758  // a message and stop:
759  m_log->message(1,
760  "PISM WARNING: '%s' does not seem to be a PISM output file.\n"
761  " If it is, please make sure that the 'source' global attribute contains the string \"PISM\".\n",
762  opts.filename.c_str());
763  }
764  }
765  }
766 
767  {
768  // A single record of a time-dependent variable cannot exceed 2^32-4
769  // bytes in size. See the NetCDF User's Guide
770  // <http://www.unidata.ucar.edu/software/netcdf/docs/netcdf.html#g_t64-bit-Offset-Limitations>.
771  // Here we use "long int" to avoid integer overflow.
772  const long int two_to_thirty_two = 4294967296L;
773  const long int
774  Mx = m_grid->Mx(),
775  My = m_grid->My(),
776  Mz = m_grid->Mz();
777  std::string output_format = m_config->get_string("output.format");
778  if (Mx * My * Mz * sizeof(double) > two_to_thirty_two - 4 and
779  (output_format == PISM_NETCDF3)) {
780  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
781  "The computational grid is too big to fit in a NetCDF-3 file.\n"
782  "Each 3D variable requires %lu Mb.\n"
783  "Please use '-o_format pnetcdf or -o_format netcdf4_parallel to proceed.",
784  Mx * My * Mz * sizeof(double) / (1024 * 1024));
785  }
786  }
787 
788  m_output_vars = output_variables(m_config->get_string("output.size"));
789 
790 #if (Pism_USE_PROJ==1)
791  {
792  std::string proj_string = m_grid->get_mapping_info().proj;
793  if (not proj_string.empty()) {
794  m_output_vars.insert("lon_bnds");
795  m_output_vars.insert("lat_bnds");
796  }
797  }
798 #endif
799 
800  init_calving();
801  init_frontal_melt();
802  init_front_retreat();
803  init_diagnostics();
804  init_snapshots();
805  init_backups();
806  init_timeseries();
807  init_extras();
808 
809  // a report on whether PISM-PIK modifications of IceModel are in use
810  {
811  std::vector<std::string> pik_methods;
812  if (m_config->get_flag("geometry.part_grid.enabled")) {
813  pik_methods.push_back("part_grid");
814  }
815  if (m_config->get_flag("geometry.remove_icebergs")) {
816  pik_methods.push_back("kill_icebergs");
817  }
818 
819  if (not pik_methods.empty()) {
820  m_log->message(2,
821  "* PISM-PIK mass/geometry methods are in use: %s\n",
822  join(pik_methods, ", ").c_str());
823  }
824  }
825

827  {
828  // reset: this gives diagnostics a chance to capture the current state of the model at the
829  // beginning of the run
830  for (auto d : m_diagnostics) {
831  d.second->reset();
832  }
833 
834  // read in the state (accumulators) if we are re-starting a run
835  if (opts.type == INIT_RESTART) {
836  File file(m_grid->com, opts.filename, PISM_GUESS, PISM_READONLY);
837  for (auto d : m_diagnostics) {
838  d.second->init(file, opts.record);
839  }
840  }
841  }
842 
843  if (m_surface_input_for_hydrology) {
844  ForcingOptions surface_input(*m_ctx, "hydrology.surface_input");
845  m_surface_input_for_hydrology->init(surface_input.filename,
846  surface_input.periodic);
847  }
848

851  m_fracture->initialize();
