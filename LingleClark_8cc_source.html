38  m_total_displacement(m_grid, "bed_displacement", WITHOUT_GHOSTS),
39  m_relief(m_grid, "bed_relief", WITHOUT_GHOSTS),
40  m_load_thickness(grid, "load_thickness", WITHOUT_GHOSTS),
41  m_elastic_displacement(grid, "elastic_bed_displacement", WITHOUT_GHOSTS) {
42 
43  m_time_name = m_config->get_string("time.dimension_name") + "_lingle_clark";
44  m_t_last = m_grid->ctx()->time()->current();
45  m_update_interval = m_config->get_number("bed_deformation.lc.update_interval", "seconds");
46  m_t_eps = m_config->get_number("time_stepping.resolution", "seconds");
47 
48  if (m_update_interval < 1.0) {
49  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
50  "invalid bed_deformation.lc.update_interval = %f seconds",
51  m_update_interval);
52  }
53 
54  // A work vector. This storage is used to put thickness change on rank 0 and to get the plate
55  // displacement change back.

57  "total (viscous and elastic) displacement "
58  "in the Lingle-Clark bed deformation model",
59  "meters", "meters", "", 0);
60 
61  m_work0 = m_total_displacement.allocate_proc0_copy();
62 
63  m_relief.set_attrs("internal",
64  "bed relief relative to the modeled bed displacement",
65  "meters", "meters", "", 0);
66 
67  bool use_elastic_model = m_config->get_flag("bed_deformation.lc.elastic_model");
68 
69  m_elastic_displacement.set_attrs("model state",
70  "elastic part of the displacement in the "
71  "Lingle-Clark bed deformation model; "
72  "see :cite:`BLKfastearth`", "meters", "meters", "", 0);
73  m_elastic_displacement0 = m_elastic_displacement.allocate_proc0_copy();
74 
75  const int
76  Mx = m_grid->Mx(),
77  My = m_grid->My(),
78  Z = m_config->get_number("bed_deformation.lc.grid_size_factor"),
79  Nx = Z*(Mx - 1) + 1,
80  Ny = Z*(My - 1) + 1;
81 
82  const double
83  Lx = Z * (m_grid->x0() - m_grid->x(0)),
84  Ly = Z * (m_grid->y0() - m_grid->y(0));
85

87  Lx, Ly,
88  m_grid->x0(), m_grid->y0(),
89  Nx, Ny, CELL_CORNER, NOT_PERIODIC);

92  "viscous_bed_displacement", WITHOUT_GHOSTS));
93  m_viscous_displacement->set_attrs("model state",

97  "meters", "meters", "", 0);
98

100  m_viscous_displacement->metadata().x().set_name("x_lc");
101  m_viscous_displacement->metadata().y().set_name("y_lc");

104  m_viscous_displacement->metadata()["coordinates"] = "";

108  ParallelSection rank0(m_grid->com);

126 /*!
127  * Initialize the model by computing the viscous bed displacement using uplift and the elastic
128  * response using ice thickness.
129  *
130  * Then compute the bed relief as the difference between bed elevation and total bed displacement.
131  *
132  * This method has to initialize m_viscous_displacement, m_elastic_displacement,
133  * m_total_displacement, and m_relief.
134  */
135 void LingleClark::bootstrap_impl(const IceModelVec2S &bed_elevation,
136  const IceModelVec2S &bed_uplift,
137  const IceModelVec2S &ice_thickness,
138  const IceModelVec2S &sea_level_elevation) {
139  m_t_last = m_grid->ctx()->time()->current();
140 
141  m_topg_last.copy_from(bed_elevation);
142 
143  compute_load(bed_elevation, ice_thickness, sea_level_elevation,
144  m_load_thickness);
145 
146  std::shared_ptr<petsc::Vec> thickness0 = m_load_thickness.allocate_proc0_copy();
147 
148  // initialize the plate displacement
149  {
150  bed_uplift.put_on_proc0(*m_work0);
151  m_load_thickness.put_on_proc0(*thickness0);
152 
153  ParallelSection rank0(m_grid->com);
154  try {
155  if (m_grid->rank() == 0) {
156  PetscErrorCode ierr = 0;
157 
158  m_serial_model->bootstrap(*thickness0, *m_work0);
159 
160  ierr = VecCopy(m_serial_model->total_displacement(), *m_work0);
161  PISM_CHK(ierr, "VecCopy");
162 
163  ierr = VecCopy(m_serial_model->viscous_displacement(), *m_viscous_displacement0);
164  PISM_CHK(ierr, "VecCopy");
165 
166  ierr = VecCopy(m_serial_model->elastic_displacement(), *m_elastic_displacement0);
167  PISM_CHK(ierr, "VecCopy");
168  }
169  } catch (...) {
170  rank0.failed();
171  }
172  rank0.check();
173  }
174 
175  m_viscous_displacement->get_from_proc0(*m_viscous_displacement0);
176 
177  m_elastic_displacement.get_from_proc0(*m_elastic_displacement0);
178 
179  m_total_displacement.get_from_proc0(*m_work0);
180 
181  // compute bed relief
182  m_topg.add(-1.0, m_total_displacement, m_relief);
183 }

190 IceModelVec2S::Ptr LingleClark::elastic_load_response_matrix() const {
191  IceModelVec2S::Ptr result(new IceModelVec2S(m_extended_grid, "lrm", WITHOUT_GHOSTS));
192 
193  int
194  Nx = m_extended_grid->Mx(),
195  Ny = m_extended_grid->My();
196 
197  auto lrm0 = result->allocate_proc0_copy();
198 
199  {
200  ParallelSection rank0(m_grid->com);
201  try {
202  if (m_grid->rank() == 0) {
203  std::vector<std::complex<double> > array(Nx * Ny);
204 
205  m_serial_model->compute_load_response_matrix((fftw_complex*)array.data());
206 
207  get_real_part((fftw_complex*)array.data(), 1.0, Nx, Ny, Nx, Ny, 0, 0, *lrm0);
208  }
209  } catch (...) {
210  rank0.failed();
211  }
212  rank0.check();
213  }
214 
215  result->get_from_proc0(*lrm0);
216 
217  return result;

220 /*! Initialize the Lingle-Clark bed deformation model.
221  *
222  * Inputs:
223  *
224  * - bed topography,
225  * - ice thickness,
226  * - plate displacement (either read from a file or bootstrapped using uplift) and
227  * possibly re-gridded.
228  */
229 void LingleClark::init_impl(const InputOptions &opts, const IceModelVec2S &ice_thickness,
230  const IceModelVec2S &sea_level_elevation) {
231  m_log->message(2, "* Initializing the Lingle-Clark bed deformation model...\n");
232 
233  if (opts.type == INIT_RESTART or opts.type == INIT_BOOTSTRAP) {
234  File input_file(m_grid->com, opts.filename, PISM_NETCDF3, PISM_READONLY);
235 
236  if (input_file.find_variable(m_time_name)) {
237  input_file.read_variable(m_time_name, {0}, {1}, &m_t_last);
238  } else {
239  m_t_last = m_grid->ctx()->time()->current();
240  }
241  } else {
242  m_t_last = m_grid->ctx()->time()->current();
243  }
244 
245  // Initialize bed topography and uplift maps.
246  BedDef::init_impl(opts, ice_thickness, sea_level_elevation);
247 
248  m_topg_last.copy_from(m_topg);
249 
250  if (opts.type == INIT_RESTART) {
251  // Set viscous displacement by reading from the input file.
252  m_viscous_displacement->read(opts.filename, opts.record);
253  // Set elastic displacement by reading from the input file.
254  m_elastic_displacement.read(opts.filename, opts.record);
255  } else if (opts.type == INIT_BOOTSTRAP) {
256  this->bootstrap(m_topg, m_uplift, ice_thickness, sea_level_elevation);
257  } else {
258  // do nothing
259  }
260 
261  // Try re-gridding plate_displacement.
262  regrid("Lingle-Clark bed deformation model",
263  *m_viscous_displacement, REGRID_WITHOUT_REGRID_VARS);
264  regrid("Lingle-Clark bed deformation model",
265  m_elastic_displacement, REGRID_WITHOUT_REGRID_VARS);
266 
267  compute_load(m_topg, ice_thickness, sea_level_elevation,
268  m_load_thickness);
269 
270  // Now that viscous displacement and elastic displacement are finally initialized,
271  // put them on rank 0 and initialize the serial model itself.
272  {
273  m_viscous_displacement->put_on_proc0(*m_viscous_displacement0);
274  m_elastic_displacement.put_on_proc0(*m_work0);
275 
276  ParallelSection rank0(m_grid->com);
277  try {
278  if (m_grid->rank() == 0) { // only processor zero does the work
279  PetscErrorCode ierr = 0;
280 
281  m_serial_model->init(*m_viscous_displacement0, *m_work0);
282 
283  ierr = VecCopy(m_serial_model->total_displacement(), *m_work0);
284  PISM_CHK(ierr, "VecCopy");
285  }
286  } catch (...) {
287  rank0.failed();
288  }
289  rank0.check();
290  }
291 
292  m_total_displacement.get_from_proc0(*m_work0);
293 
294  // compute bed relief
295  m_topg.add(-1.0, m_total_displacement, m_relief);
296 }

298 MaxTimestep LingleClark::max_timestep_impl(double t) const {
299 
300  if (t < m_t_last) {
301  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
302  "time %f is less than the previous time %f",
303  t, m_t_last);
304  }
305 
306  // Find the smallest time of the form m_t_last + k * m_update_interval that is greater
307  // than t
308  double k = ceil((t - m_t_last) / m_update_interval);
309 
310  double
311  t_next = m_t_last + k * m_update_interval,
312  dt_max = t_next - t;
313 
314  if (dt_max < m_t_eps) {
315  dt_max = m_update_interval;
316  }
317 
318  return MaxTimestep(dt_max, "bed_def lc");
319 }

340 void LingleClark::step(const IceModelVec2S &ice_thickness,
341  const IceModelVec2S &sea_level_elevation,
342  double dt) {
343 
344  compute_load(m_topg, ice_thickness, sea_level_elevation,
345  m_load_thickness);
346 
347  m_load_thickness.put_on_proc0(*m_work0);
348 
349  ParallelSection rank0(m_grid->com);
350  try {
351  if (m_grid->rank() == 0) { // only processor zero does the step
352  PetscErrorCode ierr = 0;
353 
354  m_serial_model->step(dt, *m_work0);
355 
356  ierr = VecCopy(m_serial_model->total_displacement(), *m_work0);
357  PISM_CHK(ierr, "VecCopy");
358 
359  ierr = VecCopy(m_serial_model->viscous_displacement(), *m_viscous_displacement0);
360  PISM_CHK(ierr, "VecCopy");
361 
362  ierr = VecCopy(m_serial_model->elastic_displacement(), *m_elastic_displacement0);
363  PISM_CHK(ierr, "VecCopy");
364  }
365  } catch (...) {
366  rank0.failed();
367  }
368  rank0.check();
369 
370  m_viscous_displacement->get_from_proc0(*m_viscous_displacement0);
371 
372  m_elastic_displacement.get_from_proc0(*m_elastic_displacement0);
373 
374  m_total_displacement.get_from_proc0(*m_work0);
375 
376  // Update bed elevation using bed displacement and relief.
377  {
378  m_total_displacement.add(1.0, m_relief, m_topg);
379  // Increment the topg state counter. SIAFD relies on this!
380  m_topg.inc_state_counter();
381  }
382 
383  //! Finally, we need to update bed uplift and topg_last.
384  compute_uplift(m_topg, m_topg_last, dt, m_uplift);
385  m_topg_last.copy_from(m_topg);
386 }

388 //! Update the Lingle-Clark bed deformation model.
389 void LingleClark::update_impl(const IceModelVec2S &ice_thickness,
390  const IceModelVec2S &sea_level_elevation,
391  double t, double dt) {
392 
393  double
394  t_next = m_t_last + m_update_interval,
395  t_final = t + dt;
396 
397  if (t_final < m_t_last) {
398  throw RuntimeError(PISM_ERROR_LOCATION, "cannot go back in time");
399  }
400 
401  if (std::abs(t_next - t_final) < m_t_eps) { // reached the next update time
402  double dt_beddef = t_final - m_t_last;
403  step(ice_thickness, sea_level_elevation, dt_beddef);
404  m_t_last = t_final;
405  }
406 }
