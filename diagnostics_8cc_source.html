80 IceModelVec::Ptr IceMarginPressureDifference::compute_impl() const {
81 
82  IceModelVec2S::Ptr result(new IceModelVec2S(m_grid, "ice_margin_pressure_difference", WITHOUT_GHOSTS));
83  result->metadata(0) = m_vars[0];
84 
85  IceModelVec2CellType mask(m_grid, "mask", WITH_GHOSTS);
86 
87  auto
88  &H = model->geometry().ice_thickness,
89  &bed = model->geometry().bed_elevation,
90  &sea_level = model->geometry().sea_level_elevation;
91 
92  {
93  const double H_threshold = m_config->get_number("stress_balance.ice_free_thickness_standard");
94  GeometryCalculator gc(*m_config);
95  gc.set_icefree_thickness(H_threshold);
96 
97  gc.compute_mask(sea_level, bed, H, mask);
98  }
99 
100  const double
101  rho_ice = m_config->get_number("constants.ice.density"),
102  rho_ocean = m_config->get_number("constants.sea_water.density"),
103  g = m_config->get_number("constants.standard_gravity");
104 
105  IceModelVec::AccessList list{&H, &bed, &mask, &sea_level, result.get()};
106 
107  ParallelSection loop(m_grid->com);
108  try {
109  for (Points p(*m_grid); p; p.next()) {
110  const int i = p.i(), j = p.j();
111 
112  double delta_p = 0.0;
113  if (mask.grounded_ice(i, j) and grid_edge(*m_grid, i, j)) {
114  delta_p = 0.0;
115  } else if (mask.icy(i, j) and mask.next_to_ice_free_ocean(i, j)) {
116  double
117  P_ice = 0.5 * rho_ice * g * H(i, j),
118  P_water = average_water_column_pressure(H(i, j), bed(i, j), sea_level(i, j),
119  rho_ice, rho_ocean, g);
120 
121  delta_p = P_ice - P_water;
122  } else {
123  delta_p = m_fill_value;
124  }
125 
126  (*result)(i, j) = delta_p;
127  }
128  } catch (...) {
129  loop.failed();
130  }
131  loop.check();
132 
133 
134  return result;
135 }

227 //! \brief Computes vertically-averaged ice hardness.
228 IceModelVec::Ptr HardnessAverage::compute_impl() const {
229 
230  const rheology::FlowLaw *flow_law = model->stress_balance()->shallow()->flow_law().get();
231  if (flow_law == NULL) {
232  flow_law = model->stress_balance()->modifier()->flow_law().get();
233  if (flow_law == NULL) {
234  throw RuntimeError(PISM_ERROR_LOCATION,
235  "Can't compute vertically-averaged hardness: no flow law is used.");
236  }
237  }
238 
239  IceModelVec2S::Ptr result(new IceModelVec2S(m_grid, "hardav", WITHOUT_GHOSTS));
240  result->metadata() = m_vars[0];
241 
242  const IceModelVec2CellType &cell_type = model->geometry().cell_type;
243 
244  const IceModelVec3& ice_enthalpy = model->energy_balance_model()->enthalpy();
245  const IceModelVec2S& ice_thickness = model->geometry().ice_thickness;
246 
247  IceModelVec::AccessList list{&cell_type, &ice_enthalpy, &ice_thickness, result.get()};
248  ParallelSection loop(m_grid->com);
249  try {
250  for (Points p(*m_grid); p; p.next()) {
251  const int i = p.i(), j = p.j();
252 
253  const double *Eij = ice_enthalpy.get_column(i,j);
254  const double H = ice_thickness(i,j);
255  if (cell_type.icy(i, j)) {
256  (*result)(i,j) = rheology::averaged_hardness(*flow_law,
257  H, m_grid->kBelowHeight(H),
258  &(m_grid->z()[0]), Eij);
259  } else { // put negative value below valid range
260  (*result)(i,j) = m_fill_value;
261  }
262  }
263  } catch (...) {
264  loop.failed();
265  }
266  loop.check();
267 
268  return result;
269 }

870 /*!
871  * Uses linear interpolation to go beyond vertical grid resolution.
872  */
873 IceModelVec::Ptr TemperateIceThicknessBasal::compute_impl() const {
874 
875  IceModelVec2S::Ptr result(new IceModelVec2S(m_grid, "tempicethk_basal", WITHOUT_GHOSTS));
876  result->metadata(0) = m_vars[0];
877 
878  EnthalpyConverter::Ptr EC = model->ctx()->enthalpy_converter();
879 
880  const IceModelVec2CellType &cell_type = model->geometry().cell_type;
881  const IceModelVec3& ice_enthalpy = model->energy_balance_model()->enthalpy();
882  const IceModelVec2S& ice_thickness = model->geometry().ice_thickness;
883 
884  IceModelVec::AccessList list{&cell_type, result.get(), &ice_thickness, &ice_enthalpy};
885 
886  ParallelSection loop(m_grid->com);
887  try {
888  for (Points p(*m_grid); p; p.next()) {
889  const int i = p.i(), j = p.j();
890 
891  double H = ice_thickness(i,j);
892 
893  // if we have no ice, go on to the next grid point (this cell will be
894  // marked as "missing" later)
895  if (cell_type.ice_free(i, j)) {
896  (*result)(i,j) = m_fill_value;
897  continue;
898  }
899 
900  const double *Enth = ice_enthalpy.get_column(i,j);
901 
902  unsigned int ks = m_grid->kBelowHeight(H);
903 
904  unsigned int k = 0;
905  double pressure = EC->pressure(H - m_grid->z(k));
906  while (k <= ks) { // FIXME issue #15
907  pressure = EC->pressure(H - m_grid->z(k));
908 
909  if (EC->is_temperate_relaxed(Enth[k],pressure)) {
910  k++;
911  } else {
912  break;
913  }
914  }
915  // after this loop 'pressure' is equal to the pressure at the first level
916  // that is cold
917 
918  // no temperate ice at all; go to the next grid point
919  if (k == 0) {
920  (*result)(i,j) = 0.0;
921  continue;
922  }
923 
924  // the whole column is temperate (except, possibly, some ice between
925  // z(ks) and the total thickness; we ignore it)
926  if (k == ks + 1) {
927  (*result)(i,j) = m_grid->z(ks);
928  continue;
929  }
930 
931  double
932  pressure_0 = EC->pressure(H - m_grid->z(k-1)),
933  dz = m_grid->z(k) - m_grid->z(k-1),
934  slope1 = (Enth[k] - Enth[k-1]) / dz,
935  slope2 = (EC->enthalpy_cts(pressure) - EC->enthalpy_cts(pressure_0)) / dz;
936 
937  if (slope1 != slope2) {
938  (*result)(i,j) = m_grid->z(k-1) +
939  (EC->enthalpy_cts(pressure_0) - Enth[k-1]) / (slope1 - slope2);
940 
941  // check if the resulting thickness is valid:
942  (*result)(i,j) = std::max((*result)(i,j), m_grid->z(k-1));
943  (*result)(i,j) = std::min((*result)(i,j), m_grid->z(k));
944  } else {
945  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "Linear interpolation of the thickness of"
946  " the basal temperate layer failed:\n"
947  "(i=%d, j=%d, k=%d, ks=%d)\n",
948  i, j, k, ks);
949  }
950  }
951  } catch (...) {
952  loop.failed();
953  }
954  loop.check();
955 
956 
957  return result;
958 }

1537 /*!
1538  * Return total mass change due to one of the terms in the mass continuity equation.
1539  *
1540  * Possible terms are
1541  *
1542  * - SMB: surface mass balance
1543  * - BMB: basal mass balance
1544  * - FLOW: ice flow
1545  * - ERROR: numerical flux needed to preserve non-negativity of thickness
1546  *
1547  * This computation can be restricted to grounded and floating areas
1548  * using the `area` argument.
1549  *
1550  * - BOTH: include all contributions
1551  * - GROUNDED: include grounded areas only
1552  * - SHELF: include floating areas only
1553  *
1554  * When computing mass changes due to flow it is important to remember
1555  * that ice mass in a cell can be represented by its thickness *or* an
1556  * "area specific volume". Transferring mass from one representation
1557  * to the other does not change the mass in a cell. This explains the
1558  * special case used when `term == FLOW`. (Note that surface and basal
1559  * mass balances do not affect the area specific volume field.)
1560  */
1561 double mass_change(const IceModel *model, TermType term, AreaType area) {
1562  const IceGrid &grid = *model->grid();
1563  const Config &config = *grid.ctx()->config();
1564 
1565  const double
1566  ice_density = config.get_number("constants.ice.density"),
1567  cell_area = grid.cell_area();
1568 
1569  const IceModelVec2CellType &cell_type = model->geometry().cell_type;
1570 
1571  const IceModelVec2S *thickness_change = nullptr;
1572 
1573  switch (term) {
1574  case FLOW:
1575  thickness_change = &model->geometry_evolution().thickness_change_due_to_flow();
1576  break;
1577  case SMB:
1578  thickness_change = &model->geometry_evolution().top_surface_mass_balance();
1579  break;
1580  case BMB:
1581  thickness_change = &model->geometry_evolution().bottom_surface_mass_balance();
1582  break;
1583  case ERROR:
1584  thickness_change = &model->geometry_evolution().conservation_error();
1585  break;
1586  default:
1587  // can't happen
1588  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "invalid term type");
1589  }
1590 
1591  const IceModelVec2S &dV_flow = model->geometry_evolution().area_specific_volume_change_due_to_flow();
1592 
1593  IceModelVec::AccessList list{&cell_type, thickness_change};
1594 
1595  if (term == FLOW) {
1596  list.add(dV_flow);
1597  }
1598 
1599  double volume_change = 0.0;
1600  for (Points p(grid); p; p.next()) {
1601  const int i = p.i(), j = p.j();
1602 
1603  if ((area == BOTH) or
1604  (area == GROUNDED and cell_type.grounded(i, j)) or
1605  (area == SHELF and cell_type.ocean(i, j))) {
1606 
1607  double dV = term == FLOW ? dV_flow(i, j) : 0.0;
1608 
1609  // m^3 = m^2 * m
1610  volume_change += cell_area * ((*thickness_change)(i, j) + dV);
1611  }
1612  }
1613 
1614  // (kg / m^3) * m^3 = kg
1615  return ice_density * GlobalSum(grid.com, volume_change);
1616 }

2442  auto m = mask.star(i, j);
2443  const unsigned int
2444  east = ice_free(m.e) ? 0 : 1,
2445  west = ice_free(m.w) ? 0 : 1,
2446  south = ice_free(m.s) ? 0 : 1,
2447  north = ice_free(m.n) ? 0 : 1;
2448 
2449  double *viscosity = result->get_column(i, j);
2450 
2451  if (ice_free(m.ij)) {
2452  result->set_column(i, j, m_fill_value);
2453  continue;
2454  }
2455 
2456  for (unsigned int k = 0; k < Mz; ++k) {
2457  const double depth = H - z[k];
2458 
2459  if (depth < 0.0) {
2460  viscosity[k] = m_fill_value;
2461  continue;
2462  }
2463 
2464  // EC->pressure() handles negative depths correctly
2465  const double pressure = EC->pressure(depth);
2466 
2467  const double hardness = flow_law.hardness(E[k], pressure);
2468 
2469  double u_x = 0.0, v_x = 0.0, w_x = 0.0;
2470  if (west + east > 0) {
2471  const double D = 1.0 / (dx * (west + east));
2472  u_x = D * (west * (u[k] - u_w[k]) + east * (u_e[k] - u[k]));
2473  v_x = D * (west * (v[k] - v_w[k]) + east * (v_e[k] - v[k]));
2474  w_x = D * (west * (w[k] - w_w[k]) + east * (w_e[k] - w[k]));
2475  }
2476 
2477  double u_y = 0.0, v_y = 0.0, w_y = 0.0;
2478  if (south + north > 0) {
2479  const double D = 1.0 / (dy * (south + north));
2480  u_y = D * (south * (u[k] - u_s[k]) + north * (u_n[k] - u[k]));
2481  v_y = D * (south * (v[k] - v_s[k]) + north * (v_n[k] - v[k]));
2482  w_y = D * (south * (w[k] - w_s[k]) + north * (w_n[k] - w[k]));
2483  }
2484 
2485  double
2486  u_z = 0.0,
2487  v_z = 0.0,
2488  w_z = 0.0;
2489 
2490  if (k == 0) {
2491  const double dz = z[1] - z[0];
2492  u_z = (u[1] - u[0]) / dz;
2493  v_z = (v[1] - v[0]) / dz;
2494  w_z = (w[1] - w[0]) / dz;
2495  } else if (k == Mz - 1) {
2496  const double dz = z[Mz - 1] - z[Mz - 2];
2497  u_z = (u[Mz - 1] - u[Mz - 2]) / dz;
2498  v_z = (v[Mz - 1] - v[Mz - 2]) / dz;
2499  w_z = (w[Mz - 1] - w[Mz - 2]) / dz;
2500  } else {
2501  const double
2502  dz_p = z[k + 1] - z[k],
2503  dz_m = z[k] - z[k - 1];
2504  u_z = 0.5 * ((u[k + 1] - u[k]) / dz_p + (u[k] - u[k - 1]) / dz_m);
2505  v_z = 0.5 * ((v[k + 1] - v[k]) / dz_p + (v[k] - v[k - 1]) / dz_m);
2506  w_z = 0.5 * ((w[k + 1] - w[k]) / dz_p + (w[k] - w[k - 1]) / dz_m);
2507  }
2508 
2509  // These should be "epsilon dot", but that's just too long.
2510  const double
2511  eps_xx = u_x,
2512  eps_yy = v_y,
2513  eps_zz = w_z,
2514  eps_xy = 0.5 * (u_y + v_x),
2515  eps_xz = 0.5 * (u_z + w_x),
2516  eps_yz = 0.5 * (v_z + w_y);
2517 
2518  // The second invariant of the 3D strain rate tensor; see equation 4.8 in [@ref
2519  // GreveBlatter2009]. Unlike secondInvariant_2D(), this code does not make assumptions about
2520  // the input velocity field: we do not ignore w_x and w_y and do not assume that u_z and v_z
2521  // are zero.
2522  const double
2523  gamma = (square(eps_xx) + square(eps_yy) + square(eps_zz) +
2524  2.0 * (square(eps_xy) + square(eps_xz) + square(eps_yz)));
2525 
2526  double nu = 0.0;
2527  // Note: in PISM gamma has an extra factor of 1/2; compare to
2528  flow_law.effective_viscosity(hardness, 0.5 * gamma, &nu, NULL);
2529 
2530  viscosity[k] = nu;
2531  }
2532  }
2533  } catch (...) {
2534  loop.failed();
2535  }
2536  loop.check();
2537 
2538  return result;
2539 }
