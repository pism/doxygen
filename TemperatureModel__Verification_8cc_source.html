77 void TemperatureModel_Verification::initTestFG() {
78 
79  IceModelVec::AccessList list{&m_ice_temperature};
80 
81  const double time = m_testname == 'F' ? 0.0 : m_grid->ctx()->time()->current();
82  const double A = m_testname == 'F' ? 0.0 : ApforG;
83 
84  for (Points p(*m_grid); p; p.next()) {
85  const int i = p.i(), j = p.j();
86 
87  // avoid singularity at origin
88  const double r = std::max(radius(*m_grid, i, j), 1.0);
89 
90  if (r > LforFG - 1.0) { // if (essentially) outside of sheet
91  m_ice_temperature.set_column(i, j, Tmin + ST * r);
92  } else {
93  TestFGParameters P = exactFG(time, r, m_grid->z(), A);
94  m_ice_temperature.set_column(i, j, &P.T[0]);
95  }
96  }
97 }

99 void TemperatureModel_Verification::initTestsKO() {
100 
101  const unsigned int Mz = m_grid->Mz();
102 
103  std::vector<double> T_column(Mz);
104 
105  const double time = m_grid->ctx()->time()->current();
106 
107  // evaluate exact solution in a column; all columns are the same
108  for (unsigned int k = 0; k < Mz; k++) {
109  if (m_testname == 'K') {
110  T_column[k] = exactK(time, m_grid->z(k), m_bedrock_is_ice).T;
111  } else {
112  T_column[k] = exactO(m_grid->z(k)).TT;
113  }
114  }
115 
116  // fill m_ice_temperature
117  IceModelVec::AccessList list(m_ice_temperature);
118 
119  ParallelSection loop(m_grid->com);
120  try {
121  for (Points p(*m_grid); p; p.next()) {
122  m_ice_temperature.set_column(p.i(), p.j(), &T_column[0]);
123  }
124  } catch (...) {
125  loop.failed();
126  }
127  loop.check();
128 }
