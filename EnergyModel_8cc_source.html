94 bool marginal(const IceModelVec2S &thickness, int i, int j, double threshold) {
95  int
96  n = j + 1,
97  e = i + 1,
98  s = j - 1,
99  w = i - 1;
100 
101  const double
102  N = thickness(i, n),
103  E = thickness(e, j),
104  S = thickness(i, s),
105  W = thickness(w, j),
106  NW = thickness(w, n),
107  SW = thickness(w, s),
108  NE = thickness(e, n),
109  SE = thickness(e, s);
110 
111  return ((E < threshold) or
112  (NE < threshold) or
113  (N < threshold) or
114  (NW < threshold) or
115  (W < threshold) or
116  (SW < threshold) or
117  (S < threshold) or
118  (SE < threshold));
119 }

159 void EnergyModel::init_enthalpy(const File &input_file, bool do_regrid, int record) {
160 
161  if (input_file.find_variable("enthalpy")) {
162  if (do_regrid) {
163  m_ice_enthalpy.regrid(input_file, CRITICAL);
164  } else {
165  m_ice_enthalpy.read(input_file, record);
166  }
167  } else if (input_file.find_variable("temp")) {
168  IceModelVec3
169  &temp = m_work,
170  &liqfrac = m_ice_enthalpy;
171 
172  {
173  temp.set_name("temp");
174  temp.metadata(0).set_name("temp");
175  temp.set_attrs("temporary", "ice temperature",
176  "Kelvin", "Kelvin", "land_ice_temperature", 0);
177 
178  if (do_regrid) {
179  temp.regrid(input_file, CRITICAL);
180  } else {
181  temp.read(input_file, record);
182  }
183  }
184 
185  const IceModelVec2S & ice_thickness = *m_grid->variables().get_2d_scalar("land_ice_thickness");
186 
187  if (input_file.find_variable("liqfrac")) {
188  SpatialVariableMetadata enthalpy_metadata = m_ice_enthalpy.metadata();
189 
190  liqfrac.set_name("liqfrac");
191  liqfrac.metadata(0).set_name("liqfrac");
192  liqfrac.set_attrs("temporary", "ice liquid water fraction",
193  "1", "1", "", 0);
194 
195  if (do_regrid) {
196  liqfrac.regrid(input_file, CRITICAL);
197  } else {
198  liqfrac.read(input_file, record);
199  }
200 
201  m_ice_enthalpy.set_name(enthalpy_metadata.get_name());
202  m_ice_enthalpy.metadata() = enthalpy_metadata;
203 
204  m_log->message(2,
205  " - Computing enthalpy using ice temperature,"
206  " liquid water fraction and thickness...\n");
207  compute_enthalpy(temp, liqfrac, ice_thickness, m_ice_enthalpy);
208  } else {
209  m_log->message(2,
210  " - Computing enthalpy using ice temperature and thickness "
211  "and assuming zero liquid water fraction...\n");
212  compute_enthalpy_cold(temp, ice_thickness, m_ice_enthalpy);
213  }
214  } else {
215  throw RuntimeError::formatted(PISM_ERROR_LOCATION,
216  "neither enthalpy nor temperature was found in '%s'.\n",
217  input_file.filename().c_str());
218  }
219 }

225 void EnergyModel::regrid_enthalpy() {
226 
227  auto regrid_filename = m_config->get_string("input.regrid.file");
228  auto regrid_vars = set_split(m_config->get_string("input.regrid.vars"), ',');
229 
230 
231  if (regrid_filename.empty()) {
232  return;
233  }
234 
235  std::string enthalpy_name = m_ice_enthalpy.metadata().get_name();
236 
237  if (regrid_vars.empty() or member(enthalpy_name, regrid_vars)) {
238  File regrid_file(m_grid->com, regrid_filename, PISM_GUESS, PISM_READONLY);
239  init_enthalpy(regrid_file, true, 0);
240  }

270 void EnergyModel::update(double t, double dt, const Inputs &inputs) {
271  // reset standard out flags at the beginning of every time step
272  m_stdout_flags = "";
273  m_stats = EnergyModelStats();
274 
275  const Profiling &profiling = m_grid->ctx()->profiling();
276 
277  profiling.begin("ice_energy");
278  {
279  // this call should fill m_work with new values of enthalpy
280  this->update_impl(t, dt, inputs);
281 
282  m_ice_enthalpy.copy_from(m_work);
283  }
284  profiling.end("ice_energy");
285 
286  // globalize m_stats and update m_stdout_flags
287  {
288  m_stats.sum(m_grid->com);
289 
290  if (m_stats.reduced_accuracy_counter > 0.0) { // count of when BOMBPROOF switches to lower accuracy
291  const double reduced_accuracy_percentage = 100.0 * (m_stats.reduced_accuracy_counter / (m_grid->Mx() * m_grid->My()));
292  const double reporting_threshold = 5.0; // only report if above 5%
293 
294  if (reduced_accuracy_percentage > reporting_threshold and m_log->get_threshold() > 2) {
295  m_stdout_flags = (pism::printf(" [BPsacr=%.4f%%] ", reduced_accuracy_percentage) +
296  m_stdout_flags);
297  }
298  }
299 
300  if (m_stats.bulge_counter > 0) {
301  // count of when advection bulges are limited; frequently it is identically zero
302  m_stdout_flags = (pism::printf(" BULGE=%d ", m_stats.bulge_counter) +
303  m_stdout_flags);
304  }
305  }
306 }
