58 void InitializationHelper::init_impl(const Geometry &geometry) {
59  m_input_model->init(geometry);
60 
61  InputOptions opts = process_input_options(m_grid->com, m_config);
62 
63  if (opts.type == INIT_RESTART) {
64  m_log->message(2, "* Reading effective ocean model outputs from '%s' for re-starting...\n",
65  opts.filename.c_str());
66 
67  File file(m_grid->com, opts.filename, PISM_GUESS, PISM_READONLY);
68  const unsigned int time_length = file.nrecords();
69  const unsigned int last_record = time_length > 0 ? time_length - 1 : 0;
70 
71  m_water_column_pressure->read(file, last_record);
72  m_shelf_base_mass_flux->read(file, last_record);
73  m_shelf_base_temperature->read(file, last_record);
74 
75  file.close();
76  } else {
77  m_log->message(2, "* Performing a 'fake' ocean model time-step for bootstrapping...\n");
78 
79  init_step(this, geometry, *m_grid->ctx()->time());
80  }
81 
82  // Support regridding. This is needed to ensure that initialization using "-i" is equivalent to
83  // "-i ... -bootstrap -regrid_file ..."
84  {
85  regrid("ocean model initialization helper", *m_water_column_pressure,
86  REGRID_WITHOUT_REGRID_VARS);
87  regrid("ocean model initialization helper", *m_shelf_base_mass_flux,
88  REGRID_WITHOUT_REGRID_VARS);
89  regrid("ocean model initialization helper", *m_shelf_base_temperature,
90  REGRID_WITHOUT_REGRID_VARS);
91  }
92 }
