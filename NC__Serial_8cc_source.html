297 //! \brief Get variable data.
298 void NC_Serial::get_var_double(const std::string &variable_name,
299  const std::vector<unsigned int> &start_input,
300  const std::vector<unsigned int> &count_input,
301  const std::vector<unsigned int> &imap_input, double *ip,
302  bool transposed) const {
303  std::vector<unsigned int> start = start_input;
304  std::vector<unsigned int> count = count_input;
305  std::vector<unsigned int> imap = imap_input;
306  const int start_tag = 1,
307  count_tag = 2,
308  data_tag = 3,
309  imap_tag = 4,
310  chunk_size_tag = 5;
311  int stat = NC_NOERR, com_size, ndims = static_cast<int>(start.size());
312  std::vector<double> processor_0_buffer;
313  MPI_Status mpi_stat;
314  unsigned int local_chunk_size = 1,
315  processor_0_chunk_size = 0;
316 
317  if (not transposed) {
318  imap.resize(ndims);
319  }
320 
321  // get the size of the communicator
322  MPI_Comm_size(m_com, &com_size);
323 
324  // compute the size of a local chunk
325  for (int k = 0; k < ndims; ++k) {
326  local_chunk_size *= count[k];
327  }
328 
329  // compute the maximum and send it to processor 0; this is the size of the
330  // buffer processor 0 will need
331  MPI_Reduce(&local_chunk_size, &processor_0_chunk_size, 1, MPI_UNSIGNED, MPI_MAX, 0, m_com);
332 
333  // now we need to send start, count and imap data to processor 0 and receive data
334  if (m_rank == 0) {
335  // Note: this could be optimized: if processor_0_chunk_size <=
336  // max(local_chunk_size) we can avoid allocating this buffer. The inner for
337  // loop will have to be re-ordered, though.
338  processor_0_buffer.resize(processor_0_chunk_size);
339 
340  // MPI calls below require C datatypes (so that we don't have to worry
341  // about sizes of size_t and ptrdiff_t), so we make local copies of start,
342  // count, and imap to use in the nc_get_varm_double() call.
343  std::vector<size_t> nc_start(ndims), nc_count(ndims);
344  std::vector<ptrdiff_t> nc_imap(ndims), nc_stride(ndims);
345  int varid;
346 
347  stat = nc_inq_varid(m_file_id, variable_name.c_str(), &varid);
348  check_and_abort(m_com, PISM_ERROR_LOCATION, stat);
349 
350  for (int r = 0; r < com_size; ++r) {
351 
352  if (r != 0) {
353  // Note: start, count, imap, and local_chunk_size on processor zero are
354  // used *before* they get overwritten by these calls
355  MPI_Recv(&start[0], ndims, MPI_UNSIGNED, r, start_tag, m_com, &mpi_stat);
356  MPI_Recv(&count[0], ndims, MPI_UNSIGNED, r, count_tag, m_com, &mpi_stat);
357  MPI_Recv(&imap[0], ndims, MPI_UNSIGNED, r, imap_tag, m_com, &mpi_stat);
358  MPI_Recv(&local_chunk_size, 1, MPI_UNSIGNED, r, chunk_size_tag, m_com, &mpi_stat);
359  }
360 
361  // This for loop uses start, count and imap passed in as arguments when r
362  // == 0. For r > 0 they are overwritten by MPI_Recv calls above.
363  for (int k = 0; k < ndims; ++k) {
364  nc_start[k] = start[k];
365  nc_count[k] = count[k];
366  nc_imap[k] = imap[k];
367  nc_stride[k] = 1; // fill with ones; this way it works even with
368  // NetCDF versions with a bug affecting the
369  // stride == NULL case.
370  }
371 
372  if (transposed) {
373  stat = nc_get_varm_double(m_file_id, varid, &nc_start[0], &nc_count[0], &nc_stride[0], &nc_imap[0],
374  &processor_0_buffer[0]);
375  } else {
376  stat = nc_get_vara_double(m_file_id, varid, &nc_start[0], &nc_count[0],
377  &processor_0_buffer[0]);
378  }
379  check_and_abort(m_com, PISM_ERROR_LOCATION, stat);
380 
381  if (r != 0) {
382  MPI_Send(&processor_0_buffer[0], local_chunk_size, MPI_DOUBLE, r, data_tag, m_com);
383  } else {
384  for (unsigned int k = 0; k < local_chunk_size; ++k) {
385  ip[k] = processor_0_buffer[k];
386  }
387  }
388  } // end of the for loop
389 
390  } else {
391  MPI_Send(&start[0], ndims, MPI_UNSIGNED, 0, start_tag, m_com);
392  MPI_Send(&count[0], ndims, MPI_UNSIGNED, 0, count_tag, m_com);
393  MPI_Send(&imap[0], ndims, MPI_UNSIGNED, 0, imap_tag, m_com);
394  MPI_Send(&local_chunk_size, 1, MPI_UNSIGNED, 0, chunk_size_tag, m_com);
395 
396  MPI_Recv(ip, local_chunk_size, MPI_DOUBLE, 0, data_tag, m_com, &mpi_stat);
397  }
398 }

402  const std::vector<unsigned int> &count_input,
403  const double *op) const {

405  std::vector<unsigned int> start = start_input;
406  std::vector<unsigned int> count = count_input;

408  count_tag = 2,

412  std::vector<double> processor_0_buffer;

415  processor_0_chunk_size = 0;
416 
417  // get the size of the communicator

419 
420  // compute the size of a local chunk
421  for (int k = 0; k < ndims; ++k) {
422  local_chunk_size *= count[k];
423  }
424

426  // buffer processor 0 will need
427  MPI_Reduce(&local_chunk_size, &processor_0_chunk_size, 1, MPI_UNSIGNED, MPI_MAX, 0, m_com);
428 
429  // now we need to send start and count data to processor 0 and receive data
430  if (m_rank == 0) {
431  processor_0_buffer.resize(processor_0_chunk_size);
432 
433  // MPI calls below require C datatypes (so that we don't have to worry about sizes of

472  } // end of the for loop
