29 /*!
30  * Label connected components in a mask stored in an IceModelVec2Int.
31  *
32  * This function allocates a copy on rank 0 and so should not be used if that is a
33  * problem.
34  */
35 void label_components(IceModelVec2Int &mask, bool identify_icebergs, double mask_grounded) {
36  auto mask_p0 = mask.allocate_proc0_copy();
37 
38  mask.put_on_proc0(*mask_p0);
39 
40  auto grid = mask.grid();
41 
42  ParallelSection rank0(grid->com);
43  try {
44  if (grid->rank() == 0) {
45  petsc::VecArray array(*mask_p0);
46  label_connected_components(array.get(),
47  static_cast<int>(grid->My()),
48  static_cast<int>(grid->Mx()),
49  identify_icebergs,
50  static_cast<int>(mask_grounded));
51  }
52  } catch (...) {
53  rank0.failed();
54  }
55  rank0.check();
56 
57  mask.get_from_proc0(*mask_p0);
58 }
