349 void IBIceModel::prepare_initial_outputs() {
350  double ice_density = m_config->get_number("constants.ice.density", "kg m-3");
351 
352  const IceModelVec3 &ice_enthalpy = m_energy_model->enthalpy();
353 
354  AccessList access{ &ice_enthalpy, &M1, &M2, &H1, &H2, &V1, &V2, &m_geometry.ice_thickness };
355  for (int i = m_grid->xs(); i < m_grid->xs() + m_grid->xm(); ++i) {
356  for (int j = m_grid->ys(); j < m_grid->ys() + m_grid->ym(); ++j) {
357  double const *Enth = ice_enthalpy.get_column(i, j);
358 
359  // Top Layer
360  int const ks = m_grid->kBelowHeight(m_geometry.ice_thickness(i, j));
361  V1(i, j) = m_geometry.ice_thickness(i, j) - m_grid->z(ks); // [m^3 m-2]
362  M1(i, j) = V1(i, j) * ice_density; // [kg m-2] = [m^3 m-2] [kg m-3]
363  H1(i, j) = Enth[ks] * M1(i, j); // [J m-2] = [J kg-1] [kg m-2]
364 
365  // Second layer
366  int const ks2 = ks - 1;
367  if (ks2 >= 0) {
368  V2(i, j) = m_grid->z(ks) - m_grid->z(ks2);
369  M2(i, j) = V2(i, j) * ice_density;
370  H2(i, j) = Enth[ks2] * M2(i, j);
371  } else {
372  // There is no second layer
373  V2(i, j) = 0;
374  M2(i, j) = 0;
375  H2(i, j) = 0;
376  }
377  }

422 void IBIceModel::compute_enth2(pism::IceModelVec2S &enth2, pism::IceModelVec2S &mass2) {
423  // getInternalColumn() is allocated already
424  double ice_density = m_config->get_number("constants.ice.density", "kg m-3");
425 
426  const IceModelVec3 *ice_enthalpy = &m_energy_model->enthalpy();
427 
428  AccessList access{ &m_geometry.ice_thickness, ice_enthalpy, &enth2, &mass2 };
429  for (int i = m_grid->xs(); i < m_grid->xs() + m_grid->xm(); ++i) {
430  for (int j = m_grid->ys(); j < m_grid->ys() + m_grid->ym(); ++j) {
431  enth2(i, j) = 0;
432  mass2(i, j) = 0;
433 
434  // count all ice, including cells that have so little they
435  // are considered "ice-free"
436  if (m_geometry.ice_thickness(i, j) > 0) {
437  const int ks = m_grid->kBelowHeight(m_geometry.ice_thickness(i, j));
438  double const *Enth = ice_enthalpy->get_column(i, j);
439  for (int k = 0; k < ks; ++k) {
440  double dz = (m_grid->z(k + 1) - m_grid->z(k));
441  enth2(i, j) += Enth[k] * dz; // m J / kg
442  }
443 
444  // Do the last layer a bit differently
445  double dz = (m_geometry.ice_thickness(i, j) - m_grid->z(ks));
446  enth2(i, j) += Enth[ks] * dz;
447  enth2(i, j) *= ice_density; // --> J/m^2
448  mass2(i, j) = m_geometry.ice_thickness(i, j) * ice_density; // --> kg/m^2
449  }
450  }
451  }
