32 WeatherStation::WeatherStation(IceGrid::ConstPtr grid)
33  : AtmosphereModel(grid) {
34 
35  m_log->message(2,
36  "* Using the constant-in-space atmosphere model\n"
37  " for use with scalar data from one weather station\n"
38  " combined with lapse rate corrections...\n");
39 
40  auto filename = m_config->get_string("atmosphere.one_station.file");
41 
42  if (filename.empty()) {
43  throw RuntimeError(PISM_ERROR_LOCATION,
44  "atmosphere.one_station.file cannot be empty");
45  }
46 
47  m_log->message(2,
48  " - Reading air temperature and precipitation from '%s'...\n",
49  filename.c_str());
50 
51  auto &ctx = *grid->ctx();
52 
53  bool periodic = false;
54 
55  m_precipitation_timeseries = std::make_shared<ScalarForcing>(ctx,
56  filename,
57  "precipitation",
58  "kg m-2 second-1",
59  "kg m-2 year-1",
60  "ice-equivalent precipitation rate",
61  periodic);
62 
63  m_air_temp_timeseries = std::make_shared<ScalarForcing>(ctx,
64  filename,
65  "air_temp",
66  "Kelvin",
67  "Kelvin",
68  "near-surface air temperature",
69  periodic);
70 
71  m_precipitation = allocate_precipitation(grid);
72  m_temperature = allocate_temperature(grid);
73 }

108 void WeatherStation::init_timeseries_impl(const std::vector<double> &ts) const {
109  size_t N = ts.size();
110 
111  m_precip_values.resize(N);
112  m_air_temp_values.resize(N);
113 
114  for (unsigned int k = 0; k < N; ++k) {
115  m_precip_values[k] = m_precipitation_timeseries->value(ts[k]);
116  m_air_temp_values[k] = m_air_temp_timeseries->value(ts[k]);
117  }
