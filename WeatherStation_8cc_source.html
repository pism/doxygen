56 void WeatherStation::init_impl(const Geometry &geometry) {
57  (void) geometry;
58 
59  m_log->message(2,
60  "* Initializing the constant-in-space atmosphere model\n"
61  " for use with scalar data from one weather station\n"
62  " combined with lapse rate corrections...\n");
63 
64  auto filename = m_config->get_string("atmosphere.one_station.file");
65 
66  if (filename.empty()) {
67  throw RuntimeError(PISM_ERROR_LOCATION,
68  "atmosphere.one_station.file cannot be empty");
69  }
70 
71  m_log->message(2,
72  " - Reading air temperature and precipitation from '%s'...\n",
73  filename.c_str());
74 
75  File file(m_grid->com, filename, PISM_NETCDF3, PISM_READONLY);
76  {
77  m_precipitation_timeseries.read(file, *m_grid->ctx()->time(), *m_grid->ctx()->log());
78  m_air_temp_timeseries.read(file, *m_grid->ctx()->time(), *m_grid->ctx()->log());
79  }
80 }

87 void WeatherStation::update_impl(const Geometry &geometry, double t, double dt) {
88  (void) geometry;
89 
90  double one_week = 7 * 24 * 60 * 60;
91  unsigned int N = (unsigned int)(ceil(dt / one_week)); // one point per week
92 
93  m_precipitation->set(m_precipitation_timeseries.average(t, dt, N));
94 
95  m_temperature->set(m_air_temp_timeseries.average(t, dt, N));
96 }

114 void WeatherStation::init_timeseries_impl(const std::vector<double> &ts) const {
115  size_t N = ts.size();
116 
117  m_precip_values.resize(N);
118  m_air_temp_values.resize(N);
119 
120  for (unsigned int k = 0; k < N; ++k) {
121  m_precip_values[k] = m_precipitation_timeseries(ts[k]);
122  m_air_temp_values[k] = m_air_temp_timeseries(ts[k]);
123  }
124 }
