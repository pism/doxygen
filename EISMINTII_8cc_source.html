116 void EISMINTII::initialize_using_formulas() {
117  using std::pow;
118  using std::sqrt;
119 
120  // center of the accumulation and surface temperature patterns
121  double cx = 0.0, cy = 0.0;
122  if (m_experiment == 'E') {
123  cx += 100.0e3;
124  cy += 100.0e3;
125  }
126 
127  IceModelVec::AccessList list{m_temperature.get(), m_mass_flux.get()};
128 
129  for (Points p(*m_grid); p; p.next()) {
130  const int i = p.i(), j = p.j();
131 
132  const double r = sqrt(pow(m_grid->x(i) - cx, 2) + pow(m_grid->y(j) - cy, 2));
133 
134  // accumulation (formula (7) in [Payne et al 2000])
135  (*m_mass_flux)(i,j) = std::min(m_M_max, m_S_b * (m_R_el-r));
136 
137  // surface temperature (formula (8) in [Payne et al 2000])
138  (*m_temperature)(i,j) = m_T_min + m_S_T * r;
139  }
140 
141  // convert from "m second-1" to "kg m-2 s-1"
142  m_mass_flux->scale(m_config->get_number("constants.ice.density"));
143 }
