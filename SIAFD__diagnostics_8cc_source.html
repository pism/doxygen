142 static void copy_staggered_vec(const IceModelVec2Stag &input, IceModelVec2Stag &output) {
143  IceGrid::ConstPtr grid = output.grid();
144 
145  IceModelVec::AccessList list{ &input, &output };
146 
147  for (Points p(*grid); p; p.next()) {
148  const int i = p.i(), j = p.j();
149 
150  output(i, j, 0) = input(i, j, 0);
151  output(i, j, 1) = input(i, j, 1);
152  }
153 }
