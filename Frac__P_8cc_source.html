30  : AtmosphereModel(grid, in) {
31 
32  std::string
33  prefix = "atmosphere.frac_P",

35  long_name = "precipitation multiplier, pure fraction",
36  units = "1";

38  ForcingOptions opt(*m_grid->ctx(), prefix);

40  // will be closed at the end of scope

63  m_2d_scaling->set_attrs("climate_forcing",

113 void Frac_P::update_impl(const Geometry &geometry, double t, double dt) {
114  m_input_model->update(geometry, t, dt);
115  m_precipitation->copy_from(m_input_model->mean_precipitation());
116 
117  if (m_1d_scaling) {
118  m_precipitation->scale(m_1d_scaling->value(t + 0.5 * dt));
119  }
120 
121  if (m_2d_scaling) {
122  m_2d_scaling->update(t, dt);
123  m_2d_scaling->average(t, dt);
124 
125  IceModelVec2S &P = *m_precipitation;
126  IceModelVec2T &S = *m_2d_scaling;
127 
128  IceModelVec::AccessList list{&P, &S};
129 
130  for (Points p(*m_grid); p; p.next()) {
131  const int i = p.i(), j = p.j();
132 
133  P(i, j) *= S(i, j);
134  }
135  }
136 }
