366 //! \brief Set the vertical levels in the ice according to values in `Mz` (number of levels), `Lz`
367 //! (domain height), `spacing` (quadratic or equal) and `lambda` (quadratic spacing parameter).
368 /*!
369  - When `vertical_spacing == EQUAL`, the vertical grid in the ice is equally spaced:
370  `zlevels[k] = k dz` where `dz = Lz / (Mz - 1)`.
371  - When `vertical_spacing == QUADRATIC`, the spacing is a quadratic function. The intent
372  is that the spacing is smaller near the base than near the top.
373 
374  In particular, if
375  \f$\zeta_k = k / (\mathtt{Mz} - 1)\f$ then `zlevels[k] = Lz *`
376  ((\f$\zeta_k\f$ / \f$\lambda\f$) * (1.0 + (\f$\lambda\f$ - 1.0)
377  * \f$\zeta_k\f$)) where \f$\lambda\f$ = 4. The value \f$\lambda\f$
378  indicates the slope of the quadratic function as it leaves the base.
379  Thus a value of \f$\lambda\f$ = 4 makes the spacing about four times finer
380  at the base than equal spacing would be.
381  */
382 std::vector<double> IceGrid::compute_vertical_levels(double new_Lz, unsigned int new_Mz,
383  SpacingType spacing, double lambda) {
384 
385  if (new_Mz < 2) {
386  throw RuntimeError(PISM_ERROR_LOCATION, "Mz must be at least 2");
387  }
388 
389  if (new_Lz <= 0) {
390  throw RuntimeError(PISM_ERROR_LOCATION, "Lz must be positive");
391  }
392 
393  if (spacing == QUADRATIC and lambda <= 0) {
394  throw RuntimeError(PISM_ERROR_LOCATION, "lambda must be positive");
395  }
396 
397  std::vector<double> result(new_Mz);
398 
399  // Fill the levels in the ice:
400  switch (spacing) {
401  case EQUAL: {
402  double dz = new_Lz / ((double) new_Mz - 1);
403 
404  // Equal spacing
405  for (unsigned int k=0; k < new_Mz - 1; k++) {
406  result[k] = dz * ((double) k);
407  }
408  result[new_Mz - 1] = new_Lz; // make sure it is exactly equal
409  break;
410  }
411  case QUADRATIC: {
412  // this quadratic scheme is an attempt to be less extreme in the fineness near the base.
413  for (unsigned int k=0; k < new_Mz - 1; k++) {
414  const double zeta = ((double) k) / ((double) new_Mz - 1);
415  result[k] = new_Lz * ((zeta / lambda) * (1.0 + (lambda - 1.0) * zeta));
416  }
417  result[new_Mz - 1] = new_Lz; // make sure it is exactly equal
418  break;
419  }
420  default:
421  throw RuntimeError(PISM_ERROR_LOCATION, "spacing can not be UNKNOWN");
422  }
423 
424  return result;
425 }

443 //! \brief Computes the number of processors in the X- and Y-directions.
444 static void compute_nprocs(unsigned int Mx, unsigned int My, unsigned int size,
445  unsigned int &Nx, unsigned int &Ny) {
446 
447  if (My <= 0) {
448  throw RuntimeError(PISM_ERROR_LOCATION, "'My' is invalid.");
449  }
450 
451  Nx = (unsigned int)(0.5 + sqrt(((double)Mx)*((double)size)/((double)My)));
452  Ny = 0;
453 
454  if (Nx == 0) {
455  Nx = 1;
456  }
457 
458  while (Nx > 0) {
459  Ny = size / Nx;
460  if (Nx*Ny == (unsigned int)size) {
461  break;
462  }
463  Nx--;
464  }
465 
466  if (Mx > My and Nx < Ny) {
467  // Swap Nx and Ny
468  int tmp = Nx;
469  Nx = Ny;
470  Ny = tmp;
471  }
472 
473  if ((Mx / Nx) < 2) { // note: integer division
474  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "Can't split %d grid points into %d parts (X-direction).",
475  Mx, (int)Nx);
476  }
477 
478  if ((My / Ny) < 2) { // note: integer division
479  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "Can't split %d grid points into %d parts (Y-direction).",
480  My, (int)Ny);
481  }
482 }

520 //! Compute processor ownership ranges using the grid size, MPI communicator size, and command-line
521 //! options `-Nx`, `-Ny`, `-procs_x`, `-procs_y`.
522 static OwnershipRanges compute_ownership_ranges(unsigned int Mx,
523  unsigned int My,
524  unsigned int size) {
525  OwnershipRanges result;
526 
527  unsigned int Nx_default, Ny_default;
528  compute_nprocs(Mx, My, size, Nx_default, Ny_default);
529 
530  // check -Nx and -Ny
531  options::Integer Nx("-Nx", "Number of processors in the x direction", Nx_default);
532  options::Integer Ny("-Ny", "Number of processors in the y direction", Ny_default);
533 
534  // validate results (compute_nprocs checks its results, but we also need to validate command-line
535  // options)
536  if ((Mx / Nx) < 2) {
537  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "Can't split %d grid points into %d parts (X-direction).",
538  Mx, (int)Nx);
539  }
540 
541  if ((My / Ny) < 2) {
542  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "Can't split %d grid points into %d parts (Y-direction).",
543  My, (int)Ny);
544  }
545 
546  if (Nx * Ny != (int)size) {
547  throw RuntimeError::formatted(PISM_ERROR_LOCATION, "Nx * Ny has to be equal to %d.", size);
548  }
549 
550  // check -procs_x and -procs_y
551  options::IntegerList procs_x("-procs_x", "Processor ownership ranges (x direction)", {});
552  options::IntegerList procs_y("-procs_y", "Processor ownership ranges (y direction)", {});
553 
554  if (procs_x.is_set()) {
555  if (procs_x->size() != (unsigned int)Nx) {
556  throw RuntimeError(PISM_ERROR_LOCATION, "-Nx has to be equal to the -procs_x size.");
557  }
558 
559  result.x.resize(procs_x->size());
560  for (unsigned int k = 0; k < procs_x->size(); ++k) {
561  result.x[k] = procs_x[k];
562  }
563 
564  } else {
565  result.x = ownership_ranges(Mx, Nx);
566  }
567 
568  if (procs_y.is_set()) {
569  if (procs_y->size() != (unsigned int)Ny) {
570  throw RuntimeError(PISM_ERROR_LOCATION, "-Ny has to be equal to the -procs_y size.");
571  }
572 
573  result.y.resize(procs_y->size());
574  for (unsigned int k = 0; k < procs_y->size(); ++k) {
575  result.y[k] = procs_y[k];
576  }
577  } else {
578  result.y = ownership_ranges(My, Ny);
579  }
580 
581  if (result.x.size() * result.y.size() != size) {
582  throw RuntimeError(PISM_ERROR_LOCATION, "length(procs_x) * length(procs_y) != MPI size");
583  }
584 
585  return result;
586 }

597 //! Compute grid coordinates for one direction (X or Y).
598 static std::vector<double> compute_coordinates(unsigned int M, double delta,
599  double v_min, double v_max,
600  bool cell_centered) {
601  std::vector<double> result(M);
602 
603  // Here v_min, v_max define the extent of the computational domain,
604  // which is not necessarily the same thing as the smallest and
605  // largest values of grid coordinates.
606  if (cell_centered) {
607  for (unsigned int i = 0; i < M; ++i) {
608  result[i] = v_min + (i + 0.5) * delta;
609  }
610  result[M - 1] = v_max - 0.5*delta;
611  } else {
612  for (unsigned int i = 0; i < M; ++i) {
613  result[i] = v_min + i * delta;
614  }
615  result[M - 1] = v_max;
616  }
617  return result;
618 }

620 //! Compute horizontal spacing parameters `dx` and `dy` and grid coordinates using `Mx`, `My`, `Lx`, `Ly` and periodicity.
621 /*!
622 The grid used in PISM, in particular the PETSc DAs used here, are periodic in x and y.
623 This means that the ghosted values ` foo[i+1][j], foo[i-1][j], foo[i][j+1], foo[i][j-1]`
624 for all 2D Vecs, and similarly in the x and y directions for 3D Vecs, are always available.
625 That is, they are available even if i,j is a point at the edge of the grid. On the other
626 hand, by default, `dx` is the full width `2 * Lx` divided by `Mx - 1`.
627 This means that we conceive of the computational domain as starting at the `i = 0`
628 grid location and ending at the `i = Mx - 1` grid location, in particular.
629 This idea is not quite compatible with the periodic nature of the grid.
630 
631 The upshot is that if one computes in a truly periodic way then the gap between the
632 `i = 0` and `i = Mx - 1` grid points should \em also have width `dx`.
633 Thus we compute `dx = 2 * Lx / Mx`.
634  */
635 void IceGrid::Impl::compute_horizontal_coordinates() {
636 
637  bool cell_centered = registration == CELL_CENTER;
638 
639  dx = compute_horizontal_spacing(Lx, Mx, cell_centered);
640 
641  dy = compute_horizontal_spacing(Ly, My, cell_centered);
642 
643  cell_area = dx * dy;
644 
645  double
646  x_min = x0 - Lx,
647  x_max = x0 + Lx;
648 
649  x = compute_coordinates(Mx, dx,
650  x_min, x_max,
651  cell_centered);
652 
653  double
654  y_min = y0 - Ly,
655  y_max = y0 + Ly;
656 
657  y = compute_coordinates(My, dy,
658  y_min, y_max,
659  cell_centered);
660 }

744 //! \brief Computes indices of grid points to the lower left and upper right from (X,Y).
745 /*!
746  * \code
747  * 3 2
748  * o-------o
749  * | |
750  * | + |
751  * o-------o
752  * 0 1
753  * \endcode
754  *
755  * If "+" is the point (X,Y), then (i_left, j_bottom) corresponds to
756  * point "0" and (i_right, j_top) corresponds to point "2".
757  *
758  * Does not check if the resulting indexes are in the current
759  * processor's domain. Ensures that computed indexes are within the
760  * grid.
761  */
762 void IceGrid::compute_point_neighbors(double X, double Y,
763  int &i_left, int &i_right,
764  int &j_bottom, int &j_top) const {
765  i_left = (int)floor((X - m_impl->x[0])/m_impl->dx);
766  j_bottom = (int)floor((Y - m_impl->y[0])/m_impl->dy);
767 
768  i_right = i_left + 1;
769  j_top = j_bottom + 1;
770 
771  i_left = std::max(i_left, 0);
772  i_right = std::max(i_right, 0);
773 
774  i_left = std::min(i_left, (int)m_impl->Mx - 1);
775  i_right = std::min(i_right, (int)m_impl->Mx - 1);
776 
777  j_bottom = std::max(j_bottom, 0);
778  j_top = std::max(j_top, 0);
779 
780  j_bottom = std::min(j_bottom, (int)m_impl->My - 1);
781  j_top = std::min(j_top, (int)m_impl->My - 1);
782 }

790 //! \brief Compute 4 interpolation weights necessary for linear interpolation
791 //! from the current grid. See compute_point_neighbors for the ordering of
792 //! neighbors.
793 std::vector<double> IceGrid::compute_interp_weights(double X, double Y) const{
794  int i_left = 0, i_right = 0, j_bottom = 0, j_top = 0;
795  // these values (zeros) are used when interpolation is impossible
796  double alpha = 0.0, beta = 0.0;
797 
798  compute_point_neighbors(X, Y, i_left, i_right, j_bottom, j_top);
799 
800  if (i_left != i_right) {
801  assert(m_impl->x[i_right] - m_impl->x[i_left] != 0.0);
802  alpha = (X - m_impl->x[i_left]) / (m_impl->x[i_right] - m_impl->x[i_left]);
803  }
804 
805  if (j_bottom != j_top) {
806  assert(m_impl->y[j_top] - m_impl->y[j_bottom] != 0.0);
807  beta = (Y - m_impl->y[j_bottom]) / (m_impl->y[j_top] - m_impl->y[j_bottom]);
808  }
809 
810  return {(1.0 - alpha) * (1.0 - beta),
811  alpha * (1.0 - beta),
812  alpha * beta,
813  (1.0 - alpha) * beta};
814 }
