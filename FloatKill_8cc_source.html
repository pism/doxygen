61 void FloatKill::update(IceModelVec2CellType &mask, IceModelVec2S &ice_thickness) {
62 
63  IceModelVec::AccessList list{&mask, &ice_thickness};
64 
65  const bool dont_calve_near_grounded_ice = not m_calve_near_grounding_line;
66 
67  for (Points p(*m_grid); p; p.next()) {
68  const int i = p.i(), j = p.j();
69 
70  if (mask.floating_ice(i, j)) {
71  if (m_margin_only and not mask.next_to_ice_free_ocean(i, j)) {
72  continue;
73  }
74 
75  if (dont_calve_near_grounded_ice and mask.next_to_grounded_ice(i, j)) {
76  continue;
77  }
78 
79  ice_thickness(i, j) = 0.0;
80  mask(i, j) = MASK_ICE_FREE_OCEAN;
81  }
82  }
83 
84  mask.update_ghosts();
85  ice_thickness.update_ghosts();
86 }
