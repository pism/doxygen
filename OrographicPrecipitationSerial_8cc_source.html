44 OrographicPrecipitationSerial::OrographicPrecipitationSerial(const Config &config,
45  int Mx, int My,
46  double dx, double dy,
47  int Nx, int Ny)
48  : m_Mx(Mx), m_My(My), m_Nx(Nx), m_Ny(Ny) {
49 
50  m_eps = 1.0e-18;
51 
52  // derive more parameters
53  {
54  m_i0_offset = (Nx - Mx) / 2;
55  m_j0_offset = (Ny - My) / 2;
56 
57  m_kx = fftfreq(m_Nx, dx / (2.0 * M_PI));
58  m_ky = fftfreq(m_Ny, dy / (2.0 * M_PI));
59  }
60 
61  {
62  m_background_precip_pre = config.get_number("atmosphere.orographic_precipitation.background_precip_pre", "mm/s");
63  m_background_precip_post = config.get_number("atmosphere.orographic_precipitation.background_precip_post", "mm/s");
64 
65  m_precip_scale_factor = config.get_number("atmosphere.orographic_precipitation.scale_factor");
66  m_tau_c = config.get_number("atmosphere.orographic_precipitation.conversion_time");
67  m_tau_f = config.get_number("atmosphere.orographic_precipitation.fallout_time");
68  m_Hw = config.get_number("atmosphere.orographic_precipitation.water_vapor_scale_height");
69  m_Nm = config.get_number("atmosphere.orographic_precipitation.moist_stability_frequency");
70  m_wind_speed = config.get_number("atmosphere.orographic_precipitation.wind_speed");
71  m_wind_direction = config.get_number("atmosphere.orographic_precipitation.wind_direction");
72  m_gamma = config.get_number("atmosphere.orographic_precipitation.lapse_rate");
73  m_Theta_m = config.get_number("atmosphere.orographic_precipitation.moist_adiabatic_lapse_rate");
74  m_rho_Sref = config.get_number("atmosphere.orographic_precipitation.reference_density");
75  m_latitude = config.get_number("atmosphere.orographic_precipitation.coriolis_latitude");
76  m_truncate = config.get_flag("atmosphere.orographic_precipitation.truncate");
77 
78 
79  // derived constants
80  m_f = 2.0 * 7.2921e-5 * sin(m_latitude * M_PI / 180.0);
81 
82  m_u = -sin(m_wind_direction * 2.0 * M_PI / 360.0) * m_wind_speed;
83  m_v = -cos(m_wind_direction * 2.0 * M_PI / 360.0) * m_wind_speed;
84 
85  m_Cw = m_rho_Sref * m_Theta_m / m_gamma;
86  }
87 
88  // memory allocation
89  {
90  PetscErrorCode ierr = 0;
91 
92  // precipitation
93  ierr = VecCreateSeq(PETSC_COMM_SELF, m_Mx * m_My, m_precipitation.rawptr());
94  PISM_CHK(ierr, "VecCreateSeq");
95 
96  // FFTW arrays
97  m_fftw_input = (fftw_complex *)fftw_malloc(sizeof(fftw_complex) * m_Nx * m_Ny);
98  m_fftw_output = (fftw_complex *)fftw_malloc(sizeof(fftw_complex) * m_Nx * m_Ny);
99  m_G_hat = (fftw_complex *)fftw_malloc(sizeof(fftw_complex) * m_Nx * m_Ny);
100 
101  // FFTW plans
102  m_dft_forward = fftw_plan_dft_2d(m_Nx, m_Ny, m_fftw_input, m_fftw_output,
103  FFTW_FORWARD, FFTW_ESTIMATE);
104  m_dft_inverse = fftw_plan_dft_2d(m_Nx, m_Ny, m_fftw_input, m_fftw_output,
105  FFTW_BACKWARD, FFTW_ESTIMATE);
106 
107  // Note: FFTW is weird. If a malloc() call fails it will just call
108  // abort() on you without giving you a chance to recover or tell the
109  // user what happened. This is why we don't check return values of
110  // fftw_malloc() and fftw_plan_dft_2d() calls here...
111  //
112  // (Constantine Khroulev, February 1, 2015)
113  }
114 
115  // initialize the Gaussian filter
116  {
117  FFTWArray G_hat(m_G_hat, m_Nx, m_Ny);
118  double sigma = config.get_number("atmosphere.orographic_precipitation.smoothing_standard_deviation");
119 
120  if (sigma > 0.0) {
121  FFTWArray
122  fftw_output(m_fftw_output, m_Nx, m_Ny),
123  fftw_input(m_fftw_input, m_Nx, m_Ny);
124 
125  int
126  Nx2 = Nx / 2,
127  Ny2 = Ny / 2;
128 
129  double sum = 0.0;
130  for (int i = 0; i < m_Nx; i++) {
131  for (int j = 0; j < m_Ny; j++) {
132  int
133  p = i <= Nx2 ? i : m_Nx - i,
134  q = j <= Ny2 ? j : m_Ny - j;
135  double
136  x = p * dx,
137  y = q * dy;
138 
139  double G = std::exp(-0.5 * (x * x + y * y) / (sigma * sigma));
140  sum += G;
141 
142  fftw_input(i, j) = G;
143  }
144  }
145 
146  // normalize:
147  assert(sum > 0.0);
148  for (int i = 0; i < m_Nx; i++) {
149  for (int j = 0; j < m_Ny; j++) {
150  fftw_input(i, j) /= sum;
151  }
152  }
153 
154  // compute FFT of the Gaussian
155  fftw_execute(m_dft_forward);
156 
157  // copy to m_G_hat
158  for (int i = 0; i < m_Nx; i++) {
159  for (int j = 0; j < m_Ny; j++) {
160  G_hat(i, j) = fftw_output(i, j);
161  }
162  }
163 
164  } else {
165  // fill m_G_hat with ones to disable smoothing
166  for (int i = 0; i < m_Nx; i++) {
167  for (int j = 0; j < m_Ny; j++) {
168  G_hat(i, j) = 1.0;
169  }
170  }
171  }
172  }
173 }

190 /*!
191  * Update precipitation.
192  *
193  * @param[in] surface_elevation surface on the physical (Mx*My) grid
194  */
195 void OrographicPrecipitationSerial::update(petsc::Vec &surface_elevation) {
196  // solves:
197  // Phat(k,l) = (Cw * i * sigma * Hhat(k,l)) /
198  // (1 - i * m * Hw) * (1 + i * sigma * tauc) * (1 + i * sigma * tauf);
199  // see equation (49) in
200  // R. B. Smith and I. Barstad, 2004:
201  // A Linear Theory of Orographic Precipitation. J. Atmos. Sci. 61, 1377-1391.
202 
203  std::complex<double> I(0.0, 1.0);
204 
205  // Compute fft2(surface_elevation)
206  {
207  clear_fftw_array(m_fftw_input, m_Nx, m_Ny);
208  set_real_part(surface_elevation,
209  1.0,
210  m_Mx, m_My,
211  m_Nx, m_Ny,
212  m_i0_offset, m_j0_offset,
213  m_fftw_input);
214  fftw_execute(m_dft_forward);
215  }
216 
217  {
218  FFTWArray
219  fftw_output(m_fftw_output, m_Nx, m_Ny),
220  fftw_input(m_fftw_input, m_Nx, m_Ny),
221  G_hat(m_G_hat, m_Nx, m_Ny);
222 
223  for (int i = 0; i < m_Nx; i++) {
224  const double kx = m_kx[i];
225  for (int j = 0; j < m_Ny; j++) {
226  const double ky = m_ky[j];
227 
228  // FFT(h) * FFT(Gaussian), i.e. FFT(smoothed ice surface elevation)
229  const auto &h_hat = fftw_output(i, j) * G_hat(i, j);
230 
231  double sigma = m_u * kx + m_v * ky;
232 
233  // See equation (6) in [@ref SmithBarstadBonneau2005]
234  std::complex<double> m;
235  {
236  double denominator = sigma * sigma - m_f * m_f;
237 
238  // avoid dividing by zero:
239  if (fabs(denominator) < m_eps) {
240  denominator = denominator >= 0 ? m_eps : -m_eps;
241  }
242 
243  double m_squared = (m_Nm * m_Nm - sigma * sigma) * (kx * kx + ky * ky) / denominator;
244 
245  // Note: this is a *complex* square root.
246  m = std::sqrt(std::complex<double>(m_squared));
247 
248  if (m_squared >= 0.0 and sigma != 0.0) {
249  m *= sigma > 0.0 ? 1.0 : -1.0;
250  }
251  }
252 
253  // avoid dividing by zero:
254  double delta = 0.0;
255  if (std::abs(1.0 - I * m * m_Hw) < m_eps) {
256  delta = m_eps;
257  }
258 
259  // See equation (49) in [@ref SmithBarstad2004] or equation (3) in [@ref
260  // SmithBarstadBonneau2005].
261  auto P_hat = h_hat * (m_Cw * I * sigma /
262  ((1.0 - I * m * m_Hw + delta) *
263  (1.0 + I * sigma * m_tau_c) *
264  (1.0 + I * sigma * m_tau_f)));
265  // Note: sigma, m_tau_c, and m_tau_f are purely real, so the second and the third
266  // factors in the denominator are never zero.
267  //
268  // The first factor (1 - i m H_w) *could* be zero. Here we check if it is and
269  // "regularize" if necessary.
270 
271  fftw_input(i, j) = P_hat;
272  }
273  }
274  }
275 
276  fftw_execute(m_dft_inverse);
277 
278  // get m_fftw_output and put it into m_precipitation
279  get_real_part(m_fftw_output,
280  1.0 / (m_Nx * m_Ny),
281  m_Mx, m_My,
282  m_Nx, m_Ny,
283  m_i0_offset, m_j0_offset,
284  m_precipitation);
285 
286  petsc::VecArray2D p(m_precipitation, m_Mx, m_My);
287  for (int i = 0; i < m_Mx; i++) {
288  for (int j = 0; j < m_My; j++) {
289  p(i, j) += m_background_precip_pre;
290  if (m_truncate) {
291  p(i, j) = std::max(p(i, j), 0.0);
292  }
293  p(i, j) *= m_precip_scale_factor;
294  p(i, j) += m_background_precip_post;
295  }
296  }
297 }
