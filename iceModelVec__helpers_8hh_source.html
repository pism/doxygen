29 //! \brief Computes result = x + alpha * y, where x, y, and z are 2D
30 //! IceModelVecs (scalar or vector).
31 /*!
32  * This implementation tries to be smart about handling IceModelVecs with and
33  * without ghosts and with different stencil widths.
34  *
35  * This template function was written to re-use this code for both
36  * IceModelVec2S and IceModel2V.
37  *
38  * This cannot go into a protected member IceModelVec because
39  * IceModelVec2S::operator() and IceModelVec2V::operator() return different
40  * types.
41  *
42  * Note: this code uses overloaded operators (Vector2::operator*, etc).
43  */
44 template<class V>
45 void add_2d(const V* const x, double alpha,
46  const V* const y,
47  V* const result) {
48 
49  if (x == nullptr or y == nullptr or result == nullptr) {
50  throw RuntimeError(PISM_ERROR_LOCATION, "one of the arguments is null");
51  }
52 
53  int stencil = 0;
54  bool scatter = false;
55  compute_params(x, y, result, stencil, scatter);
56 
57  IceModelVec::AccessList list{x, y, result};
58  for (PointsWithGhosts p(*result->grid(), stencil); p; p.next()) {
59  const int i = p.i(), j = p.j();
60 
61  (*result)(i, j) = (*x)(i, j) + (*y)(i, j) * alpha;
62  }
63 
64  if (scatter) {
65  result->update_ghosts();
66  }
67 
68  result->inc_state_counter();
69 }
