33 void IceModel::view_field(const IceModelVec *field) {
34  unsigned int viewer_size = (unsigned int)m_config->get_number("output.runtime.viewer.size");
35 
36  unsigned int dims = field->ndims();
37 
38  if (dims != 2) {
39  throw RuntimeError(PISM_ERROR_LOCATION, "map-plane views of 3D quantities are not supported.");
40  }
41 
42  if (field->ndof() == 1) { // scalar fields
43  std::string name = field->metadata().get_string("short_name");
44  auto viewer = m_viewers[name];
45 
46  if (not viewer) {
47  m_viewers[name].reset(new petsc::Viewer(m_grid->com, name, viewer_size, m_grid->Lx(), m_grid->Ly()));
48  viewer = m_viewers[name];
49  }
50 
51  const IceModelVec2S *v2d = dynamic_cast<const IceModelVec2S*>(field);
52  if (v2d == NULL) {
53  throw RuntimeError(PISM_ERROR_LOCATION, "get_ndims() returns GRID_2D but dynamic_cast gives a NULL");
54  }
55 
56  v2d->view(viewer, std::shared_ptr<petsc::Viewer>());
57 
58  } else if (field->ndof() == 2) { // vector fields
59  std::string
60  name_1 = field->metadata(0).get_string("short_name"),
61  name_2 = field->metadata(1).get_string("short_name");
62  auto
63  v1 = m_viewers[name_1],
64  v2 = m_viewers[name_2];
65 
66  if (not v1) {
67  m_viewers[name_1].reset(new petsc::Viewer(m_grid->com, name_1, viewer_size, m_grid->Lx(), m_grid->Ly()));
68  v1 = m_viewers[name_1];
69  }
70 
71  if (not v2) {
72  m_viewers[name_2].reset(new petsc::Viewer(m_grid->com, name_2, viewer_size, m_grid->Lx(), m_grid->Ly()));
73  v2 = m_viewers[name_2];
74  }
75 
76  const IceModelVec2 *v2d = dynamic_cast<const IceModelVec2*>(field);
77  if (v2d == NULL) {
78  throw RuntimeError(PISM_ERROR_LOCATION, "get_ndims() returns GRID_2D but dynamic_cast gives a NULL");
79  }
80 
81  v2d->view(v1, v2);
82  }
83 }

89 void IceModel::update_viewers() {
90 
91  auto viewers = set_split(m_config->get_string("output.runtime.viewer.variables"), ',');
92 
93  // map-plane viewers
94  for (auto v : viewers) {
95  if (m_grid->variables().is_available(v)) {
96  this->view_field(m_grid->variables().get(v));
97  } else {
98  // if not found, try to compute:
99  auto diag = m_diagnostics.find(v);
100 
101  if (diag != m_diagnostics.end()) {
102  this->view_field(diag->second->compute().get());
103  }
104  }
105  }
106 }
