33 void IceModel::view_field(const IceModelVec *field) {
34  unsigned int viewer_size = (unsigned int)m_config->get_number("output.runtime.viewer.size");
35 
36  if (field->ndims() != 2) {
37  throw RuntimeError(PISM_ERROR_LOCATION,
38  "map-plane views of 3D quantities are not supported.");
39  }
40 
41  auto name = field->get_name();
42 
43  if (m_viewers[name].empty()) {
44  for (size_t k = 0; k < field->ndof(); ++k) {
45  auto dof_name = field->metadata(k).get_string("short_name");
46  auto v = std::make_shared<petsc::Viewer>(m_grid->com,
47  dof_name, viewer_size,
48  m_grid->Lx(), m_grid->Ly());
49  m_viewers[name].emplace_back(v);
50  }
51  }
52 
53  field->view(m_viewers[name]);
54 }

56 //! Update the runtime graphical viewers.
57 /*!
58 Most viewers are updated by this routine, but some other are updated elsewhere.
59  */
60 void IceModel::update_viewers() {
61 
62  auto viewers = set_split(m_config->get_string("output.runtime.viewer.variables"), ',');
63 
64  // map-plane viewers
65  for (auto v : viewers) {
66  if (m_grid->variables().is_available(v)) {
67  this->view_field(m_grid->variables().get(v));
68  } else {
69  // if not found, try to compute:
70  auto diag = m_diagnostics.find(v);
71 
72  if (diag != m_diagnostics.end()) {
73  this->view_field(diag->second->compute().get());
74  }
75  }
76  }
77 }
